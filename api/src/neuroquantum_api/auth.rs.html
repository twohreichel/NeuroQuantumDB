48    pub fn new_with_path(db_path: &str) -> Result<Self, String> {
49        // Ensure directory exists
50        if let Some(parent) = std::path::Path::new(db_path).parent() {
51            std::fs::create_dir_all(parent)
52                .map_err(|e| format!("Failed to create storage directory: {e}"))?;
53        }
54
55        let storage = ApiKeyStorage::new(db_path)
56            .map_err(|e| format!("Failed to initialize API key storage: {e}"))?;
57
58        // Cleanup expired keys on startup
59        if let Err(e) = storage.cleanup_expired_keys() {
60            warn!("Failed to cleanup expired keys: {}", e);
61        }
62
63        let service = Self {
64            storage,
65            usage_tracking: HashMap::new(),
66            key_generation_tracking: HashMap::new(),
67        };
68
69        info!("üîß AuthService initialized with persistent storage");
70
71        // Check if we have admin keys
72        if !service.has_admin_keys() {
73            warn!("‚ö†Ô∏è  No admin keys found!");
74            warn!("üí° Run 'neuroquantum-api init' to create your first admin key");
75        }
76
77        Ok(service)
78    }

111    pub fn generate_api_key(
112        &mut self,
113        name: String,
114        permissions: Vec<String>,
115        expiry_hours: Option<u32>,
116        rate_limit_per_hour: Option<u32>,
117    ) -> Result<ApiKey, String> {
118        let key = format!("nqdb_{}", Uuid::new_v4().to_string().replace('-', ""));
119
120        // Use lower cost for tests to speed up execution
121        #[cfg(test)]
122        let cost = TEST_BCRYPT_COST;
123        #[cfg(not(test))]
124        let cost = DEFAULT_COST;
125
126        let key_hash = hash(&key, cost).map_err(|e| format!("Failed to hash API key: {e}"))?;
127
128        let expires_at = match expiry_hours {
129            | Some(hours) => Utc::now() + chrono::Duration::hours(i64::from(hours)),
130            | None => Utc::now() + chrono::Duration::days(30), // Default 30 days
131        };
132
133        let api_key = ApiKey {
134            key: key.clone(),
135            name: name.clone(),
136            permissions,
137            expires_at,
138            created_at: Utc::now(),
139            last_used: None,
140            usage_count: 0,
141            rate_limit_per_hour,
142        };
143
144        // Store in persistent database
145        self.storage
146            .store_key(&api_key, &key_hash)
147            .map_err(|e| format!("Failed to store API key: {e}"))?;
148
149        self.usage_tracking.insert(key.clone(), Vec::new());
150
151        info!("üîë Generated new API key: {} for {}", &key[..12], name);
152        Ok(api_key)
153    }

155    pub async fn validate_api_key(&self, key: &str) -> Option<ApiKey> {
156        // Retrieve from persistent storage
157        let (api_key_data, stored_hash) = match self.storage.get_key(key) {
158            | Ok(Some(data)) => data,
159            | Ok(None) => {
160                warn!("API key not found: {}", &key[..8.min(key.len())]);
161                return None;
162            },
163            | Err(e) => {
164                warn!("Failed to retrieve API key: {}", e);
165                return None;
166            },
167        };
168
169        // Verify the key hash for additional security
170        // NOTE: bcrypt::verify is designed to be constant-time and resistant to timing attacks
171        // The comparison takes the same amount of time regardless of where differences occur
172        if !verify(key, &stored_hash).unwrap_or(false) {
173            warn!("API key hash verification failed: {}", &key[..8]);
174            return None;
175        }
176
177        // Check if key is expired
178        if self.is_key_expired(&api_key_data) {
179            warn!("API key expired: {}", &key[..8]);
180            return None;
181        }
182
183        // Check rate limiting
184        if self.is_rate_limited(key) {
185            warn!("Rate limit exceeded for API key: {}", &key[..8]);
186            return None;
187        }
188
189        // Update last used timestamp in database
190        if let Err(e) = self.storage.update_usage(key) {
191            warn!("Failed to update API key usage: {}", e);
192        }
193
194        // Return updated key with current timestamp
195        let mut updated_key = api_key_data;
196        updated_key.last_used = Some(Utc::now());
197        updated_key.usage_count += 1;
198
199        Some(updated_key)
200    }

208    pub fn check_endpoint_permission(&self, api_key: &ApiKey, path: &str) -> bool {
209        // Define permission mappings for different endpoints
210        let required_permission = match path {
211            | p if p.starts_with("/api/v1/neuromorphic") => "neuromorphic",
212            | p if p.starts_with("/api/v1/quantum") => "quantum",
213            | p if p.starts_with("/api/v1/dna") => "dna",
214            | p if p.starts_with("/api/v1/admin") => "admin",
215            | p if p.starts_with("/metrics") => "admin",
216            | p if p.contains("/query") || p.contains("/search") => "read",
217            | p if p.contains("/train") || p.contains("/optimize") || p.contains("/compress") => {
218                "write"
219            },
220            | _ => "read", // Default to read permission
221        };
222
223        api_key
224            .permissions
225            .contains(&required_permission.to_string())
226            || api_key.permissions.contains(&"admin".to_string())
227    }

229    fn is_rate_limited(&self, key: &str) -> bool {
230        // Get API key from storage to check rate limit
231        if let Ok(Some((api_key, _))) = self.storage.get_key(key) {
232            if let Some(rate_limit) = api_key.rate_limit_per_hour {
233                if let Some(usage_times) = self.usage_tracking.get(key) {
234                    let one_hour_ago = Utc::now() - chrono::Duration::hours(1);
235                    let recent_usage = usage_times
236                        .iter()
237                        .filter(|&&time| time > one_hour_ago)
238                        .count();
239
240                    return recent_usage >= rate_limit as usize;
241                }
242            }
243        }
244        false
245    }

278    /// Check if API key generation is rate limited for a given IP address
279    /// Default: Max 5 key generations per hour per IP
280    pub fn check_key_generation_rate_limit(&self, ip_address: &str) -> Result<(), String> {
281        const MAX_GENERATIONS_PER_HOUR: usize = 5;
282
283        if let Some(generation_times) = self.key_generation_tracking.get(ip_address) {
284            let one_hour_ago = Utc::now() - chrono::Duration::hours(1);
285            let recent_generations = generation_times
286                .iter()
287                .filter(|&&time| time > one_hour_ago)
288                .count();
289
290            if recent_generations >= MAX_GENERATIONS_PER_HOUR {
291                warn!(
292                    "‚ö†Ô∏è API key generation rate limit exceeded for IP: {} ({}/{} in last hour)",
293                    ip_address, recent_generations, MAX_GENERATIONS_PER_HOUR
294                );
295                return Err(format!(
296                    "Rate limit exceeded: Maximum {MAX_GENERATIONS_PER_HOUR} key generations per hour. Try again later."
297                ));
298            }
299        }
300        Ok(())
301    }

303    /// Track API key generation attempt from an IP address
304    pub fn track_key_generation(&mut self, ip_address: &str) {
305        let entry = self
306            .key_generation_tracking
307            .entry(ip_address.to_string())
308            .or_default();
309        entry.push(Utc::now());
310
311        // Clean up old entries (older than 24 hours) to prevent memory growth
312        let cutoff = Utc::now() - chrono::Duration::hours(24);
313        entry.retain(|&time| time > cutoff);
314    }
