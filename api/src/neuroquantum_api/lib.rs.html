82/// Application state shared across handlers
83#[derive(Clone)]
84pub struct AppState {
85    pub db: Arc<tokio::sync::RwLock<NeuroQuantumDB>>,
86    pub qsql_engine: Arc<tokio::sync::Mutex<neuroquantum_qsql::QSQLEngine>>,
87    pub auth_service: AuthService,
88    pub jwt_service: JwtService,
89    pub rate_limit_service: RateLimitService,
90    pub websocket_service: Arc<WebSocketService>,
91    pub eeg_service: Arc<RwLock<EEGAuthService>>,
92    pub config: ApiConfig,
93}

95impl AppState {
96    pub async fn new(config: ApiConfig) -> Result<Self> {
97        // Initialize the database using the new builder pattern with compile-time guarantees
98        let db = NeuroQuantumDBBuilder::new()
99            .build()
100            .await
101            .map_err(|e| anyhow::anyhow!("Failed to initialize database: {e}"))?;
102
103        let auth_service = AuthService::new()
104            .map_err(|e| anyhow::anyhow!("Failed to initialize auth service: {e}"))?;
105
106        // Warn if no admin keys exist - database needs initialization
107        if !auth_service.has_admin_keys() {
108            tracing::warn!(
109                "‚ö†Ô∏è  No admin keys found! Run 'neuroquantum-api init' to create the first admin key."
110            );
111            tracing::warn!(
112                "‚ö†Ô∏è  The API server will start but authentication will fail until initialized."
113            );
114        }
115
116        let jwt_service = JwtService::new(config.jwt.secret.as_bytes());
117
118        let rate_limit_config = RateLimitConfig {
119            requests_per_window: config.rate_limit.requests_per_hour,
120            window_size_seconds: 3600,
121            burst_allowance: config.rate_limit.burst_allowance,
122            redis_url: config.redis.as_ref().map(|r| r.url.clone()),
123            fallback_to_memory: true,
124        };
125        let rate_limit_service = RateLimitService::new(rate_limit_config).await?;
126
127        // Get the storage engine Arc from the database for sharing with QSQL engine
128        let storage_engine_arc = db.storage_engine_arc();
129
130        // Wrap the database in Arc<RwLock> for shared access
131        let db_arc = Arc::new(tokio::sync::RwLock::new(db));
132
133        // Initialize QSQL engine with the shared storage engine
134        let qsql_engine = neuroquantum_qsql::QSQLEngine::with_storage(storage_engine_arc)
135            .map_err(|e| anyhow::anyhow!("Failed to initialize QSQL engine: {e}"))?;
136        let qsql_engine_arc = Arc::new(tokio::sync::Mutex::new(qsql_engine));
137
138        tracing::info!("üîó QSQL engine initialized with shared storage engine");
139
140        // Initialize WebSocket service with QSQL engine
141        let ws_config = ConnectionConfig::default();
142        let connection_manager = Arc::new(ConnectionManager::new(ws_config));
143        let pubsub_manager = Arc::new(PubSubManager::new());
144        let websocket_service = Arc::new(WebSocketService::with_qsql_engine(
145            connection_manager,
146            pubsub_manager,
147            qsql_engine_arc.clone(),
148        ));
149
150        // Initialize EEG authentication service with shared state
151        // Default sampling rate 256 Hz is standard for clinical EEG
152        let eeg_service = EEGAuthService::new(256.0)
153            .map_err(|e| anyhow::anyhow!("Failed to initialize EEG auth service: {e}"))?;
154        let eeg_service_arc = Arc::new(RwLock::new(eeg_service));
155        tracing::info!("üß† EEG authentication service initialized");
156
157        Ok(Self {
158            db: db_arc,
159            qsql_engine: qsql_engine_arc,
160            auth_service,
161            jwt_service,
162            rate_limit_service,
163            websocket_service,
164            eeg_service: eeg_service_arc,
165            config,
166        })
167    }
168}

203/// üìä Prometheus metrics endpoint (public - no authentication required)
204pub async fn metrics() -> HttpResponse {
205    match crate::metrics::render_metrics() {
206        | Ok(metrics_text) => HttpResponse::Ok()
207            .content_type("text/plain; version=0.0.4; charset=utf-8")
208            .body(metrics_text),
209        | Err(e) => {
210            tracing::error!("Failed to render metrics: {}", e);
211            HttpResponse::InternalServerError().json(serde_json::json!({
212                "error": "Failed to collect metrics",
213                "details": e.to_string()
214            }))
215        },
216    }
217}

219/// üîç WebSocket handler for real-time communication (requires authentication)
220pub async fn websocket_handler(
221    req: actix_web::HttpRequest,
222    stream: actix_web::web::Payload,
223    state: web::Data<AppState>,
224) -> Result<HttpResponse, actix_web::Error> {
225    use tracing::error;
226    use websocket::ConnectionMetadata;
227
228    // Check if user is authenticated (JWT token should be in extensions from middleware)
229    let extensions = req.extensions();
230    let user_id = if let Some(token) = extensions.get::<error::AuthToken>() {
231        Some(token.sub.clone())
232    } else if let Some(api_key) = extensions.get::<auth::ApiKey>() {
233        Some(api_key.name.clone())
234    } else {
235        return Ok(HttpResponse::Unauthorized().json(serde_json::json!({
236            "error": "Authentication required for WebSocket connection",
237            "code": "WEBSOCKET_AUTH_REQUIRED"
238        })));
239    };
240
241    // Get connection info
242    let remote_addr = req
243        .connection_info()
244        .realip_remote_addr()
245        .unwrap_or("unknown")
246        .to_string();
247    let user_agent = req
248        .headers()
249        .get("user-agent")
250        .and_then(|v| v.to_str().ok())
251        .map(std::string::ToString::to_string);
252
253    // Create connection metadata
254    let mut metadata = ConnectionMetadata::new(remote_addr);
255    metadata.user_id = user_id;
256    metadata.user_agent = user_agent;
257
258    // Handle WebSocket upgrade
259    let (response, session, msg_stream) = actix_ws::handle(&req, stream)?;
260
261    // Record WebSocket connection metrics
262    crate::metrics::record_websocket_connection("connected");
263
264    // Spawn handler task
265    let ws_service = state.websocket_service.clone();
266    actix_web::rt::spawn(async move {
267        if let Err(e) = ws_service
268            .handle_connection(session, msg_stream, metadata)
269            .await
270        {
271            error!("WebSocket connection error: {:?}", e);
272            crate::metrics::record_websocket_connection("error");
273        } else {
274            crate::metrics::record_websocket_connection("disconnected");
275        }
276    });
277
278    Ok(response)
279}

281/// Configure application routes and middleware
282pub fn configure_app(
283    app_state: AppState,
284) -> App<
285    impl actix_web::dev::ServiceFactory<
286        actix_web::dev::ServiceRequest,
287        Config = (),
288        Response = actix_web::dev::ServiceResponse<impl MessageBody>,
289        Error = actix_web::Error,
290        InitError = (),
291    >,
292> {
293    // Create Prometheus metrics
294    let prometheus = PrometheusMetricsBuilder::new("neuroquantum_api")
295        .endpoint("/internal/metrics")
296        .build()
297        .expect("Prometheus metrics builder should succeed");
298
299    let cors_origins = app_state.config.cors.allowed_origins.clone();
300    #[allow(unused_variables)]
301    let tracing_enabled = app_state.config.tracing.enabled;
302
303    App::new()
304        // Add application state
305        .app_data(web::Data::new(app_state.clone()))
306        .app_data(web::Data::new(app_state.db.clone()))
307        .app_data(web::Data::new(app_state.auth_service.clone()))
308        .app_data(web::Data::new(app_state.jwt_service.clone()))
309        .app_data(web::Data::new(app_state.rate_limit_service.clone()))
310        .app_data(web::Data::new(app_state.config))
311        // Add tracing middleware (the middleware itself checks if tracing is enabled)
312        .wrap(middleware::tracing_middleware())
313        // Add other middleware
314        .wrap(prometheus)
315        .wrap(Logger::default())
316        .wrap(Compress::default())
317        .wrap({
318            let mut cors = Cors::default()
319                .allowed_origin("http://localhost:3000")
320                .allowed_origin("http://localhost:8080");
321
322            // Add configured origins
323            for origin in &cors_origins {
324                cors = cors.allowed_origin(origin);
325            }
326
327            cors.allowed_methods(vec!["GET", "POST", "PUT", "DELETE", "PATCH", "OPTIONS"])
328                .allowed_headers(vec![
329                    "Authorization",
330                    "Content-Type",
331                    "X-API-Key",
332                    "X-Request-ID",
333                    "X-Quantum-Level"
334                ])
335                .expose_headers(vec![
336                    "X-RateLimit-Limit",
337                    "X-RateLimit-Remaining",
338                    "X-RateLimit-Reset"
339                ])
340                .max_age(3600)
341        })
342
343        // OpenAPI Documentation
344        .service(
345            SwaggerUi::new("/api-docs/{_:.*}")
346                .url("/api-docs/openapi.json", ApiDoc::openapi())
347        )
348
349        // Health and system endpoints
350        .route("/health", web::get().to(health_check))
351        .route("/metrics", web::get().to(metrics))
352        .route("/ws", web::get().to(websocket_handler))
353
354        // API v1 routes
355        .service(
356            web::scope("/api/v1")
357                // Authentication routes (public)
358                .service(
359                    web::scope("/auth")
360                        .route("/login", web::post().to(handlers::login))
361                        .route("/refresh", web::post().to(handlers::refresh_token))
362                )
363
364                // Protected admin routes
365                .service(
366                    web::scope("/auth")
367                        .wrap(middleware::auth_middleware())
368                        .route("/generate-key", web::post().to(handlers::generate_api_key))
369                        .route("/revoke-key", web::post().to(handlers::revoke_api_key))
370                )
371
372                // Protected API routes (require authentication)
373                .service(
374                    web::scope("")
375                        .wrap(middleware::auth_middleware())
376
377                        // Generic SQL query endpoint
378                        .route("/query", web::post().to(handlers::execute_sql_query))
379
380                        // CRUD Operations
381                        .service(
382                            web::scope("/tables")
383                                .route("", web::post().to(handlers::create_table))
384                                .route("/{table_name}/data", web::post().to(handlers::insert_data))
385                                .route("/{table_name}/query", web::post().to(handlers::query_data))
386                                .route("/{table_name}/data", web::put().to(handlers::update_data))
387                                .route("/{table_name}/data", web::delete().to(handlers::delete_data))
388                        )
389
390                        // Advanced Features
391                        .service(
392                            web::scope("/neural")
393                                .route("/train", web::post().to(handlers::train_neural_network))
394                                .route("/train/{network_id}", web::get().to(handlers::get_training_status))
395                        )
396                        .service(
397                            web::scope("/quantum")
398                                .route("/search", web::post().to(handlers::quantum_search))
399                        )
400                        .service(
401                            web::scope("/dna")
402                                .route("/compress", web::post().to(handlers::compress_dna))
403                                .route("/decompress", web::post().to(handlers::decompress_dna))
404                        )
405
406                        // Biometric Authentication
407                        .service(
408                            web::scope("/biometric")
409                                // New documented endpoints at /biometric/enroll and /biometric/verify
410                                .route("/enroll", web::post().to(handlers::biometric_enroll))
411                                .route("/verify", web::post().to(handlers::biometric_verify))
412                                // EEG-specific endpoints under /biometric/eeg/
413                                .service(
414                                    web::scope("/eeg")
415                                        .route("/enroll", web::post().to(handlers::eeg_enroll))
416                                        .route("/authenticate", web::post().to(handlers::eeg_authenticate))
417                                        .route("/update", web::post().to(handlers::eeg_update_signature))
418                                        .route("/users", web::get().to(handlers::eeg_list_users))
419                                )
420                        )
421
422                        // Monitoring
423                        .service(
424                            web::scope("/stats")
425                                .route("/performance", web::get().to(handlers::get_performance_stats))
426                        )
427
428                        // Index Advisor
429                        .service(
430                            web::scope("/advisor")
431                                .route("/indexes", web::get().to(handlers::get_index_recommendations))
432                                .route("/indexes/statistics", web::delete().to(handlers::clear_index_advisor_statistics))
433                        )
434                )
435        )
436}

438/// Start the HTTP server with the given configuration
439pub async fn start_server(config: ApiConfig) -> Result<()> {
440    let bind_address = format!("{}:{}", config.server.host, config.server.port);
441    let app_state = AppState::new(config.clone()).await?;
442
443    info!("üöÄ Starting NeuroQuantumDB API Server on {}", bind_address);
444    info!(
445        "üìñ API Documentation available at: http://{}/api-docs/",
446        bind_address
447    );
448    info!(
449        "üè• Health check available at: http://{}/health",
450        bind_address
451    );
452    info!("üìä Metrics available at: http://{}/metrics", bind_address);
453
454    HttpServer::new(move || configure_app(app_state.clone()))
455        .bind(&bind_address)?
456        .run()
457        .await?;
458
459    Ok(())
460}
