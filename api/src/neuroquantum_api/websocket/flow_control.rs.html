1//! Flow Control and Backpressure Management
2//!
3//! Implements automatic backpressure handling to prevent overwhelming slow clients
4//! with query results. Monitors buffer sizes and applies throttling strategies.
5//!
6//! # Features
7//!
8//! - **Buffer Monitoring**: Track client receive buffer fill levels
9//! - **Automatic Throttling**: Slow down when buffer exceeds threshold
10//! - **Drop Policies**: Configurable strategies when buffer is full
11//! - **Metrics**: Real-time backpressure statistics
12//! - **Fair Scheduling**: Prevent fast clients from starving slow ones
13//!
14//! # Architecture
15//!
16//! ```text
17//! Query Streamer
18//!      ‚îÇ
19//!      ‚îú‚îÄ‚ñ∫ FlowController
20//!      ‚îÇ        ‚îÇ
21//!      ‚îÇ        ‚îú‚îÄ‚ñ∫ Monitor buffer size
22//!      ‚îÇ        ‚îú‚îÄ‚ñ∫ Apply throttling
23//!      ‚îÇ        ‚îî‚îÄ‚ñ∫ Track metrics
24//!      ‚îÇ             ‚îÇ
25//!      ‚îÇ             ‚ñº
26//!      ‚îî‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚ñ∫ WebSocket Send
27//!                   (with backpressure)
28//! ```

80/// Policy for dropping messages when buffer is full
81#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
82#[serde(rename_all = "snake_case")]
83pub enum DropPolicy {
84    /// Drop oldest messages first (FIFO)
85    DropOldest,
86
87    /// Drop newest messages (reject new)
88    DropNewest,
89
90    /// Block until space is available
91    Block,
92
93    /// Drop all and reset
94    DropAll,
95}

165/// Flow controller for managing backpressure
166pub struct FlowController {
167    config: FlowControlConfig,
168    stats: Arc<RwLock<FlowControlStats>>,
169    last_send: Arc<RwLock<Instant>>,
170}
171
172impl FlowController {
173    /// Create a new flow controller
174    pub fn new(config: FlowControlConfig) -> Self {
175        info!(
176            "‚úÖ FlowController initialized (max_buffer: {}, threshold: {:.1}%)",
177            config.max_buffer_size,
178            config.backpressure_threshold * 100.0
179        );
180
181        let stats = FlowControlStats {
182            max_buffer_size: config.max_buffer_size,
183            ..Default::default()
184        };
185
186        Self {
187            config,
188            stats: Arc::new(RwLock::new(stats)),
189            last_send: Arc::new(RwLock::new(Instant::now())),
190        }
191    }
192
193    /// Check if we can send a message
194    pub async fn can_send(&self, buffer_size: usize) -> bool {
195        let fill_percentage = buffer_size as f32 / self.config.max_buffer_size as f32;
196
197        // Update stats
198        let mut stats = self.stats.write().await;
199        stats.buffer_size = buffer_size;
200        stats.fill_percentage = fill_percentage;
201
202        if fill_percentage >= self.config.pause_threshold {
203            stats.flow_state = FlowState::Paused;
204            false
205        } else if fill_percentage >= self.config.backpressure_threshold {
206            stats.flow_state = FlowState::Throttled;
207            true
208        } else {
209            stats.flow_state = FlowState::Normal;
210            true
211        }
212    }
213
214    /// Calculate delay before sending next batch
215    pub async fn calculate_delay(&self, buffer_size: usize) -> Duration {
216        let fill_percentage = buffer_size as f32 / self.config.max_buffer_size as f32;
217
218        if !self.config.adaptive_throttling {
219            return self.config.min_batch_delay;
220        }
221
222        if fill_percentage < self.config.backpressure_threshold {
223            // No backpressure
224            self.config.min_batch_delay
225        } else if fill_percentage >= self.config.pause_threshold {
226            // Paused - wait longer
227            let mut stats = self.stats.write().await;
228            stats.pause_events += 1;
229            self.config.pause_duration
230        } else {
231            // Throttled - linear interpolation between min and max delay
232            let throttle_factor = (fill_percentage - self.config.backpressure_threshold)
233                / (self.config.pause_threshold - self.config.backpressure_threshold);
234
235            let delay_ms = throttle_factor.mul_add(
236                self.config.max_batch_delay.as_millis() as f32
237                    - self.config.min_batch_delay.as_millis() as f32,
238                self.config.min_batch_delay.as_millis() as f32,
239            );
240
241            let delay = Duration::from_millis(delay_ms as u64);
242
243            let mut stats = self.stats.write().await;
244            stats.throttle_events += 1;
245            stats.current_throttle_delay_ms = delay_ms as u64;
246
247            debug!(
248                "üêå Throttling: buffer {}% full, delay {}ms",
249                fill_percentage * 100.0,
250                delay_ms
251            );
252
253            delay
254        }
255    }
256
257    /// Record a sent message
258    pub async fn record_sent(&self) {
259        let mut stats = self.stats.write().await;
260        stats.messages_sent += 1;
261        *self.last_send.write().await = Instant::now();
262    }
263
264    /// Record a dropped message
265    pub async fn record_dropped(&self) {
266        let mut stats = self.stats.write().await;
267        stats.messages_dropped += 1;
268
269        if stats.messages_dropped % 100 == 0 {
270            warn!(
271                "‚ö†Ô∏è  {} messages dropped due to backpressure",
272                stats.messages_dropped
273            );
274        }
275    }
276
277    /// Handle buffer full condition
278    pub async fn handle_buffer_full(&self, buffer_size: usize) -> FlowAction {
279        let mut stats = self.stats.write().await;
280        stats.buffer_size = buffer_size;
281        stats.flow_state = FlowState::Dropping;
282
283        match self.config.drop_policy {
284            | DropPolicy::DropOldest => {
285                debug!("üìâ Buffer full, dropping oldest messages");
286                FlowAction::DropOldest
287            },
288            | DropPolicy::DropNewest => {
289                debug!("üìâ Buffer full, dropping newest messages");
290                FlowAction::DropNewest
291            },
292            | DropPolicy::Block => {
293                debug!("‚è∏Ô∏è  Buffer full, blocking");
294                stats.pause_events += 1;
295                FlowAction::Block(self.config.pause_duration)
296            },
297            | DropPolicy::DropAll => {
298                warn!("üóëÔ∏è  Buffer full, dropping all messages");
299                FlowAction::DropAll
300            },
301        }
302    }
303
304    /// Wait for appropriate throttle delay
305    pub async fn wait_if_needed(&self, buffer_size: usize) {
306        let delay = self.calculate_delay(buffer_size).await;
307
308        if delay > Duration::from_millis(0) {
309            let pause_start = Instant::now();
310            tokio::time::sleep(delay).await;
311
312            let mut stats = self.stats.write().await;
313            stats.total_pause_time_ms += pause_start.elapsed().as_millis() as u64;
314        }
315    }
316
317    /// Get current statistics
318    pub async fn get_stats(&self) -> FlowControlStats {
319        self.stats.read().await.clone()
320    }
321
322    /// Reset statistics
323    pub async fn reset_stats(&self) {
324        let mut stats = self.stats.write().await;
325        *stats = FlowControlStats {
326            max_buffer_size: stats.max_buffer_size,
327            ..Default::default()
328        };
329        info!("üìä Flow control stats reset");
330    }
331
332    /// Check if connection is healthy (not constantly dropping)
333    pub async fn is_healthy(&self) -> bool {
334        let stats = self.stats.read().await;
335
336        if stats.messages_sent == 0 {
337            return true;
338        }
339
340        let drop_rate = stats.messages_dropped as f32 / stats.messages_sent as f32;
341        drop_rate < 0.1 // Less than 10% drop rate
342    }
343
344    /// Get flow control recommendation
345    pub async fn get_recommendation(&self, buffer_size: usize) -> FlowRecommendation {
346        let fill_percentage = buffer_size as f32 / self.config.max_buffer_size as f32;
347        let stats = self.stats.read().await;
348
349        if fill_percentage >= self.config.pause_threshold {
350            FlowRecommendation {
351                action: "pause".to_string(),
352                reason: "Buffer critically full".to_string(),
353                buffer_fill: fill_percentage,
354                suggested_action: "Wait for client to catch up or increase buffer size".to_string(),
355            }
356        } else if fill_percentage >= self.config.backpressure_threshold {
357            FlowRecommendation {
358                action: "throttle".to_string(),
359                reason: "Buffer filling up".to_string(),
360                buffer_fill: fill_percentage,
361                suggested_action: "Reduce send rate".to_string(),
362            }
363        } else if stats.messages_dropped > 100 {
364            FlowRecommendation {
365                action: "warning".to_string(),
366                reason: format!("{} messages dropped", stats.messages_dropped),
367                buffer_fill: fill_percentage,
368                suggested_action: "Client may be too slow, consider disconnecting".to_string(),
369            }
370        } else {
371            FlowRecommendation {
372                action: "ok".to_string(),
373                reason: "Normal flow".to_string(),
374                buffer_fill: fill_percentage,
375                suggested_action: "Continue".to_string(),
376            }
377        }
378    }
379}

383pub enum FlowAction {

386
387    /// Drop oldest messages
388    DropOldest,

390    /// Drop newest messages
391    DropNewest,
392
393    /// Drop all messages in buffer

396    /// Block for specified duration

409/// Buffered sender with flow control
410pub struct FlowControlledSender<T> {
411    controller: Arc<FlowController>,
412    buffer: Arc<RwLock<Vec<T>>>,
413}
414
415impl<T: Clone> FlowControlledSender<T> {
416    /// Create a new flow-controlled sender
417    #[must_use]
418    pub fn new(config: FlowControlConfig) -> Self {
419        Self {
420            controller: Arc::new(FlowController::new(config)),
421            buffer: Arc::new(RwLock::new(Vec::new())),
422        }
423    }
424
425    /// Send a message with flow control
426    pub async fn send(&self, message: T) -> Result<(), String> {
427        let buffer_size = self.buffer.read().await.len();
428
429        // Check if we can send
430        if !self.controller.can_send(buffer_size).await {
431            // Buffer is paused, wait
432            self.controller.wait_if_needed(buffer_size).await;
433        }
434
435        // Add to buffer
436        let mut buffer = self.buffer.write().await;
437
438        if buffer.len() >= self.controller.config.max_buffer_size {
439            // Handle buffer full
440            let action = self.controller.handle_buffer_full(buffer.len()).await;
441
442            match action {
443                | FlowAction::DropOldest => {
444                    if !buffer.is_empty() {
445                        buffer.remove(0);
446                        self.controller.record_dropped().await;
447                    }
448                },
449                | FlowAction::DropNewest => {
450                    self.controller.record_dropped().await;
451                    return Err("Message dropped (buffer full)".to_string());
452                },
453                | FlowAction::DropAll => {
454                    let dropped = buffer.len();
455                    buffer.clear();
456                    for _ in 0..dropped {
457                        self.controller.record_dropped().await;
458                    }
459                },
460                | FlowAction::Block(duration) => {
461                    drop(buffer); // Release lock while waiting
462                    tokio::time::sleep(duration).await;
463                    // Retry with Box::pin to avoid infinite recursion
464                    return Box::pin(self.send(message)).await;
465                },
466                | FlowAction::Continue => {},
467            }
468        }
469
470        buffer.push(message);
471        self.controller.record_sent().await;
472
473        Ok(())
474    }
475
476    /// Drain messages from buffer
477    pub async fn drain(&self, count: usize) -> Vec<T> {
478        let mut buffer = self.buffer.write().await;
479        let drain_count = count.min(buffer.len());
480        buffer.drain(0..drain_count).collect()
481    }
482
483    /// Get current buffer size
484    pub async fn buffer_size(&self) -> usize {
485        self.buffer.read().await.len()
486    }
487
488    /// Get flow control statistics
489    pub async fn stats(&self) -> FlowControlStats {
490        self.controller.get_stats().await
491    }
492
493    /// Check if sender is healthy
494    pub async fn is_healthy(&self) -> bool {
495        self.controller.is_healthy().await
496    }
