68/// IIR Filter coefficients for digital filtering
69/// Represents a transfer function H(z) = B(z)/A(z) where:
70/// B(z) = b\[0\] + b\[1\]*z^-1 + b\[2\]*z^-2 + ...
71/// A(z) = a\[0\] + a\[1\]*z^-1 + a\[2\]*z^-2 + ...
72#[derive(Debug, Clone)]
73pub struct IIRCoefficients {
74    /// Numerator coefficients (feedforward)
75    pub b: Vec<f32>,
76    /// Denominator coefficients (feedback), a\[0\] is typically 1.0
77    pub a: Vec<f32>,
78}

80/// Cascaded biquad (second-order section) filter structure
81/// This representation is numerically more stable than a single high-order filter
82/// because it avoids coefficient explosion from polynomial multiplication.
83/// Each section is a 2nd-order IIR filter, and they are applied sequentially.
84#[derive(Debug, Clone)]
85pub struct CascadedBiquads {
86    /// Vector of 2nd-order (biquad) filter sections
87    pub sections: Vec<IIRCoefficients>,
88}
89
90impl CascadedBiquads {
91    /// Apply cascaded biquad filter to signal
92    /// Each section is applied sequentially for numerical stability
93    #[must_use]
94    pub fn apply(&self, signal: &[f32]) -> Vec<f32> {
95        let mut result = signal.to_vec();
96        for section in &self.sections {
97            result = section.apply(&result);
98        }
99        result
100    }
101
102    /// Apply zero-phase filtering using cascaded biquads
103    /// Each section applies filtfilt independently for maximum stability
104    #[must_use]
105    pub fn filtfilt(&self, signal: &[f32]) -> Vec<f32> {
106        let mut result = signal.to_vec();
107        for section in &self.sections {
108            result = section.filtfilt(&result);
109        }
110        result
111    }
112}

114impl IIRCoefficients {
115    /// Apply IIR filter to signal using Direct Form II Transposed
116    /// This implementation is numerically stable for higher-order filters
117    #[must_use]
118    pub fn apply(&self, signal: &[f32]) -> Vec<f32> {
119        if signal.is_empty() {
120            return vec![];
121        }
122
123        let n = signal.len();
124        let mut output = vec![0.0f32; n];
125
126        // Filter state (delay line)
127        let num_states = self.b.len().max(self.a.len()) - 1;
128        let mut state = vec![0.0f32; num_states];
129
130        for i in 0..n {
131            // Calculate output
132            let y = self.b[0].mul_add(signal[i], state.first().copied().unwrap_or(0.0));
133
134            // Update states (Direct Form II Transposed)
135            for j in 0..num_states {
136                let b_term = if j + 1 < self.b.len() {
137                    self.b[j + 1] * signal[i]
138                } else {
139                    0.0
140                };
141                let a_term = if j + 1 < self.a.len() {
142                    self.a[j + 1] * y
143                } else {
144                    0.0
145                };
146                let next_state = if j + 1 < num_states {
147                    state[j + 1]
148                } else {
149                    0.0
150                };
151                state[j] = b_term - a_term + next_state;
152            }
153
154            output[i] = y;
155        }
156
157        output
158    }
159
160    /// Apply zero-phase filtering (forward-backward, equivalent to scipy.signal.filtfilt)
161    /// This eliminates phase distortion, which is critical for EEG analysis
162    /// where timing relationships between frequency components matter.
163    #[must_use]
164    pub fn filtfilt(&self, signal: &[f32]) -> Vec<f32> {
165        if signal.len() < 12 {
166            // Need enough samples for edge padding
167            return signal.to_vec();
168        }
169
170        let n = signal.len();
171        // Pad signal at edges to reduce transient effects
172        // Use reflection padding (like scipy's 'odd' mode)
173        let pad_len = (3 * self.b.len().max(self.a.len())).min(n - 1);
174        let mut padded = Vec::with_capacity(n + 2 * pad_len);
175
176        // Reflect left edge: 2*signal[0] - signal[pad_len..1]
177        for i in (1..=pad_len).rev() {
178            padded.push(2.0f32.mul_add(signal[0], -signal[i]));
179        }
180
181        // Original signal
182        padded.extend_from_slice(signal);
183
184        // Reflect right edge: 2*signal[n-1] - signal[n-2..n-pad_len-1]
185        for i in 1..=pad_len {
186            padded.push(2.0f32.mul_add(signal[n - 1], -signal[n - 1 - i]));
187        }
188
189        // Forward pass
190        let forward = self.apply(&padded);
191
192        // Reverse
193        let reversed: Vec<f32> = forward.into_iter().rev().collect();
194
195        // Backward pass
196        let backward = self.apply(&reversed);
197
198        // Reverse again and extract original signal region
199        let result: Vec<f32> = backward.into_iter().rev().collect();
200
201        // Extract the central portion (remove padding)
202        // The original signal starts at index pad_len
203        result.into_iter().skip(pad_len).take(n).collect()
204    }
205}

207/// Butterworth filter design using bilinear transformation
208/// Implements proper IIR filter coefficient calculation for medical-grade EEG filtering
209#[derive(Debug, Clone)]
210pub struct ButterworthDesign {
211    /// Sampling frequency in Hz
212    pub sampling_rate: f32,
213}
214
215impl ButterworthDesign {
216    #[must_use]
217    pub const fn new(sampling_rate: f32) -> Self {
218        Self { sampling_rate }
219    }
220
221    /// Design a 2nd-order lowpass Butterworth filter section (biquad)
222    /// Uses bilinear transformation from analog prototype
223    ///
224    /// The transfer function of a 2nd-order lowpass Butterworth filter is:
225    /// H(s) = `ω_c²` / (s² + √`2·ω_c·s` + `ω_c²`)
226    ///
227    /// After bilinear transformation with frequency prewarping:
228    #[must_use]
229    pub fn lowpass_biquad(&self, cutoff: f32) -> IIRCoefficients {
230        let nyquist = self.sampling_rate / 2.0;
231        // Clamp cutoff to safe range (0.1% to 45% of Nyquist for guaranteed stability)
232        // Higher frequencies require special handling due to bilinear transform behavior
233        let safe_cutoff = cutoff.min(nyquist * 0.45);
234        let normalized_cutoff = (safe_cutoff / nyquist).clamp(0.001, 0.45);
235
236        // Pre-warp the cutoff frequency for bilinear transform
237        let omega = (PI * normalized_cutoff).tan();
238
239        // Butterworth polynomial coefficients for 2nd order
240        let sqrt2 = std::f32::consts::SQRT_2;
241
242        // Compute intermediate values
243        let c = 1.0 / omega; // c = 1/tan(ωc*T/2)
244        let c2 = c * c;
245
246        // Using the standard biquad formulas with normalization
247        let norm = 1.0 / (sqrt2.mul_add(c, 1.0) + c2);
248
249        // Digital filter coefficients (normalized by a0)
250        let b0 = norm;
251        let b1 = 2.0 * norm;
252        let b2 = norm;
253
254        let a1 = 2.0 * (1.0 - c2) * norm;
255        let a2 = (sqrt2.mul_add(-c, 1.0) + c2) * norm;
256
257        IIRCoefficients {
258            b: vec![b0, b1, b2],
259            a: vec![1.0, a1, a2],
260        }
261    }
262
263    /// Design a 2nd-order highpass Butterworth filter section (biquad)
264    ///
265    /// The transfer function of a 2nd-order highpass Butterworth filter is:
266    /// H(s) = s² / (s² + √`2·ω_c·s` + `ω_c²`)
267    #[must_use]
268    pub fn highpass_biquad(&self, cutoff: f32) -> IIRCoefficients {
269        let nyquist = self.sampling_rate / 2.0;
270        // Clamp cutoff to safe range
271        let normalized_cutoff = (cutoff / nyquist).clamp(0.001, 0.45);
272
273        // Pre-warp the cutoff frequency
274        let omega = (PI * normalized_cutoff).tan();
275        let sqrt2 = std::f32::consts::SQRT_2;
276
277        // Compute intermediate values using the standard highpass formulas
278        let c = 1.0 / omega;
279        let c2 = c * c;
280        let norm = 1.0 / (sqrt2.mul_add(c, 1.0) + c2);
281
282        // Digital highpass coefficients (normalized)
283        let b0 = c2 * norm;
284        let b1 = -2.0 * c2 * norm;
285        let b2 = c2 * norm;
286
287        let a1 = 2.0 * (1.0 - c2) * norm;
288        let a2 = (sqrt2.mul_add(-c, 1.0) + c2) * norm;
289
290        IIRCoefficients {
291            b: vec![b0, b1, b2],
292            a: vec![1.0, a1, a2],
293        }
294    }
295
296    /// Design a bandpass Butterworth filter by cascading lowpass and highpass
297    /// The order parameter determines the steepness of the filter rolloff
298    /// Returns cascaded biquad sections for numerical stability
299    #[must_use]
300    pub fn bandpass(&self, low_cutoff: f32, high_cutoff: f32, order: usize) -> CascadedBiquads {
301        // For a bandpass, we cascade highpass (for low cutoff) with lowpass (for high cutoff)
302        // Each 2nd-order section contributes 12 dB/octave rolloff
303
304        let num_sections = order.max(1);
305
306        let mut sections = Vec::new();
307
308        // Add highpass sections for low cutoff
309        for _ in 0..num_sections {
310            sections.push(self.highpass_biquad(low_cutoff));
311        }
312
313        // Add lowpass sections for high cutoff
314        for _ in 0..num_sections {
315            sections.push(self.lowpass_biquad(high_cutoff));
316        }
317
318        CascadedBiquads { sections }
319    }
320
321    /// Design a notch (band-stop) filter using a 2nd-order IIR notch
322    /// Useful for removing power line interference (50Hz or 60Hz)
323    #[must_use]
324    pub fn notch(&self, center_freq: f32, q_factor: f32) -> IIRCoefficients {
325        let nyquist = self.sampling_rate / 2.0;
326        let normalized_freq = (center_freq / nyquist).clamp(0.001, 0.999);
327
328        // Digital notch filter design
329        let omega_0 = PI * normalized_freq;
330        let cos_omega = omega_0.cos();
331        let sin_omega = omega_0.sin();
332        let alpha = sin_omega / (2.0 * q_factor);
333
334        // Normalize by a0
335        let a0 = 1.0 + alpha;
336
337        let b0 = 1.0 / a0;
338        let b1 = -2.0 * cos_omega / a0;
339        let b2 = 1.0 / a0;
340
341        let a1 = -2.0 * cos_omega / a0;
342        let a2 = (1.0 - alpha) / a0;
343
344        IIRCoefficients {
345            b: vec![b0, b1, b2],
346            a: vec![1.0, a1, a2],
347        }
348    }
349}

371/// Digital filter for EEG signal processing with real IIR Butterworth implementation

409        let designer = ButterworthDesign::new(sampling_rate);

417            sampling_rate: Some(sampling_rate),

421    /// Create a notch filter (e.g., for 50/60Hz power line interference)

423    pub fn notch(frequency: f32) -> Self {
424        Self {
425            filter_type: FilterType::Notch,
426            cutoff_low: frequency - 1.0,
427            cutoff_high: frequency + 1.0,
428            order: 2,
429            coefficients: None,
430            sampling_rate: None,
431        }
432    }

434    /// Create a notch filter with pre-computed coefficients
435    #[must_use]
436    pub fn notch_with_rate(frequency: f32, sampling_rate: f32) -> Self {
437        let designer = ButterworthDesign::new(sampling_rate);
438        // Q factor of 30 gives a narrow notch (about 1.7Hz bandwidth at 50Hz)
439        let coefficients = FilterCoefficients::Single(designer.notch(frequency, 30.0));
440        Self {
441            filter_type: FilterType::Notch,
442            cutoff_low: frequency - 1.0,
443            cutoff_high: frequency + 1.0,
444            order: 2,
445            coefficients: Some(coefficients),
446            sampling_rate: Some(sampling_rate),
447        }
448    }

450    /// Apply filter to signal using real IIR Butterworth filtering
451    /// Uses zero-phase filtering (filtfilt) to eliminate phase distortion.
452    /// If the filter was created with a sampling rate (e.g., via `bandpass_with_rate`),
453    /// that rate is used. Otherwise, defaults to 256 Hz (common for EEG).
454    #[must_use]
455    pub fn apply(&self, signal: &[f32]) -> Vec<f32> {
456        // Use stored sampling rate if available, otherwise default to 256 Hz (common EEG rate)
457        let rate = self.sampling_rate.unwrap_or(256.0);
458        self.apply_with_rate(signal, rate)
459    }
460
461    /// Apply filter with explicit sampling rate
462    #[must_use]
463    pub fn apply_with_rate(&self, signal: &[f32], sampling_rate: f32) -> Vec<f32> {
464        if signal.len() < 12 {
465            return signal.to_vec();
466        }
467
468        // Use pre-computed coefficients if available, otherwise compute them
469        let coefficients = if let Some(ref coef) = self.coefficients {
470            coef.clone()
471        } else {
472            let designer = ButterworthDesign::new(sampling_rate);
473            match self.filter_type {
474                | FilterType::Bandpass => FilterCoefficients::Cascaded(designer.bandpass(
475                    self.cutoff_low,
476                    self.cutoff_high,
477                    self.order,
478                )),
479                | FilterType::Notch => FilterCoefficients::Single(
480                    designer.notch(f32::midpoint(self.cutoff_low, self.cutoff_high), 30.0),
481                ),
482                | FilterType::Lowpass => {
483                    FilterCoefficients::Single(designer.lowpass_biquad(self.cutoff_high))
484                },
485                | FilterType::Highpass => {
486                    FilterCoefficients::Single(designer.highpass_biquad(self.cutoff_low))
487                },
488            }
489        };
490
491        // Apply zero-phase filtering for medical-grade signal processing
492        coefficients.filtfilt(signal)
493    }
494}

496/// FFT Analyzer for frequency domain analysis using optimized rustfft (O(n log n))
497#[derive(Debug, Clone)]
498pub struct FFTAnalyzer {
499    sampling_rate: f32,
500}
501
502impl FFTAnalyzer {
503    #[must_use]
504    pub const fn new(sampling_rate: f32) -> Self {
505        Self { sampling_rate }
506    }
507
508    /// Perform FFT and extract power spectrum using rustfft (O(n log n) complexity)
509    ///
510    /// This implementation uses the Cooley-Tukey FFT algorithm via rustfft,
511    /// which is significantly faster than naive DFT for large signals.
512    /// For EEG signals with typical lengths of 512-8192 samples, this provides
513    /// a ~10-100x speedup compared to naive O(n²) DFT.
514    #[must_use]
515    pub fn analyze(&self, signal: &[f32]) -> FrequencySpectrum {
516        let n = signal.len();
517        if n == 0 {
518            return FrequencySpectrum {
519                spectrum: vec![],
520                sampling_rate: self.sampling_rate,
521            };
522        }
523
524        // Create FFT planner (reuses internal buffers for efficiency)
525        let mut planner = FftPlanner::<f32>::new();
526        let fft = planner.plan_fft_forward(n);
527
528        // Convert real signal to complex (imaginary part = 0)
529        let mut buffer: Vec<Complex<f32>> = signal.iter().map(|&x| Complex::new(x, 0.0)).collect();
530
531        // Perform FFT in-place
532        fft.process(&mut buffer);
533
534        // Extract power spectrum (magnitude squared, normalized)
535        // We only need the first half due to Nyquist theorem for real signals
536        let normalization = 1.0 / (n as f32);
537        let power_spectrum: Vec<f32> = buffer
538            .iter()
539            .take(n / 2)
540            .map(|c| c.norm() * normalization)
541            .collect();
542
543        FrequencySpectrum {
544            spectrum: power_spectrum,
545            sampling_rate: self.sampling_rate,
546        }
547    }
548
549    /// Perform FFT with Hann window for improved frequency resolution
550    ///
551    /// The Hann window reduces spectral leakage, which is important for
552    /// accurate EEG band power estimation. This is particularly useful
553    /// for distinguishing closely spaced frequency components in brainwaves.
554    #[must_use]
555    pub fn analyze_windowed(&self, signal: &[f32]) -> FrequencySpectrum {
556        let n = signal.len();
557        if n == 0 {
558            return FrequencySpectrum {
559                spectrum: vec![],
560                sampling_rate: self.sampling_rate,
561            };
562        }
563
564        // Apply Hann window to reduce spectral leakage
565        let windowed: Vec<f32> = signal
566            .iter()
567            .enumerate()
568            .map(|(i, &x)| {
569                let window_coeff = 0.5 * (1.0 - (2.0 * PI * i as f32 / (n - 1) as f32).cos());
570                x * window_coeff
571            })
572            .collect();
573
574        // Create FFT planner
575        let mut planner = FftPlanner::<f32>::new();
576        let fft = planner.plan_fft_forward(n);
577
578        // Convert to complex
579        let mut buffer: Vec<Complex<f32>> =
580            windowed.iter().map(|&x| Complex::new(x, 0.0)).collect();
581
582        // Perform FFT
583        fft.process(&mut buffer);
584
585        // Extract power spectrum with coherent gain compensation for Hann window
586        let normalization = 2.0 / (n as f32); // Factor of 2 for Hann window coherent gain
587        let power_spectrum: Vec<f32> = buffer
588            .iter()
589            .take(n / 2)
590            .map(|c| c.norm() * normalization)
591            .collect();
592
593        FrequencySpectrum {
594            spectrum: power_spectrum,
595            sampling_rate: self.sampling_rate,
596        }
597    }
598
599    /// Extract power in specific frequency band
600    #[must_use]
601    pub fn band_power(&self, spectrum: &FrequencySpectrum, band: FrequencyBand) -> f32 {
602        let (low, high) = band.range();
603        let freq_resolution = self.sampling_rate / (spectrum.spectrum.len() as f32 * 2.0);
604
605        let low_idx = (low / freq_resolution) as usize;
606        let high_idx = (high / freq_resolution) as usize;
607
608        spectrum
609            .spectrum
610            .iter()
611            .skip(low_idx)
612            .take(high_idx - low_idx)
613            .sum::<f32>()
614    }
615}

649impl EEGFeatures {
650    /// Calculate similarity with another feature set (0.0 - 1.0)
651    #[must_use]
652    pub fn similarity(&self, other: &Self) -> f32 {
653        let features_self = self.to_vector();
654        let features_other = other.to_vector();
655
656        // Cosine similarity
657        let dot_product: f32 = features_self
658            .iter()
659            .zip(features_other.iter())
660            .map(|(a, b)| a * b)
661            .sum();
662
663        let norm_self: f32 = features_self.iter().map(|x| x * x).sum::<f32>().sqrt();
664        let norm_other: f32 = features_other.iter().map(|x| x * x).sum::<f32>().sqrt();
665
666        if norm_self == 0.0 || norm_other == 0.0 {
667            return 0.0;
668        }
669
670        (dot_product / (norm_self * norm_other)).clamp(0.0, 1.0)
671    }
672
673    /// Convert features to normalized vector
674    fn to_vector(&self) -> Vec<f32> {
675        vec![
676            self.delta_power,
677            self.theta_power,
678            self.alpha_power,
679            self.beta_power,
680            self.gamma_power,
681            self.alpha_beta_ratio,
682            self.theta_alpha_ratio,
683            self.mean_amplitude,
684            self.std_deviation,
685        ]
686    }
687}

700/// EEG Signal Processor - Main component for processing EEG data
701#[derive(Debug, Clone)]
702pub struct EEGProcessor {
703    sampling_rate: f32,
704    filters: Vec<DigitalFilter>,
705    feature_extractor: FFTAnalyzer,
706    min_samples: usize,
707    signal_quality_threshold: f32,
708}
709
710impl EEGProcessor {
711    /// Create new EEG processor with specified sampling rate
712    pub fn new(sampling_rate: f32) -> Result<Self, EEGError> {
713        if !(128.0..=2048.0).contains(&sampling_rate) {
714            return Err(EEGError::InvalidSamplingRate(sampling_rate));
715        }
716
717        // Create pre-computed Butterworth filters for optimal performance
718        // Using proper IIR filters with zero-phase filtering for medical-grade EEG processing
719        let filters = vec![
720            // Notch filters for power line interference removal (narrow band rejection)
721            DigitalFilter::notch_with_rate(50.0, sampling_rate), // Europe (50Hz)
722            DigitalFilter::notch_with_rate(60.0, sampling_rate), // US (60Hz)
723            // Bandpass filter to isolate EEG frequency range (0.5-100Hz)
724            // Using 2nd-order Butterworth for gentle rolloff that preserves signal shape
725            DigitalFilter::bandpass_with_rate(0.5, 100.0, 2, sampling_rate),
726        ];
727
728        Ok(Self {
729            sampling_rate,
730            filters,
731            feature_extractor: FFTAnalyzer::new(sampling_rate),
732            min_samples: (sampling_rate * 2.0) as usize, // Minimum 2 seconds of data
733            signal_quality_threshold: 50.0, // 50% minimum quality (adjusted for synthetic signals)
734        })
735    }
736
737    /// Process raw EEG data and extract features
738    pub fn process_raw_eeg(&self, raw_data: &[f32]) -> Result<EEGFeatures, EEGError> {
739        // 1. Validate input data
740        if raw_data.len() < self.min_samples {
741            return Err(EEGError::InsufficientData {
742                got: raw_data.len(),
743                needed: self.min_samples,
744            });
745        }
746
747        debug!(
748            "Processing {} EEG samples at {}Hz",
749            raw_data.len(),
750            self.sampling_rate
751        );
752
753        // 2. Noise reduction and filtering using real IIR Butterworth filters
754        // Each filter uses zero-phase filtering (filtfilt) to eliminate phase distortion
755        let mut filtered_signal = raw_data.to_vec();
756        for filter in &self.filters {
757            filtered_signal = filter.apply_with_rate(&filtered_signal, self.sampling_rate);
758        }
759
760        // 3. Calculate signal quality
761        let signal_quality = self.calculate_signal_quality(raw_data, &filtered_signal);
762        if signal_quality < self.signal_quality_threshold {
763            return Err(EEGError::PoorSignalQuality(signal_quality));
764        }
765
766        // 4. Frequency domain analysis (FFT)
767        let spectrum = self.feature_extractor.analyze(&filtered_signal);
768
769        // 5. Feature extraction (band powers)
770        let delta_power = self
771            .feature_extractor
772            .band_power(&spectrum, FrequencyBand::Delta);
773        let theta_power = self
774            .feature_extractor
775            .band_power(&spectrum, FrequencyBand::Theta);
776        let alpha_power = self
777            .feature_extractor
778            .band_power(&spectrum, FrequencyBand::Alpha);
779        let beta_power = self
780            .feature_extractor
781            .band_power(&spectrum, FrequencyBand::Beta);
782        let gamma_power = self
783            .feature_extractor
784            .band_power(&spectrum, FrequencyBand::Gamma);
785
786        // 6. Statistical features
787        let mean_amplitude = filtered_signal.iter().sum::<f32>() / filtered_signal.len() as f32;
788        let variance = filtered_signal
789            .iter()
790            .map(|x| (x - mean_amplitude).powi(2))
791            .sum::<f32>()
792            / filtered_signal.len() as f32;
793        let std_deviation = variance.sqrt();
794
795        // 7. Calculate ratios (important for user identification)
796        let alpha_beta_ratio = if beta_power > 0.0 {
797            alpha_power / beta_power
798        } else {
799            0.0
800        };
801
802        let theta_alpha_ratio = if alpha_power > 0.0 {
803            theta_power / alpha_power
804        } else {
805            0.0
806        };
807
808        Ok(EEGFeatures {
809            delta_power,
810            theta_power,
811            alpha_power,
812            beta_power,
813            gamma_power,
814            mean_amplitude,
815            std_deviation,
816            signal_quality,
817            alpha_beta_ratio,
818            theta_alpha_ratio,
819            timestamp: chrono::Utc::now(),
820        })
821    }
822
823    /// Extract unique user signature from EEG features
824    pub fn extract_user_signature(
825        &self,
826        user_id: String,
827        eeg_features: &EEGFeatures,
828    ) -> Result<UserSignature, EEGError> {
829        // Validate feature quality
830        if eeg_features.signal_quality < self.signal_quality_threshold {
831            return Err(EEGError::PoorSignalQuality(eeg_features.signal_quality));
832        }
833
834        info!("🧠 Extracting brain signature for user: {}", user_id);
835
836        Ok(UserSignature {
837            user_id,
838            feature_template: eeg_features.clone(),
839            created_at: chrono::Utc::now(),
840            last_updated: chrono::Utc::now(),
841            enrollment_count: 1,
842            authentication_threshold: 0.85, // 85% similarity required by default
843        })
844    }
845
846    /// Calculate signal quality (0-100%)
847    fn calculate_signal_quality(&self, raw: &[f32], filtered: &[f32]) -> f32 {
848        if raw.is_empty() || filtered.is_empty() {
849            return 0.0;
850        }
851
852        // Calculate signal power
853        let signal_power: f32 = filtered.iter().map(|x| x * x).sum::<f32>() / filtered.len() as f32;
854
855        // Calculate noise
856        let noise: Vec<f32> = raw
857            .iter()
858            .zip(filtered.iter())
859            .map(|(r, f)| r - f)
860            .collect();
861        let noise_power: f32 = noise.iter().map(|x| x * x).sum::<f32>() / noise.len() as f32;
862
863        if noise_power == 0.0 || signal_power < 0.0001 {
864            return 100.0;
865        }
866
867        // Calculate SNR (Signal-to-Noise Ratio) in dB
868        let snr_db = 10.0 * (signal_power / noise_power.max(0.0001)).log10();
869
870        // Convert SNR to quality percentage (0-100%)
871        // Adjusted: Good EEG typically has SNR > 10dB for synthetic signals
872        // Map 0dB = 50%, 10dB = 75%, 20dB+ = 100%
873        let quality = if snr_db < 0.0 {
874            (snr_db / 10.0).mul_add(50.0, 50.0) // 0-50% for negative SNR
875        } else if snr_db < 20.0 {
876            (snr_db / 20.0).mul_add(50.0, 50.0) // 50-100% for 0-20dB
877        } else {
878            100.0
879        };
880
881        quality.clamp(0.0, 100.0)
882    }
883}

885/// EEG-based authentication service
886#[derive(Debug, Clone)]
887pub struct EEGAuthService {
888    processor: EEGProcessor,
889    user_signatures: HashMap<String, UserSignature>,
890    max_enrollment_samples: usize,
891}
892
893impl EEGAuthService {
894    /// Create new EEG authentication service
895    pub fn new(sampling_rate: f32) -> Result<Self, EEGError> {
896        Ok(Self {
897            processor: EEGProcessor::new(sampling_rate)?,
898            user_signatures: HashMap::new(),
899            max_enrollment_samples: 5,
900        })
901    }
902
903    /// Enroll a new user with their EEG signature
904    pub fn enroll_user(
905        &mut self,
906        user_id: String,
907        raw_eeg: &[f32],
908    ) -> Result<UserSignature, EEGError> {
909        let features = self.processor.process_raw_eeg(raw_eeg)?;
910        let signature = self
911            .processor
912            .extract_user_signature(user_id.clone(), &features)?;
913
914        self.user_signatures
915            .insert(user_id.clone(), signature.clone());
916        info!("✅ User enrolled successfully: {}", user_id);
917
918        Ok(signature)
919    }
920
921    /// Update user signature with additional EEG sample (improves accuracy)
922    pub fn update_signature(&mut self, user_id: &str, raw_eeg: &[f32]) -> Result<(), EEGError> {
923        let features = self.processor.process_raw_eeg(raw_eeg)?;
924
925        if let Some(signature) = self.user_signatures.get_mut(user_id) {
926            if signature.enrollment_count < self.max_enrollment_samples {
927                // Average the features for better template
928                signature.feature_template.delta_power =
929                    f32::midpoint(signature.feature_template.delta_power, features.delta_power);
930                signature.feature_template.theta_power =
931                    f32::midpoint(signature.feature_template.theta_power, features.theta_power);
932                signature.feature_template.alpha_power =
933                    f32::midpoint(signature.feature_template.alpha_power, features.alpha_power);
934                signature.feature_template.beta_power =
935                    f32::midpoint(signature.feature_template.beta_power, features.beta_power);
936                signature.feature_template.gamma_power =
937                    f32::midpoint(signature.feature_template.gamma_power, features.gamma_power);
938
939                signature.enrollment_count += 1;
940                signature.last_updated = chrono::Utc::now();
941
942                debug!(
943                    "Updated signature for {} (enrollment count: {})",
944                    user_id, signature.enrollment_count
945                );
946            }
947            Ok(())
948        } else {
949            Err(EEGError::SignatureNotFound(user_id.to_string()))
950        }
951    }
952
953    /// Authenticate user with EEG data
954    pub fn authenticate(
955        &self,
956        user_id: &str,
957        raw_eeg: &[f32],
958    ) -> Result<AuthenticationResult, EEGError> {
959        let features = self.processor.process_raw_eeg(raw_eeg)?;
960
961        let signature = self
962            .user_signatures
963            .get(user_id)
964            .ok_or_else(|| EEGError::SignatureNotFound(user_id.to_string()))?;
965
966        let similarity = features.similarity(&signature.feature_template);
967
968        // Use constant-time threshold check to prevent timing attacks
969        // This prevents attackers from learning how close they are to the threshold
970        let authenticated =
971            constant_time_threshold_check(similarity, signature.authentication_threshold);
972
973        if authenticated {
974            info!(
975                "🔐 EEG authentication successful for {}: {:.2}% match",
976                user_id,
977                similarity * 100.0
978            );
979        } else {
980            warn!(
981                "❌ EEG authentication failed for {}: {:.2}% match (threshold: {:.2}%)",
982                user_id,
983                similarity * 100.0,
984                signature.authentication_threshold * 100.0
985            );
986        }
987
988        Ok(AuthenticationResult {
989            user_id: user_id.to_string(),
990            authenticated,
991            similarity_score: similarity,
992            threshold: signature.authentication_threshold,
993            timestamp: chrono::Utc::now(),
994        })
995    }

1000        self.user_signatures.get(user_id)
1001    }

1005        self.user_signatures.remove(user_id).is_some()
1006    }

1008    /// List all enrolled users

1011        self.user_signatures.keys().cloned().collect()
1012    }
