2//! Prometheus metrics collection for `NeuroQuantumDB` API
3//!
4//! This module provides comprehensive metrics for monitoring API performance,
5//! database operations, and system health.
6//!
7//! # Metric Registration
8//!
9//! All metrics are registered at startup using static initialization.
10//! If metric registration fails, the application should fail fast during startup
11//! rather than continue with incomplete monitoring. This is an acceptable use
12//! of `.expect()` as it occurs before the application serves any requests.
13
14#![allow(clippy::expect_used)] // Startup-only metric registration - fail-fast is acceptable
15

282        .with_label_values(&[status])

284

286        ACTIVE_CONNECTIONS.inc();

288        ACTIVE_CONNECTIONS.dec();
289    }

330/// Update system metrics (memory, temperature, etc.)
331pub fn update_system_metrics() {
332    use sysinfo::System;
333
334    let mut sys = System::new_all();
335    sys.refresh_all();
336
337    // Update memory usage
338    let used_memory = sys.used_memory();
339    MEMORY_USAGE_BYTES.set(used_memory as f64);
340
341    // Update system temperature (if available)
342    // Note: On Raspberry Pi, this reads from /sys/class/thermal/thermal_zone0/temp
343    #[cfg(target_os = "linux")]
344    {
345        if let Ok(temp_str) = std::fs::read_to_string("/sys/class/thermal/thermal_zone0/temp") {
346            if let Ok(temp_millidegrees) = temp_str.trim().parse::<f64>() {
347                let temp_celsius = temp_millidegrees / 1000.0;
348                SYSTEM_TEMPERATURE_CELSIUS.set(temp_celsius);
349            }
350        }
351    }
352}

354/// Get the server uptime in seconds
355static START_TIME: std::sync::LazyLock<SystemTime> = std::sync::LazyLock::new(SystemTime::now);
356
357pub fn get_uptime_seconds() -> f64 {
358    START_TIME.elapsed().map(|d| d.as_secs_f64()).unwrap_or(0.0)
359}

361/// Get system temperature in Celsius (Linux-specific, returns None on other platforms)
362#[must_use]
363#[allow(clippy::missing_const_for_fn)] // Cannot be const: performs I/O operations
364pub fn get_system_temperature() -> Option<f32> {
365    #[cfg(target_os = "linux")]
366    {
367        std::fs::read_to_string("/sys/class/thermal/thermal_zone0/temp")
368            .ok()
369            .and_then(|s| s.trim().parse::<f32>().ok())
370            .map(|t| t / 1000.0)
371    }
372    #[cfg(not(target_os = "linux"))]
373    {
374        None
375    }
376}

378/// Get average query response time in milliseconds from histogram metrics
379pub fn get_average_query_time_ms() -> f32 {
380    let sum = QUERY_RESPONSE_TIME_SECONDS
381        .with_label_values(&["crud"])
382        .get_sample_sum();
383    let count = QUERY_RESPONSE_TIME_SECONDS
384        .with_label_values(&["crud"])
385        .get_sample_count();
386
387    if count > 0 {
388        (sum / count as f64 * 1000.0) as f32
389    } else {
390        1.0 // Default 1ms if no queries recorded
391    }
392}

394/// Render all metrics in Prometheus text format
395pub fn render_metrics() -> Result<String, Box<dyn std::error::Error>> {
396    // Update system metrics before rendering
397    update_system_metrics();
398
399    let encoder = TextEncoder::new();
400    let metric_families = prometheus::gather();
401    let mut buffer = Vec::new();
402    encoder.encode(&metric_families, &mut buffer)?;
403
404    Ok(String::from_utf8(buffer)?)
