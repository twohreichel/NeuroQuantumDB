97/// Lazily initialized exponentiation and logarithm tables.
98pub fn get_exp_log() -> &'static ExpLog {
99    #[cfg(feature = "std")]
100    {
101        static EXP_LOG: LazyLock<ExpLog> = LazyLock::new(initialize_exp_log);
102        &EXP_LOG
103    }
104    #[cfg(not(feature = "std"))]
105    {
106        static EXP_LOG: OnceBox<ExpLog> = OnceBox::new();
107        EXP_LOG.get_or_init(|| Box::new(initialize_exp_log()))
108    }
109}

170/// Calculates `x * log_m` using [`Exp`] and [`Log`] tables.
171#[inline(always)]
172pub fn mul(x: GfElement, log_m: GfElement, exp: &Exp, log: &Log) -> GfElement {
173    if x == 0 {
174        0
175    } else {
176        exp[utils::add_mod(log[x as usize], log_m) as usize]
177    }
178}

183#[allow(clippy::needless_range_loop)]
184fn initialize_exp_log() -> ExpLog {
185    let mut exp = Box::new([0; GF_ORDER]);
186    let mut log = Box::new([0; GF_ORDER]);
187
188    // GENERATE LFSR TABLE
189
190    let mut state = 1;
191    for i in 0..GF_MODULUS {
192        exp[state] = i;
193        state <<= 1;
194        if state >= GF_ORDER {
195            state ^= GF_POLYNOMIAL;
196        }
197    }
198    exp[0] = GF_MODULUS;
199
200    // CONVERT TO CANTOR BASIS
201
202    log[0] = 0;
203    for i in 0..GF_BITS {
204        let width = 1usize << i;
205        for j in 0..width {
206            log[j + width] = log[j] ^ CANTOR_BASIS[i];
207        }
208    }
209
210    for i in 0..GF_ORDER {
211        log[i] = exp[log[i] as usize];
212    }
213
214    for i in 0..GF_ORDER {
215        exp[log[i] as usize] = i as GfElement;
216    }
217
218    exp[GF_MODULUS as usize] = exp[0];
219
220    ExpLog { exp, log }
221}

223fn initialize_log_walsh() -> Box<LogWalsh> {
224    let log = get_exp_log().log.as_slice();
225
226    let mut log_walsh: Box<LogWalsh> = Box::new([0; GF_ORDER]);
227
228    log_walsh.copy_from_slice(log);
229    log_walsh[0] = 0;
230    fwht::fwht(log_walsh.as_mut(), GF_ORDER);
231
232    log_walsh
233}

235fn initialize_mul16() -> Box<Mul16> {
236    let exp = &get_exp_log().exp;
237    let log = &get_exp_log().log;
238    let mut mul16 = vec![[[0; 16]; 4]; GF_ORDER];
239
240    for log_m in 0..=GF_MODULUS {
241        let lut = &mut mul16[log_m as usize];
242        for i in 0..16 {
243            lut[0][i] = mul(i as GfElement, log_m, exp, log);
244            lut[1][i] = mul((i << 4) as GfElement, log_m, exp, log);
245            lut[2][i] = mul((i << 8) as GfElement, log_m, exp, log);
246            lut[3][i] = mul((i << 12) as GfElement, log_m, exp, log);
247        }
248    }
249
250    mul16.into_boxed_slice().try_into().unwrap()
251}

253fn initialize_mul128() -> Box<Mul128> {
254    // Based on:
255    // https://github.com/catid/leopard/blob/22ddc7804998d31c8f1a2617ee720e063b1fa6cd/LeopardFF16.cpp#L375
256    let exp = &get_exp_log().exp;
257    let log = &get_exp_log().log;
258
259    let mut mul128 = vec![
260        Multiply128lutT {
261            lo: [0; 4],
262            hi: [0; 4],
263        };
264        GF_ORDER
265    ];
266
267    for log_m in 0..=GF_MODULUS {
268        for i in 0..=3 {
269            let mut prod_lo = [0u8; 16];
270            let mut prod_hi = [0u8; 16];
271            for x in 0..16 {
272                let prod = mul((x << (i * 4)) as GfElement, log_m, exp, log);
273                prod_lo[x] = prod as u8;
274                prod_hi[x] = (prod >> 8) as u8;
275            }
276            mul128[log_m as usize].lo[i] = u128::from_le_bytes(prod_lo);
277            mul128[log_m as usize].hi[i] = u128::from_le_bytes(prod_hi);
278        }
279    }
280
281    mul128.into_boxed_slice().try_into().unwrap()
282}

284#[allow(clippy::needless_range_loop)]
285fn initialize_skew() -> Box<Skew> {
286    let exp = &get_exp_log().exp;
287    let log = &get_exp_log().log;
288
289    let mut skew = Box::new([0; GF_MODULUS as usize]);
290
291    let mut temp = [0; GF_BITS - 1];
292
293    for i in 1..GF_BITS {
294        temp[i - 1] = 1 << i;
295    }
296
297    for m in 0..GF_BITS - 1 {
298        let step: usize = 1 << (m + 1);
299
300        skew[(1 << m) - 1] = 0;
301
302        for i in m..GF_BITS - 1 {
303            let s: usize = 1 << (i + 1);
304            let mut j = (1 << m) - 1;
305            while j < s {
306                skew[j + s] = skew[j] ^ temp[i];
307                j += step;
308            }
309        }
310
311        temp[m] = GF_MODULUS - log[mul(temp[m], log[(temp[m] ^ 1) as usize], exp, log) as usize];
312
313        for i in m + 1..GF_BITS - 1 {
314            let sum = utils::add_mod(log[(temp[i] ^ 1) as usize], temp[m]);
315            temp[i] = mul(temp[i], sum, exp, log);
316        }
317    }
318
319    for i in 0..GF_MODULUS as usize {
320        skew[i] = log[skew[i] as usize];
321    }
322
323    skew
