5//! This crate provides to an interface into the linux `procfs` filesystem, usually mounted at
6//! `/proc`.
7//!
8//! This is a pseudo-filesystem which is available on most every linux system and provides an
9//! interface to kernel data structures.
10//!
11//!
12//! # Kernel support
13//!
14//! Not all fields/data are available in each kernel.  Some fields were added in specific kernel
15//! releases, and other fields are only present in certain kernel configuration options are
16//! enabled.  These are represented as `Option` fields in this crate.
17//!
18//! This crate aims to support all 2.6 kernels (and newer).  WSL2 is also supported.
19//!
20//! # Documentation
21//!
22//! In almost all cases, the documentation is taken from the
23//! [`proc.5`](http://man7.org/linux/man-pages/man5/proc.5.html) manual page.  This means that
24//! sometimes the style of writing is not very "rusty", or may do things like reference related files
25//! (instead of referencing related structs).  Contributions to improve this are welcome.
26//!
27//! # Panicing
28//!
29//! While previous versions of the library could panic, this current version aims to be panic-free
30//! in a many situations as possible.  Whenever the procfs crate encounters a bug in its own
31//! parsing code, it will return an [`InternalError`](enum.ProcError.html#variant.InternalError) error.  This should be considered a
32//! bug and should be [reported](https://github.com/eminence/procfs).  If you encounter a panic,
33//! please report that as well.
34//!
35//! # Cargo features
36//!
37//! The following cargo features are available:
38//!
39//! * `chrono` -- Default.  Optional.  This feature enables a few methods that return values as `DateTime` objects.
40//! * `flate2` -- Default.  Optional.  This feature enables parsing gzip compressed `/proc/config.gz` file via the `procfs::kernel_config` method.
41//! * `backtrace` -- Optional.  This feature lets you get a stack trace whenever an `InternalError` is raised.
42//!
43//! # Examples
44//!
45//! Examples can be found in the various modules shown below, or in the
46//! [examples](https://github.com/eminence/procfs/tree/master/procfs/examples) folder of the code repository.
47//!

138macro_rules! wrap_io_error {
139    ($path:expr, $expr:expr) => {
140        match $expr {
141            Ok(v) => Ok(v),
142            Err(e) => {
143                let kind = e.kind();
144                Err(::std::io::Error::new(
145                    kind,
146                    crate::IoErrorWrapper {
147                        path: $path.to_owned(),
148                        inner: e.into(),
149                    },
150                ))
151            }
152        }
153    };
154}

156pub(crate) fn read_file<P: AsRef<Path>>(path: P) -> ProcResult<String> {
157    std::fs::read_to_string(path.as_ref())
158        .map_err(|e| e.into())
159        .error_path(path.as_ref())
160}
161
162pub(crate) fn write_file<P: AsRef<Path>, T: AsRef<[u8]>>(path: P, buf: T) -> ProcResult<()> {
163    std::fs::write(path.as_ref(), buf)
164        .map_err(|e| e.into())
165        .error_path(path.as_ref())
166}

168pub(crate) fn read_value<P, T, E>(path: P) -> ProcResult<T>
169where
170    P: AsRef<Path>,
171    T: FromStr<Err = E>,
172    ProcError: From<E>,
173{
174    read_file(path).and_then(|s| s.trim().parse().map_err(ProcError::from))
175}
176
177pub(crate) fn write_value<P: AsRef<Path>, T: fmt::Display>(path: P, value: T) -> ProcResult<()> {
178    write_file(path, value.to_string().as_bytes())
179}

207/// A wrapper around a `File` that remembers the name of the path
208struct FileWrapper {
209    inner: File,
210    path: PathBuf,
211}
212
213impl FileWrapper {
214    fn open<P: AsRef<Path>>(path: P) -> Result<FileWrapper, io::Error> {
215        let p = path.as_ref();
216        let f = wrap_io_error!(p, File::open(p))?;
217        Ok(FileWrapper {
218            inner: f,
219            path: p.to_owned(),
220        })
221    }
222    fn open_at<P, Q, Fd: AsFd>(root: P, dirfd: Fd, path: Q) -> Result<FileWrapper, io::Error>
223    where
224        P: AsRef<Path>,
225        Q: AsRef<Path>,
226    {
227        use rustix::fs::{Mode, OFlags};
228
229        let p = root.as_ref().join(path.as_ref());
230        let fd = wrap_io_error!(
231            p,
232            rustix::fs::openat(dirfd, path.as_ref(), OFlags::RDONLY | OFlags::CLOEXEC, Mode::empty())
233        )?;
234        Ok(FileWrapper {
235            inner: File::from(fd),
236            path: p,
237        })
238    }
239
240    /// Returns the inner file
241    fn inner(self) -> File {
242        self.inner
243    }
244}
245
246impl Read for FileWrapper {
247    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
248        wrap_io_error!(self.path, self.inner.read(buf))
249    }
250    fn read_to_end(&mut self, buf: &mut Vec<u8>) -> io::Result<usize> {
251        wrap_io_error!(self.path, self.inner.read_to_end(buf))
252    }
253    fn read_to_string(&mut self, buf: &mut String) -> io::Result<usize> {
254        wrap_io_error!(self.path, self.inner.read_to_string(buf))
255    }
256    fn read_exact(&mut self, buf: &mut [u8]) -> io::Result<()> {
257        wrap_io_error!(self.path, self.inner.read_exact(buf))
258    }
259}
260
261impl Seek for FileWrapper {
262    fn seek(&mut self, pos: io::SeekFrom) -> io::Result<u64> {
263        wrap_io_error!(self.path, self.inner.seek(pos))
264    }
265}

275/// The boot time of the system, as a `DateTime` object.
276///
277/// This is calculated from `/proc/stat`.
278///
279/// This function requires the "chrono" features to be enabled (which it is by default).
280#[cfg(feature = "chrono")]
281pub fn boot_time() -> ProcResult<DateTime<chrono::Local>> {
282    use chrono::TimeZone;
283    let secs = boot_time_secs()?;
284
285    let date_time = expect!(chrono::Local.timestamp_opt(secs as i64, 0).single());
286
287    Ok(date_time)
288}

290/// The boottime of the system, in seconds since the epoch
291///
292/// This is calculated from `/proc/stat`.
293///
294#[cfg_attr(
295    not(feature = "chrono"),
296    doc = "If you compile with the optional `chrono` feature, you can use the `boot_time()` method to get the boot time as a `DateTime` object."
297)]
298#[cfg_attr(
299    feature = "chrono",
300    doc = "See also [boot_time()] to get the boot time as a `DateTime`"
301)]
302pub fn boot_time_secs() -> ProcResult<u64> {
303    BOOT_TIME.with(|x| {
304        let mut btime = x.borrow_mut();
305        if let Some(btime) = *btime {
306            Ok(btime)
307        } else {
308            // KernelStats doesn't call `boot_time_secs()`, so it's safe to call
309            // KernelStats::current() (which uses the local system info).
310            let stat = KernelStats::current()?;
311            *btime = Some(stat.btime);
312            Ok(stat.btime)
313        }
314    })
315}

317thread_local! {
318    static BOOT_TIME : std::cell::RefCell<Option<u64>> = std::cell::RefCell::new(None);
319}

335    /// Returns a configuration options used to build the currently running kernel
336    ///
337    /// If CONFIG_KCONFIG_PROC is available, the config is read from `/proc/config.gz`.
338    /// Else look in `/boot/config-$(uname -r)` or `/boot/config` (in that order).
339    ///
340    /// # Notes
341    /// Reading the compress `/proc/config.gz` is only supported if the `flate2` feature is enabled
342    /// (which it is by default).
343    #[cfg_attr(feature = "flate2", doc = "The flate2 feature is currently enabled")]
344    #[cfg_attr(not(feature = "flate2"), doc = "The flate2 feature is NOT currently enabled")]
345    fn current() -> ProcResult<Self> {
346        let reader: Box<dyn Read> = if Path::new(PROC_CONFIG_GZ).exists() && cfg!(feature = "flate2") {
347            #[cfg(feature = "flate2")]
348            {
349                let file = FileWrapper::open(PROC_CONFIG_GZ)?;
350                let decoder = flate2::read::GzDecoder::new(file);
351                Box::new(decoder)
352            }
353            #[cfg(not(feature = "flate2"))]
354            {
355                unreachable!("flate2 feature not enabled")
356            }
357        } else {
358            let kernel = rustix::system::uname();
359
360            let filename = format!("{}-{}", BOOT_CONFIG, kernel.release().to_string_lossy());
361
362            match FileWrapper::open(filename) {
363                Ok(file) => Box::new(BufReader::new(file)),
364                Err(e) => match e.kind() {
365                    io::ErrorKind::NotFound => {
366                        let file = FileWrapper::open(BOOT_CONFIG)?;
367                        Box::new(file)
368                    }
369                    _ => return Err(e.into()),
370                },
371            }
372        };
373
374        Self::from_read(reader)
375    }
376}

724    fn test_meminfo() {
725        // TRAVIS
726        // we don't have access to the kernel_config on travis, so skip that test there

728            Ok(ref s) if s == "true" => return,
729            _ => {}
730        }
731
732        let kernel = KernelVersion::current().unwrap();
733        let config = KernelConfig::current().ok();
734
735        let meminfo = Meminfo::current().unwrap();
736        println!("{:#?}", meminfo);
737
738        // for the fields that are only present in some kernel versions, make sure our
739        // actual kernel agrees
740
741        if kernel >= KernelVersion::new(3, 14, 0) {
742            assert!(meminfo.mem_available.is_some());
743        }
744
745        if kernel >= KernelVersion::new(2, 6, 28) {
746            assert!(meminfo.active_anon.is_some());
747            assert!(meminfo.inactive_anon.is_some());
748            assert!(meminfo.active_file.is_some());
749            assert!(meminfo.inactive_file.is_some());
750        } else {
751            assert!(meminfo.active_anon.is_none());
752            assert!(meminfo.inactive_anon.is_none());
753            assert!(meminfo.active_file.is_none());
754            assert!(meminfo.inactive_file.is_none());
755        }
756
757        if kernel >= KernelVersion::new(2, 6, 28)
758            && kernel <= KernelVersion::new(2, 6, 30)
759            && meminfo.unevictable.is_some()
760        {
761            if let Some(KernelConfig(ref config)) = config {
762                assert!(config.get("CONFIG_UNEVICTABLE_LRU").is_some());
763            }
764        }
765
766        if kernel >= KernelVersion::new(2, 6, 19)
767            && config
768                .as_ref()
769                .map_or(false, |KernelConfig(cfg)| cfg.contains_key("CONFIG_HIGHMEM"))
770        {
771            assert!(meminfo.high_total.is_some());
772            assert!(meminfo.high_free.is_some());
773            assert!(meminfo.low_total.is_some());
774            assert!(meminfo.low_free.is_some());
775        } else {
776            assert!(meminfo.high_total.is_none());
777            assert!(meminfo.high_free.is_none());
778            assert!(meminfo.low_total.is_none());
779            assert!(meminfo.low_free.is_none());
780        }
781
782        // Possible bug in procfs documentation:
783        // The man page says that MmapCopy requires CONFIG_MMU, but if you look at the
784        // source, MmapCopy is only included if CONFIG_MMU is *missing*:
785        // https://github.com/torvalds/linux/blob/v4.17/fs/proc/meminfo.c#L80
786        //if kernel >= KernelVersion::new(2, 6, 29) && config.contains_key("CONFIG_MMU") {
787        //    assert!(meminfo.mmap_copy.is_some());
788        //} else {
789        //    assert!(meminfo.mmap_copy.is_none());
790        //}
791
792        if kernel >= KernelVersion::new(2, 6, 18) {
793            assert!(meminfo.anon_pages.is_some());
794            assert!(meminfo.page_tables.is_some());
795            assert!(meminfo.nfs_unstable.is_some());
796            assert!(meminfo.bounce.is_some());
797        } else {
798            assert!(meminfo.anon_pages.is_none());
799            assert!(meminfo.page_tables.is_none());
800            assert!(meminfo.nfs_unstable.is_none());
801            assert!(meminfo.bounce.is_none());
802        }
803
804        if kernel >= KernelVersion::new(2, 6, 32) {
805            assert!(meminfo.shmem.is_some());
806            assert!(meminfo.kernel_stack.is_some());
807        } else {
808            assert!(meminfo.shmem.is_none());
809            assert!(meminfo.kernel_stack.is_none());
810        }
811
812        if kernel >= KernelVersion::new(2, 6, 19) {
813            assert!(meminfo.s_reclaimable.is_some());
814            assert!(meminfo.s_unreclaim.is_some());
815        } else {
816            assert!(meminfo.s_reclaimable.is_none());
817            assert!(meminfo.s_unreclaim.is_none());
818        }
819
820        if kernel >= KernelVersion::new(2, 6, 27)
821            && config
822                .as_ref()
823                .map_or(false, |KernelConfig(cfg)| cfg.contains_key("CONFIG_QUICKLIST"))
824        {
825            assert!(meminfo.quicklists.is_some());
826        } else {
827            assert!(meminfo.quicklists.is_none());
828        }
829
830        if kernel >= KernelVersion::new(2, 6, 26) {
831            assert!(meminfo.writeback_tmp.is_some());
832        } else {
833            assert!(meminfo.writeback_tmp.is_none());
834        }
835
836        if kernel >= KernelVersion::new(2, 6, 10) {
837            assert!(meminfo.commit_limit.is_some());
838        } else {
839            assert!(meminfo.commit_limit.is_none());
840        }
841
842        if kernel >= KernelVersion::new(2, 6, 32)
843            && config.as_ref().map_or(
844                std::path::Path::new("/proc/kpagecgroup").exists(),
845                |KernelConfig(cfg)| cfg.contains_key("CONFIG_MEMORY_FAILURE"),
846            )
847        {
848            assert!(meminfo.hardware_corrupted.is_some());
849        } else {
850            assert!(meminfo.hardware_corrupted.is_none());
851        }

853        if kernel >= KernelVersion::new(2, 6, 38)
854            && config.as_ref().map_or(false, |KernelConfig(cfg)| {
855                cfg.contains_key("CONFIG_TRANSPARENT_HUGEPAGE")

865            && config.as_ref().map_or(true, |KernelConfig(cfg)| {
866                cfg.contains_key("CONFIG_TRANSPARENT_HUGEPAGE")
867            })
868        {
869            assert!(meminfo.shmem_hugepages.is_some());
870            assert!(meminfo.shmem_pmd_mapped.is_some());

894            assert!(meminfo.hugepagesize.is_some());
895        } else {

897            assert!(meminfo.hugepages_free.is_none());
898            assert!(meminfo.hugepagesize.is_none());
899        }

902            && config

904                .map_or(true, |KernelConfig(cfg)| cfg.contains_key("CONFIG_HUGETLB_PAGE"))
905        {

909        }

912            && config
913                .as_ref()
914                .map_or(true, |KernelConfig(cfg)| cfg.contains_key("CONFIG_HUGETLB_PAGE"))
915        {
916            assert!(meminfo.hugepages_surp.is_some());
