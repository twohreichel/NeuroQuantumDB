1//! Functions and structs related to process information
2//!
3//! The primary source of data for functions in this module is the files in a `/proc/<pid>/`
4//! directory.  If you have a process ID, you can use
5//! [`Process::new(pid)`](struct.Process.html#method.new), otherwise you can get a
6//! list of all running processes using [`all_processes()`](fn.all_processes.html).
7//!
8//! In case you have procfs filesystem mounted to a location other than `/proc`,
9//! use [`Process::new_with_root()`](struct.Process.html#method.new_with_root).
10//!
11//! # Examples
12//!
13//! Here's a small example that prints out all processes that are running on the same tty as the calling
14//! process.  This is very similar to what "ps" does in its default mode.  You can run this example
15//! yourself with:
16//!
17//! > cargo run --example=ps
18//!
19//! ```rust
20//! let me = procfs::process::Process::myself().unwrap();
21//! let me_stat = me.stat().unwrap();
22//! let tps = procfs::ticks_per_second();
23//!
24//! println!("{: >10} {: <8} {: >8} {}", "PID", "TTY", "TIME", "CMD");
25//!
26//! let tty = format!("pty/{}", me_stat.tty_nr().1);
27//! for prc in procfs::process::all_processes().unwrap() {
28//!     let Ok(prc) = prc else {
29//!         // process vanished
30//!         continue;
31//!     };
32//!     let Ok(stat) = prc.stat() else {
33//!         // process vanished
34//!         continue;
35//!     };
36//!     if let Ok(stat) = prc.stat() {
37//!         if stat.tty_nr == me_stat.tty_nr {
38//!             // total_time is in seconds
39//!             let total_time =
40//!                 (stat.utime + stat.stime) as f32 / (tps as f32);
41//!             println!(
42//!                 "{: >10} {: <8} {: >8} {}",
43//!                 stat.pid, tty, total_time, stat.comm
44//!             );
45//!         }
46//!     }
47//! }
48//! ```
49//!
50//! Here's a simple example of how you could get the total memory used by the current process.
51//! There are several ways to do this.  For a longer example, see the `examples/self_memory.rs`
52//! file in the git repository.  You can run this example with:
53//!
54//! > cargo run --example=self_memory
55//!
56//! ```rust
57//! # use procfs::process::Process;
58//! let me = Process::myself().unwrap();
59//! let me_stat = me.stat().unwrap();
60//! let page_size = procfs::page_size();
61//!
62//! println!("== Data from /proc/self/stat:");
63//! println!("Total virtual memory used: {} bytes", me_stat.vsize);
64//! println!("Total resident set: {} pages ({} bytes)", me_stat.rss, me_stat.rss as u64 * page_size);
65//! ```

98    /// The mode (read/write permissions) for an open file descriptor
99    ///
100    /// This is represented as `u16` since the values of these bits are
101    /// [documented] to be within the `u16` range.
102    ///
103    /// [documented]: https://man7.org/linux/man-pages/man2/chmod.2.html#DESCRIPTION
104    #[cfg_attr(feature = "serde1", derive(Serialize, Deserialize))]
105    #[derive(Copy, Clone, Debug, Hash, Eq, PartialEq, PartialOrd, Ord)]
106    pub struct FDPermissions: u16 {
107        const READ = Mode::RUSR.bits() as u16;
108        const WRITE = Mode::WUSR.bits() as u16;
109        const EXECUTE = Mode::XUSR.bits() as u16;
110    }
111}

113/// See the [Process::fd()] method
114#[derive(Clone)]
115#[cfg_attr(feature = "serde1", derive(Serialize, Deserialize))]
116pub struct FDInfo {
117    /// The file descriptor
118    pub fd: i32,
119    /// The permission bits for this FD
120    ///
121    /// **Note**: this field is only the owner read/write/execute bits.  All the other bits
122    /// (include filetype bits) are masked out.  See also the `mode()` method.
123    pub mode: u16,
124    pub target: FDTarget,
125}
126
127impl FDInfo {
128    /// Gets a file descriptor from a raw fd
129    pub fn from_raw_fd(pid: i32, raw_fd: i32) -> ProcResult<Self> {
130        Self::from_raw_fd_with_root("/proc", pid, raw_fd)
131    }
132
133    /// Gets a file descriptor from a raw fd based on a specified `/proc` path
134    pub fn from_raw_fd_with_root(root: impl AsRef<Path>, pid: i32, raw_fd: i32) -> ProcResult<Self> {
135        let path = root.as_ref().join(pid.to_string()).join("fd").join(raw_fd.to_string());
136        let link = wrap_io_error!(path, read_link(&path))?;
137        let md = wrap_io_error!(path, path.symlink_metadata())?;
138        let link_os: &OsStr = link.as_ref();
139        Ok(Self {
140            fd: raw_fd,
141            mode: ((md.mode() as RawMode) & Mode::RWXU.bits()) as u16,
142            target: expect!(FDTarget::from_str(expect!(link_os.to_str()))),
143        })
144    }
145
146    /// Gets a file descriptor from a directory fd and a path relative to it.
147    ///
148    /// `base` is the path to the directory fd, and is used for error messages.
149    fn from_process_at<P: AsRef<Path>, Q: AsRef<Path>>(
150        base: P,
151        dirfd: BorrowedFd,
152        path: Q,
153        fd: i32,
154    ) -> ProcResult<Self> {
155        let p = path.as_ref();
156        let root = base.as_ref().join(p);
157        // for 2.6.39 <= kernel < 3.6 fstat doesn't support O_PATH see https://github.com/eminence/procfs/issues/265
158        let flags = match Version::cached() {
159            Ok(v) if v < KernelVersion::new(3, 6, 0) => OFlags::NOFOLLOW | OFlags::CLOEXEC,
160            Ok(_) => OFlags::NOFOLLOW | OFlags::PATH | OFlags::CLOEXEC,
161            Err(_) => OFlags::NOFOLLOW | OFlags::PATH | OFlags::CLOEXEC,
162        };
163        let file = wrap_io_error!(root, rustix::fs::openat(dirfd, p, flags, Mode::empty()))?;
164        let link = rustix::fs::readlinkat(&file, "", Vec::new()).map_err(io::Error::from)?;
165        let md =
166            rustix::fs::statat(&file, "", AtFlags::SYMLINK_NOFOLLOW | AtFlags::EMPTY_PATH).map_err(io::Error::from)?;
167
168        let link_os = link.to_string_lossy();
169        let target = FDTarget::from_str(link_os.as_ref())?;
170        Ok(FDInfo {
171            fd,
172            mode: (md.st_mode & Mode::RWXU.bits()) as u16,
173            target,
174        })
175    }
176
177    /// Gets the read/write mode of this file descriptor as a bitfield
178    pub fn mode(&self) -> FDPermissions {
179        FDPermissions::from_bits_truncate(self.mode)
180    }
181}

193/// Represents a process in `/proc/<pid>`.
194///
195/// **Note** The `Process` struct holds an open file descriptor to its `/proc/<pid>` directory.
196/// This makes it possible to construct a `Process` object and then later call the various methods
197/// on it without a risk of inadvertently getting information from the wrong process (due to PID
198/// reuse).
199///
200/// However the downside is that holding a lot of `Process` objects might cause the process to run
201/// out of file descriptors.
202///
203/// For use cases that don't involve holding a lot of `Process` objects, no special handler is
204/// needed.  But if you do hold a lot of these objects (for example if you're writing a `ps`
205/// or `top` -like program), you'll likely want to gather all of the necessary info from `Process`
206/// object into a new struct and then drop the `Process` object
207///
208#[derive(Debug)]
209pub struct Process {
210    fd: OwnedFd,
211    pub pid: i32,
212    pub(crate) root: PathBuf,
213}
214

217    /// Returns a `Process` based on a specified PID.
218    ///
219    /// This can fail if the process doesn't exist, or if you don't have permission to access it.
220    pub fn new(pid: i32) -> ProcResult<Process> {
221        let root = PathBuf::from("/proc").join(pid.to_string());
222        Self::new_with_root(root)
223    }
224
225    /// Returns a `Process` based on a specified `/proc/<pid>` path.
226    pub fn new_with_root(root: PathBuf) -> ProcResult<Process> {
227        // for 2.6.39 <= kernel < 3.6 fstat doesn't support O_PATH see https://github.com/eminence/procfs/issues/265
228        let flags = match Version::cached() {
229            Ok(v) if v < KernelVersion::new(3, 6, 0) => OFlags::DIRECTORY | OFlags::CLOEXEC,
230            Ok(_) => OFlags::PATH | OFlags::DIRECTORY | OFlags::CLOEXEC,
231            Err(_) => OFlags::PATH | OFlags::DIRECTORY | OFlags::CLOEXEC,
232        };
233        let file = wrap_io_error!(root, rustix::fs::openat(rustix::fs::CWD, &root, flags, Mode::empty()))?;
234
235        let pidres = root
236            .as_path()
237            .components()
238            .last()
239            .and_then(|c| match c {
240                std::path::Component::Normal(s) => Some(s),
241                _ => None,
242            })
243            .and_then(|s| s.to_string_lossy().parse::<i32>().ok())
244            .or_else(|| {
245                rustix::fs::readlinkat(rustix::fs::CWD, &root, Vec::new())
246                    .ok()
247                    .and_then(|s| s.to_string_lossy().parse::<i32>().ok())
248            });
249        let pid = match pidres {
250            Some(pid) => pid,
251            None => return Err(ProcError::NotFound(Some(root))),
252        };
253
254        Ok(Process { fd: file, pid, root })
255    }

257    /// Returns a `Process` for the currently running process.

260    pub fn myself() -> ProcResult<Process> {
261        let root = PathBuf::from("/proc/self");
262        Self::new_with_root(root)
263    }

348    /// Gets the current environment for the process.  This is done by reading the
349    /// `/proc/pid/environ` file.
350    pub fn environ(&self) -> ProcResult<HashMap<OsString, OsString>> {
351        use std::os::unix::ffi::OsStrExt;
352
353        let mut map = HashMap::new();
354
355        let mut file = FileWrapper::open_at(&self.root, &self.fd, "environ")?;
356        let mut buf = Vec::new();
357        file.read_to_end(&mut buf)?;
358
359        for slice in buf.split(|b| *b == 0) {
360            // slice will be in the form key=var, so split on the first equals sign
361            let mut split = slice.splitn(2, |b| *b == b'=');
362            if let (Some(k), Some(v)) = (split.next(), split.next()) {
363                map.insert(OsStr::from_bytes(k).to_os_string(), OsStr::from_bytes(v).to_os_string());
364            };
365            //let env = OsStr::from_bytes(slice);
366        }
367
368        Ok(map)
369    }

450    /// Gets the number of open file descriptors for a process
451    ///
452    /// Calling this function is more efficient than calling `fd().unwrap().count()`
453    pub fn fd_count(&self) -> ProcResult<usize> {
454        // Use fast path if available (Linux v6.2): https://github.com/torvalds/linux/commit/f1f1f2569901
455        let stat = wrap_io_error!(
456            self.root.join("fd"),
457            rustix::fs::statat(&self.fd, "fd", AtFlags::empty())
458        )?;
459        if stat.st_size > 0 {
460            return Ok(stat.st_size as usize);
461        }
462
463        let fds = wrap_io_error!(
464            self.root.join("fd"),
465            rustix::fs::openat(
466                &self.fd,
467                "fd",
468                OFlags::RDONLY | OFlags::DIRECTORY | OFlags::CLOEXEC,
469                Mode::empty()
470            )
471        )?;
472        let fds = wrap_io_error!(self.root.join("fd"), rustix::fs::Dir::read_from(fds))?;
473        Ok(fds.count())
474    }

500    /// Lists which memory segments are written to the core dump in the event that a core dump is performed.
501    ///
502    /// By default, the following bits are set:
503    /// 0, 1, 4 (if the CONFIG_CORE_DUMP_DEFAULT_ELF_HEADERS kernel configuration option is enabled), and 5.
504    /// This default can be modified at boot time using the core dump_filter boot option.
505    ///
506    /// This function will return `Err(ProcError::NotFound)` if the `coredump_filter` file can't be
507    /// found.  If it returns `Ok(None)` then the process has no coredump_filter
508    pub fn coredump_filter(&self) -> ProcResult<Option<CoredumpFlags>> {
509        let mut file = FileWrapper::open_at(&self.root, &self.fd, "coredump_filter")?;
510        let mut s = String::new();
511        file.read_to_string(&mut s)?;
512        if s.trim().is_empty() {
513            return Ok(None);
514        }
515        let flags = from_str!(u32, &s.trim(), 16, pid: self.pid);
516        Ok(Some(expect!(CoredumpFlags::from_bits(flags))))
517    }

529    /// Get the process's auxiliary vector
530    ///
531    /// (since 2.6.0-test7)
532    pub fn auxv(&self) -> ProcResult<HashMap<u64, u64>> {
533        let mut file = FileWrapper::open_at(&self.root, &self.fd, "auxv")?;
534        let mut map = HashMap::new();
535
536        let mut buf = Vec::new();
537        let bytes_read = file.read_to_end(&mut buf)?;
538        if bytes_read == 0 {
539            // some kernel processes won't have any data for their auxv file
540            return Ok(map);
541        }
542        buf.truncate(bytes_read);
543        let mut file = std::io::Cursor::new(buf);
544
545        let mut buf = 0usize.to_ne_bytes();
546        loop {
547            file.read_exact(&mut buf)?;
548            let key = usize::from_ne_bytes(buf) as u64;
549            file.read_exact(&mut buf)?;
550            let value = usize::from_ne_bytes(buf) as u64;
551            if key == 0 && value == 0 {
552                break;
553            }
554            map.insert(key, value);
555        }
556
557        Ok(map)
558    }

649    /// Iterate over all the [`Task`]s (aka Threads) in this process
650    ///
651    /// Note that the iterator does not receive a snapshot of tasks, it is a
652    /// lazy iterator over whatever happens to be running when the iterator
653    /// gets there, see the examples below.
654    ///
655    /// # Examples
656    ///
657    /// ## Simple iteration over subtasks
658    ///
659    /// If you want to get the info that most closely matches what was running
660    /// when you call `tasks` you should collect them as quikcly as possible,
661    /// and then run processing over that collection:
662    ///
663    /// ```
664    /// # use std::thread;
665    /// # use std::sync::mpsc::channel;
666    /// # use procfs::process::Process;
667    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
668    /// # let (finish_tx, finish_rx) = channel();
669    /// # let (start_tx, start_rx) = channel();
670    /// let name = "testing:example";
671    /// let t = thread::Builder::new().name(name.to_string())
672    ///   .spawn(move || { // do work
673    /// #     start_tx.send(()).unwrap();
674    /// #     finish_rx.recv().expect("valid channel");
675    ///   })?;
676    /// # start_rx.recv()?;
677    ///
678    /// let proc = Process::myself()?;
679    ///
680    /// // Collect a snapshot
681    /// let threads: Vec<_> = proc.tasks()?.flatten().map(|t| t.stat().unwrap().comm).collect();
682    /// threads.iter().find(|s| &**s == name).expect("thread should exist");
683    ///
684    /// # finish_tx.send(());
685    /// # t.join().unwrap();
686    /// # Ok(())
687    /// # }
688    /// ```
689    ///
690    /// ## The TaskIterator is lazy
691    ///
692    /// This means both that tasks that stop before you get to them in
693    /// iteration will not be there, and that new tasks that are created after
694    /// you start the iterator *will* appear.
695    ///
696    /// ```
697    /// # use std::thread;
698    /// # use std::sync::mpsc::channel;
699    /// # use procfs::process::Process;
700    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
701    /// let proc = Process::myself()?;
702    ///
703    /// // Task iteration is lazy
704    /// let mut task_iter = proc.tasks()?.flatten().map(|t| t.stat().unwrap().comm);
705    ///
706    /// # let (finish_tx, finish_rx) = channel();
707    /// # let (start_tx, start_rx) = channel();
708    /// let name = "testing:lazy";
709    /// let t = thread::Builder::new().name(name.to_string())
710    ///   .spawn(move || { // do work
711    /// #     start_tx.send(()).unwrap();
712    /// #     finish_rx.recv().expect("valid channel");
713    ///   })?;
714    /// # start_rx.recv()?;
715    ///
716    /// task_iter.find(|s| &**s == name).expect("thread should exist");
717    ///
718    /// # finish_tx.send(());
719    /// # t.join().unwrap();
720    /// # Ok(())
721    /// # }
722    /// ```
723    ///
724    /// Tasks that stop while you're iterating may or may not appear:
725    ///
726    /// ```
727    /// # use std::thread;
728    /// # use std::sync::mpsc::channel;
729    /// # use procfs::process::Process;
730    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
731    /// # let (finish_tx, finish_rx) = channel();
732    /// # let (start_tx, start_rx) = channel();
733    /// let name = "testing:stopped";
734    /// let t = thread::Builder::new().name(name.to_string())
735    ///   .spawn(move || { // do work
736    /// #     start_tx.send(()).unwrap();
737    /// #     finish_rx.recv().expect("valid channel");
738    ///   })?;
739    /// # start_rx.recv()?;
740    ///
741    /// let proc = Process::myself()?;
742    ///
743    /// // Task iteration is lazy
744    /// let mut task_iter = proc.tasks()?.flatten().map(|t| t.stat().unwrap().comm);
745    ///
746    /// # finish_tx.send(());
747    /// t.join().unwrap();
748    ///
749    /// // It's impossible to know if this is going to be gone
750    /// let _ = task_iter.find(|s| &**s == name).is_some();
751    /// # Ok(())
752    /// # }
753    /// ```
754    pub fn tasks(&self) -> ProcResult<TasksIter> {
755        let task_path = self.root.join("task");
756        let dir_fd = wrap_io_error!(
757            &task_path,
758            rustix::fs::openat(
759                &self.fd,
760                "task",
761                OFlags::RDONLY | OFlags::DIRECTORY | OFlags::CLOEXEC,
762                Mode::empty()
763            )
764        )?;
765        let dir = wrap_io_error!(&task_path, rustix::fs::Dir::read_from(&dir_fd))?;
766        Ok(TasksIter {
767            pid: self.pid,
768            inner: dir,
769            inner_fd: dir_fd,
770            root: task_path,
771        })
772    }

828    /// Opens a file to the process's memory (`/proc/<pid>/mem`).
829    ///
830    /// Note: you cannot start reading from the start of the file.  You must first seek to
831    /// a mapped page.  See [Process::maps].
832    ///
833    /// Permission to access this file is governed by a ptrace access mode PTRACE_MODE_ATTACH_FSCREDS check
834    ///
835    /// # Example
836    ///
837    /// Find the offset of the "hello" string in the process's stack, and compare it to the
838    /// pointer of the variable containing "hello"
839    ///
840    /// ```rust
841    /// # use std::io::{Read, Seek, SeekFrom};
842    /// # use procfs::process::{MMapPath, Process};
843    /// let me = Process::myself().unwrap();
844    /// let mut mem = me.mem().unwrap();
845    /// let maps = me.maps().unwrap();
846    ///
847    /// let hello = "hello".to_string();
848    ///
849    /// for map in maps {
850    ///     if map.pathname == MMapPath::Heap {
851    ///         mem.seek(SeekFrom::Start(map.address.0)).unwrap();
852    ///         let mut buf = vec![0; (map.address.1 - map.address.0) as usize];
853    ///         mem.read_exact(&mut buf).unwrap();
854    ///         let idx = buf.windows(5).position(|p| p == b"hello").unwrap();
855    ///         assert_eq!(map.address.0 + idx as u64, hello.as_ptr() as u64);
856    ///     }
857    /// }
858    /// ```
859    pub fn mem(&self) -> ProcResult<File> {
860        let file = FileWrapper::open_at(&self.root, &self.fd, "mem")?;
861        Ok(file.inner())
862    }

891/// The result of [`Process::fd`], iterates over all fds in a process
892#[derive(Debug)]
893pub struct FDsIter {
894    inner: rustix::fs::Dir,
895    inner_fd: rustix::fd::OwnedFd,
896    root: PathBuf,
897}
898
899impl std::iter::Iterator for FDsIter {
900    type Item = ProcResult<FDInfo>;
901    fn next(&mut self) -> Option<ProcResult<FDInfo>> {
902        loop {
903            match self.inner.next() {
904                Some(Ok(entry)) => {
905                    let name = entry.file_name().to_string_lossy();
906                    if let Ok(fd) = RawFd::from_str(&name) {
907                        if let Ok(info) = FDInfo::from_process_at(&self.root, self.inner_fd.as_fd(), name.as_ref(), fd)
908                        {
909                            break Some(Ok(info));
910                        }
911                    }
912                }
913                Some(Err(e)) => break Some(Err(io::Error::from(e).into())),
914                None => break None,
915            }
916        }
917    }
918}

944                None => break None,

950/// Return a iterator of all processes
951///
952/// If a process can't be constructed for some reason, it will be returned as an `Err(ProcError)`
953///
954/// See also some important docs on the [ProcessesIter] struct.
955///
956/// Error handling example
957/// ```
958/// # use procfs::process::Process;
959/// let all_processes: Vec<Process> = procfs::process::all_processes()
960/// .expect("Can't read /proc")
961/// .filter_map(|p| match p {
962///     Ok(p) => Some(p),                                                // happy path
963///     Err(e) => match e {
964///         procfs::ProcError::NotFound(_) => None,                      // process vanished during iteration, ignore it
965///         procfs::ProcError::Io(e, path) => None,                      // can match on path to decide if we can continue
966///         x => {
967///             println!("Can't read process due to error {x:?}"); // some unknown error
968///             None
969///         }
970///     },
971/// })
972/// .collect();
973/// ```
974pub fn all_processes() -> ProcResult<ProcessesIter> {
975    all_processes_with_root("/proc")
976}

978/// Return a list of all processes based on a specified `/proc` path
979///
980/// See [all_processes] for details and examples
981///
982/// See also some important docs on the [ProcessesIter] struct.
983pub fn all_processes_with_root(root: impl AsRef<Path>) -> ProcResult<ProcessesIter> {
984    let root = root.as_ref();
985    let dir = wrap_io_error!(
986        root,
987        rustix::fs::openat(
988            rustix::fs::CWD,
989            root,
990            OFlags::RDONLY | OFlags::DIRECTORY | OFlags::CLOEXEC,
991            Mode::empty()
992        )
993    )?;
994    let dir = wrap_io_error!(root, rustix::fs::Dir::read_from(dir))?;
995    Ok(ProcessesIter {
996        root: PathBuf::from(root),
997        inner: dir,
998    })
999}

1001/// An iterator over all processes in the system.
1002///
1003/// **Note** This is a *lazy* iterator (like most iterators in rust).  You will likely want to consume
1004/// this iterator as quickly as possible if you want a "snapshot" of the system (though it won't be a
1005/// true snapshot).  Another important thing to keep in mind is that the [`Process`] struct holds an
1006/// open file descriptor to its corresponding `/proc/<pid>` directory.  See the docs for [`Process`]
1007/// for more information.
1008#[derive(Debug)]
1009pub struct ProcessesIter {
1010    root: PathBuf,
1011    inner: rustix::fs::Dir,
1012}
1013
1014impl std::iter::Iterator for ProcessesIter {
1015    type Item = ProcResult<Process>;
1016    fn next(&mut self) -> Option<ProcResult<Process>> {
1017        loop {
1018            match self.inner.next() {
1019                Some(Ok(entry)) => {
1020                    if let Ok(pid) = i32::from_str(&entry.file_name().to_string_lossy()) {
1021                        break Some(Process::new_with_root(self.root.join(pid.to_string())));
1022                    }
1023                }
1024                Some(Err(e)) => break Some(Err(io::Error::from(e).into())),
1025                None => break None,
1026            }
1027        }
1028    }
