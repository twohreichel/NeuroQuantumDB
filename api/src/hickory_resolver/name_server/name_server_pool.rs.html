60impl<P> NameServerPool<P>

64    pub(crate) fn from_config_with_provider(
65        config: &ResolverConfig,
66        options: ResolverOpts,
67        conn_provider: P,
68    ) -> Self {
69        let datagram_conns: Vec<NameServer<P>> = config
70            .name_servers()
71            .iter()
72            .filter(|ns_config| ns_config.protocol.is_datagram())
73            .map(|ns_config| {
74                #[cfg(feature = "dns-over-rustls")]
75                let ns_config = {
76                    let mut ns_config = ns_config.clone();
77                    ns_config.tls_config = config.client_config().clone();
78                    ns_config
79                };
80                #[cfg(not(feature = "dns-over-rustls"))]
81                let ns_config = { ns_config.clone() };
82
83                NameServer::new(ns_config, options.clone(), conn_provider.clone())
84            })
85            .collect();
86
87        let stream_conns: Vec<NameServer<P>> = config
88            .name_servers()
89            .iter()
90            .filter(|ns_config| ns_config.protocol.is_stream())
91            .map(|ns_config| {
92                #[cfg(feature = "dns-over-rustls")]
93                let ns_config = {
94                    let mut ns_config = ns_config.clone();
95                    ns_config.tls_config = config.client_config().clone();
96                    ns_config
97                };
98                #[cfg(not(feature = "dns-over-rustls"))]
99                let ns_config = { ns_config.clone() };
100
101                NameServer::new(ns_config, options.clone(), conn_provider.clone())
102            })
103            .collect();
104
105        Self {
106            datagram_conns: Arc::from(datagram_conns),
107            stream_conns: Arc::from(stream_conns),
108            options,
109        }
110    }

112    /// Construct a NameServerPool from a set of name server configs
113    pub fn from_config(
114        name_servers: NameServerConfigGroup,
115        options: ResolverOpts,
116        conn_provider: P,
117    ) -> Self {
118        let map_config_to_ns =
119            |ns_config| NameServer::new(ns_config, options.clone(), conn_provider.clone());
120
121        let (datagram, stream): (Vec<_>, Vec<_>) = name_servers
122            .into_inner()
123            .into_iter()
124            .partition(|ns| ns.protocol.is_datagram());
125
126        let datagram_conns: Vec<_> = datagram.into_iter().map(map_config_to_ns).collect();
127        let stream_conns: Vec<_> = stream.into_iter().map(map_config_to_ns).collect();
128
129        Self {
130            datagram_conns: Arc::from(datagram_conns),
131            stream_conns: Arc::from(stream_conns),
132            options,
133        }
134    }

163    async fn try_send(
164        opts: ResolverOpts,
165        conns: Arc<[NameServer<P>]>,
166        request: DnsRequest,
167    ) -> Result<DnsResponse, ResolveError> {
168        let mut conns: Vec<NameServer<P>> = conns.to_vec();
169
170        match opts.server_ordering_strategy {
171            // select the highest priority connection
172            //   reorder the connections based on current view...
173            //   this reorders the inner set
174            ServerOrderingStrategy::QueryStatistics => conns.sort_unstable(),
175            ServerOrderingStrategy::UserProvidedOrder => {}
176        }
177        let request_loop = request.clone();
178
179        parallel_conn_loop(conns, request_loop, opts).await
180    }
181}

183impl<P> DnsHandle for NameServerPool<P>
184where
185    P: ConnectionProvider + 'static,
186{
187    type Response = Pin<Box<dyn Stream<Item = Result<DnsResponse, ResolveError>> + Send>>;
188    type Error = ResolveError;
189
190    fn send<R: Into<DnsRequest>>(&self, request: R) -> Self::Response {
191        let opts = self.options.clone();
192        let request = request.into();
193        let datagram_conns = Arc::clone(&self.datagram_conns);
194        let stream_conns = Arc::clone(&self.stream_conns);
195        // TODO: remove this clone, return the Message in the error?
196        let tcp_message = request.clone();
197
198        // TODO: limited to only when mDNS is enabled, but this should probably always be enforced?
199        let mdns = Local::NotMdns(request);
200
201        // local queries are queried through mDNS
202        if mdns.is_local() {
203            return mdns.take_stream();
204        }
205
206        // TODO: should we allow mDNS to be used for standard lookups as well?
207
208        // it wasn't a local query, continue with standard lookup path
209        let request = mdns.take_request();
210        Box::pin(once(async move {
211            debug!("sending request: {:?}", request.queries());
212
213            // First try the UDP connections
214            let udp_res = match Self::try_send(opts.clone(), datagram_conns, request).await {
215                Ok(response) if response.truncated() => {
216                    debug!("truncated response received, retrying over TCP");
217                    Ok(response)
218                }
219                Err(e) if opts.try_tcp_on_error || e.is_no_connections() => {
220                    debug!("error from UDP, retrying over TCP: {}", e);
221                    Err(e)
222                }
223                result => return result,
224            };
225
226            if stream_conns.is_empty() {
227                debug!("no TCP connections available");
228                return udp_res;
229            }
230
231            // Try query over TCP, as response to query over UDP was either truncated or was an
232            // error.
233            let tcp_res = Self::try_send(opts, stream_conns, tcp_message).await;
234
235            let tcp_err = match tcp_res {
236                res @ Ok(..) => return res,
237                Err(e) => e,
238            };
239
240            // Even if the UDP result was truncated, return that
241            let udp_err = match udp_res {
242                Ok(response) => return Ok(response),
243                Err(e) => e,
244            };
245
246            match udp_err.cmp_specificity(&tcp_err) {
247                Ordering::Greater => Err(udp_err),
248                _ => Err(tcp_err),
249            }
250        }))
251    }
252}

254// TODO: we should be able to have a self-referential future here with Pin and not require cloned conns
255/// An async function that will loop over all the conns with a max parallel request count of ops.num_concurrent_req
256async fn parallel_conn_loop<P>(
257    mut conns: Vec<NameServer<P>>,
258    request: DnsRequest,
259    opts: ResolverOpts,
260) -> Result<DnsResponse, ResolveError>
261where
262    P: ConnectionProvider + 'static,
263{
264    let mut err = ResolveError::no_connections();
265    // If the name server we're trying is giving us backpressure by returning ProtoErrorKind::Busy,
266    // we will first try the other name servers (as for other error types). However, if the other
267    // servers are also busy, we're going to wait for a little while and then retry each server that
268    // returned Busy in the previous round. If the server is still Busy, this continues, while
269    // the backoff increases exponentially (by a factor of 2), until it hits 300ms, in which case we
270    // give up. The request might still be retried by the caller (likely the DnsRetryHandle).
271    //
272    // TODO: more principled handling of timeouts. Currently, timeouts appear to be handled mostly
273    // close to the connection, which means the top level resolution might take substantially longer
274    // to fire than the timeout configured in `ResolverOpts`.
275    let mut backoff = Duration::from_millis(20);
276    let mut busy = SmallVec::<[NameServer<P>; 2]>::new();
277
278    loop {
279        let request_cont = request.clone();
280
281        // construct the parallel requests, 2 is the default
282        let mut par_conns = SmallVec::<[NameServer<P>; 2]>::new();
283        let count = conns.len().min(opts.num_concurrent_reqs.max(1));
284
285        // Shuffe DNS NameServers to avoid overloads to the first configured ones
286        if opts.shuffle_dns_servers {
287            for _ in 0..count {
288                let idx = rng().gen_range(0..conns.len());
289
290                // UNWRAP: swap_remove has an implicit panicking bounds check. This should
291                // never fail because we check that conns is not empty and generate the idx
292                // to explicitly be in range.
293                par_conns.push(conns.swap_remove(idx));
294            }
295        } else {
296            for conn in conns.drain(..count) {
297                par_conns.push(conn);
298            }
299        }
300
301        if par_conns.is_empty() {
302            if !busy.is_empty() && backoff < Duration::from_millis(300) {
303                <<P as ConnectionProvider>::RuntimeProvider as RuntimeProvider>::Timer::delay_for(
304                    backoff,
305                )
306                .await;
307                conns.extend(busy.drain(..));
308                backoff *= 2;
309                continue;
310            }
311            return Err(err);
312        }
313
314        let mut requests = par_conns
315            .into_iter()
316            .map(move |conn| {
317                conn.send(request_cont.clone())
318                    .first_answer()
319                    .map(|result| result.map_err(|e| (conn, e)))
320            })
321            .collect::<FuturesUnordered<_>>();
322
323        while let Some(result) = requests.next().await {
324            let (conn, e) = match result {
325                Ok(sent) => return Ok(sent),
326                Err((conn, e)) => (conn, e),
327            };
328
329            match e.kind() {
330                ResolveErrorKind::NoRecordsFound { trusted, .. } if *trusted => {
331                    return Err(e);
332                }
333                ResolveErrorKind::Proto(e) if e.is_busy() => {
334                    busy.push(conn);
335                }
336                _ if err.cmp_specificity(&e) == Ordering::Less => {
337                    err = e;
338                }
339                _ => {}
340            }
341        }
342    }
343}

345#[allow(clippy::large_enum_variant)]
346pub(crate) enum Local {
347    #[allow(dead_code)]
348    ResolveStream(Pin<Box<dyn Stream<Item = Result<DnsResponse, ResolveError>> + Send>>),
349    NotMdns(DnsRequest),
350}
351
352impl Local {
353    fn is_local(&self) -> bool {
354        matches!(*self, Self::ResolveStream(..))
355    }
356
357    /// Takes the stream
358    ///
359    /// # Panics
360    ///
361    /// Panics if this is in fact a Local::NotMdns
362    fn take_stream(self) -> Pin<Box<dyn Stream<Item = Result<DnsResponse, ResolveError>> + Send>> {
363        match self {
364            Self::ResolveStream(future) => future,
365            _ => panic!("non Local queries have no future, see take_message()"),
366        }
367    }
368
369    /// Takes the message
370    ///
371    /// # Panics
372    ///
373    /// Panics if this is in fact a Local::ResolveStream
374    fn take_request(self) -> DnsRequest {
375        match self {
376            Self::NotMdns(request) => request,
377            _ => panic!("Local queries must be polled, see take_future()"),
378        }
379    }
380}
381
382impl Stream for Local {
383    type Item = Result<DnsResponse, ResolveError>;
384
385    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
386        match *self {
387            Self::ResolveStream(ref mut ns) => ns.as_mut().poll_next(cx),
388            // TODO: making this a panic for now
389            Self::NotMdns(..) => panic!("Local queries that are not mDNS should not be polled"), //Local::NotMdns(message) => return Err(ResolveErrorKind::Message("not mDNS")),
390        }
391    }
392}
