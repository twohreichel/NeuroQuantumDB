339/// Builder for cluster configuration.
340#[derive(Debug, Default)]
341pub struct ClusterConfigBuilder {
342    config: ClusterConfig,
343}
344
345impl ClusterConfig {
346    /// Create a new builder for cluster configuration.
347    #[must_use]
348    pub fn builder() -> ClusterConfigBuilder {
349        ClusterConfigBuilder::default()
350    }
351
352    /// Validate the configuration.
353    pub fn validate(&self) -> ClusterResult<()> {
354        if self.node_id == 0 {
355            return Err(ClusterError::ConfigError(
356                "Node ID must be greater than 0".into(),
357            ));
358        }
359
360        if self.raft.election_timeout_min >= self.raft.election_timeout_max {
361            return Err(ClusterError::ConfigError(
362                "Election timeout min must be less than max".into(),
363            ));
364        }
365
366        if self.raft.heartbeat_interval >= self.raft.election_timeout_min {
367            return Err(ClusterError::ConfigError(
368                "Heartbeat interval must be less than election timeout min".into(),
369            ));
370        }
371
372        if self.sharding.replication_factor == 0 {
373            return Err(ClusterError::ConfigError(
374                "Replication factor must be greater than 0".into(),
375            ));
376        }
377
378        if self.network.enable_tls {
379            if self.network.tls_cert_path.is_none() {
380                return Err(ClusterError::ConfigError(
381                    "TLS enabled but no certificate path provided".into(),
382                ));
383            }
384            if self.network.tls_key_path.is_none() {
385                return Err(ClusterError::ConfigError(
386                    "TLS enabled but no key path provided".into(),
387                ));
388            }
389        }
390
391        Ok(())
392    }
393}
