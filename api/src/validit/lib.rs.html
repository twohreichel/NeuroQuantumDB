1//! Validate variable internal state when the variable is accessed.
2//!
3//! - Implement trait `Validate` for a type `T` to define how to validate internal state of `T`.
4//! - Wrapper struct `Valid<T: Validate>` implements `Deref` and `DerefMut` traits, and validates
5//!   the internal state when the variable is accessed.
6//!
7//! For example, If in your program you have a struct `Lt5 { v: u64 }` and you want to make sure
8//! that `v` is always less than to `5`, you can implement `Validate` trait for `Foo` and use
9//! `less!` macro to validate `a`.
10//! ```
11//! # use std::error::Error;
12//! # use std::panic::catch_unwind;
13//! # use validit::Validate;
14//! # use validit::ValidateExt;
15//! struct Lt5 { v: u64 }
16//!
17//! impl Validate for Lt5 {
18//!     fn validate(&self) -> Result<(), Box<dyn Error>> {
19//!         validit::less!(self.v, 5);
20//!         Ok(())
21//!     }
22//! }
23//!
24//! # fn main() {
25//! let v1 = Lt5 { v: 1 }.valid();
26//! let _x = v1.v; // Good
27//!
28//! let v6 = Lt5 { v: 6 }.valid();
29//! let res = catch_unwind(|| {
30//!     let _x = v6.v; // panic: panicked at 'invalid state: expect: self.v(6) < 5(5) ...
31//! });
32//! assert!(res.is_err());
33//! # }
34//! ```
