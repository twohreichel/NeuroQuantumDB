19/// A prepared statement with its cached query plan
20#[derive(Debug, Clone)]
21pub struct PreparedStatement {
22    /// Name of the prepared statement
23    pub name: String,
24    /// The original SQL statement (with parameter placeholders)
25    pub statement: Statement,
26    /// Cached query plan (if available)
27    pub cached_plan: Option<Arc<QueryPlan>>,
28    /// Number of parameters expected
29    pub parameter_count: usize,
30    /// Parameter names for named parameters
31    pub parameter_names: Vec<String>,
32    /// Statistics for this prepared statement
33    pub stats: PreparedStatementStats,
34    /// When this statement was prepared
35    pub created_at: Instant,
36}

38/// Statistics for a prepared statement
39#[derive(Debug, Clone, Default, Serialize, Deserialize)]
40pub struct PreparedStatementStats {
41    /// Number of times this statement has been executed
42    pub execution_count: u64,
43    /// Total execution time across all executions
44    pub total_execution_time: Duration,
45    /// Average execution time
46    pub average_execution_time: Duration,
47    /// Cache hits (when cached plan was used)
48    pub cache_hits: u64,
49}

51/// Session-scoped prepared statement manager
52///
53/// Each session/connection should have its own `PreparedStatementManager`
54/// for proper isolation of prepared statements.
55#[derive(Debug, Default)]
56pub struct PreparedStatementManager {
57    /// Map of statement name to prepared statement
58    statements: HashMap<String, PreparedStatement>,
59    /// Maximum number of prepared statements allowed per session
60    max_statements: usize,
61    /// Total number of executions across all statements
62    total_executions: u64,
63}
64
65impl PreparedStatementManager {
66    /// Create a new `PreparedStatementManager` with default settings
67    #[must_use]
68    pub fn new() -> Self {
69        Self {
70            statements: HashMap::new(),
71            max_statements: 1000, // Default limit
72            total_executions: 0,
73        }
74    }
75
76    /// Create a new `PreparedStatementManager` with a custom statement limit
77    #[must_use]
78    pub fn with_max_statements(max_statements: usize) -> Self {
79        Self {
80            statements: HashMap::new(),
81            max_statements,
82            total_executions: 0,
83        }
84    }
85
86    /// Prepare a statement and store it
87    ///
88    /// Returns an error if a statement with the same name already exists
89    /// or if the maximum number of statements has been reached.
90    pub fn prepare(&mut self, prepare_stmt: &PrepareStatement) -> QSQLResult<()> {
91        // Check if statement already exists
92        if self.statements.contains_key(&prepare_stmt.name) {
93            return Err(QSQLError::PreparedStatementError {
94                message: format!(
95                    "Prepared statement '{}' already exists. Use DEALLOCATE first.",
96                    prepare_stmt.name
97                ),
98            });
99        }
100
101        // Check statement limit
102        if self.statements.len() >= self.max_statements {
103            return Err(QSQLError::PreparedStatementError {
104                message: format!(
105                    "Maximum number of prepared statements ({}) reached",
106                    self.max_statements
107                ),
108            });
109        }
110
111        // Count parameters in the statement
112        let (param_count, param_names) = count_parameters(&prepare_stmt.statement);
113
114        let prepared = PreparedStatement {
115            name: prepare_stmt.name.clone(),
116            statement: (*prepare_stmt.statement).clone(),
117            cached_plan: None,
118            parameter_count: param_count,
119            parameter_names: param_names,
120            stats: PreparedStatementStats::default(),
121            created_at: Instant::now(),
122        };
123
124        info!(
125            "Prepared statement '{}' with {} parameters",
126            prepare_stmt.name, param_count
127        );
128
129        self.statements.insert(prepare_stmt.name.clone(), prepared);
130        Ok(())
131    }
132
133    /// Get a prepared statement by name
134    #[must_use]
135    pub fn get(&self, name: &str) -> Option<&PreparedStatement> {
136        self.statements.get(name)
137    }

139    /// Get a mutable reference to a prepared statement by name
140    pub fn get_mut(&mut self, name: &str) -> Option<&mut PreparedStatement> {
141        self.statements.get_mut(name)
142    }

144    /// Bind parameters to a prepared statement and return the bound statement
145    ///
146    /// This replaces parameter placeholders with actual values.
147    pub fn bind_parameters(
148        &self,
149        name: &str,
150        execute_stmt: &ExecuteStatement,
151    ) -> QSQLResult<Statement> {
152        let prepared =
153            self.statements
154                .get(name)
155                .ok_or_else(|| QSQLError::PreparedStatementError {
156                    message: format!("Prepared statement '{name}' not found"),
157                })?;
158
159        // Build parameter map from positional and named parameters
160        let mut param_values: HashMap<ParameterRef, Expression> = HashMap::new();
161
162        // Add positional parameters ($1, $2, etc.)
163        for (idx, value) in execute_stmt.parameters.iter().enumerate() {
164            param_values.insert(ParameterRef::Positional((idx + 1) as u32), value.clone());
165        }
166
167        // Add named parameters (:name)
168        for (name, value) in &execute_stmt.named_parameters {
169            param_values.insert(ParameterRef::Named(name.clone()), value.clone());
170        }
171
172        // Check that we have all required parameters
173        let provided_count = execute_stmt.parameters.len() + execute_stmt.named_parameters.len();
174        if provided_count < prepared.parameter_count {
175            return Err(QSQLError::PreparedStatementError {
176                message: format!(
177                    "Prepared statement '{}' requires {} parameters, but {} were provided",
178                    name, prepared.parameter_count, provided_count
179                ),
180            });
181        }
182
183        // Clone the statement and substitute parameters
184        let bound_statement = substitute_parameters(&prepared.statement, &param_values)?;
185
186        debug!(
187            "Bound {} parameters to prepared statement '{}'",
188            provided_count, name
189        );
190
191        Ok(bound_statement)
192    }

194    /// Record execution statistics for a prepared statement
195    pub fn record_execution(&mut self, name: &str, execution_time: Duration) {
196        if let Some(prepared) = self.statements.get_mut(name) {
197            prepared.stats.execution_count += 1;
198            prepared.stats.total_execution_time += execution_time;
199            prepared.stats.average_execution_time = Duration::from_nanos(
200                (prepared.stats.total_execution_time.as_nanos()
201                    / u128::from(prepared.stats.execution_count)) as u64,
202            );
203            self.total_executions += 1;
204        }
205    }

207    /// Deallocate a prepared statement
208    pub fn deallocate(&mut self, name: &str) -> QSQLResult<()> {
209        if self.statements.remove(name).is_some() {
210            info!("Deallocated prepared statement '{}'", name);
211            Ok(())
212        } else {
213            Err(QSQLError::PreparedStatementError {
214                message: format!("Prepared statement '{name}' not found"),
215            })
216        }
217    }
218
219    /// Deallocate all prepared statements
220    pub fn deallocate_all(&mut self) {
221        let count = self.statements.len();
222        self.statements.clear();
223        info!("Deallocated {} prepared statements", count);
224    }
225
226    /// Handle a DEALLOCATE statement
227    pub fn handle_deallocate(&mut self, deallocate_stmt: &DeallocateStatement) -> QSQLResult<()> {
228        match &deallocate_stmt.name {
229            | Some(name) => self.deallocate(name),
230            | None => {
231                self.deallocate_all();
232                Ok(())
233            },
234        }
235    }

278/// Count the number of parameters in a statement and collect parameter names
279fn count_parameters(statement: &Statement) -> (usize, Vec<String>) {
280    let mut positional_max = 0u32;
281    let mut named_params = Vec::new();
282
283    fn visit_expression(
284        expr: &Expression,
285        positional_max: &mut u32,
286        named_params: &mut Vec<String>,
287    ) {
288        match expr {
289            | Expression::Parameter(ParameterRef::Positional(idx)) => {
290                if *idx > *positional_max {
291                    *positional_max = *idx;
292                }
293            },
294            | Expression::Parameter(ParameterRef::Named(name)) => {
295                if !named_params.contains(name) {
296                    named_params.push(name.clone());
297                }
298            },
299            | Expression::BinaryOp { left, right, .. } => {
300                visit_expression(left, positional_max, named_params);
301                visit_expression(right, positional_max, named_params);
302            },
303            | Expression::UnaryOp { operand, .. } => {
304                visit_expression(operand, positional_max, named_params);
305            },
306            | Expression::FunctionCall { args, .. } => {
307                for arg in args {
308                    visit_expression(arg, positional_max, named_params);
309                }
310            },
311            | Expression::InList { expr, list, .. } => {
312                visit_expression(expr, positional_max, named_params);
313                for item in list {
314                    visit_expression(item, positional_max, named_params);
315                }
316            },
317            | Expression::Case {
318                when_clauses,
319                else_result,
320            } => {
321                for (cond, result) in when_clauses {
322                    visit_expression(cond, positional_max, named_params);
323                    visit_expression(result, positional_max, named_params);
324                }
325                if let Some(else_expr) = else_result {
326                    visit_expression(else_expr, positional_max, named_params);
327                }
328            },
329            | _ => {},
330        }
331    }
332
333    fn visit_statement(stmt: &Statement, positional_max: &mut u32, named_params: &mut Vec<String>) {
334        match stmt {
335            | Statement::Select(select) => {
336                if let Some(where_clause) = &select.where_clause {
337                    visit_expression(where_clause, positional_max, named_params);
338                }
339                if let Some(having) = &select.having {
340                    visit_expression(having, positional_max, named_params);
341                }
342                for item in &select.select_list {
343                    if let crate::ast::SelectItem::Expression { expr, .. } = item {
344                        visit_expression(expr, positional_max, named_params);
345                    }
346                }
347            },
348            | Statement::Insert(insert) => {
349                for row in &insert.values {
350                    for expr in row {
351                        visit_expression(expr, positional_max, named_params);
352                    }
353                }
354            },
355            | Statement::Update(update) => {
356                for assignment in &update.assignments {
357                    visit_expression(&assignment.value, positional_max, named_params);
358                }
359                if let Some(where_clause) = &update.where_clause {
360                    visit_expression(where_clause, positional_max, named_params);
361                }
362            },
363            | Statement::Delete(delete) => {
364                if let Some(where_clause) = &delete.where_clause {
365                    visit_expression(where_clause, positional_max, named_params);
366                }
367            },
368            | _ => {},
369        }
370    }
371
372    visit_statement(statement, &mut positional_max, &mut named_params);
373
374    let total_count = positional_max as usize + named_params.len();
375    (total_count, named_params)
376}

378/// Substitute parameters in a statement with actual values
379fn substitute_parameters(
380    statement: &Statement,
381    params: &HashMap<ParameterRef, Expression>,
382) -> QSQLResult<Statement> {
383    fn substitute_expression(
384        expr: &Expression,
385        params: &HashMap<ParameterRef, Expression>,
386    ) -> QSQLResult<Expression> {
387        match expr {
388            | Expression::Parameter(param_ref) => params.get(param_ref).cloned().ok_or_else(|| {
389                let param_str = match param_ref {
390                    | ParameterRef::Positional(idx) => format!("${idx}"),
391                    | ParameterRef::Named(name) => format!(":{name}"),
392                };
393                QSQLError::PreparedStatementError {
394                    message: format!("Missing parameter value for {param_str}"),
395                }
396            }),
397            | Expression::BinaryOp {
398                left,
399                operator,
400                right,
401            } => Ok(Expression::BinaryOp {
402                left: Box::new(substitute_expression(left, params)?),
403                operator: operator.clone(),
404                right: Box::new(substitute_expression(right, params)?),
405            }),
406            | Expression::UnaryOp { operator, operand } => Ok(Expression::UnaryOp {
407                operator: operator.clone(),
408                operand: Box::new(substitute_expression(operand, params)?),
409            }),
410            | Expression::FunctionCall { name, args } => {
411                let new_args: Result<Vec<_>, _> = args
412                    .iter()
413                    .map(|arg| substitute_expression(arg, params))
414                    .collect();
415                Ok(Expression::FunctionCall {
416                    name: name.clone(),
417                    args: new_args?,
418                })
419            },
420            | Expression::InList {
421                expr,
422                list,
423                negated,
424            } => {
425                let new_list: Result<Vec<_>, _> = list
426                    .iter()
427                    .map(|item| substitute_expression(item, params))
428                    .collect();
429                Ok(Expression::InList {
430                    expr: Box::new(substitute_expression(expr, params)?),
431                    list: new_list?,
432                    negated: *negated,
433                })
434            },
435            | Expression::Case {
436                when_clauses,
437                else_result,
438            } => {
439                let new_when: QSQLResult<Vec<_>> = when_clauses
440                    .iter()
441                    .map(|(cond, result)| {
442                        let new_cond = substitute_expression(cond, params)?;
443                        let new_result = substitute_expression(result, params)?;
444                        Ok((Box::new(new_cond), Box::new(new_result)))
445                    })
446                    .collect();
447                let new_else = else_result
448                    .as_ref()
449                    .map(|e| substitute_expression(e, params).map(Box::new))
450                    .transpose()?;
451                Ok(Expression::Case {
452                    when_clauses: new_when?,
453                    else_result: new_else,
454                })
455            },
456            // Other expression types are returned as-is
457            | other => Ok(other.clone()),
458        }
459    }
460
461    fn substitute_statement(
462        stmt: &Statement,
463        params: &HashMap<ParameterRef, Expression>,
464    ) -> QSQLResult<Statement> {
465        match stmt {
466            | Statement::Select(select) => {
467                let mut new_select = select.clone();
468                if let Some(where_clause) = &select.where_clause {
469                    new_select.where_clause = Some(substitute_expression(where_clause, params)?);
470                }
471                if let Some(having) = &select.having {
472                    new_select.having = Some(substitute_expression(having, params)?);
473                }
474                // Substitute in select list
475                new_select.select_list = select
476                    .select_list
477                    .iter()
478                    .map(|item| match item {
479                        | crate::ast::SelectItem::Expression { expr, alias } => {
480                            Ok(crate::ast::SelectItem::Expression {
481                                expr: substitute_expression(expr, params)?,
482                                alias: alias.clone(),
483                            })
484                        },
485                        | other => Ok(other.clone()),
486                    })
487                    .collect::<QSQLResult<Vec<_>>>()?;
488                Ok(Statement::Select(new_select))
489            },
490            | Statement::Insert(insert) => {
491                let mut new_insert = insert.clone();
492                new_insert.values = insert
493                    .values
494                    .iter()
495                    .map(|row| {
496                        row.iter()
497                            .map(|expr| substitute_expression(expr, params))
498                            .collect::<QSQLResult<Vec<_>>>()
499                    })
500                    .collect::<QSQLResult<Vec<_>>>()?;
501                Ok(Statement::Insert(new_insert))
502            },
503            | Statement::Update(update) => {
504                let mut new_update = update.clone();
505                new_update.assignments = update
506                    .assignments
507                    .iter()
508                    .map(|a| {
509                        Ok(crate::ast::Assignment {
510                            column: a.column.clone(),
511                            value: substitute_expression(&a.value, params)?,
512                        })
513                    })
514                    .collect::<QSQLResult<Vec<_>>>()?;
515                if let Some(where_clause) = &update.where_clause {
516                    new_update.where_clause = Some(substitute_expression(where_clause, params)?);
517                }
518                Ok(Statement::Update(new_update))
519            },
520            | Statement::Delete(delete) => {
521                let mut new_delete = delete.clone();
522                if let Some(where_clause) = &delete.where_clause {
523                    new_delete.where_clause = Some(substitute_expression(where_clause, params)?);
524                }
525                Ok(Statement::Delete(new_delete))
526            },
527            // Other statement types are returned as-is
528            | other => Ok(other.clone()),
529        }
530    }
531
532    substitute_statement(statement, params)
533}
