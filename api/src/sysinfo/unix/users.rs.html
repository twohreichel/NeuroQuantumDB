20pub(crate) struct UserInner {
21    pub(crate) uid: Uid,
22    pub(crate) gid: Gid,
23    pub(crate) name: String,
24    c_user: Vec<u8>,
25}

27impl UserInner {
28    pub(crate) fn new(uid: Uid, gid: Gid, name: String) -> Self {
29        let mut c_user = name.as_bytes().to_vec();
30        c_user.push(0);
31        Self {
32            uid,
33            gid,
34            name,
35            c_user,
36        }
37    }
38
39    pub(crate) fn id(&self) -> &Uid {
40        &self.uid
41    }
42
43    pub(crate) fn group_id(&self) -> Gid {
44        self.gid
45    }
46
47    pub(crate) fn name(&self) -> &str {
48        &self.name
49    }
50
51    pub(crate) fn groups(&self) -> Vec<Group> {
52        unsafe { get_user_groups(self.c_user.as_ptr() as *const _, self.gid.0 as _) }
53    }

56pub(crate) unsafe fn get_group_name(
57    id: libc::gid_t,
58    buffer: &mut Vec<libc::c_char>,
59) -> Option<String> {
60    let mut g = std::mem::MaybeUninit::<libc::group>::uninit();
61    let mut tmp_ptr = std::ptr::null_mut();
62    let mut last_errno = 0;
63
64    unsafe {
65        loop {
66            if retry_eintr!(set_to_0 => last_errno => getgrgid_r(
67                id as _,
68                g.as_mut_ptr() as _,
69                buffer.as_mut_ptr(),
70                buffer.capacity() as _,
71                &mut tmp_ptr as _
72            )) != 0
73            {
74                // If there was not enough memory, we give it more.
75                if last_errno == libc::ERANGE as libc::c_int {
76                    // Needs to be updated for `Vec::reserve` to actually add additional capacity.
77                    // In here it's "fine" since we never read from `buffer`.
78                    buffer.set_len(buffer.capacity());
79                    buffer.reserve(2048);
80                    continue;
81                }
82                return None;
83            }
84            break;
85        }
86        let g = g.assume_init();
87        super::utils::cstr_to_rust(g.gr_name)
88    }
89}

91pub(crate) unsafe fn get_user_groups(
92    name: *const libc::c_char,
93    group_id: libc::gid_t,
94) -> Vec<Group> {
95    let mut buffer = Vec::with_capacity(2048);
96    let mut groups = Vec::with_capacity(256);
97
98    loop {
99        unsafe {
100            let mut nb_groups = groups.capacity();
101            if getgrouplist(
102                name,
103                group_id as _,
104                groups.as_mut_ptr(),
105                &mut nb_groups as *mut _ as *mut _,
106            ) == -1
107            {
108                // Ensure the length matches the number of returned groups.
109                // Needs to be updated for `Vec::reserve` to actually add additional capacity.
110                groups.set_len(nb_groups as _);
111                groups.reserve(256);
112                continue;
113            }
114            groups.set_len(nb_groups as _);
115            return groups
116                .iter()
117                .filter_map(|group_id| {
118                    let name = get_group_name(*group_id as _, &mut buffer)?;
119                    Some(Group {
120                        inner: crate::GroupInner::new(Gid(*group_id as _), name),
121                    })
122                })
123                .collect();
124        }
125    }
126}

128pub(crate) fn get_users(users: &mut Vec<User>) {
129    fn filter(shell: *const std::ffi::c_char, uid: u32) -> bool {
130        !endswith(shell, b"/false") && !endswith(shell, b"/uucico") && uid < 65536
131    }
132
133    users.clear();
134
135    let mut users_map = std::collections::HashMap::with_capacity(10);
136
137    unsafe {
138        setpwent();
139        loop {
140            let pw = getpwent();
141            if pw.is_null() {
142                // The call was interrupted by a signal, retrying.
143                if std::io::Error::last_os_error().kind() == std::io::ErrorKind::Interrupted {
144                    continue;
145                }
146                break;
147            }
148
149            if !filter((*pw).pw_shell, (*pw).pw_uid) {
150                // This is not a "real" or "local" user.
151                continue;
152            }
153            if let Some(name) = crate::unix::utils::cstr_to_rust((*pw).pw_name) {
154                if users_map.contains_key(&name) {
155                    continue;
156                }
157
158                let uid = (*pw).pw_uid;
159                let gid = (*pw).pw_gid;
160                users_map.insert(name, (Uid(uid), Gid(gid)));
161            }
162        }
163        endpwent();
164    }
165    for (name, (uid, gid)) in users_map {
166        users.push(User {
167            inner: UserInner::new(uid, gid, name),
168        });
169    }
170}

172fn endswith(s1: *const std::ffi::c_char, s2: &[u8]) -> bool {
173    if s1.is_null() {
174        return false;
175    }
176    unsafe {
177        let mut len = libc::strlen(s1) as isize - 1;
178        let mut i = s2.len() as isize - 1;
179        while len >= 0 && i >= 0 && *s1.offset(len) == s2[i as usize] as std::ffi::c_char {
180            i -= 1;
181            len -= 1;
182        }
183        i == -1
184    }
