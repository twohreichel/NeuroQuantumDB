311    /// Check query cache for existing result
312    fn check_cache(&self, cache_key: &str) -> Option<CachedResult> {
313        if let Ok(mut cache) = self.query_cache.write() {
314            if let Some(cached) = cache.get_mut(cache_key) {
315                // Check if cache entry has expired (TTL: 300 seconds / 5 minutes)
316                const CACHE_TTL_SECONDS: u64 = 300;
317                let current_time = std::time::SystemTime::now()
318                    .duration_since(std::time::UNIX_EPOCH)
319                    .unwrap_or_default()
320                    .as_secs();
321
322                if current_time - cached.created_at_secs > CACHE_TTL_SECONDS {
323                    // Cache entry expired, remove it
324                    cache.remove(cache_key);
325
326                    if let Ok(mut stats) = self.query_statistics.write() {
327                        stats.cache_misses += 1;
328                    }
329                    return None;
330                }
331
332                // Update access statistics
333                cached.access_count += 1;
334                cached.last_accessed_secs = current_time;
335
336                // Update statistics
337                if let Ok(mut stats) = self.query_statistics.write() {
338                    stats.cache_hits += 1;
339                }
340
341                return Some(cached.clone());
342            }
343            // Cache miss
344            if let Ok(mut stats) = self.query_statistics.write() {
345                stats.cache_misses += 1;
346            }
347        }
348        None
349    }

351    /// Cache query result with LRU eviction
352    fn cache_result(&self, cache_key: String, result: &QueryResult) {
353        if let Ok(mut cache) = self.query_cache.write() {
354            let current_time = std::time::SystemTime::now()
355                .duration_since(std::time::UNIX_EPOCH)
356                .unwrap_or_default()
357                .as_secs();
358
359            let cached = CachedResult {
360                result: result.clone(),
361                created_at_secs: current_time,
362                access_count: 1,
363                last_accessed_secs: current_time,
364            };
365
366            // LRU eviction: if cache is too large, remove least recently used
367            const MAX_CACHE_SIZE: usize = 1000;
368            if cache.len() >= MAX_CACHE_SIZE {
369                // Find and remove the least recently accessed entry
370                if let Some(lru_key) = cache
371                    .iter()
372                    .min_by_key(|(_, v)| v.last_accessed_secs)
373                    .map(|(k, _)| k.clone())
374                {
375                    cache.remove(&lru_key);
376                }
377            }
378
379            cache.insert(cache_key, cached);
380        }
381    }

527    /// Generate learning spikes for neural plasticity
528    fn generate_learning_spikes(&self, pathway: &[u64], matched_nodes: &[u64]) -> CoreResult<()> {
529        let current_time = Instant::now();
530
531        // Acquire write lock for spike patterns
532        if let Ok(mut spike_patterns) = self.spike_patterns.write() {
533            // Generate spikes for nodes in the activation pathway
534            for &node_id in pathway {
535                spike_patterns
536                    .entry(node_id)
537                    .or_insert_with(Vec::new)
538                    .push(current_time);
539            }
540
541            // Generate output spikes for matched nodes (slightly later for causality)
542            for &node_id in matched_nodes {
543                spike_patterns
544                    .entry(node_id)
545                    .or_insert_with(Vec::new)
546                    .push(current_time + std::time::Duration::from_micros(10));
547            }
548
549            // Apply STDP learning based on spike timing
550            if let Ok(mut learning_engine) = self.learning_engine.write() {
551                if let Ok(mut network) = self.network.write() {
552                    // Clone spike_patterns to avoid holding multiple locks
553                    let spikes_clone: HashMap<u64, Vec<Instant>> = spike_patterns
554                        .iter()
555                        .map(|(k, v)| (*k, v.clone()))
556                        .collect();
557
558                    drop(spike_patterns); // Release lock before STDP
559
560                    // Apply spike-timing-dependent plasticity
561                    let _ = learning_engine.apply_stdp(&mut network, &spikes_clone);
562                }
563            } else {
564                // Clean up old spikes (keep only recent history)
565                let cutoff_time = current_time
566                    .checked_sub(std::time::Duration::from_secs(1))
567                    .unwrap();
568                for spike_list in spike_patterns.values_mut() {
569                    spike_list.retain(|&t| t > cutoff_time);
570                }
571            }
572        }
573
574        Ok(())
575    }

601    /// Returns true if no efficient index lookup is available for this field
602    fn is_full_scan_likely(&self, field: &str, operator: &str) -> bool {
603        // Fields that typically don't have indexes in neuromorphic systems
604        let unindexed_patterns = [
605            "description",
606            "content",
607            "text",
608            "body",
609            "notes",
610            "metadata",

612        ];
613
614        // Check if the field name matches common unindexed patterns
615        let field_lower = field.to_lowercase();
616        for pattern in &unindexed_patterns {
617            if field_lower.contains(pattern) {
618                return true;
619            }
620        }
621
622        // LIKE queries with leading wildcards always cause full scans
623        if operator == "LIKE" {
624            return true;
625        }
626
627        // NOT operators typically cause full scans
628        if operator.starts_with("NOT") || operator == "!=" || operator == "<>" {
629            return true;
630        }
631
632        false
633    }

676    /// Determine the best index type for a field based on usage patterns
677    fn suggest_index_type(&self, field: &str, operators: &[&str]) -> SuggestedIndexType {
678        let field_lower = field.to_lowercase();
679
680        // DNA sequence fields benefit from K-mer indexes
681        if field_lower.contains("dna")
682            || field_lower.contains("sequence")
683            || field_lower.contains("genome")
684        {
685            return SuggestedIndexType::DnaKmer;
686        }
687
688        // Neural/embedding fields benefit from similarity indexes
689        if field_lower.contains("neural")
690            || field_lower.contains("embedding")
691            || field_lower.contains("vector")
692        {
693            return SuggestedIndexType::NeuralSimilarity;
694        }
695
696        // Quantum state fields benefit from entanglement indexes
697        if field_lower.contains("quantum") || field_lower.contains("entangle") {
698            return SuggestedIndexType::QuantumEntanglement;
699        }
700
701        // Check operators to determine BTree vs Hash
702        let has_range_ops = operators
703            .iter()
704            .any(|op| matches!(*op, ">" | "<" | ">=" | "<=" | "BETWEEN"));
705
706        if has_range_ops {
707            SuggestedIndexType::BTree
708        } else {
709            // Pure equality checks benefit from hash indexes
710            SuggestedIndexType::Hash
711        }
712    }

760    /// Generate comprehensive optimization suggestions for a query
761    ///
762    /// Analyzes the query structure, conditions, and neural pathway to provide
763    /// actionable optimization recommendations with estimated performance benefits.
764    ///
765    /// # Arguments
766    /// * `query` - The query to analyze for optimization opportunities
767    ///
768    /// # Returns
769    /// A vector of `OptimizationSuggestion` sorted by priority and estimated improvement
770    #[must_use]
771    pub fn generate_optimization_suggestions(&self, query: &Query) -> Vec<OptimizationSuggestion> {
772        let mut suggestions = Vec::new();
773
774        // Track fields that appear in conditions for index analysis
775        let mut field_operators: HashMap<String, Vec<&str>> = HashMap::new();
776
777        // Analyze each condition for optimization opportunities
778        for condition in &query.conditions {
779            field_operators
780                .entry(condition.field.clone())
781                .or_default()
782                .push(&condition.operator);
783        }
784
785        // Generate index suggestions for frequently used fields
786        for (field, operators) in &field_operators {
787            if self.is_full_scan_likely(field, operators.first().unwrap_or(&"=")) {
788                // Find the corresponding condition to get its weight
789                let weight = query
790                    .conditions
791                    .iter()
792                    .find(|c| &c.field == field)
793                    .map_or(0.5, |c| c.weight);
794
795                let (improvement, confidence) =
796                    self.estimate_index_benefit(field, operators.first().unwrap_or(&"="), weight);
797                let index_type = self.suggest_index_type(field, &operators.clone());
798
799                suggestions.push(
800                    OptimizationSuggestion::new(
801                        OptimizationSuggestionType::CreateIndex,
802                        format!(
803                            "Create {} index on field '{}' to avoid full table scan. \
804                             This field is used with operators: {}",
805                            format!("{index_type:?}").to_lowercase(),
806                            field,
807                            operators.join(", ")
808                        ),
809                        vec![field.clone()],
810                        improvement,
811                        confidence,
812                        2,
813                    )
814                    .with_index_type(index_type)
815                    .with_metadata("operators", &operators.join(",")),
816                );
817            }
818        }
819
820        // Suggest composite index for multiple conditions on same query
821        if query.conditions.len() >= 2 {
822            let fields: Vec<String> = query.conditions.iter().map(|c| c.field.clone()).collect();
823
824            // Calculate combined weight
825            let total_weight: f32 = query.conditions.iter().map(|c| c.weight).sum();
826            let avg_weight = total_weight / query.conditions.len() as f32;
827
828            // Composite index benefit increases with number of conditions
829            let condition_bonus = ((query.conditions.len() - 1) as f32 * 0.1).min(0.3);
830            let improvement = avg_weight.mul_add(0.4, condition_bonus).min(0.8);
831
832            suggestions.push(
833                OptimizationSuggestion::new(
834                    OptimizationSuggestionType::CreateCompositeIndex,
835                    format!(
836                        "Consider creating a composite index on fields ({}) for this \
837                         multi-condition query. Composite indexes can significantly improve \
838                         performance when querying multiple fields together.",
839                        fields.join(", ")
840                    ),
841                    fields,
842                    improvement,
843                    0.7,
844                    3,
845                )
846                .with_index_type(SuggestedIndexType::BTree)
847                .with_metadata("condition_count", &query.conditions.len().to_string()),
848            );
849        }
850
851        // Suggest batch processing for queries with many target nodes
852        if query.target_nodes.len() > 10 {
853            let batch_count = (query.target_nodes.len() / 10).max(2);
854            let improvement = ((query.target_nodes.len() as f32 - 10.0) / 100.0).min(0.5);
855
856            suggestions.push(
857                OptimizationSuggestion::new(
858                    OptimizationSuggestionType::BatchProcessing,
859                    format!(
860                        "Query targets {} nodes. Consider breaking into {} batches of ~10 nodes \
861                         each for better parallelization and memory efficiency. This allows \
862                         the neuromorphic processor to activate optimal neural pathways per batch.",
863                        query.target_nodes.len(),
864                        batch_count
865                    ),
866                    vec!["target_nodes".to_string()],
867                    improvement,
868                    0.65,
869                    4,
870                )
871                .with_metadata("node_count", &query.target_nodes.len().to_string())
872                .with_metadata("suggested_batches", &batch_count.to_string()),
873            );
874        }
875
876        // Check for neural pathway optimization opportunities
877        if let Some(neural_suggestion) = self.analyze_neural_pathway_efficiency(query) {
878            suggestions.push(neural_suggestion);
879        }
880
881        // Suggest neuromorphic caching for high-priority queries
882        if query.priority > 200 {
883            suggestions.push(
884                OptimizationSuggestion::new(
885                    OptimizationSuggestionType::NeuromorphicCaching,
886                    format!(
887                        "High-priority query (priority={}) would benefit from extended \
888                         neuromorphic caching. Consider enabling persistent spike pattern \
889                         caching to speed up repeated similar queries.",
890                        query.priority
891                    ),
892                    vec!["query_cache".to_string()],
893                    0.25,
894                    0.6,
895                    5,
896                )
897                .with_metadata("priority", &query.priority.to_string()),
898            );
899        }
900
901        // Suggest query restructuring for complex queries with many conditions
902        if query.conditions.len() > 5 {
903            suggestions.push(
904                OptimizationSuggestion::new(
905                    OptimizationSuggestionType::RestructureQuery,
906                    format!(
907                        "Query has {} conditions. Consider restructuring into multiple \
908                         simpler queries with UNION or using subqueries. Complex queries \
909                         can overwhelm the neural pathway activation algorithm.",
910                        query.conditions.len()
911                    ),
912                    query.conditions.iter().map(|c| c.field.clone()).collect(),
913                    0.35,
914                    0.55,
915                    4,
916                )
917                .with_metadata("condition_count", &query.conditions.len().to_string()),
918            );
919        }
920
921        // Sort suggestions by priority (ascending) then by estimated improvement (descending)
922        suggestions.sort_by(|a, b| {
923            a.priority.cmp(&b.priority).then(
924                b.estimated_improvement
925                    .partial_cmp(&a.estimated_improvement)
926                    .unwrap_or(std::cmp::Ordering::Equal),
927            )
928        });
929
930        suggestions
931    }

933    /// Update query processing statistics
934    fn update_statistics(&self, result: &QueryResult) {
935        if let Ok(mut stats) = self.query_statistics.write() {
936            stats.total_queries += 1;
937
938            // Update average processing time
939            let total_time = stats.average_processing_time_ns * (stats.total_queries - 1)
940                + result.execution_time_ns;
941            stats.average_processing_time_ns = total_time / stats.total_queries;
942
943            // Track activation score as optimization event if above threshold
944            if result.activation_score > 0.7 {
945                stats.optimization_events += 1;
946            }
947        }
948    }
