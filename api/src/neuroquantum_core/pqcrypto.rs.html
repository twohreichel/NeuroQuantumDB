7/// Implements NIST post-quantum standards:

19/// Type aliases for ML-KEM-768 (NIST Security Level 3)
20type MlKemDecapsulationKey = <MlKem768 as KemCore>::DecapsulationKey;
21type MlKemEncapsulationKey = <MlKem768 as KemCore>::EncapsulationKey;

23/// ML-KEM-768 ciphertext size in bytes (1088 bytes)
24const MLKEM768_CIPHERTEXT_SIZE: usize = 1088;
25/// ML-KEM-768 shared secret size in bytes (32 bytes)

49/// Post-quantum cryptographic key manager
50///
51/// Uses ML-KEM-768 (`RustCrypto` implementation) for key encapsulation and
52/// ML-DSA-65 (pqcrypto implementation) for digital signatures.
53/// Both provide NIST Security Level 3.
54#[derive(Clone)]
55#[allow(clippy::struct_field_names)]
56pub struct PQCryptoManager {
57    // ML-KEM (Kyber) keys for key encapsulation using RustCrypto ml-kem
58    mlkem_encapsulation_key: Arc<MlKemEncapsulationKey>,
59    mlkem_decapsulation_key: Arc<MlKemDecapsulationKey>,
60
61    // ML-DSA (Dilithium) keys for digital signatures
62    mldsa_public_key: Arc<mldsa65::PublicKey>,
63    mldsa_secret_key: Arc<mldsa65::SecretKey>,
64}
65
66/// Quantum-resistant token claims
67#[derive(Debug, Clone, Serialize, Deserialize)]
68pub struct QuantumTokenClaims {
69    pub user_id: String,
70    pub session_id: String,
71    pub timestamp: i64,
72    pub quantum_signature: String, // Base64-encoded ML-DSA signature
73    pub kem_ciphertext: String,    // Base64-encoded ML-KEM ciphertext
74}
75
76impl PQCryptoManager {
77    /// Create a new post-quantum crypto manager with generated keys
78    pub fn new() -> Self {
79        let mut rng = rand::thread_rng();
80
81        // Generate ML-KEM-768 key pair (NIST Security Level 3) using RustCrypto
82        let (dk, ek) = MlKem768::generate(&mut rng);
83
84        // Generate ML-DSA-65 key pair (NIST Security Level 3)
85        let (mldsa_pk, mldsa_sk) = mldsa65::keypair();
86
87        // ML-KEM-768 key sizes are fixed by the standard
88        // Encapsulation key: 1184 bytes, Decapsulation key: 2400 bytes
89        tracing::info!(
90            "üîê Generated post-quantum key pairs: ML-KEM-768 (ek: 1184 bytes, dk: 2400 bytes), ML-DSA-65 (pk: {} bytes, sk: {} bytes)",
91            mldsa_pk.as_bytes().len(),
92            mldsa_sk.as_bytes().len()
93        );
94
95        Self {
96            mlkem_encapsulation_key: Arc::new(ek),
97            mlkem_decapsulation_key: Arc::new(dk),
98            mldsa_public_key: Arc::new(mldsa_pk),
99            mldsa_secret_key: Arc::new(mldsa_sk),
100        }
101    }
102
103    /// Get the ML-KEM encapsulation (public) key as base64
104    #[must_use]
105    pub fn get_mlkem_public_key_base64(&self) -> String {
106        let encoded = self.mlkem_encapsulation_key.as_bytes();
107        base64::Engine::encode(&base64::engine::general_purpose::STANDARD, encoded)
108    }
109
110    /// Get the ML-DSA public key as base64
111    #[must_use]
112    pub fn get_mldsa_public_key_base64(&self) -> String {
113        base64::Engine::encode(
114            &base64::engine::general_purpose::STANDARD,
115            self.mldsa_public_key.as_bytes(),
116        )
117    }
118
119    /// Sign a message using ML-DSA (Dilithium)
120    #[must_use]
121    pub fn sign_message(&self, message: &[u8]) -> Vec<u8> {
122        let signed_msg = mldsa65::sign(message, &self.mldsa_secret_key);
123        // SignedMessage is a wrapper, convert to bytes using the trait
124        use pqcrypto_traits::sign::SignedMessage;
125        signed_msg.as_bytes().to_vec()
126    }
127
128    /// Verify a signature using ML-DSA (Dilithium)
129    pub fn verify_signature(&self, signed_message_bytes: &[u8]) -> Result<Vec<u8>, PQCryptoError> {
130        // Create SignedMessage from bytes using the trait
131        use pqcrypto_traits::sign::SignedMessage;
132        let signed_msg = mldsa65::SignedMessage::from_bytes(signed_message_bytes)
133            .map_err(|_| PQCryptoError::SignatureVerificationFailed)?;
134
135        // Open (verify) the signed message and extract original message
136        let opened_msg = mldsa65::open(&signed_msg, &self.mldsa_public_key)
137            .map_err(|_| PQCryptoError::SignatureVerificationFailed)?;
138
139        Ok(opened_msg)
140    }
141
142    /// Encapsulate a shared secret using ML-KEM (Kyber)
143    /// Returns (`ciphertext_bytes`, `shared_secret_bytes`)
144    ///
145    /// The ciphertext can be transmitted to the key holder who can then
146    /// decapsulate it using their decapsulation key to obtain the same shared secret.
147    /// For ML-KEM-768, ciphertext is 1088 bytes and shared secret is 32 bytes.
148    ///
149    /// # Panics
150    ///
151    /// Panics if ML-KEM encapsulation fails with a valid key, which should never happen
152    /// according to the algorithm specification. This is a catastrophic cryptographic failure.
153    #[allow(clippy::expect_used)] // Encapsulation with valid key should never fail per spec
154    #[must_use]
155    pub fn encapsulate(&self) -> (Vec<u8>, Vec<u8>) {
156        let mut rng = rand::thread_rng();
157
158        // Encapsulate using the encapsulation key
159        let (ct, shared_secret) = self
160            .mlkem_encapsulation_key
161            .encapsulate(&mut rng)
162            .expect("ML-KEM encapsulation should not fail with valid key");
163
164        // Convert to bytes - Ciphertext and SharedKey are Array types that implement AsRef<[u8]>
165        let ct_bytes: Vec<u8> = AsRef::<[u8]>::as_ref(&ct).to_vec();
166        let ss_bytes: Vec<u8> = AsRef::<[u8]>::as_ref(&shared_secret).to_vec();
167
168        (ct_bytes, ss_bytes)
169    }
170
171    /// Decapsulate a shared secret using ML-KEM (Kyber)
172    ///
173    /// Takes ciphertext bytes (as returned by encapsulate) and returns the shared secret.
174    /// This correctly deserializes the ciphertext and performs proper decapsulation
175    /// using the decapsulation (secret) key.
176    pub fn decapsulate(&self, ciphertext_bytes: &[u8]) -> Result<Vec<u8>, PQCryptoError> {
177        // Validate ciphertext size for ML-KEM-768 (1088 bytes)
178        if ciphertext_bytes.len() != MLKEM768_CIPHERTEXT_SIZE {
179            return Err(PQCryptoError::InvalidCiphertext(format!(
180                "Invalid ciphertext length: expected {} bytes, got {} bytes",
181                MLKEM768_CIPHERTEXT_SIZE,
182                ciphertext_bytes.len()
183            )));
184        }
185
186        // Deserialize the ciphertext from bytes using TryFrom
187        // Ciphertext<MlKem768> is a hybrid_array::Array with fixed size 1088
188        let ct: Ciphertext<MlKem768> = ciphertext_bytes.try_into().map_err(|_| {
189            PQCryptoError::InvalidCiphertext("Failed to parse ciphertext bytes".to_string())
190        })?;
191
192        // Decapsulate using the decapsulation key
193        let shared_secret = self
194            .mlkem_decapsulation_key
195            .decapsulate(&ct)
196            .map_err(|()| {
197                PQCryptoError::DecapsulationFailed(
198                    "ML-KEM decapsulation failed - possibly corrupted ciphertext".to_string(),
199                )
200            })?;
201
202        let shared_secret_bytes = AsRef::<[u8]>::as_ref(&shared_secret).to_vec();
203
204        // Validate shared secret size (should always be 32 bytes for ML-KEM-768)
205        if shared_secret_bytes.len() != MLKEM768_SHARED_SECRET_SIZE {
206            return Err(PQCryptoError::DecapsulationFailed(format!(
207                "Invalid shared secret size: expected {} bytes, got {} bytes",
208                MLKEM768_SHARED_SECRET_SIZE,
209                shared_secret_bytes.len()
210            )));
211        }
212
213        Ok(shared_secret_bytes)
214    }

216    /// Generate quantum token claims with signatures
217    pub fn generate_quantum_claims(
218        &self,
219        user_id: &str,
220        session_id: &str,
221    ) -> Result<QuantumTokenClaims, PQCryptoError> {
222        let timestamp = chrono::Utc::now().timestamp();
223        let message = format!("{user_id}:{session_id}:{timestamp}");
224
225        // Sign the message with ML-DSA
226        let signature = self.sign_message(message.as_bytes());
227        let quantum_signature =
228            base64::Engine::encode(&base64::engine::general_purpose::STANDARD, &signature);
229
230        // Generate KEM ciphertext
231        let (ciphertext, _shared_secret) = self.encapsulate();
232        let kem_ciphertext =
233            base64::Engine::encode(&base64::engine::general_purpose::STANDARD, &ciphertext);
234
235        Ok(QuantumTokenClaims {
236            user_id: user_id.to_string(),
237            session_id: session_id.to_string(),
238            timestamp,
239            quantum_signature,
240            kem_ciphertext,
241        })
242    }
243
244    /// Verify quantum token claims
245    pub fn verify_quantum_claims(&self, claims: &QuantumTokenClaims) -> Result<(), PQCryptoError> {
246        // Reconstruct the original message
247        let message = format!(
248            "{}:{}:{}",
249            claims.user_id, claims.session_id, claims.timestamp
250        );
251
252        // Decode signature from base64
253        let signature = base64::Engine::decode(
254            &base64::engine::general_purpose::STANDARD,
255            &claims.quantum_signature,
256        )
257        .map_err(|e| PQCryptoError::EncodingError(format!("Invalid base64: {e}")))?;
258
259        // Verify the signature
260        let verified_msg = self.verify_signature(&signature)?;
261
262        // Check if the verified message matches
263        if verified_msg != message.as_bytes() {
264            return Err(PQCryptoError::SignatureVerificationFailed);
265        }
266
267        Ok(())
268    }
