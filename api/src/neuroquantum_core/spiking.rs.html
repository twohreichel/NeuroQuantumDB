1//! # Spiking Neural Network Models
2//!
3//! This module implements biologically accurate spiking neural network models,
4//! with the Izhikevich neuron model as the primary implementation.
5//!
6//! ## Izhikevich Model
7//!
8//! The Izhikevich model (2003) provides a computationally efficient yet
9//! biologically accurate representation of cortical neurons. It can reproduce
10//! the firing patterns of all known types of cortical neurons using only
11//! two coupled differential equations.
12//!
13//! ### Equations
14//!
15//! ```text
16//! dv/dt = 0.04vÂ² + 5v + 140 - u + I
17//! du/dt = a(bv - u)
18//!
19//! if v >= 30 mV then v := c, u := u + d
20//! ```
21//!
22//! Where:
23//! - `v` is the membrane potential (mV)
24//! - `u` is the membrane recovery variable
25//! - `I` is the synaptic current or injected current
26//! - `a`, `b`, `c`, `d` are dimensionless parameters
27//!
28//! ### Neuron Types
29//!
30//! | Type | a | b | c | d | Description |
31//! |------|---|---|---|---|-------------|
32//! | RS   | 0.02 | 0.2 | -65 | 8 | Regular Spiking (excitatory) |
33//! | IB   | 0.02 | 0.2 | -55 | 4 | Intrinsically Bursting |
34//! | CH   | 0.02 | 0.2 | -50 | 2 | Chattering |
35//! | FS   | 0.1  | 0.2 | -65 | 2 | Fast Spiking (inhibitory) |
36//! | TC   | 0.02 | 0.25 | -65 | 0.05 | Thalamocortical |
37//! | RZ   | 0.1  | 0.26 | -65 | 2 | Resonator |
38//! | LTS  | 0.02 | 0.25 | -65 | 2 | Low-Threshold Spiking |
39//!
40//! ## References
41//!
42//! - Izhikevich, E.M. (2003). Simple model of spiking neurons.
43//!   IEEE Transactions on Neural Networks, 14(6):1569-1572.
44//! - Izhikevich, E.M. (2004). Which model to use for cortical spiking neurons?
45//!   IEEE Transactions on Neural Networks, 15(5):1063-1070.

55/// Cortical neuron types based on the Izhikevich model.
56///
57/// Each type has characteristic firing patterns observed in real cortical neurons.
58#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
59pub enum IzhikevichNeuronType {
60    /// Regular Spiking (RS) - Most common excitatory neurons in cortex.
61    /// Pyramidal cells in layers II-VI exhibit this pattern.
62    /// Characterized by adaptation: decreasing firing frequency with sustained input.
63    RegularSpiking,
64
65    /// Intrinsically Bursting (IB) - Layer V pyramidal neurons.
66    /// Fire an initial burst of spikes followed by regular spiking.
67    /// Important for initiating and propagating cortical activity.
68    IntrinsicallyBursting,
69
70    /// Chattering (CH) - Fast rhythmic bursting.
71    /// Found in superficial layers, generate gamma oscillations.
72    /// Bursts at 25-70 Hz with 2-5 spikes per burst.
73    Chattering,
74
75    /// Fast Spiking (FS) - `GABAergic` interneurons (basket cells, chandelier cells).
76    /// High-frequency firing without adaptation.
77    /// Critical for cortical inhibition and gamma rhythms.
78    FastSpiking,
79
80    /// Thalamocortical (TC) - Thalamic relay neurons.
81    /// Can fire in two modes: tonic (depolarized) and burst (hyperpolarized).
82    /// Essential for sensory processing and sleep rhythms.
83    Thalamocortical,
84
85    /// Resonator (RZ) - Subthreshold oscillations.
86    /// Prefer inputs at specific frequencies, enabling frequency selectivity.
87    /// Found in various brain regions including basal ganglia.
88    Resonator,
89
90    /// Low-Threshold Spiking (LTS) - Martinotti cells.
91    /// Fire at low frequencies with adaptation.
92    /// Target distal dendrites of pyramidal cells.
93    LowThresholdSpiking,
94}

97    /// Get the characteristic parameters (a, b, c, d) for this neuron type.

100    /// from Izhikevich's original publications and subsequent refinements.
101    #[must_use]
102    pub const fn parameters(&self) -> IzhikevichParameters {
103        match self {
104            | Self::RegularSpiking => IzhikevichParameters {
105                a: 0.02,
106                b: 0.2,
107                c: -65.0,
108                d: 8.0,
109            },
110            | Self::IntrinsicallyBursting => IzhikevichParameters {
111                a: 0.02,
112                b: 0.2,
113                c: -55.0,
114                d: 4.0,
115            },
116            | Self::Chattering => IzhikevichParameters {
117                a: 0.02,
118                b: 0.2,
119                c: -50.0,
120                d: 2.0,
121            },
122            | Self::FastSpiking => IzhikevichParameters {
123                a: 0.1,
124                b: 0.2,
125                c: -65.0,
126                d: 2.0,
127            },
128            | Self::Thalamocortical => IzhikevichParameters {
129                a: 0.02,
130                b: 0.25,
131                c: -65.0,
132                d: 0.05,
133            },
134            | Self::Resonator => IzhikevichParameters {
135                a: 0.1,

137                c: -65.0,
138                d: 2.0,
139            },
140            | Self::LowThresholdSpiking => IzhikevichParameters {

146        }

152        matches!(
153            self,
154            Self::RegularSpiking
155                | Self::IntrinsicallyBursting
156                | Self::Chattering
157                | Self::Thalamocortical

159    }

164        matches!(self, Self::FastSpiking | Self::LowThresholdSpiking)
165    }

168/// Izhikevich model parameters.
169///
170/// These four dimensionless parameters completely determine the neuron's dynamics.
171#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
172pub struct IzhikevichParameters {
173    /// Time scale of recovery variable `u`.
174    /// Smaller values result in slower recovery.
175    /// Typical range: 0.02 (slow) to 0.1 (fast)
176    pub a: f64,
177
178    /// Sensitivity of `u` to subthreshold fluctuations of `v`.
179    /// Greater values couple `u` and `v` more strongly.
180    /// Typical range: 0.2 to 0.26
181    pub b: f64,
182
183    /// After-spike reset value of membrane potential `v`.
184    /// Unit: mV
185    /// Typical range: -65 to -50 mV
186    pub c: f64,
187
188    /// After-spike increment of recovery variable `u`.
189    /// Represents the total amount of outward minus inward currents
190    /// activated during the spike and affecting after-spike behavior.
191    /// Typical range: 0.05 to 8
192    pub d: f64,
193}
194
195impl Default for IzhikevichParameters {
196    fn default() -> Self {
197        // Default to Regular Spiking parameters
198        IzhikevichNeuronType::RegularSpiking.parameters()
199    }
200}

304    /// Add synaptic input to the neuron.

321    /// Simulate one timestep of neuron dynamics.
322    ///
323    /// Uses the Euler method with 0.5 ms resolution (two 0.5 ms steps per ms).
324    /// Returns `true` if the neuron fired a spike during this timestep.
325    ///
326    /// # Arguments
327    ///
328    /// * `current_time` - Current simulation timestep
329    ///
330    /// # Details
331    ///
332    /// The equations are solved using the forward Euler method:
333    /// ```text
334    /// v += 0.5 * (0.04*v^2 + 5*v + 140 - u + I)  // twice for numerical stability
335    /// u += a * (b*v - u)
336    /// ```
337    #[instrument(level = "trace", skip(self))]
338    pub fn step(&mut self, current_time: u64) -> bool {
339        let total_input = self.current_input + self.synaptic_input;
340        self.synaptic_input = 0.0; // Reset for next timestep
341
342        let IzhikevichParameters { a, b, c, d } = self.params;
343
344        // Two half-steps for numerical stability (Euler method with dt=0.5ms)
345        for _ in 0..2 {
346            self.v += 0.5
347                * ((0.04 * self.v).mul_add(self.v, 5.0 * self.v) + 140.0 - self.u + total_input);
348        }
349        self.u += a * b.mul_add(self.v, -self.u);
350
351        // Check for spike
352        if self.v >= self.spike_threshold {
353            // Spike reset
354            self.v = c;
355            self.u += d;
356
357            // Record spike
358            self.spike_count += 1;
359            self.last_spike_time = Some(current_time);
360
361            // Update history
362            if self.spike_history.len() >= self.max_history_length {
363                self.spike_history.remove(0);
364            }
365            self.spike_history.push(current_time);
366
367            debug!(neuron_id = self.id, time = current_time, "Spike fired");
368            return true;
369        }
370
371        false
372    }
373
374    /// Get the instantaneous firing rate (Hz) based on recent spike history.
375    ///
376    /// Calculated from the inter-spike interval of the last two spikes.
377    #[must_use]
378    pub fn firing_rate(&self) -> Option<f64> {
379        if self.spike_history.len() < 2 {
380            return None;
381        }
382
383        let last = self.spike_history[self.spike_history.len() - 1];
384        let second_last = self.spike_history[self.spike_history.len() - 2];
385        let isi = (last - second_last) as f64; // Inter-spike interval in ms
386
387        if isi > 0.0 {
388            Some(1000.0 / isi) // Convert to Hz
389        } else {
390            None
391        }
392    }
393
394    /// Get the average firing rate over the entire spike history.
395    #[must_use]
396    pub fn average_firing_rate(&self, simulation_duration_ms: u64) -> f64 {
397        if simulation_duration_ms == 0 {
398            return 0.0;
399        }
400        (self.spike_count as f64 / simulation_duration_ms as f64) * 1000.0
401    }

403    /// Reset the neuron to its initial state.
404    pub fn reset(&mut self) {
405        self.v = self.params.c;
406        self.u = self.params.b * self.params.c;
407        self.spike_count = 0;
408        self.last_spike_time = None;
409        self.spike_history.clear();
410        self.synaptic_input = 0.0;
411        self.current_input = 0.0;
412    }
413
414    /// Get the spike history.
415    #[must_use]
416    pub fn spike_times(&self) -> &[u64] {
417        &self.spike_history
418    }
419
420    /// Check if the neuron is in a refractory state.
421    ///
422    /// Refractory period is approximately 2 ms after a spike.
423    #[must_use]
424    pub const fn is_refractory(&self, current_time: u64, refractory_period_ms: u64) -> bool {
425        if let Some(last_spike) = self.last_spike_time {
426            current_time.saturating_sub(last_spike) < refractory_period_ms
427        } else {
428            false
429        }
430    }

433/// Conductance-based synapse for spiking neural networks.
434///
435/// Models synaptic transmission with exponential decay dynamics,
436/// supporting both AMPA/NMDA (excitatory) and GABA (inhibitory) synapses.
437#[derive(Debug, Clone, Serialize, Deserialize)]
438pub struct SpikingSynapse {
439    /// Source neuron ID
440    pub pre_id: u64,
441
442    /// Target neuron ID
443    pub post_id: u64,
444
445    /// Synaptic weight (conductance)
446    pub weight: f64,
447
448    /// Synaptic delay (in timesteps)
449    pub delay: u64,
450
451    /// Synaptic time constant (ms)
452    pub tau: f64,
453
454    /// Current synaptic conductance
455    pub conductance: f64,
456
457    /// Reversal potential (mV)
458    /// AMPA/NMDA: 0 mV (excitatory)
459    /// GABA-A: -70 mV (inhibitory)
460    pub reversal_potential: f64,
461
462    /// Queue of pending spikes (time, weight)
463    spike_queue: Vec<(u64, f64)>,
464}
465
466impl SpikingSynapse {
467    /// Create a new excitatory synapse (AMPA-like).
468    #[must_use]
469    pub const fn excitatory(pre_id: u64, post_id: u64, weight: f64) -> Self {
470        Self {
471            pre_id,
472            post_id,
473            weight,
474            delay: 1,
475            tau: 5.0, // AMPA time constant ~5 ms
476            conductance: 0.0,
477            reversal_potential: 0.0, // Excitatory reversal potential
478            spike_queue: Vec::new(),
479        }
480    }
481
482    /// Create a new inhibitory synapse (GABA-A-like).
483    #[must_use]
484    pub const fn inhibitory(pre_id: u64, post_id: u64, weight: f64) -> Self {
485        Self {
486            pre_id,
487            post_id,
488            weight: weight.abs(), // Weight magnitude
489            delay: 1,
490            tau: 10.0, // GABA-A time constant ~10 ms
491            conductance: 0.0,
492            reversal_potential: -70.0, // Inhibitory reversal potential
493            spike_queue: Vec::new(),
494        }
495    }
496
497    /// Create a synapse with custom parameters.
498    #[must_use]
499    pub const fn with_params(
500        pre_id: u64,
501        post_id: u64,
502        weight: f64,
503        delay: u64,
504        tau: f64,
505        reversal_potential: f64,
506    ) -> Self {
507        Self {
508            pre_id,
509            post_id,
510            weight,
511            delay,
512            tau,
513            conductance: 0.0,
514            reversal_potential,
515            spike_queue: Vec::new(),
516        }
517    }
518
519    /// Receive a presynaptic spike.
520    pub fn receive_spike(&mut self, arrival_time: u64) {
521        self.spike_queue
522            .push((arrival_time + self.delay, self.weight));
523    }
524
525    /// Update synaptic conductance and return current to inject into postsynaptic neuron.
526    ///
527    /// # Arguments
528    ///
529    /// * `current_time` - Current simulation timestep
530    /// * `post_v` - Postsynaptic membrane potential (for conductance-based current)
531    pub fn update(&mut self, current_time: u64, post_v: f64) -> f64 {
532        // Process arriving spikes
533        let mut spike_contribution = 0.0;
534        self.spike_queue.retain(|(arrival_time, weight)| {
535            if *arrival_time <= current_time {
536                spike_contribution += weight;
537                false
538            } else {
539                true
540            }
541        });
542        self.conductance += spike_contribution;
543
544        // Exponential decay of conductance
545        let decay = (-1.0 / self.tau).exp();
546        self.conductance *= decay;
547
548        // Calculate synaptic current (Ohm's law: I = g * (E - V))
549        self.conductance * (self.reversal_potential - post_v)
550    }
551}

553/// Spike-Timing-Dependent Plasticity (STDP) rule.
554///
555/// Implements the classic asymmetric STDP window:
556/// - Pre-before-post: LTP (potentiation)
557/// - Post-before-pre: LTD (depression)
558#[derive(Debug, Clone, Serialize, Deserialize)]
559pub struct STDPRule {
560    /// Maximum potentiation (positive weight change)
561    pub a_plus: f64,
562
563    /// Maximum depression (negative weight change)
564    pub a_minus: f64,
565
566    /// Time constant for potentiation (ms)
567    pub tau_plus: f64,
568
569    /// Time constant for depression (ms)
570    pub tau_minus: f64,
571
572    /// Minimum weight
573    pub w_min: f64,
574
575    /// Maximum weight
576    pub w_max: f64,
577}
578
579impl Default for STDPRule {
580    fn default() -> Self {
581        Self {
582            a_plus: 0.005,
583            a_minus: 0.00525, // Slightly stronger depression for stability
584            tau_plus: 20.0,
585            tau_minus: 20.0,
586            w_min: 0.0,
587            w_max: 1.0,
588        }
589    }
590}
591
592impl STDPRule {
593    /// Calculate weight change based on spike timing difference.
594    ///
595    /// # Arguments
596    ///
597    /// * `delta_t` - Time difference (`t_post` - `t_pre`) in ms
598    /// * `current_weight` - Current synaptic weight
599    ///
600    /// # Returns
601    ///
602    /// The new weight after applying STDP
603    #[must_use]
604    pub fn apply(&self, delta_t: f64, current_weight: f64) -> f64 {
605        let dw = if delta_t > 0.0 {
606            // Pre before post: LTP
607            self.a_plus * (-delta_t / self.tau_plus).exp()
608        } else {
609            // Post before pre: LTD
610            -self.a_minus * (delta_t / self.tau_minus).exp()
611        };
612
613        // Apply weight change with soft bounds
614        let new_weight = current_weight + dw;
615        new_weight.clamp(self.w_min, self.w_max)
616    }
617}

619/// Spiking Neural Network using Izhikevich neurons.
620///
621/// This network supports real-time simulation with STDP learning.
622#[derive(Debug)]
623pub struct SpikingNeuralNetwork {
624    /// Neurons in the network
625    neurons: RwLock<HashMap<u64, IzhikevichNeuron>>,
626
627    /// Synapses indexed by postsynaptic neuron ID
628    synapses: RwLock<HashMap<u64, Vec<SpikingSynapse>>>,
629
630    /// STDP learning rule
631    pub stdp_rule: STDPRule,
632
633    /// Enable STDP learning
634    pub learning_enabled: bool,
635
636    /// Current simulation time (ms)
637    current_time: RwLock<u64>,
638
639    /// Network statistics
640    total_spikes: RwLock<u64>,
641}
642
643impl SpikingNeuralNetwork {
644    /// Create a new spiking neural network.
645    #[must_use]
646    pub fn new() -> Self {
647        Self {
648            neurons: RwLock::new(HashMap::new()),
649            synapses: RwLock::new(HashMap::new()),
650            stdp_rule: STDPRule::default(),
651            learning_enabled: true,
652            current_time: RwLock::new(0),
653            total_spikes: RwLock::new(0),
654        }
655    }
656
657    /// Add a neuron to the network.
658    pub fn add_neuron(&self, neuron: IzhikevichNeuron) -> CoreResult<()> {
659        let mut neurons = self.neurons.write().map_err(|e| {
660            CoreError::LockError(format!("Failed to acquire neurons write lock: {e}"))
661        })?;
662
663        if neurons.contains_key(&neuron.id) {
664            return Err(CoreError::InvalidOperation(format!(
665                "Neuron with ID {} already exists",
666                neuron.id
667            )));
668        }
669
670        neurons.insert(neuron.id, neuron);
671        Ok(())
672    }
673
674    /// Add multiple neurons of a specific type.
675    pub fn add_neuron_population(
676        &self,
677        start_id: u64,
678        count: u64,
679        neuron_type: IzhikevichNeuronType,
680    ) -> CoreResult<Vec<u64>> {
681        let mut neurons = self.neurons.write().map_err(|e| {
682            CoreError::LockError(format!("Failed to acquire neurons write lock: {e}"))
683        })?;
684
685        let mut ids = Vec::with_capacity(count as usize);
686        for i in 0..count {
687            let id = start_id + i;
688            if neurons.contains_key(&id) {
689                return Err(CoreError::InvalidOperation(format!(
690                    "Neuron with ID {id} already exists"
691                )));
692            }
693            neurons.insert(id, IzhikevichNeuron::new(id, neuron_type));
694            ids.push(id);
695        }
696
697        Ok(ids)
698    }
699
700    /// Connect two neurons with a synapse.
701    pub fn connect(
702        &self,
703        pre_id: u64,
704        post_id: u64,
705        weight: f64,
706        excitatory: bool,
707    ) -> CoreResult<()> {
708        // Verify neurons exist
709        {
710            let neurons = self.neurons.read().map_err(|e| {
711                CoreError::LockError(format!("Failed to acquire neurons read lock: {e}"))
712            })?;
713            if !neurons.contains_key(&pre_id) {
714                return Err(CoreError::NotFound(format!(
715                    "Presynaptic neuron {pre_id} not found"
716                )));
717            }
718            if !neurons.contains_key(&post_id) {
719                return Err(CoreError::NotFound(format!(
720                    "Postsynaptic neuron {post_id} not found"
721                )));
722            }
723        }
724
725        let synapse = if excitatory {
726            SpikingSynapse::excitatory(pre_id, post_id, weight)
727        } else {
728            SpikingSynapse::inhibitory(pre_id, post_id, weight)
729        };
730
731        let mut synapses = self.synapses.write().map_err(|e| {
732            CoreError::LockError(format!("Failed to acquire synapses write lock: {e}"))
733        })?;
734
735        synapses.entry(post_id).or_default().push(synapse);
736        Ok(())
737    }

739    /// Inject current into a specific neuron.
740    pub fn inject_current(&self, neuron_id: u64, current: f64) -> CoreResult<()> {
741        let mut neurons = self.neurons.write().map_err(|e| {
742            CoreError::LockError(format!("Failed to acquire neurons write lock: {e}"))
743        })?;
744
745        let neuron = neurons
746            .get_mut(&neuron_id)
747            .ok_or_else(|| CoreError::NotFound(format!("Neuron {neuron_id} not found")))?;
748
749        neuron.set_current(current);
750        Ok(())
751    }

753    /// Simulate one timestep of the network (1 ms).
754    ///
755    /// Returns a vector of neuron IDs that fired during this timestep.
756    #[instrument(level = "debug", skip(self))]
757    pub fn step(&self) -> CoreResult<Vec<u64>> {
758        let current_time = {
759            let mut time = self.current_time.write().map_err(|e| {
760                CoreError::LockError(format!("Failed to acquire time write lock: {e}"))
761            })?;
762            *time += 1;
763            *time
764        };
765
766        let mut fired_neurons = Vec::new();
767
768        // Phase 1: Update synapses and collect synaptic currents
769        let mut synaptic_currents: HashMap<u64, f64> = HashMap::new();
770        {
771            let neurons = self.neurons.read().map_err(|e| {
772                CoreError::LockError(format!("Failed to acquire neurons read lock: {e}"))
773            })?;
774            let mut synapses = self.synapses.write().map_err(|e| {
775                CoreError::LockError(format!("Failed to acquire synapses write lock: {e}"))
776            })?;
777
778            for (post_id, synapse_list) in synapses.iter_mut() {
779                if let Some(post_neuron) = neurons.get(post_id) {
780                    let mut total_current = 0.0;
781                    for synapse in synapse_list.iter_mut() {
782                        total_current += synapse.update(current_time, post_neuron.v);
783                    }
784                    synaptic_currents.insert(*post_id, total_current);
785                }
786            }
787        }
788
789        // Phase 2: Apply synaptic currents and step neurons
790        {
791            let mut neurons = self.neurons.write().map_err(|e| {
792                CoreError::LockError(format!("Failed to acquire neurons write lock: {e}"))
793            })?;
794
795            for (id, neuron) in neurons.iter_mut() {
796                if let Some(&current) = synaptic_currents.get(id) {
797                    neuron.add_synaptic_input(current);
798                }
799                if neuron.step(current_time) {
800                    fired_neurons.push(*id);
801                }
802            }
803        }
804
805        // Phase 3: Propagate spikes through synapses
806        {
807            let mut synapses = self.synapses.write().map_err(|e| {
808                CoreError::LockError(format!("Failed to acquire synapses write lock: {e}"))
809            })?;
810
811            for &pre_id in &fired_neurons {
812                // Find all synapses originating from this neuron
813                for synapse_list in synapses.values_mut() {
814                    for synapse in synapse_list.iter_mut() {
815                        if synapse.pre_id == pre_id {
816                            synapse.receive_spike(current_time);
817                        }
818                    }
819                }
820            }
821        }
822
823        // Phase 4: Apply STDP learning if enabled
824        if self.learning_enabled && !fired_neurons.is_empty() {
825            self.apply_stdp(&fired_neurons, current_time)?;
826        }
827
828        // Update statistics
829        {
830            let mut total = self.total_spikes.write().map_err(|e| {
831                CoreError::LockError(format!("Failed to acquire total_spikes write lock: {e}"))
832            })?;
833            *total += fired_neurons.len() as u64;
834        }
835
836        Ok(fired_neurons)
837    }

839    /// Apply STDP learning rule based on spike timing.
840    fn apply_stdp(&self, fired_neurons: &[u64], current_time: u64) -> CoreResult<()> {
841        let neurons = self.neurons.read().map_err(|e| {
842            CoreError::LockError(format!("Failed to acquire neurons read lock: {e}"))
843        })?;
844        let mut synapses = self.synapses.write().map_err(|e| {
845            CoreError::LockError(format!("Failed to acquire synapses write lock: {e}"))
846        })?;
847
848        for &post_id in fired_neurons {
849            if let Some(synapse_list) = synapses.get_mut(&post_id) {
850                for synapse in synapse_list.iter_mut() {
851                    // Get presynaptic spike time
852                    if let Some(pre_neuron) = neurons.get(&synapse.pre_id) {
853                        if let Some(pre_spike_time) = pre_neuron.last_spike_time {
854                            let delta_t = (current_time as i64 - pre_spike_time as i64) as f64;
855                            // Only apply STDP for recent spikes (within 100 ms window)
856                            if delta_t.abs() < 100.0 {
857                                synapse.weight = self.stdp_rule.apply(delta_t, synapse.weight);
858                            }
859                        }
860                    }
861                }
862            }
863        }
864
865        Ok(())
866    }

868    /// Run the simulation for a specified duration.
869    ///
870    /// # Arguments
871    ///
872    /// * `duration_ms` - Duration to simulate in milliseconds
873    ///
874    /// # Returns
875    ///
876    /// A map of neuron IDs to their spike times during the simulation
877    #[instrument(level = "info", skip(self))]
878    pub fn simulate(&self, duration_ms: u64) -> CoreResult<HashMap<u64, Vec<u64>>> {
879        let mut spike_raster: HashMap<u64, Vec<u64>> = HashMap::new();
880
881        for _ in 0..duration_ms {
882            let fired = self.step()?;
883            let time = *self.current_time.read().map_err(|e| {
884                CoreError::LockError(format!("Failed to acquire time read lock: {e}"))
885            })?;
886
887            for id in fired {
888                spike_raster.entry(id).or_default().push(time);
889            }
890        }
891
892        Ok(spike_raster)
893    }

895    /// Get network statistics.
896    pub fn statistics(&self) -> CoreResult<NetworkStatistics> {
897        let neurons = self.neurons.read().map_err(|e| {
898            CoreError::LockError(format!("Failed to acquire neurons read lock: {e}"))
899        })?;
900        let synapses = self.synapses.read().map_err(|e| {
901            CoreError::LockError(format!("Failed to acquire synapses read lock: {e}"))
902        })?;
903        let total_spikes = *self.total_spikes.read().map_err(|e| {
904            CoreError::LockError(format!("Failed to acquire total_spikes read lock: {e}"))
905        })?;
906        let current_time = *self
907            .current_time
908            .read()
909            .map_err(|e| CoreError::LockError(format!("Failed to acquire time read lock: {e}")))?;
910
911        let total_synapses: usize = synapses.values().map(std::vec::Vec::len).sum();
912        let avg_weight: f64 = if total_synapses > 0 {
913            synapses
914                .values()
915                .flat_map(|v| v.iter())
916                .map(|s| s.weight)
917                .sum::<f64>()
918                / total_synapses as f64
919        } else {
920            0.0
921        };
922
923        Ok(NetworkStatistics {
924            neuron_count: neurons.len(),
925            synapse_count: total_synapses,
926            total_spikes,
927            simulation_time_ms: current_time,
928            average_synaptic_weight: avg_weight,
929        })
930    }

934        Ok(*self
935            .current_time
936            .read()
937            .map_err(|e| CoreError::LockError(format!("Failed to acquire time read lock: {e}")))?)
938    }

940    /// Reset all neurons and simulation time.
941    pub fn reset(&self) -> CoreResult<()> {
942        {
943            let mut neurons = self.neurons.write().map_err(|e| {
944                CoreError::LockError(format!("Failed to acquire neurons write lock: {e}"))
945            })?;
946            for neuron in neurons.values_mut() {
947                neuron.reset();
948            }
949        }
950        {
951            let mut synapses = self.synapses.write().map_err(|e| {
952                CoreError::LockError(format!("Failed to acquire synapses write lock: {e}"))
953            })?;
954            for synapse_list in synapses.values_mut() {
955                for synapse in synapse_list.iter_mut() {
956                    synapse.conductance = 0.0;
957                    synapse.spike_queue.clear();
958                }
959            }
960        }
961        *self.current_time.write().map_err(|e| {
962            CoreError::LockError(format!("Failed to acquire time write lock: {e}"))
963        })? = 0;
964        *self.total_spikes.write().map_err(|e| {
965            CoreError::LockError(format!("Failed to acquire total_spikes write lock: {e}"))
966        })? = 0;
967
968        Ok(())
969    }
