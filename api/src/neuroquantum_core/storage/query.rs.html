1//! Query types for the storage engine
2//!
3//! This module defines the query structures used for CRUD operations:
4//! - `SelectQuery`: For reading data with filtering, sorting, and pagination
5//! - `InsertQuery`: For inserting new rows
6//! - `UpdateQuery`: For modifying existing rows
7//! - `DeleteQuery`: For removing rows
8//! - `AlterTableOp`: For DDL schema modifications
9
10use std::collections::HashMap;
11
12use super::types::{ColumnDefinition, DataType, Value};
13
14/// Query for selecting rows from a table
15#[derive(Debug, Clone)]
16pub struct SelectQuery {
17    pub table: String,
18    pub columns: Vec<String>,
19    pub where_clause: Option<WhereClause>,
20    pub order_by: Option<OrderBy>,
21    pub limit: Option<u64>,
22    pub offset: Option<u64>,
23}
24
25/// Query for inserting a new row
26#[derive(Debug, Clone)]
27pub struct InsertQuery {
28    pub table: String,
29    pub values: HashMap<String, Value>,
30}
31
32/// Query for updating existing rows
33#[derive(Debug, Clone)]
34pub struct UpdateQuery {
35    pub table: String,
36    pub set_values: HashMap<String, Value>,
37    pub where_clause: Option<WhereClause>,
38}
39
40/// Query for deleting rows
41#[derive(Debug, Clone)]
42pub struct DeleteQuery {
43    pub table: String,
44    pub where_clause: Option<WhereClause>,
45}
46
47/// ALTER TABLE operations for schema modifications
48#[derive(Debug, Clone)]
49pub enum AlterTableOp {
50    AddColumn {
51        column: ColumnDefinition,
52    },
53    DropColumn {
54        column_name: String,
55    },
56    RenameColumn {
57        old_name: String,
58        new_name: String,
59    },
60    ModifyColumn {
61        column_name: String,
62        new_data_type: DataType,
63    },
64}
65
66/// WHERE clause for filtering rows
67#[derive(Debug, Clone)]
68pub struct WhereClause {
69    pub conditions: Vec<Condition>,
70}
71
72/// A single filter condition
73#[derive(Debug, Clone)]
74pub struct Condition {
75    pub field: String,
76    pub operator: ComparisonOperator,
77    pub value: Value,
78}
79
80/// Comparison operators for conditions
81#[derive(Debug, Clone)]
82pub enum ComparisonOperator {
83    Equal,
84    NotEqual,
85    LessThan,
86    LessThanOrEqual,
87    GreaterThan,
88    GreaterThanOrEqual,
89    Like,
90    In,
91}
92
93/// ORDER BY clause for sorting results
94#[derive(Debug, Clone)]
95pub struct OrderBy {
96    pub field: String,
97    pub direction: SortDirection,
98}
99
100/// Sort direction for ORDER BY
101#[derive(Debug, Clone)]
102pub enum SortDirection {
103    Ascending,
104    Descending,
105}
