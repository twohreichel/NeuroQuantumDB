1//! Recovery Manager implementing ARIES algorithm
2//!
3//! Three-phase recovery:
4//! 1. Analysis: Determine which transactions were active and which pages were dirty
5//! 2. Redo: Replay all changes from the log to restore the database state
6//! 3. Undo: Roll back incomplete transactions

55    /// Perform crash recovery using ARIES algorithm
56    pub async fn recover(
57        &self,
58        wal_manager: &super::WALManager,
59        pager: Arc<PageStorageManager>,
60    ) -> Result<RecoveryStats> {
61        let start_time = std::time::Instant::now();
62        info!("üîÑ Starting ARIES recovery...");
63
64        // Phase 1: Analysis
65        info!("üìä Phase 1: Analysis");
66        let analysis_result = self.analysis_phase(wal_manager).await?;
67
68        // Log detailed transaction state information
69        let redo_txns = analysis_result.transactions_needing_redo();
70        let undo_txns = analysis_result.transactions_needing_undo();
71        info!("   - Transactions needing redo: {}", redo_txns.len());
72        info!("   - Transactions needing undo: {}", undo_txns.len());
73        info!(
74            "   - Already aborted: {}",
75            analysis_result.aborted_txns.len()
76        );
77
78        // Phase 2: Redo
79        info!("‚ôªÔ∏è Phase 2: Redo");
80        let redo_count = self
81            .redo_phase(wal_manager, &analysis_result, &pager)
82            .await?;
83
84        // Phase 3: Undo
85        info!("‚Ü©Ô∏è Phase 3: Undo");
86        let undo_count = self
87            .undo_phase(wal_manager, &analysis_result, &pager)
88            .await?;
89
90        let recovery_time_ms = start_time.elapsed().as_millis() as u64;
91
92        let stats = RecoveryStats {
93            records_analyzed: analysis_result.total_records,
94            redo_operations: redo_count,
95            undo_operations: undo_count,
96            transactions_committed: analysis_result.committed_txns.len(),
97            transactions_aborted: analysis_result.active_txns.len(),
98            recovery_time_ms,
99            checkpoint_lsn: analysis_result.checkpoint_lsn,
100        };
101
102        info!("‚úÖ Recovery completed in {}ms", recovery_time_ms);
103        info!("   - Records analyzed: {}", stats.records_analyzed);
104        info!("   - Redo operations: {}", stats.redo_operations);
105        info!("   - Undo operations: {}", stats.undo_operations);
106        info!("   - Committed: {}", stats.transactions_committed);
107        info!("   - Aborted: {}", stats.transactions_aborted);
108
109        Ok(stats)
110    }

112    /// Phase 1: Analysis - determine transaction and page state
113    ///
114    /// This phase scans the log to build:
115    /// - Transaction table: maps `tx_id` -> `TransactionState` with full ARIES tracking
116    /// - Dirty page table: maps `page_id` -> recovery LSN
117    /// - Sets of committed and aborted transactions
118    async fn analysis_phase(&self, wal_manager: &super::WALManager) -> Result<AnalysisResult> {
119        info!("Scanning log from beginning...");
120
121        // Read all log records
122        let records = wal_manager.read_log_records(1).await?;
123
124        // Full TransactionState tracking for ARIES
125        let mut active_txn_states: HashMap<TransactionId, TransactionState> = HashMap::new();
126        // Simplified LSN mapping for compatibility
127        let mut active_txns: HashMap<TransactionId, LSN> = HashMap::new();
128        let mut committed_txns: HashSet<TransactionId> = HashSet::new();
129        let mut aborted_txns: HashSet<TransactionId> = HashSet::new();
130        let mut dirty_pages: HashMap<PageId, LSN> = HashMap::new();
131        let mut checkpoint_lsn: Option<LSN> = None;
132
133        for record in &records {
134            match &record.record_type {
135                | WALRecordType::Begin { tx_id, .. } => {
136                    // Create a new TransactionState with full tracking
137                    let tx_state = TransactionState::new(*tx_id, record.lsn);
138                    active_txn_states.insert(*tx_id, tx_state);
139                    active_txns.insert(*tx_id, record.lsn);
140                    debug!("Found BEGIN for TX={}", tx_id);
141                },
142                | WALRecordType::Update { tx_id, page_id, .. } => {
143                    // Update transaction state with operation tracking
144                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
145                        tx_state.record_operation(record.lsn, Some(*page_id));
146                    } else {
147                        // Transaction started before our log scan - create state
148                        let mut tx_state = TransactionState::new(*tx_id, record.lsn);
149                        tx_state.record_operation(record.lsn, Some(*page_id));
150                        active_txn_states.insert(*tx_id, tx_state);
151                    }
152                    active_txns.insert(*tx_id, record.lsn);
153                    dirty_pages.entry(*page_id).or_insert(record.lsn);
154                    debug!("Found UPDATE for TX={}, Page={}", tx_id, page_id.0);
155                },
156                | WALRecordType::Commit { tx_id } => {
157                    // Mark transaction as committed
158                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
159                        tx_state.begin_commit();
160                        tx_state.complete_commit(record.lsn);
161                    }
162                    active_txns.remove(tx_id);
163                    committed_txns.insert(*tx_id);
164                    debug!("Found COMMIT for TX={}", tx_id);
165                },
166                | WALRecordType::Abort { tx_id } => {
167                    // Mark transaction as aborted
168                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
169                        tx_state.begin_abort();
170                        tx_state.complete_abort(record.lsn);
171                    }
172                    active_txns.remove(tx_id);
173                    aborted_txns.insert(*tx_id);
174                    debug!("Found ABORT for TX={}", tx_id);
175                },
176                | WALRecordType::CheckpointBegin { .. } => {
177                    checkpoint_lsn = Some(record.lsn);
178                    debug!("Found CHECKPOINT at LSN={}", record.lsn);
179                },
180                | WALRecordType::CheckpointEnd => {
181                    debug!("Found CHECKPOINT END at LSN={}", record.lsn);
182                },
183                | WALRecordType::CLR {
184                    tx_id,
185                    undo_next_lsn,
186                    page_id,
187                    ..
188                } => {
189                    // Update undo chain for CLR records
190                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
191                        tx_state.record_operation(record.lsn, Some(*page_id));
192                        // CLR updates the undo_next_lsn to skip already undone operations
193                        tx_state.undo_next_lsn = Some(*undo_next_lsn);
194                    }
195                    active_txns.insert(*tx_id, record.lsn);
196                    dirty_pages.entry(*page_id).or_insert(record.lsn);
197                    debug!("Found CLR for TX={}, undo_next={}", tx_id, undo_next_lsn);
198                },
199                | WALRecordType::Savepoint { tx_id, name, .. } => {
200                    // Track savepoint in transaction state
201                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
202                        tx_state.savepoints.insert(name.clone(), record.lsn);
203                        tx_state.record_operation(record.lsn, None);
204                    }
205                    active_txns.insert(*tx_id, record.lsn);
206                    debug!("Found SAVEPOINT '{}' for TX={}", name, tx_id);
207                },
208                | WALRecordType::RollbackToSavepoint {
209                    tx_id,
210                    name,
211                    target_lsn,
212                } => {
213                    // Update undo chain to reflect rollback
214                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
215                        tx_state.undo_next_lsn = Some(*target_lsn);
216                        tx_state.record_operation(record.lsn, None);
217                    }
218                    active_txns.insert(*tx_id, record.lsn);
219                    debug!(
220                        "Found ROLLBACK TO SAVEPOINT '{}' for TX={}, target_lsn={}",
221                        name, tx_id, target_lsn
222                    );
223                },
224                | WALRecordType::ReleaseSavepoint { tx_id, name } => {
225                    // Remove savepoint from transaction state
226                    if let Some(tx_state) = active_txn_states.get_mut(tx_id) {
227                        tx_state.savepoints.remove(name);
228                        tx_state.record_operation(record.lsn, None);
229                    }
230                    active_txns.insert(*tx_id, record.lsn);
231                    debug!("Found RELEASE SAVEPOINT '{}' for TX={}", name, tx_id);
232                },
233            }
234        }
235
236        info!("Analysis complete:");
237        info!("  - Active transactions: {}", active_txns.len());
238        info!("  - Committed transactions: {}", committed_txns.len());
239        info!("  - Aborted transactions: {}", aborted_txns.len());
240        info!("  - Dirty pages: {}", dirty_pages.len());
241
242        Ok(AnalysisResult {
243            active_txn_states,
244            active_txns,
245            committed_txns,
246            aborted_txns,
247            dirty_pages,
248            checkpoint_lsn,
249            total_records: records.len(),
250        })
251    }

253    /// Phase 2: Redo - replay all changes
254    async fn redo_phase(
255        &self,
256        wal_manager: &super::WALManager,
257        analysis: &AnalysisResult,
258        pager: &Arc<PageStorageManager>,
259    ) -> Result<usize> {
260        info!("Redoing changes from log...");
261
262        // Determine starting point (checkpoint or beginning)
263        let start_lsn = analysis.checkpoint_lsn.unwrap_or(1);
264        let records = wal_manager.read_log_records(start_lsn).await?;
265
266        let mut redo_count = 0;
267
268        for record in &records {
269            match &record.record_type {
270                | WALRecordType::Update {
271                    page_id,
272                    offset,
273                    after_image,
274                    ..
275                } => {
276                    // Check if page needs redo (is dirty and LSN >= recovery_lsn)
277                    if let Some(&recovery_lsn) = analysis.dirty_pages.get(page_id) {
278                        if record.lsn >= recovery_lsn {
279                            // Apply the update
280                            self.apply_redo(pager, *page_id, *offset, after_image.clone())
281                                .await?;
282                            redo_count += 1;
283                            debug!("REDO: Page={}, LSN={}", page_id.0, record.lsn);
284                        }
285                    }
286                },
287                | WALRecordType::CLR {
288                    page_id, redo_data, ..
289                } => {
290                    // Redo CLR operations
291                    self.apply_redo(pager, *page_id, 0, redo_data.clone())
292                        .await?;
293                    redo_count += 1;
294                    debug!("REDO CLR: Page={}, LSN={}", page_id.0, record.lsn);
295                },
296                | _ => {}, // Skip non-update records
297            }
298        }
299
300        info!("Redo complete: {} operations", redo_count);
301        Ok(redo_count)
302    }

304    /// Phase 3: Undo - roll back incomplete transactions
305    ///
306    /// Uses the full `TransactionState` information for optimized undo:
307    /// - Respects `undo_next_lsn` for CLR-aware recovery
308    /// - Uses `modified_pages` for selective page access
309    /// - Leverages `operation_count` for progress tracking
310    async fn undo_phase(
311        &self,
312        wal_manager: &super::WALManager,
313        analysis: &AnalysisResult,
314        pager: &Arc<PageStorageManager>,
315    ) -> Result<usize> {
316        info!("Undoing incomplete transactions...");
317
318        // Use TransactionState for transactions that need undo
319        let txns_to_undo = analysis.transactions_needing_undo();
320        if txns_to_undo.is_empty() && analysis.active_txns.is_empty() {
321            info!("No active transactions to undo");
322            return Ok(0);
323        }
324
325        let mut undo_count = 0;
326
327        // Prefer using full TransactionState when available
328        for tx_state in &txns_to_undo {
329            info!(
330                "Undoing transaction: {} (status: {}, operations: {})",
331                tx_state.tx_id, tx_state.status, tx_state.operation_count
332            );
333
334            // Use undo_next_lsn if available (CLR-aware), otherwise use last_lsn
335            let start_lsn = tx_state.undo_next_lsn.unwrap_or(tx_state.last_lsn);
336
337            let undo_ops = self
338                .undo_transaction(wal_manager, tx_state.tx_id, start_lsn, pager)
339                .await?;
340            undo_count += undo_ops;
341        }
342
343        // Fallback: Also process any transactions in active_txns not in active_txn_states
344        for (tx_id, last_lsn) in &analysis.active_txns {
345            // Skip if already processed via TransactionState
346            if analysis.active_txn_states.contains_key(tx_id) {
347                continue;
348            }
349
350            info!("Undoing transaction (legacy path): {}", tx_id);
351            let undo_ops = self
352                .undo_transaction(wal_manager, *tx_id, *last_lsn, pager)
353                .await?;
354            undo_count += undo_ops;
355        }
356
357        info!("Undo complete: {} operations", undo_count);
358        Ok(undo_count)
359    }

361    /// Undo a single transaction by following `prev_lsn` chain
362    async fn undo_transaction(
363        &self,
364        wal_manager: &super::WALManager,
365        tx_id: TransactionId,
366        mut current_lsn: LSN,
367        pager: &Arc<PageStorageManager>,
368    ) -> Result<usize> {
369        let mut undo_count = 0;
370        let records = wal_manager.read_log_records(1).await?;
371
372        // Build a map of LSN -> Record for quick lookup
373        let record_map: HashMap<LSN, &WALRecord> = records.iter().map(|r| (r.lsn, r)).collect();
374
375        // Follow the undo chain
376        while let Some(record) = record_map.get(&current_lsn) {
377            match &record.record_type {
378                | WALRecordType::Update {
379                    page_id,
380                    offset,
381                    before_image,
382                    ..
383                } => {
384                    // Apply the before image (undo the change)
385                    self.apply_redo(pager, *page_id, *offset, before_image.clone())
386                        .await?;
387                    undo_count += 1;
388                    debug!("UNDO: TX={}, Page={}, LSN={}", tx_id, page_id.0, record.lsn);
389
390                    // Write CLR (Compensation Log Record) - not implemented here for simplicity
391                },
392                | WALRecordType::Begin { .. } => {
393                    // Reached the beginning of the transaction
394                    break;
395                },
396                | _ => {},
397            }
398
399            // Move to previous LSN
400            if let Some(prev_lsn) = record.prev_lsn {
401                current_lsn = prev_lsn;
402            } else {
403                break;
404            }
405        }
406
407        Ok(undo_count)
408    }

410    /// Apply a redo operation to a page
411    async fn apply_redo(
412        &self,
413        pager: &Arc<PageStorageManager>,
414        page_id: PageId,
415        offset: usize,
416        data: Vec<u8>,
417    ) -> Result<()> {
418        // Read the page
419        let mut page = pager.read_page(page_id).await?;
420
421        // Apply the update using the write_data method
422        page.write_data(offset, &data)?;
423
424        // Write the page back
425        pager.write_page(&page).await?;
426
427        Ok(())
428    }

431/// Result of the analysis phase with full ARIES transaction tracking
432#[derive(Debug)]
433struct AnalysisResult {
434    /// Active transaction states (not committed/aborted) - full `TransactionState` for proper undo
435    active_txn_states: HashMap<TransactionId, TransactionState>,
436    /// Active transactions (not committed/aborted) - LSN mapping for compatibility
437    active_txns: HashMap<TransactionId, LSN>,
438    /// Committed transactions
439    committed_txns: HashSet<TransactionId>,
440    /// Aborted transactions
441    aborted_txns: HashSet<TransactionId>,
442    /// Dirty pages and their recovery LSN
443    dirty_pages: HashMap<PageId, LSN>,
444    /// Last checkpoint LSN (if any)
445    checkpoint_lsn: Option<LSN>,
446    /// Total records analyzed
447    total_records: usize,
448}
449
450impl AnalysisResult {
451    /// Get all transactions that need undo (active or aborting)
452    pub fn transactions_needing_undo(&self) -> Vec<&TransactionState> {
453        self.active_txn_states
454            .values()
455            .filter(|state| state.needs_undo())
456            .collect()
457    }
458
459    /// Get all transactions that need redo (committed or committing)
460    pub fn transactions_needing_redo(&self) -> Vec<&TransactionState> {
461        self.active_txn_states
462            .values()
463            .filter(|state| state.needs_redo())
464            .collect()
465    }
466}
