36/// Constant-time comparison utilities for preventing timing attacks.
37///
38/// # Security Considerations
39///
40/// Timing attacks work by measuring how long it takes to compare two values.
41/// Standard comparison operations (==, !=) will return immediately upon finding
42/// the first difference, which can leak information about:
43/// - API keys: How many characters match before the first difference
44/// - Passwords: Similar information leakage
45/// - Session tokens: Position of first mismatch
46/// - Biometric thresholds: How close a value is to passing
47///
48/// These utilities ensure that comparisons take constant time regardless of
49/// input, preventing such attacks.
50///
51/// Performs constant-time comparison of two byte slices.
52/// Returns true if they are equal, false otherwise.
53/// This function takes the same amount of time regardless of where the first
54/// difference occurs, preventing timing attacks.
55///
56/// # Examples
57///
58/// ```
59/// use neuroquantum_core::security::constant_time_compare;
60///
61/// let secret1 = b"my_secret_key_12345";
62/// let secret2 = b"my_secret_key_12345";
63/// let secret3 = b"my_secret_key_99999";
64///
65/// assert!(constant_time_compare(secret1, secret2));
66/// assert!(!constant_time_compare(secret1, secret3));
67/// ```
68#[must_use]
69pub fn constant_time_compare(a: &[u8], b: &[u8]) -> bool {
70    if a.len() != b.len() {
71        return false;
72    }
73    a.ct_eq(b).into()
74}

76/// Performs constant-time comparison of two strings.
77/// Converts strings to bytes and compares them in constant time.
78///
79/// # Examples
80///
81/// ```
82/// use neuroquantum_core::security::constant_time_compare_str;
83///
84/// let token1 = "session_abc123";
85/// let token2 = "session_abc123";
86/// let token3 = "session_xyz789";
87///
88/// assert!(constant_time_compare_str(token1, token2));
89/// assert!(!constant_time_compare_str(token1, token3));
90/// ```
91#[must_use]
92pub fn constant_time_compare_str(a: &str, b: &str) -> bool {
93    constant_time_compare(a.as_bytes(), b.as_bytes())
94}

96/// Performs constant-time threshold comparison for floating point values.
97///
98/// This is used for similarity scores in biometric authentication.
99/// Returns true if value >= threshold, using a constant-time comparison
100/// to prevent leaking information about how close the value is to the threshold.
101///
102/// # Security Considerations
103///
104/// Without constant-time comparison, an attacker could learn:
105/// - How close their biometric sample is to passing
106/// - Whether they're getting warmer or colder
107/// - Precise threshold values through binary search
108///
109/// This function prevents such leakage by always taking the same time regardless
110/// of how close the value is to the threshold.
111///
112/// # Implementation
113///
114/// 1. Converts floats to fixed-point i32 (with 10000x scale for 4 decimal precision)
115/// 2. Clamps values to valid i32 range to prevent undefined behavior
116/// 3. Computes difference: value - threshold (wrapping to handle all cases)
117/// 4. Extracts sign bit using arithmetic right shift (endian-agnostic)
118/// 5. Returns true if difference is non-negative (sign bit = 0)
119///
120/// This approach leverages two's complement representation where negative numbers
121/// have their sign bit (bit 31 for i32) set to 1. The algorithm is endian-agnostic
122/// as it uses bit shifting instead of byte array indexing.
123///
124/// # Examples
125///
126/// ```
127/// use neuroquantum_core::security::constant_time_threshold_check;
128///
129/// assert!(constant_time_threshold_check(0.90, 0.85));
130/// assert!(!constant_time_threshold_check(0.80, 0.85));
131/// ```
132#[must_use]
133pub fn constant_time_threshold_check(value: f32, threshold: f32) -> bool {
134    // Handle NaN explicitly for predictable behavior in security-critical context
135    // NaN comparisons always return false for security (fail-safe default)
136    if value.is_nan() || threshold.is_nan() {
137        return false;
138    }
139
140    // Handle infinity cases explicitly to avoid overflow issues
141    if value.is_infinite() {
142        // Positive infinity is always >= any finite threshold
143        // Negative infinity is always < any finite threshold
144        return value.is_sign_positive() && !threshold.is_infinite();
145    }
146    if threshold.is_infinite() {
147        // Value can never be >= positive infinity
148        // Value is always >= negative infinity
149        return threshold.is_sign_negative();
150    }
151
152    // Convert to fixed-point integers for constant-time comparison
153    // Use 10000 scale factor for 4 decimal places precision
154    // Clamp to valid i32 range to avoid undefined behavior from overflow
155    let value_scaled = (value * 10000.0).clamp(i32::MIN as f32, i32::MAX as f32);
156    let threshold_scaled = (threshold * 10000.0).clamp(i32::MIN as f32, i32::MAX as f32);
157    let value_fixed = value_scaled as i32;
158    let threshold_fixed = threshold_scaled as i32;
159
160    // Compute difference: value - threshold
161    let diff = value_fixed.wrapping_sub(threshold_fixed);
162
163    // Extract sign bit in an endian-agnostic way
164    // In two's complement, negative numbers have the sign bit (bit 31) set to 1
165    // Shift right by 31 to get the sign bit in the LSB position
166    let sign_bit = ((diff >> 31) & 1) as u8;
167
168    // Return true if sign bit is 0 (non-negative), false if 1 (negative)
169    // Use constant-time equality check to convert to bool
170    sign_bit.ct_eq(&0u8).into()
171}

270/// Quantum-safe cryptographic keys with automatic zeroing on drop
271#[derive(Zeroize, ZeroizeOnDrop)]
272pub struct QuantumKeys {
273    /// ML-KEM keys for key exchange
274    mlkem_public: Vec<u8>,
275    #[zeroize(skip)]
276    mlkem_secret: Vec<u8>,
277    /// ML-DSA signature keys
278    mldsa_public: Vec<u8>,
279    #[zeroize(skip)]
280    mldsa_secret: Vec<u8>,
281    /// AES-256-GCM symmetric key for data encryption
282    #[zeroize(skip)]
283    symmetric_key: [u8; 32],
284    /// Key creation timestamp
285    #[zeroize(skip)]
286    created_at: SystemTime,
287}

289impl QuantumKeys {
290    /// Generate new quantum-safe key pair
291    pub fn generate() -> Result<Self, SecurityError> {
292        info!("üîê Generating quantum-safe cryptographic keys...");
293
294        // Generate ML-KEM-1024 keys for KEM using RustCrypto ml-kem
295        let mut rng = rand::thread_rng();
296        let (mlkem_dk, mlkem_ek) = MlKem1024::generate(&mut rng);
297
298        // Generate ML-DSA-87 keys for signatures
299        let (mldsa_pk, mldsa_sk) = mldsa87::keypair();
300
301        // Generate symmetric key for AES-256-GCM
302        let mut symmetric_key = [0u8; 32];
303        use rand::RngCore;
304        rand::thread_rng().fill_bytes(&mut symmetric_key);
305
306        // Serialize ML-KEM keys to bytes
307        let mlkem_public_bytes = mlkem_ek.as_bytes().to_vec();
308        let mlkem_secret_bytes = mlkem_dk.as_bytes().to_vec();
309
310        Ok(Self {
311            mlkem_public: mlkem_public_bytes,
312            mlkem_secret: mlkem_secret_bytes,
313            mldsa_public: mldsa_pk.as_bytes().to_vec(),
314            mldsa_secret: mldsa_sk.as_bytes().to_vec(),
315            symmetric_key,
316            created_at: SystemTime::now(),
317        })
318    }
319
320    /// Check if keys need rotation
321    #[must_use]
322    pub fn needs_rotation(&self, rotation_interval: u64) -> bool {
323        if let Ok(elapsed) = SystemTime::now().duration_since(self.created_at) {
324            elapsed.as_secs() > rotation_interval
325        } else {
326            false
327        }
328    }

346    /// Encrypt data using quantum-resistant encryption
347    pub async fn quantum_encrypt(&self, data: &[u8]) -> Result<Vec<u8>, SecurityError> {
348        if !self.config.mlkem_enabled {
349            return Err(SecurityError::EncryptionNotEnabled);
350        }
351
352        let keys = self.keys.read().await;
353
354        // Use AES-256-GCM for symmetric encryption
355        let cipher = Aes256Gcm::new_from_slice(&keys.symmetric_key)
356            .map_err(|e| SecurityError::EncryptionFailed(e.to_string()))?;
357
358        // Generate random nonce
359        let mut nonce_bytes = [0u8; 12];
360        use rand::RngCore;
361        rand::thread_rng().fill_bytes(&mut nonce_bytes);
362        let nonce = Nonce::from(nonce_bytes);
363
364        // Encrypt data
365        let ciphertext = cipher
366            .encrypt(&nonce, data)
367            .map_err(|e| SecurityError::EncryptionFailed(e.to_string()))?;
368
369        // Prepend nonce to ciphertext
370        let mut result = nonce_bytes.to_vec();
371        result.extend_from_slice(&ciphertext);
372
373        debug!("‚úÖ Encrypted {} bytes of data", data.len());
374        Ok(result)
375    }
376
377    /// Decrypt data using quantum-resistant decryption
378    pub async fn quantum_decrypt(&self, encrypted_data: &[u8]) -> Result<Vec<u8>, SecurityError> {
379        if encrypted_data.len() < 12 {
380            return Err(SecurityError::InvalidEncryptedData);
381        }
382
383        let keys = self.keys.read().await;
384
385        // Extract nonce and ciphertext
386        let nonce_bytes: [u8; 12] = encrypted_data[..12]
387            .try_into()
388            .map_err(|_| SecurityError::InvalidEncryptedData)?;
389        let nonce = Nonce::from(nonce_bytes);
390        let ciphertext = &encrypted_data[12..];
391
392        // Decrypt using AES-256-GCM
393        let cipher = Aes256Gcm::new_from_slice(&keys.symmetric_key)
394            .map_err(|e| SecurityError::DecryptionFailed(e.to_string()))?;
395
396        let plaintext = cipher
397            .decrypt(&nonce, ciphertext)
398            .map_err(|e| SecurityError::DecryptionFailed(e.to_string()))?;
399
400        debug!("‚úÖ Decrypted {} bytes of data", plaintext.len());
401        Ok(plaintext)
402    }

404    /// Sign data using ML-DSA post-quantum signature
405    pub async fn sign_data(&self, data: &[u8]) -> Result<Vec<u8>, SecurityError> {
406        if !self.config.mldsa_enabled {
407            return Err(SecurityError::SignatureDisabled);
408        }
409
410        let keys = self.keys.read().await;
411
412        // Reconstruct ML-DSA secret key
413        let sk = mldsa87::SecretKey::from_bytes(&keys.mldsa_secret)
414            .map_err(|e| SecurityError::SignatureFailed(format!("Invalid secret key: {e:?}")))?;
415
416        // Sign the data
417        let signed = mldsa87::sign(data, &sk);
418
419        debug!("‚úÖ Signed {} bytes of data", data.len());
420        Ok(signed.as_bytes().to_vec())
421    }

423    /// Verify ML-DSA signature
424    pub async fn verify_signature(&self, signed_data: &[u8]) -> Result<Vec<u8>, SecurityError> {
425        let keys = self.keys.read().await;
426
427        // Reconstruct ML-DSA public key
428        let pk = mldsa87::PublicKey::from_bytes(&keys.mldsa_public).map_err(|e| {
429            SecurityError::SignatureVerificationFailed(format!("Invalid public key: {e:?}"))
430        })?;
431
432        // Reconstruct signed message
433        let signed_msg = mldsa87::SignedMessage::from_bytes(signed_data).map_err(|e| {
434            SecurityError::SignatureVerificationFailed(format!("Invalid signature: {e:?}"))
435        })?;
436
437        // Verify and extract original message
438        let message = mldsa87::open(&signed_msg, &pk).map_err(|e| {
439            SecurityError::SignatureVerificationFailed(format!("Verification failed: {e:?}"))
440        })?;
441
442        debug!("‚úÖ Signature verified successfully");
443        Ok(message)
444    }

446    /// Generate quantum-safe shared secret using ML-KEM
447    ///
448    /// This method deserializes the peer's encapsulation key from bytes,
449    /// then encapsulates a shared secret that only the peer can decapsulate.
450    pub async fn generate_shared_secret(
451        &self,
452        peer_public_key: &[u8],
453    ) -> Result<(Vec<u8>, Vec<u8>), SecurityError> {
454        // Expected encapsulation key size for ML-KEM-1024 is 1568 bytes
455        const MLKEM1024_EK_SIZE: usize = 1568;
456
457        if peer_public_key.len() != MLKEM1024_EK_SIZE {
458            return Err(SecurityError::KeyExchangeFailed(format!(
459                "Invalid public key length: expected {} bytes, got {} bytes",
460                MLKEM1024_EK_SIZE,
461                peer_public_key.len()
462            )));
463        }
464
465        // Deserialize the encapsulation key from bytes
466        type MlKem1024EncapsulationKey = <MlKem1024 as KemCore>::EncapsulationKey;
467        type MlKem1024EkArray = Encoded<MlKem1024EncapsulationKey>;
468
469        // Use TryFrom to convert slice to fixed-size array
470        let ek_array: MlKem1024EkArray = peer_public_key.try_into().map_err(|_| {
471            SecurityError::KeyExchangeFailed("Failed to parse encapsulation key bytes".to_string())
472        })?;
473        let ek = MlKem1024EncapsulationKey::from_bytes(&ek_array);
474
475        // Encapsulate a shared secret
476        let mut rng = rand::thread_rng();
477        let (ct, shared_secret) = ek.encapsulate(&mut rng).map_err(|()| {
478            SecurityError::KeyExchangeFailed("ML-KEM encapsulation failed".to_string())
479        })?;
480
481        Ok((
482            AsRef::<[u8]>::as_ref(&shared_secret).to_vec(),
483            AsRef::<[u8]>::as_ref(&ct).to_vec(),
484        ))
485    }

505    /// Decapsulate a shared secret from ML-KEM ciphertext
506    ///
507    /// Takes ciphertext bytes from `generate_shared_secret` and returns the shared secret.
508    /// This validates the ciphertext size and performs proper decapsulation using the
509    /// decapsulation (secret) key.
510    ///
511    /// For ML-KEM-1024, ciphertext must be exactly 1568 bytes.
512    pub async fn decapsulate_shared_secret(
513        &self,
514        ciphertext: &[u8],
515    ) -> Result<Vec<u8>, SecurityError> {
516        // Validate ciphertext size for ML-KEM-1024 (1568 bytes)
517        if ciphertext.len() != MLKEM1024_CIPHERTEXT_SIZE {
518            return Err(SecurityError::KeyExchangeFailed(format!(
519                "Invalid ciphertext length: expected {} bytes, got {} bytes",
520                MLKEM1024_CIPHERTEXT_SIZE,
521                ciphertext.len()
522            )));
523        }
524
525        let keys = self.keys.read().await;
526
527        // Deserialize the decapsulation key from bytes
528        type MlKem1024DecapsulationKey = <MlKem1024 as KemCore>::DecapsulationKey;
529        type MlKem1024DkArray = Encoded<MlKem1024DecapsulationKey>;
530
531        // Convert secret key bytes to fixed-size array
532        let dk_array: MlKem1024DkArray = keys.mlkem_secret.as_slice().try_into().map_err(|_| {
533            SecurityError::KeyExchangeFailed("Failed to parse decapsulation key bytes".to_string())
534        })?;
535        let dk = MlKem1024DecapsulationKey::from_bytes(&dk_array);
536
537        // Deserialize the ciphertext from bytes
538        let ct: Ciphertext<MlKem1024> = ciphertext.try_into().map_err(|_| {
539            SecurityError::KeyExchangeFailed("Failed to parse ciphertext bytes".to_string())
540        })?;
541
542        // Decapsulate using the decapsulation key
543        let shared_secret = dk.decapsulate(&ct).map_err(|()| {
544            SecurityError::KeyExchangeFailed(
545                "ML-KEM decapsulation failed - possibly corrupted ciphertext".to_string(),
546            )
547        })?;
548
549        debug!("‚úÖ Successfully decapsulated shared secret");
550        Ok(AsRef::<[u8]>::as_ref(&shared_secret).to_vec())
551    }

596    /// Register new EEG pattern for a user
597    pub async fn register_eeg_pattern(
598        &self,
599        user_id: String,
600        eeg_data: &[f32],
601    ) -> Result<(), SecurityError> {
602        if !self.config.eeg_enabled {
603            return Err(SecurityError::BiometricDisabled);
604        }
605
606        if eeg_data.len() < self.config.min_sample_length {
607            return Err(SecurityError::InsufficientBiometricData);
608        }
609
610        info!("üß† Registering EEG pattern for user: {}", user_id);
611
612        // Extract features from raw EEG data
613        let features = self.extract_eeg_features(eeg_data)?;
614        let frequency_bands = self.extract_frequency_bands(eeg_data)?;
615
616        // Calculate statistical properties
617        let mean = features.iter().sum::<f32>() / features.len() as f32;
618        let variance =
619            features.iter().map(|x| (x - mean).powi(2)).sum::<f32>() / features.len() as f32;
620        let std_dev = variance.sqrt();
621
622        let template = EEGTemplate {
623            user_id: user_id.clone(),
624            features,
625            mean,
626            std_dev,
627            frequency_bands,
628            created_at: SystemTime::now(),
629            last_used: SystemTime::now(),
630        };
631
632        let mut templates = self.eeg_templates.write().await;
633        templates.insert(user_id.clone(), template);
634
635        info!("‚úÖ EEG pattern registered for user: {}", user_id);
636        Ok(())
637    }

639    /// Authenticate user using EEG pattern matching
640    pub async fn authenticate_eeg(
641        &self,
642        user_id: &str,
643        eeg_data: &[f32],
644    ) -> Result<AuthResult, SecurityError> {
645        if !self.config.eeg_enabled {
646            return Err(SecurityError::BiometricDisabled);

652        let template = templates
653            .get_mut(user_id)
654            .ok_or_else(|| SecurityError::UserNotFound(user_id.to_string()))?;
655
656        // Extract features from current EEG sample
657        let current_features = self.extract_eeg_features(eeg_data)?;
658
659        // Calculate similarity score using cosine similarity
660        let similarity = self.cosine_similarity(&template.features, &current_features);
661
662        debug!("üìä EEG similarity score: {:.4}", similarity);
663
664        // Update last used timestamp
665        template.last_used = SystemTime::now();
666
667        // Use constant-time threshold comparison to prevent timing attacks
668        // This prevents attackers from learning how close they are to the threshold
669        if constant_time_threshold_check(similarity, self.config.similarity_threshold) {
670            info!("‚úÖ EEG authentication successful for user: {}", user_id);
671            Ok(AuthResult::Success {
672                user_id: user_id.to_string(),
673                confidence: similarity,
674                method: AuthMethod::Biometric,
675            })
676        } else {
677            warn!(
678                "‚ùå EEG authentication failed for user: {} (similarity: {:.4})",
679                user_id, similarity
680            );
681            Ok(AuthResult::Failed {
682                reason: format!("Biometric match score too low: {similarity:.4}"),
683            })
684        }
685    }

689        match self.config.feature_method {

701    /// Extract frequency domain features using FFT with rustfft
702    fn fft_features(&self, eeg_data: &[f32]) -> Result<Vec<f32>, SecurityError> {
703        let mut features = Vec::new();
704        let window_size = 256.min(eeg_data.len());
705
706        // Create FFT planner
707        let mut planner = FftPlanner::new();
708        let fft = planner.plan_fft_forward(window_size);
709
710        // Process each window with 50% overlap
711        // Ensure we process at least one window if data length equals window_size
712        let end = if eeg_data.len() == window_size {
713            window_size
714        } else {
715            eeg_data.len().saturating_sub(window_size)
716        };
717
718        for i in (0..=end.saturating_sub(window_size)).step_by((window_size / 2).max(1)) {
719            if i + window_size > eeg_data.len() {
720                break;
721            }
722            let window = &eeg_data[i..i + window_size];
723
724            // Convert to complex numbers for FFT
725            let mut buffer: Vec<Complex<f32>> =
726                window.iter().map(|&x| Complex::new(x, 0.0)).collect();
727
728            // Perform FFT
729            fft.process(&mut buffer);
730
731            // Calculate power spectrum and extract frequency band features
732            let power_spectrum: Vec<f32> = buffer
733                .iter()
734                .take(window_size / 2) // Only need first half (Nyquist)
735                .map(nalgebra::Complex::norm_sqr)
736                .collect();
737
738            // Extract standard EEG frequency bands (assuming 256 Hz sampling rate)
739            // Delta (0.5-4 Hz), Theta (4-8 Hz), Alpha (8-13 Hz), Beta (13-30 Hz), Gamma (30-50 Hz)
740            let sampling_rate = 256.0;
741            let freq_resolution = sampling_rate / window_size as f32;
742
743            let delta_band = Self::band_power(&power_spectrum, 0.5, 4.0, freq_resolution);
744            let theta_band = Self::band_power(&power_spectrum, 4.0, 8.0, freq_resolution);
745            let alpha_band = Self::band_power(&power_spectrum, 8.0, 13.0, freq_resolution);
746            let beta_band = Self::band_power(&power_spectrum, 13.0, 30.0, freq_resolution);
747            let gamma_band = Self::band_power(&power_spectrum, 30.0, 50.0, freq_resolution);
748
749            features
750                .extend_from_slice(&[delta_band, theta_band, alpha_band, beta_band, gamma_band]);
751        }
752
753        Ok(features)
754    }

756    /// Calculate power in a specific frequency band
757    fn band_power(
758        power_spectrum: &[f32],
759        low_freq: f32,
760        high_freq: f32,
761        freq_resolution: f32,
762    ) -> f32 {
763        let low_idx = (low_freq / freq_resolution) as usize;
764        let high_idx = ((high_freq / freq_resolution) as usize).min(power_spectrum.len());
765
766        if low_idx >= high_idx || high_idx > power_spectrum.len() {
767            return 0.0;
768        }
769
770        power_spectrum[low_idx..high_idx].iter().sum::<f32>() / (high_idx - low_idx) as f32
771    }
772
773    /// Extract wavelet features
774    fn wavelet_features(&self, eeg_data: &[f32]) -> Result<Vec<f32>, SecurityError> {
775        // Simplified wavelet transform (Haar wavelet)
776        let mut features = Vec::new();
777        let mut signal = eeg_data.to_vec();
778
779        while signal.len() > 1 {
780            let mut approximation = Vec::new();
781            let mut detail = Vec::new();
782
783            for i in (0..signal.len() - 1).step_by(2) {
784                let avg = f32::midpoint(signal[i], signal[i + 1]);
785                let diff = (signal[i] - signal[i + 1]) / 2.0;
786                approximation.push(avg);
787                detail.push(diff);
788            }
789
790            // Use detail coefficients as features
791            let energy: f32 = detail.iter().map(|x| x * x).sum();
792            features.push(energy);
793
794            signal = approximation;
795        }
796
797        Ok(features)
798    }

800    /// Extract frequency band powers using FFT
801    fn extract_frequency_bands(&self, eeg_data: &[f32]) -> Result<FrequencyBands, SecurityError> {
802        let window_size = eeg_data.len().min(512).next_power_of_two();
803        let mut planner = FftPlanner::new();
804        let fft = planner.plan_fft_forward(window_size);
805
806        // Pad or truncate data to window size
807        let mut buffer: Vec<Complex<f32>> = eeg_data
808            .iter()
809            .take(window_size)
810            .map(|&x| Complex::new(x, 0.0))
811            .collect();
812
813        // Pad with zeros if needed
814        buffer.resize(window_size, Complex::new(0.0, 0.0));
815
816        // Perform FFT
817        fft.process(&mut buffer);
818
819        // Calculate power spectrum
820        let power_spectrum: Vec<f32> = buffer
821            .iter()
822            .take(window_size / 2)
823            .map(nalgebra::Complex::norm_sqr)
824            .collect();
825
826        // Extract band powers (assuming 256 Hz sampling rate)
827        let sampling_rate = 256.0;
828        let freq_resolution = sampling_rate / window_size as f32;
829
830        Ok(FrequencyBands {
831            delta: Self::band_power(&power_spectrum, 0.5, 4.0, freq_resolution),
832            theta: Self::band_power(&power_spectrum, 4.0, 8.0, freq_resolution),
833            alpha: Self::band_power(&power_spectrum, 8.0, 13.0, freq_resolution),
834            beta: Self::band_power(&power_spectrum, 13.0, 30.0, freq_resolution),
835            gamma: Self::band_power(&power_spectrum, 30.0, 50.0, freq_resolution),
836        })
837    }
838
839    /// Calculate cosine similarity between two feature vectors
840    fn cosine_similarity(&self, a: &[f32], b: &[f32]) -> f32 {
841        if a.len() != b.len() {
842            return 0.0;
843        }
844
845        let dot_product: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
846        let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
847        let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
848
849        if norm_a == 0.0 || norm_b == 0.0 {
850            return 0.0;
851        }
852
853        dot_product / (norm_a * norm_b)
854    }
855}

976        let salt = SaltString::generate(&mut OsRng);
977        let argon2 = Argon2::default();
978
979        argon2
980            .hash_password(password.as_bytes(), &salt)
981            .map(|hash| hash.to_string())
982            .map_err(|e| SecurityError::PasswordHashFailed(e.to_string()))
983    }

986        let parsed_hash = PasswordHash::new(&self.password_hash)

988
989        // NOTE: Argon2::verify_password is designed to be constant-time and resistant
990        // to timing attacks. It always performs the full verification process regardless
991        // of where differences occur in the hash comparison.
992        Ok(Argon2::default()
993            .verify_password(password.as_bytes(), &parsed_hash)
994            .is_ok())
995    }

1087            }

1090                "‚ùå Authentication failed for user: {} (attempt {})",
1091                username, user.failed_attempts

1094                reason: "Invalid credentials".to_string(),

1180        resource: String,
1181        action: String,
1182        result: AuditResult,
1183        details: HashMap<String, String>,
1184    ) -> Result<(), SecurityError> {
1185        if !self.config.enabled {
1186            return Ok(());
1187        }
1188
1189        let mut logs = self.logs.write().await;
1190

1224    /// Calculate cryptographic hash of log entry
1225    fn calculate_log_hash(&self, log: &AuditLog) -> String {
1226        let mut hasher = Sha3_512::new();
1227
1228        hasher.update(log.id.as_bytes());
1229        hasher.update(
1230            log.timestamp
1231                .duration_since(UNIX_EPOCH)
1232                .map(|d| d.as_secs())
1233                .unwrap_or(0)
1234                .to_le_bytes(),
1235        );
1236        if let Some(ref user_id) = log.user_id {
1237            hasher.update(user_id.as_bytes());
1238        }
1239        hasher.update(log.resource.as_bytes());
1240        hasher.update(log.action.as_bytes());
1241        if let Some(ref prev_hash) = log.previous_hash {
1242            hasher.update(prev_hash.as_bytes());
1243        }
1244
1245        let result = hasher.finalize();
1246        hex::encode(result)
1247    }

1258    /// Verify integrity of audit log chain
1259    pub async fn verify_integrity(&self) -> Result<bool, SecurityError> {
1260        if !self.config.tamper_proof {
1261            return Ok(true);
1262        }
1263
1264        let logs = self.logs.read().await;
1265
1266        for i in 1..logs.len() {
1267            let current = &logs[i];
1268            let previous = &logs[i - 1];
1269
1270            // Verify chain
1271            if current.previous_hash.as_ref() != Some(&previous.hash) {
1272                error!("üö® Audit log chain broken at index {}", i);
1273                return Ok(false);
1274            }
1275
1276            // Verify hash
1277            let calculated_hash = self.calculate_log_hash(current);
1278            if calculated_hash != current.hash {
1279                error!("üö® Audit log hash mismatch at index {}", i);
1280                return Ok(false);
1281            }
1282        }
1283
1284        info!("‚úÖ Audit log integrity verified");
1285        Ok(true)
1286    }

1357        Self {
1358            id: Uuid::new_v4().to_string(),
1359            user_id,
1360            created_at: now,
1361            last_access: now,
1362            expires_at: now + std::time::Duration::from_secs(timeout_secs),
1363            permissions,
1364        }
1365    }

1368    pub fn is_expired(&self) -> bool {

1370    }

1373        let now = SystemTime::now();
1374        self.last_access = now;
1375        self.expires_at = now + std::time::Duration::from_secs(timeout_secs);
1376    }

1516        let result = self.rbac_manager.authenticate(username, password).await?;
1517

1520                // Create session
1521                let user = self.rbac_manager.get_user(&user_id).await?;
1522                let config = self.config.read().await;

1531                let mut sessions = self.sessions.write().await;
1532                sessions.insert(session_id.clone(), session);
1533
1534                Ok(session_id)
1535            },
1536            | AuthResult::Failed { reason } => Err(SecurityError::AuthenticationFailed(reason)),
1537        }

1542        &self,
1543        session_id: &str,
1544        required_permission: &Permission,
1545    ) -> Result<bool, SecurityError> {
1546        let mut sessions = self.sessions.write().await;
1547
1548        let session = sessions
1549            .get_mut(session_id)
1550            .ok_or(SecurityError::InvalidSession)?;
1551
1552        if session.is_expired() {
1553            sessions.remove(session_id);
1554            return Err(SecurityError::SessionExpired);
1555        }
1556
1557        let config = self.config.read().await;
1558        session.refresh(config.access_control.session_timeout);
1559
1560        Ok(session.permissions.contains(required_permission))
1561    }
