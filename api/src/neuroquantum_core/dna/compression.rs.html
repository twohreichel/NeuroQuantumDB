34    /// Compress DNA sequence using multiple algorithms
35    #[instrument(skip(self, bases))]
36    pub async fn compress_sequence(
37        &mut self,
38        bases: &[DNABase],
39    ) -> Result<CompressedSequence, DNAError> {
40        if bases.is_empty() {
41            return Ok(CompressedSequence::empty());
42        }
43
44        debug!("Starting advanced compression for {} bases", bases.len());
45
46        let mut current_bases = bases.to_vec();
47        let original_size = bases.len();
48        let mut compression_steps = Vec::new();
49
50        // Step 1: Biological pattern optimization
51        if self.config.enable_dictionary {
52            let (optimized, savings) = self.optimize_biological_patterns(&current_bases).await?;
53            current_bases = optimized;
54            compression_steps.push(CompressionStep {
55                algorithm: "BiologicalPatterns".to_string(),
56                size_before: bases.len(),
57                size_after: current_bases.len(),
58                compression_ratio: current_bases.len() as f64 / bases.len() as f64,
59                savings_bytes: savings,
60            });
61        }
62
63        // Step 2: Dictionary compression for repeated sequences
64        let (dict_compressed, dict_savings) =
65            self.apply_dictionary_compression(&current_bases).await?;
66        current_bases = dict_compressed;
67        compression_steps.push(CompressionStep {
68            algorithm: "Dictionary".to_string(),
69            size_before: compression_steps
70                .last()
71                .map_or(original_size, |s| s.size_after),
72            size_after: current_bases.len(),
73            compression_ratio: current_bases.len() as f64
74                / compression_steps
75                    .last()
76                    .map_or(original_size, |s| s.size_after) as f64,
77            savings_bytes: dict_savings,
78        });
79
80        // Step 3: Huffman coding for frequency-based compression
81        let huffman_compressed = self.apply_huffman_compression(&current_bases).await?;
82        let huffman_size = huffman_compressed.len() / 8; // Convert bits to approximate bytes
83        compression_steps.push(CompressionStep {
84            algorithm: "Huffman".to_string(),
85            size_before: current_bases.len(),
86            size_after: huffman_size,
87            compression_ratio: huffman_size as f64 / current_bases.len() as f64,
88            savings_bytes: current_bases.len().saturating_sub(huffman_size),
89        });
90
91        let final_compression_ratio = huffman_size as f64 / original_size as f64;
92
93        Ok(CompressedSequence {
94            compressed_data: huffman_compressed,
95            original_size,
96            compressed_size: huffman_size,
97            compression_ratio: final_compression_ratio,
98            compression_steps,
99            huffman_tree: self.huffman_tree.clone(),
100            pattern_dictionary: self.pattern_dictionary.clone(),
101        })
102    }

111        let mut optimized = bases.to_vec();

120        // Optimize palindromic sequences
121        total_savings += self.optimize_palindromes(&mut optimized);

123        Ok((optimized, total_savings))

127    fn optimize_complementary_pairs(&self, bases: &[DNABase]) -> usize {

130        // Look for Watson-Crick base pairs and encode them more efficiently
131        for window in bases.windows(2) {
132            if let [base1, base2] = window {
133                if self.biological_patterns.are_complementary(*base1, *base2) {
134                    // Mark complementary pairs for special encoding
135                    // This is a placeholder for more sophisticated encoding
136                    savings += 1;
137                }
138            }
139        }
140
141        savings

144    /// Optimize repetitive DNA motifs
145    async fn optimize_repetitive_motifs(
146        &mut self,
147        bases: &mut [DNABase],
148    ) -> Result<usize, DNAError> {
149        let mut savings = 0;
150        let motif_sizes = [3, 6, 9, 12]; // Common biological motif sizes
151
152        for &motif_size in &motif_sizes {
153            if bases.len() < motif_size * 2 {
154                continue;
155            }
156
157            let motif_savings = self.find_and_compress_motifs(bases, motif_size).await?;
158            savings += motif_savings;
159        }
160
161        Ok(savings)
162    }

164    /// Find and compress repetitive motifs of a specific size
165    async fn find_and_compress_motifs(
166        &mut self,
167        bases: &[DNABase],
168        motif_size: usize,
169    ) -> Result<usize, DNAError> {
170        let mut motif_counts = HashMap::new();
171
172        // Count motif frequencies in parallel
173        let motifs: Vec<_> = bases.windows(motif_size).enumerate().collect();
174
175        for (_, motif) in motifs {
176            *motif_counts.entry(motif.to_vec()).or_insert(0) += 1;
177        }
178
179        // Find motifs that appear frequently
180        let frequent_motifs: Vec<_> = motif_counts
181            .into_iter()
182            .filter(|(_, count)| *count >= 3)
183            .collect();
184
185        let mut total_savings = 0;
186
187        for (motif, count) in frequent_motifs {
188            if self.pattern_dictionary.len() >= u16::MAX as usize {
189                break;
190            }
191
192            let pattern_id = self.pattern_dictionary.len() as u16;
193            self.pattern_dictionary.insert(motif.clone(), pattern_id);
194
195            // Calculate savings: original motif size * occurrences - encoding overhead
196            total_savings += (motif_size * count).saturating_sub(count * 2); // 2 bytes per reference
197        }
198
199        Ok(total_savings)
200    }

202    /// Optimize palindromic sequences
203    fn optimize_palindromes(&self, bases: &mut [DNABase]) -> usize {
204        let mut savings = 0;
205
206        // Look for palindromic sequences (reverse complements)
207        for window_size in (4..=16).step_by(2) {
208            // Even sizes for palindromes
209            if bases.len() < window_size {
210                break;
211            }
212
213            for window in bases.windows(window_size) {
214                if self.is_palindrome(window) {
215                    // Mark palindrome for special encoding
216                    savings += window_size / 2; // Can encode half + flag
217                }
218            }
219        }
220
221        savings
222    }

224    /// Check if a sequence is a palindrome (reverse complement)
225    fn is_palindrome(&self, bases: &[DNABase]) -> bool {
226        let len = bases.len();
227        if !len.is_multiple_of(2) {
228            return false;
229        }
230
231        for i in 0..len / 2 {
232            let left = bases[i];
233            let right = bases[len - 1 - i];
234
235            if !self.biological_patterns.are_complementary(left, right) {
236                return false;
237            }
238        }
239
240        true
241    }

243    /// Apply dictionary compression to DNA sequence
244    ///
245    /// # Note on `.expect()` usage
246    ///
247    /// This function uses `.expect()` for `DNABase::from_bits()` calls where the input
248    /// is masked to 2 bits (& 0b11), making failure mathematically impossible.
249    /// All 2-bit values (0-3) map to valid DNA bases.
250    #[allow(clippy::expect_used)] // Bitwise-masked values are provably valid
251    async fn apply_dictionary_compression(
252        &mut self,
253        bases: &[DNABase],
254    ) -> Result<(Vec<DNABase>, usize), DNAError> {
255        debug!("Applying dictionary compression");
256
257        if self.pattern_dictionary.is_empty() {
258            return Ok((bases.to_vec(), 0));
259        }
260
261        let mut compressed = Vec::new();
262        let mut i = 0;
263        let mut savings = 0;
264
265        while i < bases.len() {
266            let mut matched = false;
267
268            // Try to match patterns from longest to shortest
269            let mut patterns: Vec<_> = self.pattern_dictionary.keys().collect();
270            patterns.sort_by_key(|p| std::cmp::Reverse(p.len()));
271
272            for pattern in patterns {
273                if i + pattern.len() <= bases.len()
274                    && &bases[i..i + pattern.len()] == pattern.as_slice()
275                {
276                    // Encode pattern reference
277                    if let Some(&pattern_id) = self.pattern_dictionary.get(pattern) {
278                        compressed.push(DNABase::Adenine); // Special marker
279                        compressed.push(
280                            DNABase::from_bits((pattern_id >> 8) as u8 & 0b11)
281                                .expect("valid 2-bit pattern"),
282                        );
283                        compressed.push(
284                            DNABase::from_bits((pattern_id >> 6) as u8 & 0b11)
285                                .expect("valid 2-bit pattern"),
286                        );
287                        compressed.push(
288                            DNABase::from_bits((pattern_id >> 4) as u8 & 0b11)
289                                .expect("valid 2-bit pattern"),
290                        );
291                        compressed.push(
292                            DNABase::from_bits((pattern_id >> 2) as u8 & 0b11)
293                                .expect("valid 2-bit pattern"),
294                        );
295                        compressed.push(
296                            DNABase::from_bits(pattern_id as u8 & 0b11)
297                                .expect("valid 2-bit pattern"),
298                        );
299
300                        i += pattern.len();
301                        savings += pattern.len().saturating_sub(6);
302                        matched = true;
303                        break;
304                    }
305                }
306            }
307
308            if !matched {
309                compressed.push(bases[i]);
310                i += 1;
311            }
312        }
313
314        Ok((compressed, savings))
315    }

317    /// Apply Huffman compression to DNA sequence
318    async fn apply_huffman_compression(&mut self, bases: &[DNABase]) -> Result<Vec<u8>, DNAError> {
319        debug!("Applying Huffman compression");
320
321        // Build frequency table
322        let mut frequencies = [0u64; 4];
323        for &base in bases {
324            frequencies[base as usize] += 1;
325        }
326
327        // Build Huffman tree
328        self.huffman_tree = Some(HuffmanTree::build(&frequencies));
329
330        // Encode sequence
331        let mut encoded_bits = Vec::new();
332        if let Some(ref tree) = self.huffman_tree {
333            for &base in bases {
334                let code = tree.get_code(base);
335                encoded_bits.extend(code);
336            }
337        }
338
339        // Convert bits to bytes
340        let mut bytes = Vec::new();
341        for chunk in encoded_bits.chunks(8) {
342            let mut byte = 0u8;
343            for (i, &bit) in chunk.iter().enumerate() {
344                if bit {
345                    byte |= 1 << (7 - i);
346                }
347            }
348            bytes.push(byte);
349        }
350
351        Ok(bytes)
352    }

354    /// Decompress a compressed DNA sequence
355    pub async fn decompress_sequence(
356        &self,
357        compressed: &CompressedSequence,
358    ) -> Result<Vec<DNABase>, DNAError> {
359        debug!("Decompressing DNA sequence");
360
361        // Step 1: Huffman decompression
362        let mut current_bases = if let Some(ref tree) = compressed.huffman_tree {
363            self.huffman_decompress(&compressed.compressed_data, tree, compressed.original_size)
364                .await?
365        } else {
366            return Err(DNAError::DecompressionFailed(
367                "Missing Huffman tree".to_string(),
368            ));
369        };
370
371        // Step 2: Dictionary decompression (reverse order from compression)
372        current_bases = self
373            .dictionary_decompress(&current_bases, &compressed.pattern_dictionary)
374            .await?;
375
376        // Step 3: Biological pattern deoptimization
377        current_bases = self.deoptimize_biological_patterns(&current_bases).await?;
378
379        Ok(current_bases)
380    }

382    /// Huffman decompression
383    async fn huffman_decompress(
384        &self,
385        data: &[u8],
386        tree: &HuffmanTree,
387        expected_size: usize,
388    ) -> Result<Vec<DNABase>, DNAError> {
389        let mut result = Vec::with_capacity(expected_size);
390        let mut bit_stream = BitStream::new(data);
391
392        while result.len() < expected_size {
393            if let Some(base) = tree.decode_next(&mut bit_stream)? {
394                result.push(base);
395            } else {
396                break;
397            }
398        }
399
400        Ok(result)
401    }

403    /// Dictionary decompression
404    async fn dictionary_decompress(
405        &self,
406        bases: &[DNABase],
407        dictionary: &HashMap<Vec<DNABase>, u16>,
408    ) -> Result<Vec<DNABase>, DNAError> {
409        // Build reverse dictionary
410        let reverse_dict: HashMap<u16, Vec<DNABase>> = dictionary
411            .iter()
412            .map(|(pattern, &id)| (id, pattern.clone()))
413            .collect();
414
415        let mut result = Vec::new();
416        let mut i = 0;
417
418        while i < bases.len() {
419            if i + 6 <= bases.len() && bases[i] == DNABase::Adenine {
420                // Potential pattern reference
421                let mut pattern_id = 0u16;
422                for j in 1..6 {
423                    pattern_id = (pattern_id << 2) | (bases[i + j] as u16);
424                }
425
426                if let Some(pattern) = reverse_dict.get(&pattern_id) {
427                    result.extend(pattern);
428                    i += 6;
429                } else {
430                    result.push(bases[i]);
431                    i += 1;
432                }
433            } else {
434                result.push(bases[i]);
435                i += 1;
436            }
437        }
438
439        Ok(result)
440    }

453/// Huffman tree for DNA base compression
454#[derive(Debug, Clone)]
455pub struct HuffmanTree {
456    root: HuffmanNode,
457    codes: [Vec<bool>; 4],
458}
459
460#[derive(Debug, Clone, PartialEq, Eq)]
461enum HuffmanNode {
462    Leaf {
463        base: DNABase,
464    },
465    Internal {
466        left: Box<HuffmanNode>,
467        right: Box<HuffmanNode>,
468    },
469}
470
471#[derive(Debug, Clone, PartialEq, Eq)]
472struct HuffmanNodeFreq {
473    freq: u64,
474    node: HuffmanNode,
475}
476
477impl Ord for HuffmanNodeFreq {
478    fn cmp(&self, other: &Self) -> Ordering {
479        other.freq.cmp(&self.freq) // Reverse for min-heap
480    }
481}
482
483impl PartialOrd for HuffmanNodeFreq {
484    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
485        Some(self.cmp(other))
486    }
487}
488
489impl HuffmanTree {
490    /// Build Huffman tree from base frequencies
491    ///
492    /// # Note on `.expect()` usage
493    ///
494    /// Uses `.expect()` in contexts where failure is impossible:
495    /// - Loop invariants ensure heap has sufficient elements
496    /// - Index values 0-3 always map to valid DNA bases
497    #[allow(clippy::expect_used)] // Loop invariants and valid indices guarantee success
498    #[must_use]
499    pub fn build(frequencies: &[u64; 4]) -> Self {
500        let mut heap = BinaryHeap::new();
501
502        // Create leaf nodes
503        for (i, &freq) in frequencies.iter().enumerate() {
504            if freq > 0 {
505                heap.push(HuffmanNodeFreq {
506                    freq,
507                    node: HuffmanNode::Leaf {
508                        base: DNABase::from_bits(i as u8).expect("valid DNA base index 0-3"),
509                    },
510                });
511            }
512        }
513
514        // Build tree
515        while heap.len() > 1 {
516            let node1 = heap.pop().expect("heap has at least 2 nodes in loop");
517            let node2 = heap.pop().expect("heap has at least 2 nodes in loop");
518
519            heap.push(HuffmanNodeFreq {
520                freq: node1.freq + node2.freq,
521                node: HuffmanNode::Internal {
522                    left: Box::new(node1.node),
523                    right: Box::new(node2.node),
524                },
525            });
526        }
527
528        let root = heap
529            .pop()
530            .expect("heap should have exactly one root node")
531            .node;
532        let mut tree = Self {
533            root,
534            codes: [Vec::new(), Vec::new(), Vec::new(), Vec::new()],
535        };
536
537        tree.build_codes();
538        tree
539    }
540
541    /// Build code table from tree
542    fn build_codes(&mut self) {
543        self.build_codes_recursive(&self.root.clone(), Vec::new());
544    }
545
546    fn build_codes_recursive(&mut self, node: &HuffmanNode, path: Vec<bool>) {
547        match node {
548            | HuffmanNode::Leaf { base } => {
549                self.codes[*base as usize] = path;
550            },
551            | HuffmanNode::Internal { left, right } => {
552                let mut left_path = path.clone();
553                left_path.push(false);
554                self.build_codes_recursive(left, left_path);
555
556                let mut right_path = path;
557                right_path.push(true);
558                self.build_codes_recursive(right, right_path);
559            },
560        }
561    }
562
563    /// Get Huffman code for a base
564    #[must_use]
565    pub fn get_code(&self, base: DNABase) -> &[bool] {
566        &self.codes[base as usize]
567    }
568
569    /// Decode next base from bit stream
570    pub fn decode_next(&self, bit_stream: &mut BitStream) -> Result<Option<DNABase>, DNAError> {
571        decode_from_node(&self.root, bit_stream)
572    }
573}
574
575fn decode_from_node(
576    node: &HuffmanNode,
577    bit_stream: &mut BitStream,
578) -> Result<Option<DNABase>, DNAError> {
579    match node {
580        | HuffmanNode::Leaf { base } => Ok(Some(*base)),
581        | HuffmanNode::Internal { left, right } => {
582            if let Some(bit) = bit_stream.next_bit() {
583                let child = if bit { right } else { left };
584                decode_from_node(child, bit_stream)
585            } else {
586                Ok(None)
587            }
588        },
589    }
590}

592/// Bit stream for Huffman decoding
593pub struct BitStream<'a> {
594    data: &'a [u8],
595    byte_pos: usize,
596    bit_pos: u8,
597}
598
599impl<'a> BitStream<'a> {
600    const fn new(data: &'a [u8]) -> Self {
601        Self {
602            data,
603            byte_pos: 0,
604            bit_pos: 0,
605        }
606    }
607
608    fn next_bit(&mut self) -> Option<bool> {
609        if self.byte_pos >= self.data.len() {
610            return None;
611        }
612
613        let byte = self.data[self.byte_pos];
614        let bit = (byte >> (7 - self.bit_pos)) & 1 == 1;
615
616        self.bit_pos += 1;
617        if self.bit_pos >= 8 {
618            self.bit_pos = 0;
619            self.byte_pos += 1;
620        }
621
622        Some(bit)
623    }
624}

626/// Biological pattern recognition and optimization
627#[derive(Debug)]
628struct BiologicalPatterns {
629    complement_map: HashMap<DNABase, DNABase>,
630}
631
632impl BiologicalPatterns {
633    fn new() -> Self {
634        let mut complement_map = HashMap::new();
635        complement_map.insert(DNABase::Adenine, DNABase::Thymine);
636        complement_map.insert(DNABase::Thymine, DNABase::Adenine);
637        complement_map.insert(DNABase::Guanine, DNABase::Cytosine);
638        complement_map.insert(DNABase::Cytosine, DNABase::Guanine);
639
640        Self { complement_map }
641    }
642
643    fn are_complementary(&self, base1: DNABase, base2: DNABase) -> bool {
644        self.complement_map.get(&base1) == Some(&base2)
645    }
646}
