28/// Reed-Solomon error corrector optimized for DNA data
29#[derive(Debug)]
30pub struct ReedSolomonCorrector {
31    /// Number of parity shards for error correction
32    parity_shards: usize,
33    /// Data shards per block
34    data_shards: usize,
35    /// Shard size in bytes (fixed per block for reed-solomon-simd)
36    shard_bytes: usize,
37    /// Maximum correctable errors per block
38    max_correctable_errors: usize,
39    /// Error correction statistics
40    stats: std::sync::Arc<std::sync::Mutex<ErrorCorrectionStats>>,
41}

43impl ReedSolomonCorrector {
44    /// Create a new Reed-Solomon corrector with the given error correction strength
45    #[must_use]
46    pub fn new(error_correction_strength: u8) -> Self {
47        // Map error correction strength (0-255) to Reed-Solomon parameters
48        // GF(2^8) requires data_shards + parity_shards <= 255
49        let parity_shards = (error_correction_strength as usize).clamp(1, 64);
50        // Calculate data_shards ensuring total doesn't exceed 255
51        let max_data_shards = 255 - parity_shards;
52        let data_shards = (parity_shards * 4).clamp(16, max_data_shards.min(191));
53
54        // reed-solomon-simd requires shard_bytes to be a non-zero multiple of 2
55        // We use 2 bytes per shard, padding each byte with a zero byte
56        let shard_bytes = 2;
57
58        let max_correctable_errors = parity_shards / 2;
59
60        Self {
61            parity_shards,
62            data_shards,
63            shard_bytes,
64            max_correctable_errors,
65            stats: std::sync::Arc::new(std::sync::Mutex::new(ErrorCorrectionStats::default())),
66        }
67    }

86    /// Generate Reed-Solomon parity data for the given input
87    #[instrument(skip(self, data))]
88    pub fn generate_parity(&self, data: &[u8]) -> Result<Vec<u8>, DNAError> {
89        if data.is_empty() {
90            return Ok(Vec::new());
91        }
92
93        debug!("Generating Reed-Solomon parity for {} bytes", data.len());
94
95        let mut all_parity = Vec::new();
96
97        // Process data in blocks
98        for chunk in data.chunks(self.data_shards) {
99            let parity = self.generate_parity_block(chunk)?;
100            all_parity.extend(parity);
101        }
102
103        Ok(all_parity)
104    }

106    /// Generate parity for a single data block
107    fn generate_parity_block(&self, data_block: &[u8]) -> Result<Vec<u8>, DNAError> {
108        // Pad block to data_shards size if necessary
109        let mut padded_data = data_block.to_vec();
110        padded_data.resize(self.data_shards, 0);
111
112        // Create encoder for this block
113        let mut encoder =
114            ReedSolomonEncoder::new(self.data_shards, self.parity_shards, self.shard_bytes)
115                .map_err(|e| {
116                    DNAError::ErrorCorrectionFailed(format!("Failed to create encoder: {e:?}"))
117                })?;
118
119        // Add each data shard (each byte becomes a 2-byte shard, padded with 0)
120        for &byte in &padded_data {
121            encoder.add_original_shard([byte, 0]).map_err(|e| {
122                DNAError::ErrorCorrectionFailed(format!("Failed to add data shard: {e:?}"))
123            })?;
124        }
125
126        // Encode to generate parity shards
127        let result = encoder.encode().map_err(|e| {
128            DNAError::ErrorCorrectionFailed(format!("Reed-Solomon encoding failed: {e:?}"))
129        })?;
130
131        // Extract parity data (recovery shards) - need to iterate with index
132        let mut parity = Vec::with_capacity(self.parity_shards);
133        for i in 0..self.parity_shards {
134            if let Some(recovery_shard) = result.recovery(i) {
135                parity.extend_from_slice(recovery_shard);
136            }
137        }
138
139        Ok(parity)
140    }

142    /// Correct errors in the data using Reed-Solomon decoding
143    #[instrument(skip(self, data, parity))]
144    pub fn correct_errors(&self, data: &[u8], parity: &[u8]) -> Result<(Vec<u8>, usize), DNAError> {
145        if data.is_empty() {
146            return Ok((Vec::new(), 0));
147        }
148
149        debug!(
150            "Correcting errors in {} bytes with {} parity bytes",
151            data.len(),
152            parity.len()
153        );
154
155        let mut corrected_data = Vec::with_capacity(data.len());
156        let mut total_errors_corrected = 0;
157
158        // Each parity shard is shard_bytes (2) bytes
159        let expected_parity_per_block = self.parity_shards * self.shard_bytes;
160        let mut parity_offset = 0;
161
162        // Process data in blocks
163        for chunk in data.chunks(self.data_shards) {
164            let parity_end = parity_offset + expected_parity_per_block;
165            if parity_end > parity.len() {
166                return Err(DNAError::ErrorCorrectionFailed(format!(
167                    "Insufficient parity data: need {} bytes, have {}",
168                    parity_end,
169                    parity.len()
170                )));
171            }
172
173            let block_parity = &parity[parity_offset..parity_end];
174            let (corrected_block, errors) = self.correct_errors_block(chunk, block_parity)?;
175
176            corrected_data.extend(corrected_block);
177            total_errors_corrected += errors;
178            parity_offset = parity_end;
179        }
180
181        // Remove padding from the last block
182        corrected_data.truncate(data.len());
183
184        if total_errors_corrected > 0 {
185            warn!("Reed-Solomon corrected {} errors", total_errors_corrected);
186        }
187
188        Ok((corrected_data, total_errors_corrected))
189    }

191    /// Correct errors in a single data block
192    fn correct_errors_block(
193        &self,
194        data_block: &[u8],
195        parity_block: &[u8],
196    ) -> Result<(Vec<u8>, usize), DNAError> {
197        // Pad block to data_shards size if necessary
198        let mut padded_data = data_block.to_vec();
199        padded_data.resize(self.data_shards, 0);
200
201        // Create decoder for this block
202        let mut decoder =
203            ReedSolomonDecoder::new(self.data_shards, self.parity_shards, self.shard_bytes)
204                .map_err(|e| {
205                    DNAError::ErrorCorrectionFailed(format!("Failed to create decoder: {e:?}"))
206                })?;
207
208        // Convert to shards format with Option for missing/corrupted shards
209        let mut shards: Vec<Option<Vec<u8>>> =
210            Vec::with_capacity(self.data_shards + self.parity_shards);
211
212        // Fill data shards (each byte padded to 2 bytes to match shard_bytes = 2)
213        for &byte in &padded_data {
214            shards.push(Some(vec![byte, 0]));
215        }
216
217        // Fill parity shards (parity data is already 2 bytes per shard)
218        for chunk in parity_block.chunks(2).take(self.parity_shards) {
219            if chunk.len() == 2 {
220                shards.push(Some(chunk.to_vec()));
221            } else if chunk.len() == 1 {
222                shards.push(Some(vec![chunk[0], 0]));
223            }
224        }
225
226        // Pad with None if needed
227        while shards.len() < self.data_shards + self.parity_shards {
228            shards.push(None);
229        }
230
231        // Detect errors by checking which shards are missing or corrupted
232        let errors_detected = self.detect_errors(&shards);
233
234        // Count missing shards
235        let missing_shards = shards.iter().filter(|s| s.is_none()).count();
236        let total_errors = errors_detected + missing_shards;
237
238        // Update statistics
239        {
240            let mut stats = self
241                .stats
242                .lock()
243                .unwrap_or_else(std::sync::PoisonError::into_inner);
244            stats.blocks_processed += 1;
245            stats.errors_detected += total_errors;
246        }
247
248        if missing_shards > 0 || errors_detected > 0 {
249            debug!(
250                "Detected {} missing shards and {} corrupted shards, attempting reconstruction",
251                missing_shards, errors_detected
252            );
253
254            // Update reconstruction attempt counter
255            {
256                let mut stats = self
257                    .stats
258                    .lock()
259                    .unwrap_or_else(std::sync::PoisonError::into_inner);
260                stats.reconstructions_attempted += 1;
261            }
262
263            // Mark corrupted shards as None for reconstruction
264            let corrupted_indices = self.identify_corrupted_shards(&shards);
265            for idx in corrupted_indices {
266                if idx < shards.len() {
267                    shards[idx] = None;
268                }
269            }
270
271            // Add available original shards to decoder
272            for (i, shard) in shards.iter().enumerate().take(self.data_shards) {
273                if let Some(shard_data) = shard {
274                    decoder.add_original_shard(i, shard_data).map_err(|e| {
275                        DNAError::ErrorCorrectionFailed(format!(
276                            "Failed to add original shard {i}: {e:?}"
277                        ))
278                    })?;
279                }
280            }
281
282            // Add available recovery (parity) shards to decoder
283            for (i, shard) in shards
284                .iter()
285                .enumerate()
286                .skip(self.data_shards)
287                .take(self.parity_shards)
288            {
289                if let Some(shard_data) = shard {
290                    let recovery_idx = i - self.data_shards;
291                    decoder
292                        .add_recovery_shard(recovery_idx, shard_data)
293                        .map_err(|e| {
294                            DNAError::ErrorCorrectionFailed(format!(
295                                "Failed to add recovery shard {recovery_idx}: {e:?}"
296                            ))
297                        })?;
298                }
299            }
300
301            // Attempt Reed-Solomon reconstruction
302            match decoder.decode() {
303                | Ok(result) => {
304                    debug!(
305                        "Successfully reconstructed data with {} errors/missing shards",
306                        total_errors
307                    );
308
309                    // Update successful reconstruction counter
310                    {
311                        let mut stats = self
312                            .stats
313                            .lock()
314                            .unwrap_or_else(std::sync::PoisonError::into_inner);
315                        stats.reconstructions_successful += 1;
316                        stats.errors_corrected += total_errors;
317                    }
318
319                    // Extract corrected data
320                    let mut corrected = Vec::with_capacity(data_block.len());
321                    for i in 0..self.data_shards.min(data_block.len()) {
322                        if let Some(restored_shard) = result.restored_original(i) {
323                            if !restored_shard.is_empty() {
324                                corrected.push(restored_shard[0]);
325                            }
326                        }
327                    }
328
329                    // Truncate to original length
330                    corrected.truncate(data_block.len());
331
332                    Ok((corrected, total_errors))
333                },
334                | Err(e) => Err(DNAError::ErrorCorrectionFailed(format!(
335                    "Reed-Solomon reconstruction failed: {:?}. Too many errors to correct: {} missing, {} corrupted, need at least {} valid shards",
336                    e, missing_shards, errors_detected, self.data_shards
337                ))),
338            }
339        } else {
340            // No errors detected, return original data
341            Ok((data_block.to_vec(), 0))
342        }
343    }

345    /// Detect errors in Reed-Solomon encoded data
346    /// This checks for corrupted shards using checksums and validates shard integrity
347    fn detect_errors(&self, shards: &[Option<Vec<u8>>]) -> usize {
348        let mut corrupted_count = 0;
349
350        for (idx, shard) in shards.iter().enumerate() {
351            if let Some(shard_data) = shard {
352                // Verify shard is not empty
353                if shard_data.is_empty() {
354                    corrupted_count += 1;
355                    continue;
356                }
357
358                // Verify shard has expected size
359                if shard_data.len() != self.shard_bytes {
360                    debug!(
361                        "Detected corrupted shard at index {}: expected {} bytes, got {} bytes",
362                        idx,
363                        self.shard_bytes,
364                        shard_data.len()
365                    );
366                    corrupted_count += 1;
367                    continue;
368                }
369
370                // With 2-byte shards where second byte is padding (0), only mark as
371                // corrupted if ALL bytes (including the data byte) are 0xFF
372                // A shard like [0, 0] is valid (represents the byte value 0)
373                let all_0xff = shard_data.iter().all(|&b| b == 0xFF);
374                if all_0xff {
375                    debug!(
376                        "Detected corrupted shard at index {}: all bytes are 0xFF",
377                        idx
378                    );
379                    corrupted_count += 1;
380                    continue;
381                }
382
383                // Additional validation: check for expected shard size
384                let expected_size = if idx < self.data_shards {
385                    // Data shards should have consistent size
386                    shards[0..self.data_shards]
387                        .iter()
388                        .filter_map(|s| s.as_ref())
389                        .map(std::vec::Vec::len)
390                        .max()
391                        .unwrap_or(shard_data.len())
392                } else {
393                    // Parity shards should match data shard size
394                    shards[0..self.data_shards]
395                        .iter()
396                        .filter_map(|s| s.as_ref())
397                        .map(std::vec::Vec::len)
398                        .next()
399                        .unwrap_or(shard_data.len())
400                };
401
402                // Allow some tolerance for the last shard which might be padded
403                if !shard_data.is_empty()
404                    && (shard_data.len() < expected_size / 2
405                        || shard_data.len() > expected_size * 2)
406                {
407                    debug!(
408                        "Detected size anomaly in shard {}: expected ~{} bytes, got {} bytes",
409                        idx,
410                        expected_size,
411                        shard_data.len()
412                    );
413                    corrupted_count += 1;
414                }
415            }
416        }
417
418        if corrupted_count > 0 {
419            debug!(
420                "Detected {} corrupted shards during error detection",
421                corrupted_count
422            );
423        }
424
425        corrupted_count
426    }

428    /// Identify indices of corrupted shards for reconstruction
429    fn identify_corrupted_shards(&self, shards: &[Option<Vec<u8>>]) -> Vec<usize> {
430        let mut corrupted_indices = Vec::new();
431
432        for (idx, shard) in shards.iter().enumerate() {
433            if let Some(shard_data) = shard {
434                // Check for corruption indicators
435                if shard_data.is_empty() {
436                    corrupted_indices.push(idx);
437                    continue;
438                }
439
440                // Check for suspicious patterns
441                if shard_data.len() > 1 {
442                    let first_byte = shard_data[0];
443                    let all_same = shard_data.iter().all(|&b| b == first_byte);
444                    if all_same && (first_byte == 0xFF || first_byte == 0x00) {
445                        corrupted_indices.push(idx);
446                    }
447                }
448            }
449        }
450
451        corrupted_indices
452    }

454    /// Calculate the required parity length for a given data size
455    #[must_use]
456    pub const fn calculate_parity_length(&self, data_size: usize) -> usize {
457        let blocks = data_size.div_ceil(self.data_shards);
458        // Each parity shard is shard_bytes (2) bytes
459        blocks * self.parity_shards * self.shard_bytes
460    }

462    /// Validate Reed-Solomon parameters
463    pub fn validate_parameters(&self) -> Result<(), DNAError> {
464        if self.data_shards == 0 || self.parity_shards == 0 {
465            return Err(DNAError::ErrorCorrectionFailed(
466                "Invalid Reed-Solomon parameters: zero shards".to_string(),
467            ));
468        }
469
470        if self.data_shards + self.parity_shards > 255 {
471            return Err(DNAError::ErrorCorrectionFailed(
472                "Invalid Reed-Solomon parameters: too many shards".to_string(),
473            ));
474        }
475
476        Ok(())
477    }

502    /// Correct homopolymer run errors (common in DNA sequencing)
503    fn correct_homopolymer_errors(&self, bases: &mut [DNABase]) -> usize {
504        let mut corrections = 0;
505        let max_homopolymer_length = 8; // Biological threshold
506
507        let mut i = 0;
508        while i < bases.len() {
509            let current_base = bases[i];
510            let mut run_length = 1;
511            let mut j = i + 1;
512
513            // Count run length
514            while j < bases.len() && bases[j] == current_base {
515                run_length += 1;
516                j += 1;
517            }
518
519            // If run is too long, introduce variation (simple heuristic)
520            if run_length > max_homopolymer_length {
521                // Replace every 4th base in long runs with a different base
522                for k in (i + 3..j).step_by(4) {
523                    let original = bases[k];
524                    bases[k] = self.get_alternate_base(original);
525                    corrections += 1;
526                }
527            }
528
529            i = j;
530        }
531
532        corrections
533    }

535    /// Correct GC bias errors
536    fn correct_gc_bias_errors(&self, bases: &mut [DNABase]) -> usize {
537        // GC bias correction based on local GC content analysis
538        // This analyzes windows of the sequence to detect and correct bias patterns
539        let mut corrections = 0;
540        let window_size = 20; // Analyze 20bp windows
541
542        if bases.len() < window_size {
543            return 0;
544        }
545
546        for window_start in 0..(bases.len() - window_size) {
547            let window = &bases[window_start..window_start + window_size];
548
549            // Calculate GC content in window
550            let gc_count = window
551                .iter()
552                .filter(|b| matches!(b, DNABase::Guanine | DNABase::Cytosine))
553                .count();
554
555            let gc_content = gc_count as f64 / window_size as f64;
556
557            // Detect extreme GC bias (< 20% or > 80%)
558            // Natural DNA typically has 40-60% GC content
559            if gc_content < 0.2 {
560                // Too AT-rich: likely sequencing error
561                // Look for isolated GC bases that might be errors
562                for i in window_start..window_start + window_size {
563                    if matches!(bases[i], DNABase::Guanine | DNABase::Cytosine) {
564                        // Check if surrounded by AT
565                        let prev_at =
566                            i > 0 && matches!(bases[i - 1], DNABase::Adenine | DNABase::Thymine);
567                        let next_at = i < bases.len() - 1
568                            && matches!(bases[i + 1], DNABase::Adenine | DNABase::Thymine);
569
570                        if prev_at && next_at {
571                            // Likely sequencing error - convert to AT
572                            bases[i] = if matches!(bases[i], DNABase::Guanine) {
573                                DNABase::Adenine
574                            } else {
575                                DNABase::Thymine
576                            };
577                            corrections += 1;
578                        }
579                    }
580                }
581            } else if gc_content > 0.8 {
582                // Too GC-rich: likely sequencing error
583                // Look for isolated AT bases that might be errors
584                for i in window_start..window_start + window_size {
585                    if matches!(bases[i], DNABase::Adenine | DNABase::Thymine) {
586                        // Check if surrounded by GC
587                        let prev_gc =
588                            i > 0 && matches!(bases[i - 1], DNABase::Guanine | DNABase::Cytosine);
589                        let next_gc = i < bases.len() - 1
590                            && matches!(bases[i + 1], DNABase::Guanine | DNABase::Cytosine);
591
592                        if prev_gc && next_gc {
593                            // Likely sequencing error - convert to GC
594                            bases[i] = if matches!(bases[i], DNABase::Adenine) {
595                                DNABase::Guanine
596                            } else {
597                                DNABase::Cytosine
598                            };
599                            corrections += 1;
600                        }
601                    }
602                }
603            }
604        }
605
606        corrections
607    }

619    /// Verify Reed-Solomon integrity without full correction
620    pub fn verify_integrity(&self, data: &[u8], parity: &[u8]) -> Result<bool, DNAError> {
621        if data.is_empty() && parity.is_empty() {
622            return Ok(true);
623        }
624
625        let expected_parity_len = self.calculate_parity_length(data.len());
626        if parity.len() != expected_parity_len {
627            return Ok(false);
628        }
629
630        // Quick integrity check using Reed-Solomon verification
631        // This is more efficient than full error correction
632        for (chunk, parity_chunk) in data
633            .chunks(self.data_shards)
634            .zip(parity.chunks(self.parity_shards))
635        {
636            if !self.verify_block_integrity(chunk, parity_chunk)? {
637                return Ok(false);
638            }
639        }
640
641        Ok(true)
642    }

644    /// Verify integrity of a single block
645    fn verify_block_integrity(
646        &self,
647        data_block: &[u8],
648        parity_block: &[u8],
649    ) -> Result<bool, DNAError> {
650        // Reconstruct the full shard set
651        let mut padded_data = data_block.to_vec();
652        padded_data.resize(self.data_shards, 0);
653
654        // Create encoder to regenerate parity
655        let mut encoder =
656            ReedSolomonEncoder::new(self.data_shards, self.parity_shards, self.shard_bytes)
657                .map_err(|e| {
658                    DNAError::ErrorCorrectionFailed(format!("Failed to create encoder: {e:?}"))
659                })?;
660
661        // Add each data shard (padded to 2 bytes to match shard_bytes = 2)
662        for &byte in &padded_data {
663            encoder.add_original_shard([byte, 0]).map_err(|e| {
664                DNAError::ErrorCorrectionFailed(format!("Failed to add shard: {e:?}"))
665            })?;
666        }
667
668        // Encode to generate expected parity
669        let result = encoder
670            .encode()
671            .map_err(|e| DNAError::ErrorCorrectionFailed(format!("Encoding failed: {e:?}")))?;
672
673        // Extract generated parity - need to iterate with index
674        let mut generated_parity = Vec::with_capacity(self.parity_shards);
675        for i in 0..self.parity_shards {
676            if let Some(recovery_shard) = result.recovery(i) {
677                generated_parity.extend_from_slice(recovery_shard);
678            }
679        }
680
681        // Compare with provided parity
682        let parity_matches = generated_parity
683            .iter()
684            .zip(parity_block.iter())
685            .all(|(a, b)| a == b);
686
687        Ok(parity_matches)
688    }
689}
