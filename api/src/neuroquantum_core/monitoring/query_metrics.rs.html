83/// Slow query log manager
84#[derive(Debug, Default)]
85pub struct SlowQueryLog {
86    entries: Vec<SlowQueryEntry>,
87    total_slow_queries: u64,
88    queries_by_table: HashMap<String, u64>,
89}
90
91impl SlowQueryLog {
92    #[must_use]
93    pub fn new() -> Self {
94        Self {
95            entries: Vec::new(),
96            total_slow_queries: 0,
97            queries_by_table: HashMap::new(),
98        }
99    }
100
101    /// Add a slow query entry
102    pub fn add_entry(&mut self, entry: SlowQueryEntry, max_entries: usize) {
103        self.total_slow_queries += 1;
104
105        // Extract table name from query (simplified)
106        if let Some(table) = self.extract_table_name(&entry.query_text) {
107            *self.queries_by_table.entry(table).or_insert(0) += 1;
108        }
109
110        // Keep only the most recent entries
111        if self.entries.len() >= max_entries {
112            self.entries.remove(0);
113        }
114
115        self.entries.push(entry);
116    }
117
118    /// Get all slow query entries
119    #[must_use]
120    pub fn get_entries(&self) -> &[SlowQueryEntry] {
121        &self.entries
122    }
123
124    /// Get total count of slow queries
125    #[must_use]
126    pub const fn total_slow_queries(&self) -> u64 {
127        self.total_slow_queries
128    }
129
130    /// Get slow queries by table
131    #[must_use]
132    pub const fn queries_by_table(&self) -> &HashMap<String, u64> {
133        &self.queries_by_table
134    }
135
136    /// Extract table name from query text (simplified parser)
137    fn extract_table_name(&self, query: &str) -> Option<String> {
138        let query_lower = query.to_lowercase();
139        if let Some(from_pos) = query_lower.find("from ") {
140            let after_from = &query_lower[from_pos + 5..];
141            let table_name = after_from
142                .split_whitespace()
143                .next()
144                .unwrap_or("")
145                .trim_matches(|c| c == '(' || c == ')' || c == ',');
146            if !table_name.is_empty() {
147                return Some(table_name.to_string());
148            }
149        }
150        None
151    }
152}

166/// Index usage statistics tracker
167#[derive(Debug, Default)]
168pub struct IndexUsageStats {
169    indexes: HashMap<String, IndexUsageEntry>,
170    unused_indexes: Vec<String>,
171}
172
173impl IndexUsageStats {
174    #[must_use]
175    pub fn new() -> Self {
176        Self {
177            indexes: HashMap::new(),
178            unused_indexes: Vec::new(),
179        }
180    }
181
182    /// Record index usage
183    pub fn record_index_scan(&mut self, index_name: String, table_name: String, rows_read: u64) {
184        let entry = self
185            .indexes
186            .entry(index_name.clone())
187            .or_insert_with(|| IndexUsageEntry {
188                index_name: index_name.clone(),
189                table_name,
190                total_scans: 0,
191                total_rows_read: 0,
192                avg_rows_per_scan: 0.0,
193                last_used: SystemTime::now()
194                    .duration_since(UNIX_EPOCH)
195                    .map(|d| d.as_secs())
196                    .unwrap_or(0),
197                created_at: SystemTime::now()
198                    .duration_since(UNIX_EPOCH)
199                    .map(|d| d.as_secs())
200                    .unwrap_or(0),
201            });
202
203        entry.total_scans += 1;
204        entry.total_rows_read += rows_read;
205        entry.avg_rows_per_scan = entry.total_rows_read as f64 / entry.total_scans as f64;
206        entry.last_used = SystemTime::now()
207            .duration_since(UNIX_EPOCH)
208            .map(|d| d.as_secs())
209            .unwrap_or(0);
210    }
211
212    /// Get all index usage statistics
213    #[must_use]
214    pub fn get_all_stats(&self) -> Vec<IndexUsageEntry> {
215        self.indexes.values().cloned().collect()
216    }
217
218    /// Get unused indexes (not used in last N seconds)
219    pub fn get_unused_indexes(&mut self, threshold_secs: u64) -> Vec<String> {
220        let now = SystemTime::now()
221            .duration_since(UNIX_EPOCH)
222            .map(|d| d.as_secs())
223            .unwrap_or(0);
224
225        self.unused_indexes.clear();
226
227        for (name, entry) in &self.indexes {
228            if now - entry.last_used > threshold_secs {
229                self.unused_indexes.push(name.clone());
230            }
231        }
232
233        self.unused_indexes.clone()
234    }
235
236    /// Get most frequently used indexes
237    #[must_use]
238    pub fn get_top_indexes(&self, limit: usize) -> Vec<IndexUsageEntry> {
239        let mut entries: Vec<IndexUsageEntry> = self.indexes.values().cloned().collect();
240        entries.sort_by(|a, b| b.total_scans.cmp(&a.total_scans));
241        entries.truncate(limit);
242        entries
243    }

257/// Lock type enumeration

259pub enum LockType {

261    Exclusive,
262    RowLevel,
263    TableLevel,

267impl std::fmt::Display for LockType {

273            | Self::TableLevel => write!(f, "TableLevel"),
274            | Self::IndexLevel => write!(f, "IndexLevel"),

279/// Lock contention tracker
280#[derive(Debug, Default)]
281pub struct LockContentionTracker {
282    events: Vec<LockContentionEvent>,
283    total_contentions: u64,
284    total_wait_time_ms: u64,
285    contentions_by_resource: HashMap<String, u64>,
286    contentions_by_type: HashMap<LockType, u64>,
287}
288
289impl LockContentionTracker {
290    #[must_use]
291    pub fn new() -> Self {
292        Self {
293            events: Vec::new(),
294            total_contentions: 0,
295            total_wait_time_ms: 0,
296            contentions_by_resource: HashMap::new(),
297            contentions_by_type: HashMap::new(),
298        }
299    }
300
301    /// Record a lock contention event
302    pub fn record_contention(&mut self, event: LockContentionEvent, max_events: usize) {
303        self.total_contentions += 1;
304        self.total_wait_time_ms += event.wait_time_ms;
305
306        *self
307            .contentions_by_resource
308            .entry(event.resource.clone())
309            .or_insert(0) += 1;
310
311        *self
312            .contentions_by_type
313            .entry(event.lock_type.clone())
314            .or_insert(0) += 1;
315
316        // Keep only recent events
317        if self.events.len() >= max_events {
318            self.events.remove(0);
319        }
320
321        self.events.push(event);
322    }
323
324    /// Get all contention events
325    #[must_use]
326    pub fn get_events(&self) -> &[LockContentionEvent] {
327        &self.events
328    }
329
330    /// Get total contention count
331    #[must_use]
332    pub const fn total_contentions(&self) -> u64 {
333        self.total_contentions
334    }
335
336    /// Get average wait time
337    #[must_use]
338    pub fn avg_wait_time_ms(&self) -> f64 {
339        if self.total_contentions == 0 {
340            0.0
341        } else {
342            self.total_wait_time_ms as f64 / self.total_contentions as f64
343        }
344    }
345
346    /// Get contentions by resource
347    #[must_use]
348    pub const fn contentions_by_resource(&self) -> &HashMap<String, u64> {
349        &self.contentions_by_resource
350    }
351
352    /// Get contentions by type
353    #[must_use]
354    pub const fn contentions_by_type(&self) -> &HashMap<LockType, u64> {
355        &self.contentions_by_type
356    }
357
358    /// Get most contended resources
359    #[must_use]
360    pub fn get_hot_resources(&self, limit: usize) -> Vec<(String, u64)> {
361        let mut resources: Vec<(String, u64)> = self
362            .contentions_by_resource
363            .iter()
364            .map(|(k, v)| (k.clone(), *v))
365            .collect();
366        resources.sort_by(|a, b| b.1.cmp(&a.1));
367        resources.truncate(limit);
368        resources
369    }

380/// Histogram for query execution times
381#[derive(Debug, Clone, Serialize, Deserialize)]
382pub struct QueryHistogram {
383    pub query_pattern: String,
384    pub count: u64,
385    pub total_time_ms: u64,
386    pub min_time_ms: u64,
387    pub max_time_ms: u64,
388    pub avg_time_ms: f64,
389    pub p50_time_ms: u64,
390    pub p95_time_ms: u64,
391    pub p99_time_ms: u64,
392    pub execution_times: Vec<u64>, // Keep recent execution times for percentile calculation
393}
394
395impl QueryHistogram {
396    #[must_use]
397    pub const fn new(query_pattern: String) -> Self {
398        Self {
399            query_pattern,
400            count: 0,
401            total_time_ms: 0,
402            min_time_ms: u64::MAX,
403            max_time_ms: 0,
404            avg_time_ms: 0.0,
405            p50_time_ms: 0,
406            p95_time_ms: 0,
407            p99_time_ms: 0,
408            execution_times: Vec::new(),
409        }
410    }
411
412    /// Add an execution time and recalculate statistics
413    pub fn add_execution(&mut self, time_ms: u64, max_samples: usize) {
414        self.count += 1;
415        self.total_time_ms += time_ms;
416        self.min_time_ms = self.min_time_ms.min(time_ms);
417        self.max_time_ms = self.max_time_ms.max(time_ms);
418        self.avg_time_ms = self.total_time_ms as f64 / self.count as f64;
419
420        // Keep recent execution times for percentile calculation
421        if self.execution_times.len() >= max_samples {
422            self.execution_times.remove(0);
423        }
424        self.execution_times.push(time_ms);
425
426        // Recalculate percentiles
427        self.calculate_percentiles();
428    }
429
430    /// Calculate percentile statistics
431    fn calculate_percentiles(&mut self) {
432        if self.execution_times.is_empty() {
433            return;
434        }
435
436        let mut sorted = self.execution_times.clone();
437        sorted.sort_unstable();
438
439        let len = sorted.len();
440        self.p50_time_ms = sorted[len / 2];
441        self.p95_time_ms = sorted[len.saturating_mul(95) / 100];
442        self.p99_time_ms = sorted[len.saturating_mul(99) / 100];
443    }
444}

446impl QueryExecutionStats {
447    #[must_use]
448    pub fn new() -> Self {
449        Self {
450            query_histogram: HashMap::new(),
451            total_queries: 0,
452            failed_queries: 0,
453        }
454    }
455
456    /// Record query execution
457    pub fn record_execution(&mut self, query_pattern: String, time_ms: u64, success: bool) {
458        self.total_queries += 1;
459
460        if !success {
461            self.failed_queries += 1;
462        }
463
464        let histogram = self
465            .query_histogram
466            .entry(query_pattern.clone())
467            .or_insert_with(|| QueryHistogram::new(query_pattern));
468
469        histogram.add_execution(time_ms, 1000); // Keep last 1000 executions
470    }
471
472    /// Get all query histograms
473    #[must_use]
474    pub fn get_all_histograms(&self) -> Vec<QueryHistogram> {
475        self.query_histogram.values().cloned().collect()
476    }
477
478    /// Get slowest query patterns
479    #[must_use]
480    pub fn get_slowest_patterns(&self, limit: usize) -> Vec<QueryHistogram> {
481        let mut histograms: Vec<QueryHistogram> = self.query_histogram.values().cloned().collect();
482        histograms.sort_by(|a, b| {
483            b.avg_time_ms
484                .partial_cmp(&a.avg_time_ms)
485                .unwrap_or(std::cmp::Ordering::Equal)
486        });
487        histograms.truncate(limit);
488        histograms
489    }
490
491    /// Get most frequent query patterns
492    #[must_use]
493    pub fn get_most_frequent_patterns(&self, limit: usize) -> Vec<QueryHistogram> {
494        let mut histograms: Vec<QueryHistogram> = self.query_histogram.values().cloned().collect();
495        histograms.sort_by(|a, b| b.count.cmp(&a.count));
496        histograms.truncate(limit);
497        histograms
498    }

500    /// Get total queries
501    #[must_use]
502    pub const fn total_queries(&self) -> u64 {
503        self.total_queries
504    }
505
506    /// Get failed queries count
507    #[must_use]
508    pub const fn failed_queries(&self) -> u64 {
509        self.failed_queries
510    }
511
512    /// Get error rate percentage
513    #[must_use]
514    pub fn error_rate(&self) -> f64 {
515        if self.total_queries == 0 {
516            0.0
517        } else {
518            (self.failed_queries as f64 / self.total_queries as f64) * 100.0
519        }
520    }

523impl AdvancedQueryMetrics {
524    /// Create a new advanced query metrics collector
525    #[must_use]
526    pub fn new(config: MonitoringConfig) -> Self {
527        Self {
528            slow_query_log: Arc::new(RwLock::new(SlowQueryLog::new())),
529            index_usage_stats: Arc::new(RwLock::new(IndexUsageStats::new())),
530            lock_contention_tracker: Arc::new(RwLock::new(LockContentionTracker::new())),
531            query_execution_stats: Arc::new(RwLock::new(QueryExecutionStats::new())),
532            config,
533        }
534    }
535
536    /// Record a query execution
537    pub async fn record_query_execution(&self, params: QueryExecutionParams) {
538        let execution_time_ms = params.execution_time.as_millis() as u64;
539
540        // Record execution statistics
541        let query_pattern = self.normalize_query(&params.query_text);
542        self.query_execution_stats.write().await.record_execution(
543            query_pattern,
544            execution_time_ms,
545            params.success,
546        );
547
548        // Log slow queries
549        if execution_time_ms >= self.config.slow_query_threshold_ms {
550            let table_name = self.extract_table_from_query(&params.query_text);
551            let entry = SlowQueryEntry {
552                query_id: params.query_id,
553                query_text: params.query_text,
554                execution_time_ms,
555                timestamp: SystemTime::now()
556                    .duration_since(UNIX_EPOCH)
557                    .map(|d| d.as_secs())
558                    .unwrap_or(0),
559                user: params.user,
560                rows_examined: params.rows_examined,
561                rows_returned: params.rows_returned,
562                index_used: params.index_used.clone(),
563                lock_wait_time_ms: params.lock_wait_time.as_millis() as u64,
564                execution_plan: params.execution_plan,
565            };
566
567            self.slow_query_log
568                .write()
569                .await
570                .add_entry(entry, self.config.max_slow_queries);
571
572            // Record index usage
573            if self.config.track_index_usage {
574                if let Some(index_name) = params.index_used {
575                    self.index_usage_stats.write().await.record_index_scan(
576                        index_name,
577                        table_name,
578                        params.rows_examined,
579                    );
580                }
581            }
582        }
583    }

585    /// Record lock contention event
586    pub async fn record_lock_contention(
587        &self,
588        query_id: u64,
589        resource: String,
590        lock_type: LockType,
591        wait_time: Duration,
592        holder_query_id: Option<u64>,
593    ) {
594        if !self.config.track_lock_contention {

596        }

601            lock_type,

603            timestamp: SystemTime::now()
604                .duration_since(UNIX_EPOCH)
605                .map(|d| d.as_secs())
606                .unwrap_or(0),
607            holder_query_id,
608        };
609
610        self.lock_contention_tracker
611            .write()
612            .await
613            .record_contention(event, 10000); // Keep last 10k events
614    }

661    /// Normalize query for pattern matching
662    fn normalize_query(&self, query: &str) -> String {
663        // Simplified normalization: replace literals with placeholders
664        let normalized = query
665            .to_lowercase()
666            .split_whitespace()
667            .map(|word| {
668                if word.parse::<i64>().is_ok()
669                    || word.parse::<f64>().is_ok()
670                    || (word.starts_with('\'') && word.ends_with('\''))
671                {
672                    "?"
673                } else {
674                    word
675                }
676            })
677            .collect::<Vec<_>>()
678            .join(" ");
679
680        // Truncate if too long
681        if normalized.len() > 200 {
682            format!("{}...", &normalized[..200])
683        } else {
684            normalized
685        }
686    }

688    /// Extract table name from query
689    fn extract_table_from_query(&self, query: &str) -> String {
690        let query_lower = query.to_lowercase();
691        if let Some(from_pos) = query_lower.find("from ") {
692            let after_from = &query_lower[from_pos + 5..];
693            let table_name = after_from
694                .split_whitespace()
695                .next()
696                .unwrap_or("")
697                .trim_matches(|c| c == '(' || c == ')' || c == ',');
698            if !table_name.is_empty() {
699                return table_name.to_string();
700            }
701        }
702        "unknown".to_string()
703    }

730    #[tokio::test]
731    async fn test_slow_query_logging() {
732        let config = MonitoringConfig {
733            slow_query_threshold_ms: 50,
734            max_slow_queries: 100,
735            ..Default::default()
736        };
737
738        let metrics = AdvancedQueryMetrics::new(config);
739
740        // Record a slow query
741        metrics
742            .record_query_execution(QueryExecutionParams {
743                query_id: 1,
744                query_text: "SELECT * FROM users WHERE id = 1".to_string(),
745                execution_time: Duration::from_millis(100),
746                rows_examined: 1000,
747                rows_returned: 100,
748                index_used: Some("idx_users_id".to_string()),
749                lock_wait_time: Duration::from_millis(10),

825        }
