1//! Error types for `NeuroQuantumDB`
2//!
3//! This module defines comprehensive error handling for the `NeuroQuantumDB` system,
4//! including DNA compression errors, storage errors, and system-level errors.
5
6use thiserror::Error;
7
8/// Main error type for `NeuroQuantumDB` operations
9#[derive(Debug, Error)]
10pub enum NeuroQuantumError {
11    #[error("Core system error: {0}")]
12    CoreError(String),
13
14    #[error("Storage error: {0}")]
15    StorageError(String),
16
17    #[error("Compression error: {0}")]
18    CompressionError(String),
19
20    #[error("Serialization error: {0}")]
21    SerializationError(String),
22
23    #[error("Query processing error: {0}")]
24    QueryError(String),
25
26    #[error("Learning engine error: {0}")]
27    LearningError(String),
28
29    #[error("Security error: {0}")]
30    SecurityError(String),
31
32    #[error("Configuration error: {0}")]
33    ConfigError(String),
34
35    #[error("Network error: {0}")]
36    NetworkError(String),
37
38    #[error("Authentication error: {0}")]
39    AuthError(String),
40
41    #[error("Validation error: {0}")]
42    ValidationError(String),
43
44    #[error("Resource exhausted: {0}")]
45    ResourceExhausted(String),
46
47    #[error("Operation timeout: {0}")]
48    Timeout(String),
49
50    #[error("Invalid configuration: {0}")]
51    InvalidConfig(String),
52
53    #[error("Invalid operation: {0}")]
54    InvalidOperation(String),
55
56    #[error("Lock error: {0}")]
57    LockError(String),
58
59    #[error("Not found: {0}")]
60    NotFound(String),
61
62    #[error("Transaction error: {0}")]
63    TransactionError(String),
64
65    #[error("Deadlock detected: {0}")]
66    DeadlockDetected(String),
67
68    #[error("Isolation violation: {0}")]
69    IsolationViolation(String),
70
71    #[error("Concurrent modification detected: {0}")]
72    ConcurrentModification(String),
73
74    #[error("IO error: {0}")]
75    IoError(#[from] std::io::Error),
76
77    #[error("JSON error: {0}")]
78    JsonError(#[from] serde_json::Error),
79}
80
81// Type aliases for backward compatibility with existing modules
82pub type CoreError = NeuroQuantumError;
83pub type CoreResult<T> = Result<T, NeuroQuantumError>;
84
85impl NeuroQuantumError {
86    /// Create an invalid operation error
87    #[must_use]
88    pub fn invalid_operation(msg: &str) -> Self {
89        Self::InvalidOperation(msg.to_string())
90    }
91}
92
93// Convert DNA errors to NeuroQuantumError
94impl From<crate::dna::DNAError> for NeuroQuantumError {
95    fn from(error: crate::dna::DNAError) -> Self {
96        Self::CompressionError(error.to_string())
97    }
98}
