1//! # Transverse Field Ising Model (TFIM)
2//!
3//! # ⚠️ Classical Simulation Notice
4//!
5//! **This is a CLASSICAL Monte Carlo simulation of quantum annealing dynamics.**
6//! It does NOT run on quantum hardware. The transverse field evolution and
7//! tunneling effects are approximated using classical stochastic methods.
8//!
9//! ## Technical Background
10//!
11//! The Transverse Field Ising Model (TFIM) describes a quantum system with:
12//!
13//! - Ising interactions (`J_ij`) between spins
14//! - A transverse magnetic field (Γ) inducing quantum fluctuations
15//! - External longitudinal fields (`h_i`) biasing individual spins
16//!
17//! This classical simulation approximates the quantum dynamics by:
18//!
19//! 1. Starting with a high transverse field (more "quantum" fluctuations)
20//! 2. Gradually reducing the field (classical annealing schedule)
21//! 3. Using Monte Carlo sampling to simulate thermal+quantum fluctuations
22//!
23//! ## Limitations
24//!
25//! - Cannot capture true quantum entanglement
26//! - Tunneling is approximated, not exact
27//! - No exponential quantum speedup
28//!
29//! ## When to Use
30//!
31//! - Optimization problems with Ising-like structure
32//! - Spin glass problems
33//! - Prototyping before deploying to real quantum annealers

42/// Field schedule for transverse field annealing
43#[derive(Debug, Clone, Serialize, Deserialize)]
44pub enum FieldSchedule {
45    /// Linear schedule: Γ(t) = `Γ_0` * (1 - t/T)
46    Linear,
47    /// Exponential schedule: Γ(t) = `Γ_0` * exp(-λt)
48    Exponential { decay_rate: f64 },
49    /// Polynomial schedule: Γ(t) = `Γ_0` * (1 - t/T)^p
50    Polynomial { power: f64 },
51    /// Custom schedule with explicit values
52    Custom { values: Vec<f64> },
53}
54
55/// Configuration for Transverse Field Ising Model
56#[derive(Debug, Clone, Serialize, Deserialize)]
57pub struct TransverseFieldConfig {
58    /// Initial transverse field strength
59    pub initial_field: f64,
60    /// Final transverse field strength
61    pub final_field: f64,
62    /// Number of annealing steps
63    pub num_steps: usize,
64    /// Field schedule type
65    pub field_schedule: FieldSchedule,
66    /// Temperature for thermal fluctuations
67    pub temperature: f64,
68    /// Enable quantum tunneling
69    pub quantum_tunneling: bool,
70}

85/// TFIM Problem with Ising Hamiltonian
86#[derive(Debug, Clone, Serialize, Deserialize)]
87pub struct TFIMProblem {
88    /// Number of spins
89    pub num_spins: usize,
90    /// Coupling matrix `J_ij` (interaction strengths)
91    pub couplings: DMatrix<f64>,
92    /// External field `h_i` for each spin

98/// TFIM Solution with spin configuration

135    /// Solve TFIM problem using quantum annealing with multiple retries
136    /// This is more realistic as real quantum annealers run multiple times
137    #[instrument(skip(self, problem))]
138    pub fn solve(&self, problem: &TFIMProblem) -> CoreResult<TFIMSolution> {
139        self.solve_with_retries(problem, 1)
140    }
141
142    /// Solve TFIM problem with multiple annealing runs (retries)
143    /// Returns the best solution found across all runs
144    #[instrument(skip(self, problem))]
145    pub fn solve_with_retries(
146        &self,
147        problem: &TFIMProblem,
148        num_retries: usize,
149    ) -> CoreResult<TFIMSolution> {
150        let start_time = std::time::Instant::now();
151
152        if problem.num_spins == 0 {
153            return Err(CoreError::invalid_operation("Empty TFIM problem"));
154        }
155
156        debug!(
157            "Solving TFIM problem '{}' with {} spins ({} retries)",
158            problem.name, problem.num_spins, num_retries
159        );
160
161        let mut best_solution: Option<TFIMSolution> = None;
162
163        for _retry in 0..num_retries {
164            let solution = self.solve_single_run(problem)?;
165
166            match &best_solution {
167                | None => best_solution = Some(solution),
168                | Some(current_best) => {
169                    if solution.energy < current_best.energy {
170                        best_solution = Some(solution);
171                    }
172                },
173            }
174        }
175
176        let mut final_solution = best_solution.ok_or_else(|| {
177            CoreError::invalid_operation(
178                "TFIM solving failed: no valid solution found after all retries",
179            )
180        })?;
181        final_solution.computation_time_ms = start_time.elapsed().as_secs_f64() * 1000.0;
182
183        info!(
184            "TFIM solved with {} retries: best_energy={:.4}, prob={:.4}, time={:.2}ms",
185            num_retries,
186            final_solution.energy,
187            final_solution.ground_state_prob,
188            final_solution.computation_time_ms
189        );
190
191        Ok(final_solution)
192    }

194    /// Single annealing run
195    fn solve_single_run(&self, problem: &TFIMProblem) -> CoreResult<TFIMSolution> {
196        let start_time = std::time::Instant::now();
197
198        let mut rng = rand::thread_rng();
199
200        // Initialize random spin configuration
201        let mut spins: Vec<i8> = (0..problem.num_spins)
202            .map(|_| if rng.gen::<bool>() { 1 } else { -1 })
203            .collect();
204
205        let mut current_energy = self.calculate_energy(problem, &spins);
206        let mut best_spins = spins.clone();
207        let mut best_energy = current_energy;
208        let mut tunneling_events = 0;
209
210        // Quantum annealing with transverse field
211        for step in 0..self.config.num_steps {
212            let progress = step as f64 / self.config.num_steps as f64;
213            let transverse_field = self.get_field_strength(progress);
214
215            // Quantum tunneling probability
216            let tunnel_prob = if self.config.quantum_tunneling {
217                self.tunneling_probability(transverse_field)
218            } else {
219                0.0
220            };
221
222            // Try flipping each spin
223            for i in 0..problem.num_spins {
224                let old_spin = spins[i];
225                spins[i] = -old_spin; // Flip spin
226
227                let new_energy = self.calculate_energy(problem, &spins);
228                let delta_e = new_energy - current_energy;
229
230                // Accept/reject with quantum tunneling enhancement
231                let accept = if delta_e < 0.0 {
232                    true
233                } else if rng.gen::<f64>() < tunnel_prob {
234                    tunneling_events += 1;
235                    true
236                } else {
237                    let thermal_prob = (-delta_e / self.config.temperature).exp();
238                    rng.gen::<f64>() < thermal_prob
239                };
240
241                if accept {
242                    current_energy = new_energy;
243                    if current_energy < best_energy {
244                        best_spins = spins.clone();
245                        best_energy = current_energy;
246                    }
247                } else {
248                    spins[i] = old_spin; // Revert flip
249                }
250            }
251        }
252
253        let computation_time_ms = start_time.elapsed().as_secs_f64() * 1000.0;
254
255        // Estimate ground state probability
256        let ground_state_prob =
257            self.estimate_ground_state_probability(problem, best_energy, self.config.temperature);
258
259        debug!(
260            "TFIM single run: energy={:.4}, tunneling_events={}, prob={:.4}, time={:.2}ms",
261            best_energy, tunneling_events, ground_state_prob, computation_time_ms
262        );
263
264        Ok(TFIMSolution {
265            spins: best_spins,
266            energy: best_energy,
267            ground_state_prob,
268            steps: self.config.num_steps,
269            tunneling_events,
270            computation_time_ms,
271        })
272    }

274    /// Calculate Ising Hamiltonian energy: H = -Σ `J_ij` `s_i` `s_j` - Σ `h_i` `s_i`
275    fn calculate_energy(&self, problem: &TFIMProblem, spins: &[i8]) -> f64 {
276        let mut energy = 0.0;
277
278        // Interaction terms
279        for i in 0..problem.num_spins {
280            for j in (i + 1)..problem.num_spins {
281                let coupling = problem.couplings[(i, j)];
282                energy -= coupling * f64::from(spins[i]) * f64::from(spins[j]);
283            }
284        }
285
286        // External field terms
287        for (i, &spin) in spins.iter().enumerate().take(problem.num_spins) {
288            energy -= problem.external_fields[i] * f64::from(spin);
289        }
290
291        energy
292    }

294    /// Get transverse field strength at given progress (0 to 1)
295    fn get_field_strength(&self, progress: f64) -> f64 {
296        match &self.config.field_schedule {
297            | FieldSchedule::Linear => self
298                .config
299                .initial_field
300                .mul_add(1.0 - progress, self.config.final_field * progress),
301            | FieldSchedule::Exponential { decay_rate } => self
302                .config
303                .initial_field
304                .mul_add((-decay_rate * progress).exp(), self.config.final_field),
305            | FieldSchedule::Polynomial { power } => self
306                .config
307                .initial_field
308                .mul_add((1.0 - progress).powf(*power), self.config.final_field),
309            | FieldSchedule::Custom { values } => {
310                let idx = (progress * (values.len() - 1) as f64) as usize;
311                values[idx.min(values.len() - 1)]
312            },
313        }
314    }

316    /// Calculate quantum tunneling probability based on transverse field
317    fn tunneling_probability(&self, field_strength: f64) -> f64 {
318        // Tunneling probability ∝ exp(-barrier/Γ)
319        // Higher field → higher tunneling probability
320        let barrier = 2.0; // Energy barrier height
321        let prob = (field_strength / barrier).min(1.0);
322        prob * 0.5 // Scale to reasonable range
323    }

325    /// Estimate ground state probability using Boltzmann distribution
326    fn estimate_ground_state_probability(
327        &self,
328        problem: &TFIMProblem,
329        energy: f64,
330        temperature: f64,
331    ) -> f64 {
332        // P(E) ∝ exp(-E/kT)
333        // Normalize by partition function estimate
334        let partition_approx = 2.0_f64.powi(problem.num_spins as i32);
335        let prob = (-energy / temperature).exp() / partition_approx;
336        prob.min(1.0)
337    }

339    /// Create TFIM problem from QUBO Q matrix
340    pub fn from_qubo(q_matrix: &DMatrix<f64>) -> CoreResult<TFIMProblem> {
341        let n = q_matrix.nrows();
342        if n == 0 || n != q_matrix.ncols() {
343            return Err(CoreError::invalid_operation("Invalid Q matrix"));
344        }
345
346        // Convert QUBO to Ising: x_i = (1 + s_i) / 2
347        let mut couplings = DMatrix::zeros(n, n);
348        let mut external_fields = vec![0.0; n];
349
350        for i in 0..n {
351            for j in 0..n {
352                let q_ij = q_matrix[(i, j)];
353                if i == j {
354                    external_fields[i] += q_ij / 2.0;
355                } else {
356                    couplings[(i, j)] = q_ij / 4.0;
357                }
358            }
359        }
360
361        Ok(TFIMProblem {
362            num_spins: n,
363            couplings,
364            external_fields,
365            name: "QUBO-to-Ising".to_string(),
366        })
367    }

369    /// Create fully-connected ferromagnetic Ising model
370    pub fn ferromagnetic_model(num_spins: usize, coupling: f64) -> CoreResult<TFIMProblem> {
371        if num_spins == 0 {
372            return Err(CoreError::invalid_operation("Invalid number of spins"));
373        }
374
375        let mut couplings = DMatrix::zeros(num_spins, num_spins);
376        for i in 0..num_spins {
377            for j in (i + 1)..num_spins {
378                couplings[(i, j)] = coupling;
379                couplings[(j, i)] = coupling;
380            }
381        }
382
383        Ok(TFIMProblem {
384            num_spins,
385            couplings,
386            external_fields: vec![0.0; num_spins],
387            name: "Ferromagnetic".to_string(),
388        })
389    }

391    /// Create spin glass model with random couplings
392    pub fn spin_glass_model(num_spins: usize, disorder_strength: f64) -> CoreResult<TFIMProblem> {
393        if num_spins == 0 {
394            return Err(CoreError::invalid_operation("Invalid number of spins"));
395        }
396
397        let mut rng = rand::thread_rng();
398        let mut couplings = DMatrix::zeros(num_spins, num_spins);
399
400        for i in 0..num_spins {
401            for j in (i + 1)..num_spins {
402                let coupling = (rng.gen::<f64>() - 0.5) * 2.0 * disorder_strength;
403                couplings[(i, j)] = coupling;
404                couplings[(j, i)] = coupling;
405            }
406        }
407
408        let external_fields: Vec<f64> = (0..num_spins)
409            .map(|_| (rng.gen::<f64>() - 0.5) * disorder_strength)
410            .collect();
411
412        Ok(TFIMProblem {
413            num_spins,
414            couplings,
415            external_fields,
416            name: "SpinGlass".to_string(),
417        })
418    }

431    #[test]
432    fn test_tfim_simple() {
433        let problem = TFIMProblem {
434            num_spins: 3,
435            couplings: DMatrix::from_fn(3, 3, |i, j| if i == j { 0.0 } else { 1.0 }),
436            external_fields: vec![0.0, 0.0, 0.0],
437            name: "Simple".to_string(),
438        };
439
440        // Use more annealing steps for better convergence
441        let config = TransverseFieldConfig {
442            initial_field: 10.0,
443            final_field: 0.01,
444            num_steps: 2000,
445            field_schedule: FieldSchedule::Linear,
446            temperature: 0.5, // Lower temperature for better convergence
447            quantum_tunneling: true,
448        };
449
450        let solver = TFIMSolver::with_config(config);
451
452        // Run multiple annealing attempts to ensure convergence
453        // This mimics real quantum annealing where multiple samples are taken
454        let solution = solver.solve_with_retries(&problem, 5).unwrap();
455
456        assert_eq!(solution.spins.len(), 3);
457
458        // For ferromagnetic coupling (J_ij = 1.0), optimal energy is when all spins align
459        // Energy = -J * (s0*s1 + s0*s2 + s1*s2) = -1.0 * 3 = -3.0 (all aligned)
460        // With multiple retries, we should reliably find the ground state
461        assert!(
462            solution.energy < 0.0,
463            "Ferromagnetic coupling should result in aligned spins with negative energy, got: {}",
464            solution.energy
465        );
466
467        // Check if we found the optimal solution (all spins aligned)
468        // Optimal energy for 3 spins with J=1.0 is -3.0
469        assert!(
470            solution.energy <= -2.5,
471            "Expected near-optimal energy (≤ -2.5), got: {}",
472            solution.energy
473        );
474
475        assert!(solution.ground_state_prob > 0.0);
476    }
