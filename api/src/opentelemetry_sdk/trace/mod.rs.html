61    #[test]
62    fn tracer_in_span() {
63        // Arrange
64        let exporter = InMemorySpanExporterBuilder::new().build();
65        let provider = TracerProvider::builder()
66            .with_span_processor(SimpleSpanProcessor::new(Box::new(exporter.clone())))
67            .build();
68
69        // Act
70        let tracer = provider.tracer("test_tracer");
71        tracer.in_span("span_name", |cx| {
72            let span = cx.span();
73            assert!(span.is_recording());
74            span.update_name("span_name_updated");
75            span.set_attribute(KeyValue::new("attribute1", "value1"));
76            span.add_event("test-event".to_string(), vec![]);
77        });
78
79        // Assert
80        let exported_spans = exporter
81            .get_finished_spans()
82            .expect("Spans are expected to be exported.");
83        assert_eq!(exported_spans.len(), 1);
84        let span = &exported_spans[0];
85        assert_eq!(span.name, "span_name_updated");
86        assert_eq!(span.instrumentation_scope.name(), "test_tracer");
87        assert_eq!(span.attributes.len(), 1);
88        assert_eq!(span.events.len(), 1);
89        assert_eq!(span.events[0].name, "test-event");
90        assert_eq!(span.span_context.trace_flags(), TraceFlags::SAMPLED);
91        assert!(!span.span_context.is_remote());
92        assert_eq!(span.status, Status::Unset);
93    }

205        // Arrange
206        let exporter = InMemorySpanExporterBuilder::new().build();
207        let provider = TracerProvider::builder()
208            .with_span_processor(SimpleSpanProcessor::new(Box::new(exporter.clone())))
209            .build();
210
211        // Act
212        let tracer = provider.tracer("test_tracer");
213
214        let mut events = Vec::new();
215        for _i in 0..(DEFAULT_MAX_EVENT_PER_SPAN * 2) {
216            events.push(Event::with_name("test event"))
217        }
218
219        // add events via span builder
220        let span_builder = SpanBuilder::from_name("span_name").with_events(events);
221        let mut span = tracer.build(span_builder);
222
223        // add events using span api after building the span
224        span.add_event("test event again, after span builder", Vec::new());
225        span.add_event("test event once again, after span builder", Vec::new());
226        span.end();
227
228        // Assert
229        let exported_spans = exporter
230            .get_finished_spans()
231            .expect("Spans are expected to be exported.");
232        assert_eq!(exported_spans.len(), 1);
233        let span = &exported_spans[0];
234        assert_eq!(span.name, "span_name");
235        assert_eq!(span.events.len(), DEFAULT_MAX_EVENT_PER_SPAN as usize);
236        assert_eq!(span.events.dropped_count, DEFAULT_MAX_EVENT_PER_SPAN + 2);
237    }

269        fn should_sample(
270            &self,
271            parent_context: Option<&Context>,
272            _trace_id: TraceId,
273            _name: &str,
274            _span_kind: &SpanKind,
275            _attributes: &[KeyValue],

287                trace_state,
288            }
289        }
290    }

292    #[test]
293    fn trace_state_for_record_only_sampler() {
294        let exporter = InMemorySpanExporterBuilder::new().build();
295        let provider = TracerProvider::builder()
296            .with_sampler(TestRecordOnlySampler::default())
297            .with_span_processor(SimpleSpanProcessor::new(Box::new(exporter.clone())))
298            .build();
299
300        let tracer = provider.tracer("test");
301        let trace_state = TraceState::from_key_value(vec![("foo", "bar")]).unwrap();
302
303        let parent_context = Context::new().with_span(TestSpan(SpanContext::new(
304            TraceId::from_u128(10000),
305            SpanId::from_u64(20),
306            TraceFlags::SAMPLED,
307            true,
308            trace_state.clone(),
309        )));
310
311        let span = tracer.build_with_context(
312            SpanBuilder::from_name("span")
313                .with_attributes(vec![KeyValue::new("extra_attr_key", "extra_attr_value")]),
314            &parent_context,
315        );
316        assert!(!span.span_context().trace_flags().is_sampled());
317        assert_eq!(
318            span.exported_data().unwrap().attributes,
319            vec![
320                KeyValue::new("extra_attr_key", "extra_attr_value"),
321                KeyValue::new("record_only_key", "record_only_value")
322            ]
323        );
324        assert_eq!(span.span_context().trace_state().get("foo"), Some("bar"));
325    }
