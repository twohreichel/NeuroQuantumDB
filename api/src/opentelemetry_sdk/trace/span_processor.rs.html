1//! # OpenTelemetry Span Processor Interface
2//!
3//! Span processor is an interface which allows hooks for span start and end method
4//! invocations. The span processors are invoked only when
5//! [`is_recording`] is true.
6//!
7//! Built-in span processors are responsible for batching and conversion of spans to
8//! exportable representation and passing batches to exporters.
9//!
10//! Span processors can be registered directly on SDK [`TracerProvider`] and they are
11//! invoked in the same order as they were registered.
12//!
13//! All `Tracer` instances created by a `TracerProvider` share the same span processors.
14//! Changes to this collection reflect in all `Tracer` instances.
15//!
16//! The following diagram shows `SpanProcessor`'s relationship to other components
17//! in the SDK:
18//!
19//! ```ascii
20//!   +-----+--------------+   +-----------------------+   +-------------------+
21//!   |     |              |   |                       |   |                   |
22//!   |     |              |   | (Batch)SpanProcessor  |   |    SpanExporter   |
23//!   |     |              +---> (Simple)SpanProcessor +--->  (OTLPExporter)   |
24//!   |     |              |   |                       |   |                   |
25//!   | SDK | Tracer.span()|   +-----------------------+   +-------------------+
26//!   |     | Span.end()   |
27//!   |     |              |
28//!   |     |              |
29//!   |     |              |
30//!   |     |              |
31//!   +-----+--------------+
32//! ```
33//!
34//! [`is_recording`]: opentelemetry::trace::Span::is_recording()
35//! [`TracerProvider`]: opentelemetry::trace::TracerProvider

80/// `SpanProcessor` is an interface which allows hooks for span start and end
81/// method invocations. The span processors are invoked only when is_recording
82/// is true.
83pub trait SpanProcessor: Send + Sync + std::fmt::Debug {
84    /// `on_start` is called when a `Span` is started.  This method is called
85    /// synchronously on the thread that started the span, therefore it should
86    /// not block or throw exceptions.
87    fn on_start(&self, span: &mut Span, cx: &Context);
88    /// `on_end` is called after a `Span` is ended (i.e., the end timestamp is
89    /// already set). This method is called synchronously within the `Span::end`
90    /// API, therefore it should not block or throw an exception.
91    fn on_end(&self, span: SpanData);
92    /// Force the spans lying in the cache to be exported.
93    fn force_flush(&self) -> TraceResult<()>;
94    /// Shuts down the processor. Called when SDK is shut down. This is an
95    /// opportunity for processors to do any cleanup required.
96    ///
97    /// Implementation should make sure shutdown can be called multiple times.
98    fn shutdown(&self) -> TraceResult<()>;
99    /// Set the resource for the log processor.
100    fn set_resource(&mut self, _resource: &Resource) {}
101}

103/// A [SpanProcessor] that passes finished spans to the configured
104/// `SpanExporter`, as soon as they are finished, without any batching. This is
105/// typically useful for debugging and testing. For scenarios requiring higher
106/// performance/throughput, consider using [BatchSpanProcessor].
107#[derive(Debug)]
108pub struct SimpleSpanProcessor {
109    exporter: Mutex<Box<dyn SpanExporter>>,
110}
111
112impl SimpleSpanProcessor {
113    /// Create a new [SimpleSpanProcessor] using the provided exporter.
114    pub fn new(exporter: Box<dyn SpanExporter>) -> Self {
115        Self {
116            exporter: Mutex::new(exporter),
117        }
118    }
119}
120
121impl SpanProcessor for SimpleSpanProcessor {
122    fn on_start(&self, _span: &mut Span, _cx: &Context) {
123        // Ignored
124    }
125
126    fn on_end(&self, span: SpanData) {
127        if !span.span_context.is_sampled() {
128            return;
129        }
130
131        let result = self
132            .exporter
133            .lock()
134            .map_err(|_| TraceError::Other("SimpleSpanProcessor mutex poison".into()))
135            .and_then(|mut exporter| futures_executor::block_on(exporter.export(vec![span])));
136
137        if let Err(err) = result {
138            // TODO: check error type, and log `error` only if the error is user-actiobable, else log `debug`
139            otel_debug!(
140                name: "SimpleProcessor.OnEnd.Error",
141                reason = format!("{:?}", err)
142            );
143        }
144    }
145
146    fn force_flush(&self) -> TraceResult<()> {
147        // Nothing to flush for simple span processor.
148        Ok(())
149    }
150
151    fn shutdown(&self) -> TraceResult<()> {
152        if let Ok(mut exporter) = self.exporter.lock() {
153            exporter.shutdown();
154            Ok(())
155        } else {
156            Err(TraceError::Other(
157                "SimpleSpanProcessor mutex poison at shutdown".into(),
158            ))
159        }
160    }
161
162    fn set_resource(&mut self, resource: &Resource) {
163        if let Ok(mut exporter) = self.exporter.lock() {
164            exporter.set_resource(resource);
165        }
166    }
167}

169/// A [`SpanProcessor`] that asynchronously buffers finished spans and reports
170/// them at a preconfigured interval.
171///
172/// Batch span processors need to run a background task to collect and send
173/// spans. Different runtimes need different ways to handle the background task.
174///
175/// Note: Configuring an opentelemetry `Runtime` that's not compatible with the
176/// underlying runtime can cause deadlocks (see tokio section).
177///
178/// ### Use with Tokio
179///
180/// Tokio currently offers two different schedulers. One is
181/// `current_thread_scheduler`, the other is `multiple_thread_scheduler`. Both
182/// of them default to use batch span processors to install span exporters.
183///
184/// Tokio's `current_thread_scheduler` can cause the program to hang forever if
185/// blocking work is scheduled with other tasks in the same runtime. To avoid
186/// this, be sure to enable the `rt-tokio-current-thread` feature in this crate
187/// if you are using that runtime (e.g. users of actix-web), and blocking tasks
188/// will then be scheduled on a different thread.
189///
190/// # Examples
191///
192/// This processor can be configured with an [`executor`] of your choice to
193/// batch and upload spans asynchronously when they end. If you have added a
194/// library like [`tokio`] or [`async-std`], you can pass in their respective
195/// `spawn` and `interval` functions to have batching performed in those
196/// contexts.
197///
198/// ```
199/// # #[cfg(feature="tokio")]
200/// # {
201/// use opentelemetry::global;
202/// use opentelemetry_sdk::{runtime, testing::trace::NoopSpanExporter, trace};
203/// use opentelemetry_sdk::trace::BatchConfigBuilder;
204/// use std::time::Duration;
205///
206/// #[tokio::main]
207/// async fn main() {
208///     // Configure your preferred exporter
209///     let exporter = NoopSpanExporter::new();
210///
211///     // Create a batch span processor using an exporter and a runtime
212///     let batch = trace::BatchSpanProcessor::builder(exporter, runtime::Tokio)
213///         .with_batch_config(BatchConfigBuilder::default().with_max_queue_size(4096).build())
214///         .build();
215///
216///     // Then use the `with_batch_exporter` method to have the provider export spans in batches.
217///     let provider = trace::TracerProvider::builder()
218///         .with_span_processor(batch)
219///         .build();
220///
221///     let _ = global::set_tracer_provider(provider);
222/// }
223/// # }
224/// ```
225///
226/// [`executor`]: https://docs.rs/futures/0.3/futures/executor/index.html
227/// [`tokio`]: https://tokio.rs
228/// [`async-std`]: https://async.rs
229pub struct BatchSpanProcessor<R: RuntimeChannel> {
230    message_sender: R::Sender<BatchMessage>,
231
232    // Track dropped spans
233    dropped_spans_count: AtomicUsize,
234
235    // Track the maximum queue size that was configured for this processor
236    max_queue_size: usize,
237}
238
239impl<R: RuntimeChannel> fmt::Debug for BatchSpanProcessor<R> {
240    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
241        f.debug_struct("BatchSpanProcessor")
242            .field("message_sender", &self.message_sender)
243            .finish()
244    }
245}

247impl<R: RuntimeChannel> SpanProcessor for BatchSpanProcessor<R> {
248    fn on_start(&self, _span: &mut Span, _cx: &Context) {
249        // Ignored
250    }
251
252    fn on_end(&self, span: SpanData) {
253        if !span.span_context.is_sampled() {
254            return;
255        }
256
257        let result = self.message_sender.try_send(BatchMessage::ExportSpan(span));
258
259        // If the queue is full, and we can't buffer a span
260        if result.is_err() {
261            // Increment the number of dropped spans. If this is the first time we've had to drop,
262            // emit a warning.
263            if self.dropped_spans_count.fetch_add(1, Ordering::Relaxed) == 0 {
264                otel_warn!(name: "BatchSpanProcessor.SpanDroppingStarted",
265                    message = "Beginning to drop span messages due to full/internal errors. No further log will be emitted for further drops until Shutdown. During Shutdown time, a log will be emitted with exact count of total spans dropped.");
266            }
267        }
268    }
269
270    fn force_flush(&self) -> TraceResult<()> {
271        let (res_sender, res_receiver) = oneshot::channel();
272        self.message_sender
273            .try_send(BatchMessage::Flush(Some(res_sender)))
274            .map_err(|err| TraceError::Other(err.into()))?;
275
276        futures_executor::block_on(res_receiver)
277            .map_err(|err| TraceError::Other(err.into()))
278            .and_then(|identity| identity)
279    }
280
281    fn shutdown(&self) -> TraceResult<()> {
282        let dropped_spans = self.dropped_spans_count.load(Ordering::Relaxed);
283        let max_queue_size = self.max_queue_size;
284        if dropped_spans > 0 {
285            otel_warn!(
286                name: "BatchSpanProcessor.Shutdown",
287                dropped_spans = dropped_spans,
288                max_queue_size = max_queue_size,
289                message = "Spans were dropped due to a full or closed queue. The count represents the total count of span records dropped in the lifetime of the BatchLogProcessor. Consider increasing the queue size and/or decrease delay between intervals."
290            );
291        }
292
293        let (res_sender, res_receiver) = oneshot::channel();
294        self.message_sender
295            .try_send(BatchMessage::Shutdown(res_sender))
296            .map_err(|err| TraceError::Other(err.into()))?;
297
298        futures_executor::block_on(res_receiver)
299            .map_err(|err| TraceError::Other(err.into()))
300            .and_then(|identity| identity)
301    }
302
303    fn set_resource(&mut self, resource: &Resource) {
304        let resource = Arc::new(resource.clone());
305        let _ = self
306            .message_sender
307            .try_send(BatchMessage::SetResource(resource));
308    }
309}

311/// Messages sent between application thread and batch span processor's work thread.
312// In this enum the size difference is not a concern because:
313// 1. If we wrap SpanData into a pointer, it will add overhead when processing.
314// 2. Most of the messages will be ExportSpan.
315#[allow(clippy::large_enum_variant)]
316#[derive(Debug)]
317enum BatchMessage {
318    /// Export spans, usually called when span ends
319    ExportSpan(SpanData),
320    /// Flush the current buffer to the backend, it can be triggered by
321    /// pre configured interval or a call to `force_push` function.
322    Flush(Option<oneshot::Sender<ExportResult>>),
323    /// Shut down the worker thread, push all spans in buffer to the backend.
324    Shutdown(oneshot::Sender<ExportResult>),
325    /// Set the resource for the exporter.
326    SetResource(Arc<Resource>),
327}
328
329struct BatchSpanProcessorInternal<R> {
330    spans: Vec<SpanData>,
331    export_tasks: FuturesUnordered<BoxFuture<'static, ExportResult>>,
332    runtime: R,
333    exporter: Box<dyn SpanExporter>,
334    config: BatchConfig,
335}
336
337impl<R: RuntimeChannel> BatchSpanProcessorInternal<R> {
338    async fn flush(&mut self, res_channel: Option<oneshot::Sender<ExportResult>>) {
339        let export_task = self.export();
340        let task = Box::pin(async move {
341            let result = export_task.await;
342
343            if let Some(channel) = res_channel {
344                // If a response channel is provided, attempt to send the export result through it.
345                if let Err(result) = channel.send(result) {
346                    otel_debug!(
347                        name: "BatchSpanProcessor.Flush.SendResultError",
348                        reason = format!("{:?}", result)
349                    );
350                }
351            } else if let Err(err) = result {
352                // If no channel is provided and the export operation encountered an error,
353                // log the error directly here.
354                // TODO: Consider returning the status instead of logging it.
355                otel_error!(
356                    name: "BatchSpanProcessor.Flush.ExportError",
357                    reason = format!("{:?}", err),
358                    message = "Failed during the export process"
359                );
360            }
361
362            Ok(())
363        });
364
365        if self.config.max_concurrent_exports == 1 {
366            let _ = task.await;
367        } else {
368            self.export_tasks.push(task);
369            while self.export_tasks.next().await.is_some() {}
370        }
371    }
372
373    /// Process a single message
374    ///
375    /// A return value of false indicates shutdown
376    async fn process_message(&mut self, message: BatchMessage) -> bool {
377        match message {
378            // Span has finished, add to buffer of pending spans.
379            BatchMessage::ExportSpan(span) => {
380                self.spans.push(span);
381
382                if self.spans.len() == self.config.max_export_batch_size {
383                    // If concurrent exports are saturated, wait for one to complete.
384                    if !self.export_tasks.is_empty()
385                        && self.export_tasks.len() == self.config.max_concurrent_exports
386                    {
387                        self.export_tasks.next().await;
388                    }
389
390                    let export_task = self.export();
391                    let task = async move {
392                        if let Err(err) = export_task.await {
393                            otel_error!(
394                                name: "BatchSpanProcessor.Export.Error",
395                                reason = format!("{}", err)
396                            );
397                        }
398
399                        Ok(())
400                    };
401                    // Special case when not using concurrent exports
402                    if self.config.max_concurrent_exports == 1 {
403                        let _ = task.await;
404                    } else {
405                        self.export_tasks.push(Box::pin(task));
406                    }
407                }
408            }
409            // Span batch interval time reached or a force flush has been invoked, export
410            // current spans.
411            //
412            // This is a hint to ensure that any tasks associated with Spans for which the
413            // SpanProcessor had already received events prior to the call to ForceFlush
414            // SHOULD be completed as soon as possible, preferably before returning from
415            // this method.
416            //
417            // In particular, if any SpanProcessor has any associated exporter, it SHOULD
418            // try to call the exporter's Export with all spans for which this was not
419            // already done and then invoke ForceFlush on it. The built-in SpanProcessors
420            // MUST do so. If a timeout is specified (see below), the SpanProcessor MUST
421            // prioritize honoring the timeout over finishing all calls. It MAY skip or
422            // abort some or all Export or ForceFlush calls it has made to achieve this
423            // goal.
424            //
425            // NB: `force_flush` is not currently implemented on exporters; the equivalent
426            // would be waiting for exporter tasks to complete. In the case of
427            // channel-coupled exporters, they will need a `force_flush` implementation to
428            // properly block.
429            BatchMessage::Flush(res_channel) => {
430                self.flush(res_channel).await;
431            }
432            // Stream has terminated or processor is shutdown, return to finish execution.
433            BatchMessage::Shutdown(ch) => {
434                self.flush(Some(ch)).await;
435                self.exporter.shutdown();
436                return false;
437            }
438            // propagate the resource
439            BatchMessage::SetResource(resource) => {
440                self.exporter.set_resource(&resource);
441            }
442        }
443        true
444    }
445
446    fn export(&mut self) -> BoxFuture<'static, ExportResult> {
447        // Batch size check for flush / shutdown. Those methods may be called
448        // when there's no work to do.
449        if self.spans.is_empty() {
450            return Box::pin(future::ready(Ok(())));
451        }
452
453        let export = self.exporter.export(self.spans.split_off(0));
454        let timeout = self.runtime.delay(self.config.max_export_timeout);
455        let time_out = self.config.max_export_timeout;
456
457        Box::pin(async move {
458            match future::select(export, timeout).await {
459                Either::Left((export_res, _)) => export_res,
460                Either::Right((_, _)) => ExportResult::Err(TraceError::ExportTimedOut(time_out)),
461            }
462        })
463    }
464
465    async fn run(mut self, mut messages: impl FusedStream<Item = BatchMessage> + Unpin) {
466        loop {
467            select! {
468                // FuturesUnordered implements Fuse intelligently such that it
469                // will become eligible again once new tasks are added to it.
470                _ = self.export_tasks.next() => {
471                    // An export task completed; do we need to do anything with it?
472                },
473                message = messages.next() => {
474                    match message {
475                        Some(message) => {
476                            if !self.process_message(message).await {
477                                break;
478                            }
479                        },
480                        None => break,
481                    }
482                },
483            }
484        }
485    }
486}

488impl<R: RuntimeChannel> BatchSpanProcessor<R> {
489    pub(crate) fn new(exporter: Box<dyn SpanExporter>, config: BatchConfig, runtime: R) -> Self {
490        let (message_sender, message_receiver) =
491            runtime.batch_message_channel(config.max_queue_size);
492
493        let max_queue_size = config.max_queue_size;
494
495        let inner_runtime = runtime.clone();
496        // Spawn worker process via user-defined spawn function.
497        runtime.spawn(Box::pin(async move {
498            // Timer will take a reference to the current runtime, so its important we do this within the
499            // runtime.spawn()
500            let ticker = inner_runtime
501                .interval(config.scheduled_delay)
502                .skip(1) // The ticker is fired immediately, so we should skip the first one to align with the interval.
503                .map(|_| BatchMessage::Flush(None));
504            let timeout_runtime = inner_runtime.clone();
505
506            let messages = Box::pin(stream::select(message_receiver, ticker));
507            let processor = BatchSpanProcessorInternal {
508                spans: Vec::new(),
509                export_tasks: FuturesUnordered::new(),
510                runtime: timeout_runtime,
511                config,
512                exporter,
513            };
514
515            processor.run(messages).await
516        }));
517
518        // Return batch processor with link to worker
519        BatchSpanProcessor {
520            message_sender,
521            dropped_spans_count: AtomicUsize::new(0),
522            max_queue_size,
523        }
524    }
525
526    /// Create a new batch processor builder
527    pub fn builder<E>(exporter: E, runtime: R) -> BatchSpanProcessorBuilder<E, R>
528    where
529        E: SpanExporter,
530    {
531        BatchSpanProcessorBuilder {
532            exporter,
533            config: Default::default(),
534            runtime,
535        }
536    }
537}

539/// Batch span processor configuration.
540/// Use [`BatchConfigBuilder`] to configure your own instance of [`BatchConfig`].
541#[derive(Debug)]
542pub struct BatchConfig {
543    /// The maximum queue size to buffer spans for delayed processing. If the
544    /// queue gets full it drops the spans. The default value of is 2048.
545    max_queue_size: usize,
546
547    /// The delay interval in milliseconds between two consecutive processing
548    /// of batches. The default value is 5 seconds.
549    scheduled_delay: Duration,
550
551    /// The maximum number of spans to process in a single batch. If there are
552    /// more than one batch worth of spans then it processes multiple batches
553    /// of spans one batch after the other without any delay. The default value
554    /// is 512.
555    max_export_batch_size: usize,
556
557    /// The maximum duration to export a batch of data.
558    max_export_timeout: Duration,
559
560    /// Maximum number of concurrent exports
561    ///
562    /// Limits the number of spawned tasks for exports and thus memory consumed
563    /// by an exporter. A value of 1 will cause exports to be performed
564    /// synchronously on the BatchSpanProcessor task.
565    max_concurrent_exports: usize,
566}
567
568impl Default for BatchConfig {
569    fn default() -> Self {
570        BatchConfigBuilder::default().build()
571    }
572}

632        self.max_concurrent_exports = max_concurrent_exports;
633        self
634    }

636    /// Set scheduled_delay_duration for [`BatchConfigBuilder`].
637    /// It's the delay interval in milliseconds between two consecutive processing of batches.
638    /// The default value is 5000 milliseconds.
639    pub fn with_scheduled_delay(mut self, scheduled_delay: Duration) -> Self {
640        self.scheduled_delay = scheduled_delay;
641        self
642    }

644    /// Set max_export_timeout for [`BatchConfigBuilder`].
645    /// It's the maximum duration to export a batch of data.
646    /// The The default value is 30000 milliseconds.
647    pub fn with_max_export_timeout(mut self, max_export_timeout: Duration) -> Self {
648        self.max_export_timeout = max_export_timeout;
649        self
650    }

652    /// Builds a `BatchConfig` enforcing the following invariants:
653    /// * `max_export_batch_size` must be less than or equal to `max_queue_size`.
654    pub fn build(self) -> BatchConfig {
655        // max export batch size must be less or equal to max queue size.
656        // we set max export batch size to max queue size if it's larger than max queue size.
657        let max_export_batch_size = min(self.max_export_batch_size, self.max_queue_size);
658
659        BatchConfig {
660            max_queue_size: self.max_queue_size,
661            scheduled_delay: self.scheduled_delay,
662            max_export_timeout: self.max_export_timeout,
663            max_concurrent_exports: self.max_concurrent_exports,
664            max_export_batch_size,
665        }
666    }
667
668    fn init_from_env_vars(mut self) -> Self {
669        if let Some(max_concurrent_exports) = env::var(OTEL_BSP_MAX_CONCURRENT_EXPORTS)
670            .ok()
671            .and_then(|max_concurrent_exports| usize::from_str(&max_concurrent_exports).ok())
672        {
673            self.max_concurrent_exports = max_concurrent_exports;
674        }
675
676        if let Some(max_queue_size) = env::var(OTEL_BSP_MAX_QUEUE_SIZE)
677            .ok()
678            .and_then(|queue_size| usize::from_str(&queue_size).ok())
679        {
680            self.max_queue_size = max_queue_size;
681        }
682
683        if let Some(scheduled_delay) = env::var(OTEL_BSP_SCHEDULE_DELAY)
684            .ok()
685            .and_then(|delay| u64::from_str(&delay).ok())
686        {
687            self.scheduled_delay = Duration::from_millis(scheduled_delay);
688        }
689
690        if let Some(max_export_batch_size) = env::var(OTEL_BSP_MAX_EXPORT_BATCH_SIZE)
691            .ok()
692            .and_then(|batch_size| usize::from_str(&batch_size).ok())
693        {
694            self.max_export_batch_size = max_export_batch_size;
695        }
696
697        // max export batch size must be less or equal to max queue size.
698        // we set max export batch size to max queue size if it's larger than max queue size.
699        if self.max_export_batch_size > self.max_queue_size {
700            self.max_export_batch_size = self.max_queue_size;
701        }
702
703        if let Some(max_export_timeout) = env::var(OTEL_BSP_EXPORT_TIMEOUT)
704            .ok()
705            .and_then(|timeout| u64::from_str(&timeout).ok())
706        {
707            self.max_export_timeout = Duration::from_millis(max_export_timeout);
708        }
709
710        self
711    }
