30mod tests {
31    use super::*;
32    use crate::testing::logs::InMemoryLogExporter;
33    use crate::Resource;

35    use opentelemetry::logs::{Logger, LoggerProvider as _, Severity};

37    use std::borrow::Borrow;

40    #[test]
41    fn logging_sdk_test() {
42        // Arrange
43        let resource = Resource::new(vec![
44            KeyValue::new("k1", "v1"),
45            KeyValue::new("k2", "v2"),
46            KeyValue::new("k3", "v3"),
47            KeyValue::new("k4", "v4"),
48        ]);
49        let exporter: InMemoryLogExporter = InMemoryLogExporter::default();
50        let logger_provider = LoggerProvider::builder()
51            .with_resource(resource.clone())
52            .with_log_processor(SimpleLogProcessor::new(Box::new(exporter.clone())))
53            .build();
54
55        // Act
56        let logger = logger_provider.logger("test-logger");
57        let mut log_record = logger.create_log_record();
58        log_record.set_severity_number(Severity::Error);
59        log_record.set_severity_text("Error");
60
61        // Adding attributes using a vector with explicitly constructed Key and AnyValue objects.
62        log_record.add_attributes(vec![
63            (Key::new("key1"), AnyValue::from("value1")),
64            (Key::new("key2"), AnyValue::from("value2")),
65        ]);
66
67        // Adding attributes using an array with explicitly constructed Key and AnyValue objects.
68        log_record.add_attributes([
69            (Key::new("key3"), AnyValue::from("value3")),
70            (Key::new("key4"), AnyValue::from("value4")),
71        ]);
72
73        // Adding attributes using a vector with tuple auto-conversion to Key and AnyValue.
74        log_record.add_attributes(vec![("key5", "value5"), ("key6", "value6")]);
75
76        // Adding attributes using an array with tuple auto-conversion to Key and AnyValue.
77        log_record.add_attributes([("key7", "value7"), ("key8", "value8")]);
78
79        // Adding Attributes from a HashMap
80        let mut attributes_map = HashMap::new();
81        attributes_map.insert("key9", "value9");
82        attributes_map.insert("key10", "value10");
83
84        log_record.add_attributes(attributes_map);
85
86        logger.emit(log_record);
87
88        // Assert
89        let exported_logs = exporter
90            .get_emitted_logs()
91            .expect("Logs are expected to be exported.");
92        assert_eq!(exported_logs.len(), 1);
93        let log = exported_logs
94            .first()
95            .expect("Atleast one log is expected to be present.");
96        assert_eq!(log.instrumentation.name(), "test-logger");
97        assert_eq!(log.record.severity_number, Some(Severity::Error));
98        assert_eq!(log.record.attributes_len(), 10);
99        for i in 1..=10 {
100            assert!(log.record.attributes_contains(
101                &Key::new(format!("key{}", i)),
102                &AnyValue::String(format!("value{}", i).into())
103            ));
104        }
105
106        // validate Resource
107        assert_eq!(&resource, log.resource.borrow());
108    }

110    #[test]
111    #[allow(deprecated)]
112    fn logger_attributes() {
113        let provider = LoggerProvider::builder().build();
114        let scope = InstrumentationScope::builder("test_logger")
115            .with_schema_url("https://opentelemetry.io/schema/1.0.0")
116            .with_attributes(vec![(KeyValue::new("test_k", "test_v"))])
117            .build();
118
119        let logger = provider.logger_with_scope(scope);
120        let instrumentation_scope = logger.instrumentation_scope();
121        assert_eq!(instrumentation_scope.name(), "test_logger");
122        assert_eq!(
123            instrumentation_scope.schema_url(),
124            Some("https://opentelemetry.io/schema/1.0.0")
125        );
126        assert!(instrumentation_scope
127            .attributes()
128            .eq(&[KeyValue::new("test_k", "test_v")]));
129    }
