33#[derive(Debug, Clone)]
34/// Handles the creation and coordination of [`Logger`]s.
35///
36/// All `Logger`s created by a `LoggerProvider` will share the same
37/// [`Resource`] and have their created log records processed by the
38/// configured log processors. This is a clonable handle to the `LoggerProvider`
39/// itself, and cloning it will create a new reference, not a new instance of a
40/// `LoggerProvider`. Dropping the last reference will trigger the shutdown of
41/// the provider, ensuring that all remaining logs are flushed and no further
42/// logs are processed. Shutdown can also be triggered manually by calling
43/// the [`shutdown`](LoggerProvider::shutdown) method.
44///
45/// [`Logger`]: opentelemetry::logs::Logger
46/// [`Resource`]: crate::Resource
47pub struct LoggerProvider {
48    inner: Arc<LoggerProviderInner>,
49}
50
51impl opentelemetry::logs::LoggerProvider for LoggerProvider {
52    type Logger = Logger;
53
54    fn logger(&self, name: impl Into<Cow<'static, str>>) -> Self::Logger {
55        let scope = InstrumentationScope::builder(name).build();
56        self.logger_with_scope(scope)
57    }
58
59    fn logger_with_scope(&self, scope: InstrumentationScope) -> Self::Logger {
60        // If the provider is shutdown, new logger will refer a no-op logger provider.
61        if self.inner.is_shutdown.load(Ordering::Relaxed) {
62            otel_debug!(
63                name: "LoggerProvider.NoOpLoggerReturned",
64                logger_name = scope.name(),
65            );
66            return Logger::new(scope, noop_logger_provider().clone());
67        }
68        if scope.name().is_empty() {
69            otel_info!(name: "LoggerNameEmpty",  message = "Logger name is empty; consider providing a meaningful name. Logger will function normally and the provided name will be used as-is.");
70        };
71        otel_debug!(
72            name: "LoggerProvider.NewLoggerReturned",
73            logger_name = scope.name(),
74        );
75        Logger::new(scope, self.clone())
76    }
77}

79impl LoggerProvider {
80    /// Create a new `LoggerProvider` builder.
81    pub fn builder() -> Builder {
82        Builder::default()
83    }
84
85    pub(crate) fn log_processors(&self) -> &[Box<dyn LogProcessor>] {
86        &self.inner.processors
87    }
88
89    pub(crate) fn resource(&self) -> &Resource {
90        &self.inner.resource
91    }
92
93    /// Force flush all remaining logs in log processors and return results.
94    pub fn force_flush(&self) -> Vec<LogResult<()>> {
95        self.log_processors()
96            .iter()
97            .map(|processor| processor.force_flush())
98            .collect()
99    }
100
101    /// Shuts down this `LoggerProvider`
102    pub fn shutdown(&self) -> LogResult<()> {
103        otel_debug!(
104            name: "LoggerProvider.ShutdownInvokedByUser",
105        );
106        if self
107            .inner
108            .is_shutdown
109            .compare_exchange(false, true, Ordering::SeqCst, Ordering::SeqCst)
110            .is_ok()
111        {
112            // propagate the shutdown signal to processors
113            let errs = self.inner.shutdown();
114            if errs.is_empty() {
115                Ok(())
116            } else {
117                Err(LogError::Other(format!("{errs:?}").into()))
118            }
119        } else {
120            Err(LogError::AlreadyShutdown("LoggerProvider".to_string()))
121        }
122    }
123}

125#[derive(Debug)]
126struct LoggerProviderInner {
127    processors: Vec<Box<dyn LogProcessor>>,
128    resource: Resource,
129    is_shutdown: AtomicBool,
130}
131
132impl LoggerProviderInner {
133    /// Shuts down the `LoggerProviderInner` and returns any errors.
134    pub(crate) fn shutdown(&self) -> Vec<LogError> {
135        let mut errs = vec![];
136        for processor in &self.processors {
137            if let Err(err) = processor.shutdown() {
138                // Log at debug level because:
139                //  - The error is also returned to the user for handling (if applicable)
140                //  - Or the error occurs during `LoggerProviderInner::Drop` as part of telemetry shutdown,
141                //    which is non-actionable by the user
142                match err {
143                    // specific handling for mutex poisioning
144                    LogError::MutexPoisoned(_) => {
145                        otel_debug!(
146                            name: "LoggerProvider.Drop.ShutdownMutexPoisoned",
147                        );
148                    }
149                    _ => {
150                        otel_debug!(
151                            name: "LoggerProvider.Drop.ShutdownError",
152                            error = format!("{err}")
153                        );
154                    }
155                }
156                errs.push(err);
157            }
158        }
159        errs
160    }
161}
162
163impl Drop for LoggerProviderInner {
164    fn drop(&mut self) {
165        if !self.is_shutdown.load(Ordering::Relaxed) {
166            otel_info!(
167                name: "LoggerProvider.Drop",
168                message = "Last reference of LoggerProvider dropped, initiating shutdown."
169            );
170            let _ = self.shutdown(); // errors are handled within shutdown
171        } else {
172            otel_debug!(
173                name: "LoggerProvider.Drop.AlreadyShutdown",
174                message = "LoggerProvider was already shut down; drop will not attempt shutdown again."
175            );
176        }
177    }
178}

180#[derive(Debug, Default)]
181/// Builder for provider attributes.
182pub struct Builder {
183    processors: Vec<Box<dyn LogProcessor>>,
184    resource: Option<Resource>,
185}
186
187impl Builder {
188    /// The `LogExporter` that this provider should use.
189    pub fn with_simple_exporter<T: LogExporter + 'static>(self, exporter: T) -> Self {
190        let mut processors = self.processors;
191        processors.push(Box::new(SimpleLogProcessor::new(Box::new(exporter))));
192
193        Builder { processors, ..self }
194    }
195
196    /// The `LogExporter` setup using a default `BatchLogProcessor` that this provider should use.
197    pub fn with_batch_exporter<T: LogExporter + 'static, R: RuntimeChannel>(
198        self,
199        exporter: T,
200        runtime: R,
201    ) -> Self {
202        let batch = BatchLogProcessor::builder(exporter, runtime).build();
203        self.with_log_processor(batch)
204    }
205
206    /// The `LogProcessor` that this provider should use.
207    pub fn with_log_processor<T: LogProcessor + 'static>(self, processor: T) -> Self {
208        let mut processors = self.processors;
209        processors.push(Box::new(processor));
210
211        Builder { processors, ..self }
212    }
213
214    /// The `Resource` to be associated with this Provider.
215    pub fn with_resource(self, resource: Resource) -> Self {
216        Builder {
217            resource: Some(resource),
218            ..self
219        }
220    }
221
222    /// Create a new provider from this configuration.
223    pub fn build(self) -> LoggerProvider {
224        let resource = self.resource.unwrap_or_default();
225
226        let logger_provider = LoggerProvider {
227            inner: Arc::new(LoggerProviderInner {
228                processors: self.processors,
229                resource,
230                is_shutdown: AtomicBool::new(false),
231            }),
232        };
233
234        // invoke set_resource on all the processors
235        for processor in logger_provider.log_processors() {
236            processor.set_resource(logger_provider.resource());
237        }
238
239        otel_debug!(
240            name: "LoggerProvider.Built",
241        );
242        logger_provider
243    }
244}

279impl opentelemetry::logs::Logger for Logger {
280    type LogRecord = LogRecord;
281
282    fn create_log_record(&self) -> Self::LogRecord {
283        LogRecord::default()
284    }
285
286    /// Emit a `LogRecord`.
287    fn emit(&self, mut record: Self::LogRecord) {
288        let provider = &self.provider;
289        let processors = provider.log_processors();
290
291        //let mut log_record = record;
292        if record.trace_context.is_none() {
293            let trace_context = Context::map_current(|cx| {
294                cx.has_active_span()
295                    .then(|| TraceContext::from(cx.span().span_context()))
296            });
297
298            if let Some(ref trace_context) = trace_context {
299                record.trace_context = Some(trace_context.clone());
300            }
301        }
302        if record.observed_timestamp.is_none() {
303            record.observed_timestamp = Some(SystemTime::now());
304        }
305
306        for p in processors {
307            p.emit(&mut record, &self.scope);
308        }
309    }
310
311    #[cfg(feature = "spec_unstable_logs_enabled")]
312    fn event_enabled(&self, level: Severity, target: &str) -> bool {
313        let provider = &self.provider;
314
315        let mut enabled = false;
316        for processor in provider.log_processors() {
317            enabled = enabled || processor.event_enabled(level, target, self.scope.name().as_ref());
318        }
319        enabled
320    }
