22/// Handles the creation and coordination of [Meter]s.
23///
24/// All `Meter`s created by a `MeterProvider` will be associated with the same
25/// [Resource], have the same [View]s applied to them, and have their produced
26/// metric telemetry passed to the configured [MetricReader]s. This is a
27/// clonable handle to the MeterProvider implementation itself, and cloning it
28/// will create a new reference, not a new instance of a MeterProvider. Dropping
29/// the last reference to it will trigger shutdown of the provider. Shutdown can
30/// also be triggered manually by calling the `shutdown` method.
31/// [Meter]: opentelemetry::metrics::Meter
32#[derive(Clone, Debug)]
33pub struct SdkMeterProvider {
34    inner: Arc<SdkMeterProviderInner>,
35}
36
37#[derive(Debug)]
38struct SdkMeterProviderInner {
39    pipes: Arc<Pipelines>,
40    meters: Mutex<HashMap<InstrumentationScope, Arc<SdkMeter>>>,
41    is_shutdown: AtomicBool,
42}
43
44impl Default for SdkMeterProvider {
45    fn default() -> Self {
46        SdkMeterProvider::builder().build()
47    }
48}

50impl SdkMeterProvider {
51    /// Return default [MeterProviderBuilder]
52    pub fn builder() -> MeterProviderBuilder {
53        MeterProviderBuilder::default()
54    }
55
56    /// Flushes all pending telemetry.
57    ///
58    /// There is no guaranteed that all telemetry be flushed or all resources have
59    /// been released on error.
60    ///
61    /// # Examples
62    ///
63    /// ```
64    /// use opentelemetry::{global, Context};
65    /// use opentelemetry_sdk::metrics::SdkMeterProvider;
66    ///
67    /// fn init_metrics() -> SdkMeterProvider {
68    ///     // Setup metric pipelines with readers + views, default has no
69    ///     // readers so nothing is exported.
70    ///     let provider = SdkMeterProvider::default();
71    ///
72    ///     // Set provider to be used as global meter provider
73    ///     let _ = global::set_meter_provider(provider.clone());
74    ///
75    ///     provider
76    /// }
77    ///
78    /// fn main() -> Result<(), Box<dyn std::error::Error>> {
79    ///     let provider = init_metrics();
80    ///
81    ///     // create instruments + record measurements
82    ///
83    ///     // force all instruments to flush
84    ///     provider.force_flush()?;
85    ///
86    ///     // record more measurements..
87    ///
88    ///     // shutdown ensures any cleanup required by the provider is done,
89    ///     // and also invokes shutdown on the readers.
90    ///     provider.shutdown()?;
91    ///
92    ///     Ok(())
93    /// }
94    /// ```
95    pub fn force_flush(&self) -> MetricResult<()> {
96        self.inner.force_flush()
97    }
98
99    /// Shuts down the meter provider flushing all pending telemetry and releasing
100    /// any held computational resources.
101    ///
102    /// This call is idempotent. The first call will perform all flush and releasing
103    /// operations. Subsequent calls will perform no action and will return an error
104    /// stating this.
105    ///
106    /// Measurements made by instruments from meters this MeterProvider created will
107    /// not be exported after Shutdown is called.
108    ///
109    /// There is no guaranteed that all telemetry be flushed or all resources have
110    /// been released on error.
111    pub fn shutdown(&self) -> MetricResult<()> {
112        otel_info!(
113            name: "MeterProvider.Shutdown",
114            message = "User initiated shutdown of MeterProvider."
115        );
116        self.inner.shutdown()
117    }
118}
119
120impl SdkMeterProviderInner {
121    fn force_flush(&self) -> MetricResult<()> {
122        self.pipes.force_flush()
123    }
124
125    fn shutdown(&self) -> MetricResult<()> {
126        if self
127            .is_shutdown
128            .compare_exchange(false, true, Ordering::SeqCst, Ordering::SeqCst)
129            .is_ok()
130        {
131            self.pipes.shutdown()
132        } else {
133            Err(MetricError::Other(
134                "metrics provider already shut down".into(),
135            ))
136        }
137    }
138}
139
140impl Drop for SdkMeterProviderInner {
141    fn drop(&mut self) {
142        // If user has already shutdown the provider manually by calling
143        // shutdown(), then we don't need to call shutdown again.
144        if self.is_shutdown.load(Ordering::Relaxed) {
145            otel_debug!(
146                name: "MeterProvider.Drop.AlreadyShutdown",
147                message = "MeterProvider was already shut down; drop will not attempt shutdown again."
148            );
149        } else {
150            otel_info!(
151                name: "MeterProvider.Drop",
152                message = "Last reference of MeterProvider dropped, initiating shutdown."
153            );
154            if let Err(err) = self.shutdown() {
155                otel_error!(
156                    name: "MeterProvider.Drop.ShutdownFailed",
157                    message = "Shutdown attempt failed during drop of MeterProvider.",
158                    reason = format!("{}", err)
159                );
160            } else {
161                otel_info!(
162                    name: "MeterProvider.Drop.ShutdownCompleted",
163                );
164            }
165        }
166    }
167}

169impl MeterProvider for SdkMeterProvider {
170    fn meter(&self, name: &'static str) -> Meter {
171        let scope = InstrumentationScope::builder(name).build();
172        self.meter_with_scope(scope)
173    }
174
175    fn meter_with_scope(&self, scope: InstrumentationScope) -> Meter {
176        if self.inner.is_shutdown.load(Ordering::Relaxed) {
177            otel_debug!(
178                name: "MeterProvider.NoOpMeterReturned",
179                meter_name = scope.name(),
180            );
181            return Meter::new(Arc::new(NoopMeter::new()));
182        }
183
184        if scope.name().is_empty() {
185            otel_info!(name: "MeterNameEmpty", message = "Meter name is empty; consider providing a meaningful name. Meter will function normally and the provided name will be used as-is.");
186        };
187
188        if let Ok(mut meters) = self.inner.meters.lock() {
189            if let Some(existing_meter) = meters.get(&scope) {
190                otel_debug!(
191                    name: "MeterProvider.ExistingMeterReturned",
192                    meter_name = scope.name(),
193                );
194                Meter::new(existing_meter.clone())
195            } else {
196                let new_meter = Arc::new(SdkMeter::new(scope.clone(), self.inner.pipes.clone()));
197                meters.insert(scope.clone(), new_meter.clone());
198                otel_debug!(
199                    name: "MeterProvider.NewMeterCreated",
200                    meter_name = scope.name(),
201                );
202                Meter::new(new_meter)
203            }
204        } else {
205            otel_debug!(
206                name: "MeterProvider.NoOpMeterReturned",
207                meter_name = scope.name(),
208            );
209            Meter::new(Arc::new(NoopMeter::new()))
210        }
211    }
212}

214/// Configuration options for a [MeterProvider].
215#[derive(Default)]
216pub struct MeterProviderBuilder {
217    resource: Option<Resource>,
218    readers: Vec<Box<dyn MetricReader>>,
219    views: Vec<Arc<dyn View>>,
220}
221
222impl MeterProviderBuilder {
223    /// Associates a [Resource] with a [MeterProvider].
224    ///
225    /// This [Resource] represents the entity producing telemetry and is associated
226    /// with all [Meter]s the [MeterProvider] will create.
227    ///
228    /// By default, if this option is not used, the default [Resource] will be used.
229    ///
230    /// [Meter]: opentelemetry::metrics::Meter
231    pub fn with_resource(mut self, resource: Resource) -> Self {
232        self.resource = Some(resource);
233        self
234    }
235
236    /// Associates a [MetricReader] with a [MeterProvider].
237    ///
238    /// By default, if this option is not used, the [MeterProvider] will perform no
239    /// operations; no data will be exported without a [MetricReader].
240    pub fn with_reader<T: MetricReader>(mut self, reader: T) -> Self {
241        self.readers.push(Box::new(reader));
242        self
243    }
244
245    #[cfg(feature = "spec_unstable_metrics_views")]
246    /// Associates a [View] with a [MeterProvider].
247    ///
248    /// [View]s are appended to existing ones in a [MeterProvider] if this option is
249    /// used multiple times.
250    ///
251    /// By default, if this option is not used, the [MeterProvider] will use the
252    /// default view.
253    pub fn with_view<T: View>(mut self, view: T) -> Self {
254        self.views.push(Arc::new(view));
255        self
256    }
257
258    /// Construct a new [MeterProvider] with this configuration.
259
260    pub fn build(self) -> SdkMeterProvider {
261        otel_debug!(
262            name: "MeterProvider.Building",
263            builder = format!("{:?}", &self),
264        );
265
266        let meter_provider = SdkMeterProvider {
267            inner: Arc::new(SdkMeterProviderInner {
268                pipes: Arc::new(Pipelines::new(
269                    self.resource.unwrap_or_default(),
270                    self.readers,
271                    self.views,
272                )),
273                meters: Default::default(),
274                is_shutdown: AtomicBool::new(false),
275            }),
276        };
277
278        otel_info!(
279            name: "MeterProvider.Built",
280        );
281        meter_provider
282    }
283}

285impl fmt::Debug for MeterProviderBuilder {
286    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
287        f.debug_struct("MeterProviderBuilder")
288            .field("resource", &self.resource)
289            .field("readers", &self.readers)
290            .field("views", &self.views.len())
291            .finish()
292    }

463    #[test]
464    fn test_shutdown_invoked_on_last_drop() {
465        let reader = TestMetricReader::new();
466        let provider = super::SdkMeterProvider::builder()
467            .with_reader(reader.clone())
468            .build();
469        let clone1 = provider.clone();
470        let clone2 = provider.clone();
471
472        // Initially, shutdown should not be called
473        assert!(!reader.is_shutdown());
474
475        // Drop the first clone
476        drop(clone1);
477        assert!(!reader.is_shutdown());
478
479        // Drop the second clone
480        drop(clone2);
481        assert!(!reader.is_shutdown());
482
483        // Drop the last original provider
484        drop(provider);
485        // Now the shutdown should be invoked
486        assert!(reader.is_shutdown());
487    }

489    #[test]
490    fn same_meter_reused_same_scope() {
491        let provider = super::SdkMeterProvider::builder().build();
492        let _meter1 = provider.meter("test");
493        let _meter2 = provider.meter("test");
494        assert_eq!(provider.inner.meters.lock().unwrap().len(), 1);
495
496        let scope = InstrumentationScope::builder("test")
497            .with_version("1.0.0")
498            .with_schema_url("http://example.com")
499            .build();
500
501        let _meter3 = provider.meter_with_scope(scope.clone());
502        let _meter4 = provider.meter_with_scope(scope.clone());
503        let _meter5 = provider.meter_with_scope(scope);
504        assert_eq!(provider.inner.meters.lock().unwrap().len(), 2);
505
506        // these are different meters because meter names are case sensitive
507        let make_scope = |name| {
508            InstrumentationScope::builder(name)
509                .with_version("1.0.0")
510                .with_schema_url("http://example.com")
511                .build()
512        };
513
514        let _meter6 = provider.meter_with_scope(make_scope("ABC"));
515        let _meter7 = provider.meter_with_scope(make_scope("Abc"));
516        let _meter8 = provider.meter_with_scope(make_scope("abc"));
517
518        assert_eq!(provider.inner.meters.lock().unwrap().len(), 5);
519    }
