29/// A collection of one or more aggregated time series from an [Instrument].
30///
31/// [Instrument]: crate::metrics::Instrument
32#[derive(Debug)]
33pub struct Metric {
34    /// The name of the instrument that created this data.
35    pub name: Cow<'static, str>,
36    /// The description of the instrument, which can be used in documentation.
37    pub description: Cow<'static, str>,
38    /// The unit in which the instrument reports.
39    pub unit: Cow<'static, str>,
40    /// The aggregated data from an instrument.
41    pub data: Box<dyn Aggregation>,
42}

44/// The store of data reported by an [Instrument].
45///
46/// It will be one of: [Gauge], [Sum], or [Histogram].
47///
48/// [Instrument]: crate::metrics::Instrument
49pub trait Aggregation: fmt::Debug + any::Any + Send + Sync {
50    /// Support downcasting
51    fn as_any(&self) -> &dyn any::Any;
52    /// Support downcasting during aggregation
53    fn as_mut(&mut self) -> &mut dyn any::Any;
54}

93/// DataPoint is a single data point in a time series.
94#[derive(Debug, PartialEq)]
95pub struct DataPoint<T> {
96    /// Attributes is the set of key value pairs that uniquely identify the
97    /// time series.
98    pub attributes: Vec<KeyValue>,
99    /// The time when the time series was started.
100    pub start_time: Option<SystemTime>,
101    /// The time when the time series was recorded.
102    pub time: Option<SystemTime>,
103    /// The value of this data point.
104    pub value: T,
105    /// The sampled [Exemplar]s collected during the time series.
106    pub exemplars: Vec<Exemplar<T>>,
107}

207/// A single exponential histogram data point in a time series.
208#[derive(Debug, PartialEq)]
209pub struct ExponentialHistogramDataPoint<T> {
210    /// The set of key value pairs that uniquely identify the time series.
211    pub attributes: Vec<KeyValue>,
212    /// When the time series was started.
213    pub start_time: SystemTime,
214    /// The time when the time series was recorded.
215    pub time: SystemTime,
216
217    /// The number of updates this histogram has been calculated with.
218    pub count: usize,
219    /// The minimum value recorded.
220    pub min: Option<T>,
221    /// The maximum value recorded.
222    pub max: Option<T>,
223    /// The sum of the values recorded.
224    pub sum: T,
225
226    /// Describes the resolution of the histogram.
227    ///
228    /// Boundaries are located at powers of the base, where:
229    ///
230    ///   base = 2 ^ (2 ^ -scale)
231    pub scale: i8,
232
233    /// The number of values whose absolute value is less than or equal to
234    /// `zero_threshold`.
235    ///
236    /// When `zero_threshold` is `0`, this is the number of values that cannot be
237    /// expressed using the standard exponential formula as well as values that have
238    /// been rounded to zero.
239    pub zero_count: u64,
240
241    /// The range of positive value bucket counts.
242    pub positive_bucket: ExponentialBucket,
243    /// The range of negative value bucket counts.
244    pub negative_bucket: ExponentialBucket,
245
246    /// The width of the zero region.
247    ///
248    /// Where the zero region is defined as the closed interval
249    /// [-zero_threshold, zero_threshold].
250    pub zero_threshold: f64,
251
252    /// The sampled exemplars collected during the time series.
253    pub exemplars: Vec<Exemplar<T>>,
254}
255
256impl<T: Copy> Clone for ExponentialHistogramDataPoint<T> {
257    fn clone(&self) -> Self {
258        Self {
259            attributes: self.attributes.clone(),
260            start_time: self.start_time,
261            time: self.time,
262            count: self.count,
263            min: self.min,
264            max: self.max,
265            sum: self.sum,
266            scale: self.scale,
267            zero_count: self.zero_count,
268            positive_bucket: self.positive_bucket.clone(),
269            negative_bucket: self.negative_bucket.clone(),
270            zero_threshold: self.zero_threshold,
271            exemplars: self.exemplars.clone(),
272        }
273    }
274}

276/// A set of bucket counts, encoded in a contiguous array of counts.
277#[derive(Debug, PartialEq)]
278pub struct ExponentialBucket {
279    /// The bucket index of the first entry in the `counts` vec.
280    pub offset: i32,
281
282    /// A vec where `counts[i]` carries the count of the bucket at index `offset + i`.
283    ///
284    /// `counts[i]` is the count of values greater than base^(offset+i) and less than
285    /// or equal to base^(offset+i+1).
286    pub counts: Vec<u64>,
287}
288
289impl Clone for ExponentialBucket {
290    fn clone(&self) -> Self {
291        Self {
292            offset: self.offset,
293            counts: self.counts.clone(),
294        }
295    }
296}

298/// A measurement sampled from a time series providing a typical example.
299#[derive(Debug, PartialEq)]
300pub struct Exemplar<T> {
301    /// The attributes recorded with the measurement but filtered out of the
302    /// time series' aggregated data.
303    pub filtered_attributes: Vec<KeyValue>,
304    /// The time when the measurement was recorded.
305    pub time: SystemTime,
306    /// The measured value.
307    pub value: T,
308    /// The ID of the span that was active during the measurement.
309    ///
310    /// If no span was active or the span was not sampled this will be empty.
311    pub span_id: [u8; 8],
312    /// The ID of the trace the active span belonged to during the measurement.
313    ///
314    /// If no span was active or the span was not sampled this will be empty.
315    pub trace_id: [u8; 16],
316}
317
318impl<T: Copy> Clone for Exemplar<T> {
319    fn clone(&self) -> Self {
320        Self {
321            filtered_attributes: self.filtered_attributes.clone(),
322            time: self.time,
323            value: self.value,
324            span_id: self.span_id,
325            trace_id: self.trace_id,
326        }
327    }
328}
