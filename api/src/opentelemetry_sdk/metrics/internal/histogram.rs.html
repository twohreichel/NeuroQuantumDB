13impl<T> Aggregator for Mutex<Buckets<T>>
14where
15    T: Number,
16{
17    type InitConfig = usize;
18    /// Value and bucket index
19    type PreComputedValue = (T, usize);
20
21    fn update(&self, (value, index): (T, usize)) {
22        let mut buckets = self.lock().unwrap_or_else(|err| err.into_inner());
23
24        buckets.total += value;
25        buckets.count += 1;
26        buckets.counts[index] += 1;
27        if value < buckets.min {
28            buckets.min = value;
29        }
30        if value > buckets.max {
31            buckets.max = value
32        }
33    }
34
35    fn create(count: &usize) -> Self {
36        Mutex::new(Buckets::<T>::new(*count))
37    }
38
39    fn clone_and_reset(&self, count: &usize) -> Self {
40        let mut current = self.lock().unwrap_or_else(|err| err.into_inner());
41        Mutex::new(replace(current.deref_mut(), Buckets::new(*count)))
42    }
43}

45#[derive(Default)]
46struct Buckets<T> {
47    counts: Vec<u64>,
48    count: u64,
49    total: T,
50    min: T,
51    max: T,
52}

54impl<T: Number> Buckets<T> {
55    /// returns buckets with `n` bins.
56    fn new(n: usize) -> Buckets<T> {
57        Buckets {
58            counts: vec![0; n],
59            min: T::max(),
60            max: T::min(),

62        }

66/// Summarizes a set of measurements as a histogram with explicitly defined
67/// buckets.
68pub(crate) struct Histogram<T: Number> {
69    value_map: ValueMap<Mutex<Buckets<T>>>,
70    bounds: Vec<f64>,
71    record_min_max: bool,
72    record_sum: bool,
73    start: Mutex<SystemTime>,
74}
75
76impl<T: Number> Histogram<T> {
77    #[allow(unused_mut)]
78    pub(crate) fn new(mut bounds: Vec<f64>, record_min_max: bool, record_sum: bool) -> Self {
79        #[cfg(feature = "spec_unstable_metrics_views")]
80        {
81            // TODO: When views are used, validate this upfront
82            bounds.retain(|v| !v.is_nan());
83            bounds.sort_by(|a, b| a.partial_cmp(b).expect("NaNs filtered out"));
84        }
85
86        let buckets_count = bounds.len() + 1;
87        Histogram {
88            value_map: ValueMap::new(buckets_count),
89            bounds,
90            record_min_max,
91            record_sum,
92            start: Mutex::new(SystemTime::now()),
93        }
94    }
95
96    pub(crate) fn measure(&self, measurement: T, attrs: &[KeyValue]) {
97        let f = measurement.into_float();
98        // This search will return an index in the range `[0, bounds.len()]`, where
99        // it will return `bounds.len()` if value is greater than the last element
100        // of `bounds`. This aligns with the buckets in that the length of buckets
101        // is `bounds.len()+1`, with the last bucket representing:
102        // `(bounds[bounds.len()-1], +âˆž)`.
103        let index = self.bounds.partition_point(|&x| x < f);
104
105        self.value_map.measure((measurement, index), attrs);
106    }
107
108    pub(crate) fn delta(
109        &self,
110        dest: Option<&mut dyn Aggregation>,
111    ) -> (usize, Option<Box<dyn Aggregation>>) {
112        let t = SystemTime::now();
113        let h = dest.and_then(|d| d.as_mut().downcast_mut::<data::Histogram<T>>());
114        let mut new_agg = if h.is_none() {
115            Some(data::Histogram {
116                data_points: vec![],
117                temporality: Temporality::Delta,
118            })
119        } else {
120            None
121        };
122        let h = h.unwrap_or_else(|| new_agg.as_mut().expect("present if h is none"));
123        h.temporality = Temporality::Delta;
124
125        let prev_start = self
126            .start
127            .lock()
128            .map(|mut start| replace(start.deref_mut(), t))
129            .unwrap_or(t);
130
131        self.value_map
132            .collect_and_reset(&mut h.data_points, |attributes, aggr| {
133                let b = aggr.into_inner().unwrap_or_else(|err| err.into_inner());
134                HistogramDataPoint {
135                    attributes,
136                    start_time: prev_start,
137                    time: t,
138                    count: b.count,
139                    bounds: self.bounds.clone(),
140                    bucket_counts: b.counts,
141                    sum: if self.record_sum {
142                        b.total
143                    } else {
144                        T::default()
145                    },
146                    min: if self.record_min_max {
147                        Some(b.min)
148                    } else {
149                        None
150                    },
151                    max: if self.record_min_max {
152                        Some(b.max)
153                    } else {
154                        None
155                    },
156                    exemplars: vec![],
157                }
158            });
159
160        (h.data_points.len(), new_agg.map(|a| Box::new(a) as Box<_>))
161    }
162
163    pub(crate) fn cumulative(
164        &self,
165        dest: Option<&mut dyn Aggregation>,
166    ) -> (usize, Option<Box<dyn Aggregation>>) {
167        let t = SystemTime::now();
168        let h = dest.and_then(|d| d.as_mut().downcast_mut::<data::Histogram<T>>());
169        let mut new_agg = if h.is_none() {
170            Some(data::Histogram {
171                data_points: vec![],
172                temporality: Temporality::Cumulative,
173            })
174        } else {
175            None
176        };
177        let h = h.unwrap_or_else(|| new_agg.as_mut().expect("present if h is none"));
178        h.temporality = Temporality::Cumulative;
179
180        let prev_start = self
181            .start
182            .lock()
183            .map(|s| *s)
184            .unwrap_or_else(|_| SystemTime::now());
185
186        self.value_map
187            .collect_readonly(&mut h.data_points, |attributes, aggr| {
188                let b = aggr.lock().unwrap_or_else(|err| err.into_inner());
189                HistogramDataPoint {
190                    attributes,
191                    start_time: prev_start,
192                    time: t,
193                    count: b.count,
194                    bounds: self.bounds.clone(),
195                    bucket_counts: b.counts.clone(),
196                    sum: if self.record_sum {
197                        b.total
198                    } else {
199                        T::default()
200                    },
201                    min: if self.record_min_max {
202                        Some(b.min)
203                    } else {
204                        None
205                    },
206                    max: if self.record_min_max {
207                        Some(b.max)
208                    } else {
209                        None
210                    },
211                    exemplars: vec![],
212                }
213            });
214
215        (h.data_points.len(), new_agg.map(|a| Box::new(a) as Box<_>))
216    }
217}

219#[cfg(test)]
220mod tests {
221    use super::*;
222
223    #[test]
224    fn check_buckets_are_selected_correctly() {
225        let hist = Histogram::<i64>::new(vec![1.0, 3.0, 6.0], false, false);
226        for v in 1..11 {
227            hist.measure(v, &[]);
228        }
229        let (count, dp) = hist.cumulative(None);
230        let dp = dp.unwrap();
231        let dp = dp.as_any().downcast_ref::<data::Histogram<i64>>().unwrap();
232        assert_eq!(count, 1);
233        assert_eq!(dp.data_points[0].count, 10);
234        assert_eq!(dp.data_points[0].bucket_counts.len(), 4);
235        assert_eq!(dp.data_points[0].bucket_counts[0], 1); // 1
236        assert_eq!(dp.data_points[0].bucket_counts[1], 2); // 2, 3
237        assert_eq!(dp.data_points[0].bucket_counts[2], 3); // 4, 5, 6
238        assert_eq!(dp.data_points[0].bucket_counts[3], 4); // 7, 8, 9, 10
239    }
