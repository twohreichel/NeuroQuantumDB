9/// Summarizes a set of pre-computed sums as their arithmetic sum.
10pub(crate) struct PrecomputedSum<T: Number> {
11    value_map: ValueMap<Assign<T>>,
12    monotonic: bool,
13    start: Mutex<SystemTime>,
14    reported: Mutex<HashMap<Vec<KeyValue>, T>>,
15}
16
17impl<T: Number> PrecomputedSum<T> {
18    pub(crate) fn new(monotonic: bool) -> Self {
19        PrecomputedSum {
20            value_map: ValueMap::new(()),
21            monotonic,
22            start: Mutex::new(SystemTime::now()),
23            reported: Mutex::new(Default::default()),
24        }
25    }
26
27    pub(crate) fn measure(&self, measurement: T, attrs: &[KeyValue]) {
28        // The argument index is not applicable to PrecomputedSum.
29        self.value_map.measure(measurement, attrs);
30    }
31
32    pub(crate) fn delta(
33        &self,
34        dest: Option<&mut dyn Aggregation>,
35    ) -> (usize, Option<Box<dyn Aggregation>>) {
36        let t = SystemTime::now();
37
38        let s_data = dest.and_then(|d| d.as_mut().downcast_mut::<data::Sum<T>>());
39        let mut new_agg = if s_data.is_none() {
40            Some(data::Sum {
41                data_points: vec![],
42                temporality: Temporality::Delta,
43                is_monotonic: self.monotonic,
44            })
45        } else {
46            None
47        };
48        let s_data = s_data.unwrap_or_else(|| new_agg.as_mut().expect("present if s_data is none"));
49        s_data.temporality = Temporality::Delta;
50        s_data.is_monotonic = self.monotonic;
51
52        let prev_start = self
53            .start
54            .lock()
55            .map(|mut start| replace(start.deref_mut(), t))
56            .unwrap_or(t);
57
58        let mut reported = match self.reported.lock() {
59            Ok(r) => r,
60            Err(_) => return (0, None),
61        };
62        let mut new_reported = HashMap::with_capacity(reported.len());
63
64        self.value_map
65            .collect_and_reset(&mut s_data.data_points, |attributes, aggr| {
66                let value = aggr.value.get_value();
67                new_reported.insert(attributes.clone(), value);
68                let delta = value - *reported.get(&attributes).unwrap_or(&T::default());
69                DataPoint {
70                    attributes,
71                    start_time: Some(prev_start),
72                    time: Some(t),
73                    value: delta,
74                    exemplars: vec![],
75                }
76            });
77
78        *reported = new_reported;
79        drop(reported); // drop before values guard is dropped
80
81        (
82            s_data.data_points.len(),
83            new_agg.map(|a| Box::new(a) as Box<_>),
84        )
85    }
86
87    pub(crate) fn cumulative(
88        &self,
89        dest: Option<&mut dyn Aggregation>,
90    ) -> (usize, Option<Box<dyn Aggregation>>) {
91        let t = SystemTime::now();
92
93        let s_data = dest.and_then(|d| d.as_mut().downcast_mut::<data::Sum<T>>());
94        let mut new_agg = if s_data.is_none() {
95            Some(data::Sum {
96                data_points: vec![],
97                temporality: Temporality::Cumulative,
98                is_monotonic: self.monotonic,
99            })
100        } else {
101            None
102        };
103        let s_data = s_data.unwrap_or_else(|| new_agg.as_mut().expect("present if s_data is none"));
104        s_data.temporality = Temporality::Cumulative;
105        s_data.is_monotonic = self.monotonic;
106
107        let prev_start = self.start.lock().map(|start| *start).unwrap_or(t);
108
109        self.value_map
110            .collect_readonly(&mut s_data.data_points, |attributes, aggr| DataPoint {
111                attributes,
112                start_time: Some(prev_start),
113                time: Some(t),
114                value: aggr.value.get_value(),
115                exemplars: vec![],
116            });
117
118        (
119            s_data.data_points.len(),
120            new_agg.map(|a| Box::new(a) as Box<_>),
121        )
122    }
123}
