1//! Representations of entities producing telemetry.
2//!
3//! A [Resource] is an immutable representation of the entity producing
4//! telemetry as attributes. For example, a process producing telemetry that is
5//! running in a container on Kubernetes has a Pod name, it is in a namespace
6//! and possibly is part of a Deployment which also has a name. All three of
7//! these attributes can be included in the `Resource`. Note that there are
8//! certain ["standard attributes"] that have prescribed meanings.
9//!
10//! ["standard attributes"]: https://github.com/open-telemetry/opentelemetry-specification/blob/v1.9.0/specification/resource/semantic_conventions/README.md
11//!
12//! # Resource detectors
13//!
14//! [`ResourceDetector`]s are used to detect resource from runtime or
15//! environmental variables. The following are provided by default with this
16//! SDK.
17//!
18//! - [`EnvResourceDetector`] - detect resource from environmental variables.
19//! - [`TelemetryResourceDetector`] - detect telemetry SDK's information.
20//!
21//! The OS and Process resource detectors are packaged separately in the
22//! [`opentelemetry-resource-detector` crate](https://github.com/open-telemetry/opentelemetry-rust-contrib/tree/main/opentelemetry-resource-detectors).

40/// Inner structure of `Resource` holding the actual data.
41/// This structure is designed to be shared among `Resource` instances via `Arc`.
42#[derive(Debug, Clone, PartialEq)]
43struct ResourceInner {
44    attrs: HashMap<Key, Value>,
45    schema_url: Option<Cow<'static, str>>,
46}
47
48/// An immutable representation of the entity producing telemetry as attributes.
49/// Utilizes `Arc` for efficient sharing and cloning.
50#[derive(Clone, Debug, PartialEq)]
51pub struct Resource {
52    inner: Arc<ResourceInner>,
53}
54
55impl Default for Resource {
56    fn default() -> Self {
57        Self::from_detectors(
58            Duration::from_secs(0),
59            vec![
60                Box::new(SdkProvidedResourceDetector),
61                Box::new(TelemetryResourceDetector),
62                Box::new(EnvResourceDetector::new()),
63            ],
64        )
65    }
66}

68impl Resource {
69    /// Creates an empty resource.
70    /// This is the basic constructor that initializes a resource with no attributes and no schema URL.
71    pub fn empty() -> Self {
72        Resource {
73            inner: Arc::new(ResourceInner {
74                attrs: HashMap::new(),
75                schema_url: None,
76            }),
77        }
78    }

80    /// Create a new `Resource` from key value pairs.
81    ///
82    /// Values are de-duplicated by key, and the first key-value pair with a non-empty string value
83    /// will be retained
84    pub fn new<T: IntoIterator<Item = KeyValue>>(kvs: T) -> Self {
85        let mut attrs = HashMap::new();
86        for kv in kvs {
87            attrs.insert(kv.key, kv.value);
88        }
89
90        Resource {
91            inner: Arc::new(ResourceInner {
92                attrs,
93                schema_url: None,
94            }),
95        }
96    }
97
98    /// Create a new `Resource::default()` and merge with provided key value pairs.
99    ///
100    /// Values are de-duplicated by key, and the first key-value pair with a non-empty string value
101    /// will be retained
102    pub fn new_with_defaults<T: IntoIterator<Item = KeyValue>>(keys: T) -> Self {
103        Resource::default().merge(&mut Resource::new(keys))
104    }
105
106    /// Create a new `Resource` from a key value pairs and [schema url].
107    ///
108    /// Values are de-duplicated by key, and the first key-value pair with a non-empty string value
109    /// will be retained.
110    ///
111    /// schema_url must be a valid URL using HTTP or HTTPS protocol.
112    ///
113    /// [schema url]: https://github.com/open-telemetry/opentelemetry-specification/blob/v1.9.0/specification/schemas/overview.md#schema-url
114    pub fn from_schema_url<KV, S>(kvs: KV, schema_url: S) -> Self
115    where
116        KV: IntoIterator<Item = KeyValue>,
117        S: Into<Cow<'static, str>>,
118    {
119        let schema_url_str = schema_url.into();
120        let normalized_schema_url = if schema_url_str.is_empty() {
121            None
122        } else {
123            Some(schema_url_str)
124        };
125        let mut attrs = HashMap::new();
126        for kv in kvs {
127            attrs.insert(kv.key, kv.value);
128        }
129        Resource {
130            inner: Arc::new(ResourceInner {
131                attrs,
132                schema_url: normalized_schema_url,
133            }),
134        }
135    }
136
137    /// Create a new `Resource` from resource detectors.
138    ///
139    /// timeout will be applied to each detector.
140    pub fn from_detectors(timeout: Duration, detectors: Vec<Box<dyn ResourceDetector>>) -> Self {
141        let mut resource = Resource::empty();
142        for detector in detectors {
143            let detected_res = detector.detect(timeout);
144            // This call ensures that if the Arc is not uniquely owned,
145            // the data is cloned before modification, preserving safety.
146            // If the Arc is uniquely owned, it simply returns a mutable reference to the data.
147            let inner = Arc::make_mut(&mut resource.inner);
148            for (key, value) in detected_res.into_iter() {
149                inner.attrs.insert(Key::new(key.clone()), value.clone());
150            }
151        }
152
153        resource
154    }
155
156    /// Create a new `Resource` by combining two resources.
157    ///
158    /// ### Key value pairs
159    /// Keys from the `other` resource have priority over keys from this resource, even if the
160    /// updated value is empty.
161    ///
162    /// ### [Schema url]
163    /// If both of the resource are not empty. Schema url is determined by the following rules, in order:
164    /// 1. If this resource has a schema url, it will be used.
165    /// 2. If this resource does not have a schema url, and the other resource has a schema url, it will be used.
166    /// 3. If both resources have a schema url and it's the same, it will be used.
167    /// 4. If both resources have a schema url and it's different, the schema url will be empty.
168    /// 5. If both resources do not have a schema url, the schema url will be empty.
169    ///
170    /// [Schema url]: https://github.com/open-telemetry/opentelemetry-specification/blob/v1.9.0/specification/schemas/overview.md#schema-url
171    pub fn merge<T: Deref<Target = Self>>(&self, other: T) -> Self {
172        if self.is_empty() {
173            return other.clone();
174        }
175        if other.is_empty() {
176            return self.clone();
177        }
178        let mut combined_attrs = self.inner.attrs.clone();
179        for (k, v) in other.inner.attrs.iter() {
180            combined_attrs.insert(k.clone(), v.clone());
181        }
182        // Resolve the schema URL according to the precedence rules
183        let combined_schema_url = match (&self.inner.schema_url, &other.inner.schema_url) {
184            // If both resources have a schema URL and it's the same, use it
185            (Some(url1), Some(url2)) if url1 == url2 => Some(url1.clone()),
186            // If both resources have a schema URL but they are not the same, the schema URL will be empty
187            (Some(_), Some(_)) => None,
188            // If this resource does not have a schema URL, and the other resource has a schema URL, it will be used
189            (None, Some(url)) => Some(url.clone()),
190            // If this resource has a schema URL, it will be used (covers case 1 and any other cases where `self` has a schema URL)
191            (Some(url), _) => Some(url.clone()),
192            // If both resources do not have a schema URL, the schema URL will be empty
193            (None, None) => None,
194        };
195        Resource {
196            inner: Arc::new(ResourceInner {
197                attrs: combined_attrs,
198                schema_url: combined_schema_url,
199            }),
200        }
201    }
202
203    /// Return the [schema url] of the resource. If the resource does not have a schema url, return `None`.
204    ///
205    /// [schema url]: https://github.com/open-telemetry/opentelemetry-specification/blob/v1.9.0/specification/schemas/overview.md#schema-url
206    pub fn schema_url(&self) -> Option<&str> {
207        self.inner.schema_url.as_ref().map(|s| s.as_ref())
208    }
209
210    /// Returns the number of attributes for this resource
211    pub fn len(&self) -> usize {
212        self.inner.attrs.len()
213    }
214
215    /// Returns `true` if the resource contains no attributes.
216    pub fn is_empty(&self) -> bool {
217        self.inner.attrs.is_empty()
218    }
219
220    /// Gets an iterator over the attributes of this resource.
221    pub fn iter(&self) -> Iter<'_> {
222        Iter(self.inner.attrs.iter())
223    }
224
225    /// Retrieve the value from resource associate with given key.
226    pub fn get(&self, key: Key) -> Option<Value> {
227        self.inner.attrs.get(&key).cloned()
228    }
229}

231/// An iterator over the entries of a `Resource`.
232#[derive(Debug)]
233pub struct Iter<'a>(hash_map::Iter<'a, Key, Value>);
234
235impl<'a> Iterator for Iter<'a> {
236    type Item = (&'a Key, &'a Value);
237
238    fn next(&mut self) -> Option<Self::Item> {
239        self.0.next()
240    }
241}
242
243impl<'a> IntoIterator for &'a Resource {
244    type Item = (&'a Key, &'a Value);
245    type IntoIter = Iter<'a>;
246
247    fn into_iter(self) -> Self::IntoIter {
248        Iter(self.inner.attrs.iter())
249    }
250}

252/// ResourceDetector detects OpenTelemetry resource information
253///
254/// Implementations of this trait can be passed to
255/// the [`Resource::from_detectors`] function to generate a Resource from the merged information.
256pub trait ResourceDetector {
257    /// detect returns an initialized Resource based on gathered information.
258    ///
259    /// timeout is used in case the detection operation takes too much time.
260    ///
261    /// If source information to construct a Resource is inaccessible, an empty Resource should be returned
262    ///
263    /// If source information to construct a Resource is invalid, for example,
264    /// missing required values. an empty Resource should be returned.
265    fn detect(&self, timeout: Duration) -> Resource;
266}

316    #[test]
317    fn merge_resource_schema_url() {
318        // if both resources contains key value pairs
319        let test_cases = vec![
320            (Some("http://schema/a"), None, Some("http://schema/a")),
321            (Some("http://schema/a"), Some("http://schema/b"), None),
322            (None, Some("http://schema/b"), Some("http://schema/b")),
323            (
324                Some("http://schema/a"),
325                Some("http://schema/a"),
326                Some("http://schema/a"),
327            ),
328            (None, None, None),
329        ];
330
331        for (schema_url_a, schema_url_b, expected_schema_url) in test_cases.into_iter() {
332            let resource_a = Resource::from_schema_url(
333                vec![KeyValue::new("key", "")],
334                schema_url_a.unwrap_or(""),
335            );
336            let resource_b = Resource::from_schema_url(
337                vec![KeyValue::new("key", "")],
338                schema_url_b.unwrap_or(""),
339            );
340
341            let merged_resource = resource_a.merge(&resource_b);
342            let result_schema_url = merged_resource.schema_url();
343
344            assert_eq!(
345                result_schema_url.map(|s| s as &str),
346                expected_schema_url,
347                "Merging schema_url_a {:?} with schema_url_b {:?} did not yield expected result {:?}",
348                schema_url_a, schema_url_b, expected_schema_url
349            );
350        }
351
352        // if only one resource contains key value pairs
353        let resource = Resource::from_schema_url(vec![], "http://schema/a");
354        let other_resource = Resource::new(vec![KeyValue::new("key", "")]);
355
356        assert_eq!(resource.merge(&other_resource).schema_url(), None);
357    }
