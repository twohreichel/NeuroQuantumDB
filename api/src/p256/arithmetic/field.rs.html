17/// The number of 64-bit limbs used to represent a [`FieldElement`].
18const LIMBS: usize = 4;

20/// Constant representing the modulus
21/// p = 2^{224}(2^{32} − 1) + 2^{192} + 2^{96} − 1
22pub const MODULUS: FieldElement = FieldElement([
23    0xffff_ffff_ffff_ffff,
24    0x0000_0000_ffff_ffff,
25    0x0000_0000_0000_0000,
26    0xffff_ffff_0000_0001,
27]);

29/// R = 2^256 mod p
30const R: FieldElement = FieldElement([
31    0x0000_0000_0000_0001,
32    0xffff_ffff_0000_0000,
33    0xffff_ffff_ffff_ffff,
34    0x0000_0000_ffff_fffe,
35]);

37/// R^2 = 2^512 mod p
38const R2: FieldElement = FieldElement([
39    0x0000_0000_0000_0003,
40    0xffff_fffb_ffff_ffff,
41    0xffff_ffff_ffff_fffe,
42    0x0000_0004_ffff_fffd,
43]);

45/// An element in the finite field modulo p = 2^{224}(2^{32} − 1) + 2^{192} + 2^{96} − 1.
46///
47/// The internal representation is in little-endian order. Elements are always in
48/// Montgomery form; i.e., FieldElement(a) = aR mod p, with R = 2^256.
49#[derive(Clone, Copy, Debug)]
50pub struct FieldElement(pub(crate) [u64; LIMBS]);
51

59    fn from_bytes_wide(bytes: [u8; 64]) -> Self {
60        FieldElement::montgomery_reduce(
61            u64::from_be_bytes(bytes[0..8].try_into().unwrap()),
62            u64::from_be_bytes(bytes[8..16].try_into().unwrap()),
63            u64::from_be_bytes(bytes[16..24].try_into().unwrap()),
64            u64::from_be_bytes(bytes[24..32].try_into().unwrap()),
65            u64::from_be_bytes(bytes[32..40].try_into().unwrap()),
66            u64::from_be_bytes(bytes[40..48].try_into().unwrap()),
67            u64::from_be_bytes(bytes[48..56].try_into().unwrap()),
68            u64::from_be_bytes(bytes[56..64].try_into().unwrap()),
69        )
70    }

72    /// Attempts to parse the given byte array as an SEC1-encoded field element.
73    ///
74    /// Returns None if the byte array does not contain a big-endian integer in the range
75    /// [0, p).
76    pub fn from_bytes(bytes: &FieldBytes) -> CtOption<Self> {
77        let mut w = [0u64; LIMBS];
78
79        // Interpret the bytes as a big-endian integer w.
80        w[3] = u64::from_be_bytes(bytes[0..8].try_into().unwrap());
81        w[2] = u64::from_be_bytes(bytes[8..16].try_into().unwrap());
82        w[1] = u64::from_be_bytes(bytes[16..24].try_into().unwrap());
83        w[0] = u64::from_be_bytes(bytes[24..32].try_into().unwrap());
84
85        // If w is in the range [0, p) then w - p will overflow, resulting in a borrow
86        // value of 2^64 - 1.
87        let (_, borrow) = sbb(w[0], MODULUS.0[0], 0);
88        let (_, borrow) = sbb(w[1], MODULUS.0[1], borrow);
89        let (_, borrow) = sbb(w[2], MODULUS.0[2], borrow);
90        let (_, borrow) = sbb(w[3], MODULUS.0[3], borrow);
91        let is_some = (borrow as u8) & 1;
92
93        // Convert w to Montgomery form: w * R^2 * R^-1 mod p = wR mod p
94        CtOption::new(FieldElement(w).to_montgomery(), Choice::from(is_some))
95    }

97    /// Returns the SEC1 encoding of this field element.
98    pub fn to_bytes(self) -> FieldBytes {
99        // Convert from Montgomery form to canonical form
100        let tmp = self.to_canonical();
101
102        let mut ret = FieldBytes::default();
103        ret[0..8].copy_from_slice(&tmp.0[3].to_be_bytes());
104        ret[8..16].copy_from_slice(&tmp.0[2].to_be_bytes());
105        ret[16..24].copy_from_slice(&tmp.0[1].to_be_bytes());
106        ret[24..32].copy_from_slice(&tmp.0[0].to_be_bytes());
107        ret
108    }

129    /// Returns self + rhs mod p
130    pub const fn add(&self, rhs: &Self) -> Self {
131        // Bit 256 of p is set, so addition can result in five words.
132        let (w0, carry) = adc(self.0[0], rhs.0[0], 0);
133        let (w1, carry) = adc(self.0[1], rhs.0[1], carry);
134        let (w2, carry) = adc(self.0[2], rhs.0[2], carry);
135        let (w3, w4) = adc(self.0[3], rhs.0[3], carry);
136
137        // Attempt to subtract the modulus, to ensure the result is in the field.
138        let (result, _) = Self::sub_inner(
139            w0,
140            w1,
141            w2,
142            w3,
143            w4,
144            MODULUS.0[0],
145            MODULUS.0[1],
146            MODULUS.0[2],
147            MODULUS.0[3],
148            0,
149        );
150        result
151    }

167    /// Returns self - rhs mod p
168    pub(crate) const fn informed_subtract(&self, rhs: &Self) -> (Self, u64) {
169        Self::sub_inner(
170            self.0[0], self.0[1], self.0[2], self.0[3], 0, rhs.0[0], rhs.0[1], rhs.0[2], rhs.0[3],
171            0,
172        )

174    #[inline]
175    #[allow(clippy::too_many_arguments)]
176    const fn sub_inner(
177        l0: u64,
178        l1: u64,
179        l2: u64,
180        l3: u64,
181        l4: u64,
182        r0: u64,
183        r1: u64,
184        r2: u64,
185        r3: u64,
186        r4: u64,
187    ) -> (Self, u64) {
188        let (w0, borrow) = sbb(l0, r0, 0);
189        let (w1, borrow) = sbb(l1, r1, borrow);
190        let (w2, borrow) = sbb(l2, r2, borrow);
191        let (w3, borrow) = sbb(l3, r3, borrow);
192        let (_, borrow) = sbb(l4, r4, borrow);
193
194        // If underflow occurred on the final limb, borrow = 0xfff...fff, otherwise
195        // borrow = 0x000...000. Thus, we use it as a mask to conditionally add the
196        // modulus.
197        let (w0, carry) = adc(w0, MODULUS.0[0] & borrow, 0);
198        let (w1, carry) = adc(w1, MODULUS.0[1] & borrow, carry);
199        let (w2, carry) = adc(w2, MODULUS.0[2] & borrow, carry);
200        let (w3, _) = adc(w3, MODULUS.0[3] & borrow, carry);
201
202        (FieldElement([w0, w1, w2, w3]), borrow)
203    }

205    /// Montgomery Reduction
206    ///
207    /// The general algorithm is:
208    /// ```text
209    /// A <- input (2n b-limbs)
210    /// for i in 0..n {
211    ///     k <- A[i] p' mod b
212    ///     A <- A + k p b^i
213    /// }
214    /// A <- A / b^n
215    /// if A >= p {
216    ///     A <- A - p
217    /// }
218    /// ```
219    ///
220    /// For secp256r1, we have the following simplifications:
221    ///
222    /// - `p'` is 1, so our multiplicand is simply the first limb of the intermediate A.
223    ///
224    /// - The first limb of p is 2^64 - 1; multiplications by this limb can be simplified
225    ///   to a shift and subtraction:
226    ///   ```text
227    ///       a_i * (2^64 - 1) = a_i * 2^64 - a_i = (a_i << 64) - a_i
228    ///   ```
229    ///   However, because `p' = 1`, the first limb of p is multiplied by limb i of the
230    ///   intermediate A and then immediately added to that same limb, so we simply
231    ///   initialize the carry to limb i of the intermediate.
232    ///
233    /// - The third limb of p is zero, so we can ignore any multiplications by it and just
234    ///   add the carry.
235    ///
236    /// References:
237    /// - Handbook of Applied Cryptography, Chapter 14
238    ///   Algorithm 14.32
239    ///   http://cacr.uwaterloo.ca/hac/about/chap14.pdf
240    ///
241    /// - Efficient and Secure Elliptic Curve Cryptography Implementation of Curve P-256
242    ///   Algorithm 7) Montgomery Word-by-Word Reduction
243    ///   https://csrc.nist.gov/csrc/media/events/workshop-on-elliptic-curve-cryptography-standards/documents/papers/session6-adalier-mehmet.pdf
244    #[inline]
245    #[allow(clippy::too_many_arguments)]
246    const fn montgomery_reduce(
247        r0: u64,
248        r1: u64,
249        r2: u64,
250        r3: u64,
251        r4: u64,
252        r5: u64,
253        r6: u64,
254        r7: u64,
255    ) -> Self {
256        let (r1, carry) = mac(r1, r0, MODULUS.0[1], r0);
257        let (r2, carry) = adc(r2, 0, carry);
258        let (r3, carry) = mac(r3, r0, MODULUS.0[3], carry);
259        let (r4, carry2) = adc(r4, 0, carry);
260
261        let (r2, carry) = mac(r2, r1, MODULUS.0[1], r1);
262        let (r3, carry) = adc(r3, 0, carry);
263        let (r4, carry) = mac(r4, r1, MODULUS.0[3], carry);
264        let (r5, carry2) = adc(r5, carry2, carry);
265
266        let (r3, carry) = mac(r3, r2, MODULUS.0[1], r2);
267        let (r4, carry) = adc(r4, 0, carry);
268        let (r5, carry) = mac(r5, r2, MODULUS.0[3], carry);
269        let (r6, carry2) = adc(r6, carry2, carry);
270
271        let (r4, carry) = mac(r4, r3, MODULUS.0[1], r3);
272        let (r5, carry) = adc(r5, 0, carry);
273        let (r6, carry) = mac(r6, r3, MODULUS.0[3], carry);
274        let (r7, r8) = adc(r7, carry2, carry);
275
276        // Result may be within MODULUS of the correct value
277        let (result, _) = Self::sub_inner(
278            r4,
279            r5,
280            r6,
281            r7,
282            r8,
283            MODULUS.0[0],
284            MODULUS.0[1],
285            MODULUS.0[2],
286            MODULUS.0[3],
287            0,
288        );
289        result
290    }

304    /// Returns self * rhs mod p
305    pub const fn mul(&self, rhs: &Self) -> Self {
306        // Schoolbook multiplication.
307
308        let (w0, carry) = mac(0, self.0[0], rhs.0[0], 0);
309        let (w1, carry) = mac(0, self.0[0], rhs.0[1], carry);
310        let (w2, carry) = mac(0, self.0[0], rhs.0[2], carry);
311        let (w3, w4) = mac(0, self.0[0], rhs.0[3], carry);
312
313        let (w1, carry) = mac(w1, self.0[1], rhs.0[0], 0);
314        let (w2, carry) = mac(w2, self.0[1], rhs.0[1], carry);
315        let (w3, carry) = mac(w3, self.0[1], rhs.0[2], carry);
316        let (w4, w5) = mac(w4, self.0[1], rhs.0[3], carry);
317
318        let (w2, carry) = mac(w2, self.0[2], rhs.0[0], 0);
319        let (w3, carry) = mac(w3, self.0[2], rhs.0[1], carry);
320        let (w4, carry) = mac(w4, self.0[2], rhs.0[2], carry);
321        let (w5, w6) = mac(w5, self.0[2], rhs.0[3], carry);
322
323        let (w3, carry) = mac(w3, self.0[3], rhs.0[0], 0);
324        let (w4, carry) = mac(w4, self.0[3], rhs.0[1], carry);
325        let (w5, carry) = mac(w5, self.0[3], rhs.0[2], carry);
326        let (w6, w7) = mac(w6, self.0[3], rhs.0[3], carry);
327
328        FieldElement::montgomery_reduce(w0, w1, w2, w3, w4, w5, w6, w7)
329    }

346    /// Returns `self^by`, where `by` is a little-endian integer exponent.
347    ///
348    /// **This operation is variable time with respect to the exponent.** If the exponent
349    /// is fixed, this operation is effectively constant time.
350    pub fn pow_vartime(&self, by: &[u64; 4]) -> Self {
351        let mut res = Self::one();
352        for e in by.iter().rev() {
353            for i in (0..64).rev() {
354                res = res.square();
355
356                if ((*e >> i) & 1) == 1 {
357                    res = res * self;
358                }
359            }
360        }
361        res
362    }

364    /// Returns the multiplicative inverse of self, if self is non-zero.
365    pub fn invert(&self) -> CtOption<Self> {
366        // We need to find b such that b * a ≡ 1 mod p. As we are in a prime
367        // field, we can apply Fermat's Little Theorem:
368        //
369        //    a^p         ≡ a mod p
370        //    a^(p-1)     ≡ 1 mod p
371        //    a^(p-2) * a ≡ 1 mod p
372        //
373        // Thus inversion can be implemented with a single exponentiation.
374
375        let t111 = self.mul(&self.mul(&self.square()).square());
376        let t111111 = t111.mul(t111.sqn(3));
377        let x15 = t111111.sqn(6).mul(t111111).sqn(3).mul(t111);
378        let x16 = x15.square().mul(self);
379        let i53 = x16.sqn(16).mul(x16).sqn(15);
380        let x47 = x15.mul(i53);
381        let inverse = x47
382            .mul(i53.sqn(17).mul(self).sqn(143).mul(x47).sqn(47))
383            .sqn(2)
384            .mul(self);
385
386        CtOption::new(inverse, !self.is_zero())
387    }

389    /// Returns the square root of self mod p, or `None` if no square root exists.
390    pub fn sqrt(&self) -> CtOption<Self> {
391        // We need to find alpha such that alpha^2 = beta mod p. For secp256r1,
392        // p ≡ 3 mod 4. By Euler's Criterion, beta^(p-1)/2 ≡ 1 mod p. So:
393        //
394        //     alpha^2 = beta beta^((p - 1) / 2) mod p ≡ beta^((p + 1) / 2) mod p
395        //     alpha = ± beta^((p + 1) / 4) mod p
396        //
397        // Thus sqrt can be implemented with a single exponentiation.
398
399        let t11 = self.mul(&self.square());
400        let t1111 = t11.mul(&t11.sqn(2));
401        let t11111111 = t1111.mul(t1111.sqn(4));
402        let x16 = t11111111.sqn(8).mul(t11111111);
403        let sqrt = x16
404            .sqn(16)
405            .mul(x16)
406            .sqn(32)
407            .mul(self)
408            .sqn(96)
409            .mul(self)
410            .sqn(94);
411
412        CtOption::new(
413            sqrt,
414            (&sqrt * &sqrt).ct_eq(self), // Only return Some if it's the square root.
415        )
416    }
