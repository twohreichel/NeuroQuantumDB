12/// A SIMD vector of complex numbers, stored with the real values and imaginary values interleaved.
13/// Implemented for __m128, __m128d, __m256, __m256d, but these all require the AVX instruction set.
14///
15/// The goal of this trait is to reduce code duplication by letting code be generic over the vector type
16pub trait AvxVector: Copy + Debug + Send + Sync {

21    unsafe fn half_root2() -> Self; // an entire vector filled with 0.5.sqrt()

25    unsafe fn sub(left: Self, right: Self) -> Self;

36    unsafe fn swap_complex_components(self) -> Self;

83    /// Pairwise multiply the complex numbers in `left` with the complex numbers in `right`.
84    #[inline(always)]
85    unsafe fn mul_complex(left: Self, right: Self) -> Self {
86        // Extract the real and imaginary components from left into 2 separate registers
87        let (left_real, left_imag) = Self::duplicate_complex_components(left);
88
89        // create a shuffled version of right where the imaginary values are swapped with the reals
90        let right_shuffled = Self::swap_complex_components(right);
91
92        // multiply our duplicated imaginary left vector by our shuffled right vector. that will give us the right side of the traditional complex multiplication formula
93        let output_right = Self::mul(left_imag, right_shuffled);
94
95        // use a FMA instruction to multiply together left side of the complex multiplication formula, then alternatingly add and subtract the left side from the right
96        Self::fmaddsub(left_real, right, output_right)
97    }

100    unsafe fn rotate90(self, rotation: Rotation90<Self>) -> Self {
101        // Use the pre-computed vector stored in the Rotation90 instance to negate either the reals or imaginaries
102        let negated = Self::xor(self, rotation.0);
103
104        // Our goal is to swap the reals with the imaginaries, then negate either the reals or the imaginaries, based on whether we're an inverse or not
105        Self::swap_complex_components(negated)
106    }

114    unsafe fn column_butterfly3(rows: [Self; 3], twiddles: Self) -> [Self; 3] {
115        // This algorithm is derived directly from the definition of the Dft of size 3
116        // We'd theoretically have to do 4 complex multiplications, but all of the twiddles we'd be multiplying by are conjugates of each other
117        // By doing some algebra to expand the complex multiplications and factor out the multiplications, we get this
118
119        let [mut mid1, mid2] = Self::column_butterfly2([rows[1], rows[2]]);
120        let output0 = Self::add(rows[0], mid1);
121
122        let (twiddle_real, twiddle_imag) = Self::duplicate_complex_components(twiddles);
123
124        mid1 = Self::fmadd(mid1, twiddle_real, rows[0]);
125
126        let rotation = Self::make_rotation90(FftDirection::Inverse);
127        let mid2_rotated = Self::rotate90(mid2, rotation);
128
129        let output1 = Self::fmadd(mid2_rotated, twiddle_imag, mid1);
130        let output2 = Self::fnmadd(mid2_rotated, twiddle_imag, mid1);
131
132        [output0, output1, output2]
133    }

135    #[inline(always)]
136    unsafe fn column_butterfly4(rows: [Self; 4], rotation: Rotation90<Self>) -> [Self; 4] {
137        // Algorithm: 2x2 mixed radix
138
139        // Perform the first set of size-2 FFTs.
140        let [mid0, mid2] = Self::column_butterfly2([rows[0], rows[2]]);
141        let [mid1, mid3] = Self::column_butterfly2([rows[1], rows[3]]);
142
143        // Apply twiddle factors (in this case just a rotation)
144        let mid3_rotated = mid3.rotate90(rotation);
145
146        // Transpose the data and do size-2 FFTs down the columns
147        let [output0, output1] = Self::column_butterfly2([mid0, mid1]);
148        let [output2, output3] = Self::column_butterfly2([mid2, mid3_rotated]);
149
150        // Swap outputs 1 and 2 in the output to do a square transpose
151        [output0, output2, output1, output3]
152    }

155    unsafe fn column_butterfly5(rows: [Self; 5], twiddles: [Self; 2]) -> [Self; 5] {
156        // This algorithm is derived directly from the definition of the Dft of size 5
157        // We'd theoretically have to do 16 complex multiplications for the Dft, but many of the twiddles we'd be multiplying by are conjugates of each other
158        // By doing some algebra to expand the complex multiplications and factor out the real multiplications, we get this faster formula where we only do the equivalent of 4 multiplications
159
160        // do some prep work before we can start applying twiddle factors
161        let [sum1, diff4] = Self::column_butterfly2([rows[1], rows[4]]);
162        let [sum2, diff3] = Self::column_butterfly2([rows[2], rows[3]]);
163
164        let rotation = Self::make_rotation90(FftDirection::Inverse);
165        let rotated4 = Self::rotate90(diff4, rotation);
166        let rotated3 = Self::rotate90(diff3, rotation);
167
168        // to compute the first output, compute the sum of all elements. sum1 and sum2 already have the sum of 1+4 and 2+3 respectively, so if we add them, we'll get the sum of all 4
169        let sum1234 = Self::add(sum1, sum2);
170        let output0 = Self::add(rows[0], sum1234);
171
172        // apply twiddle factors
173        let (twiddles0_re, twiddles0_im) = Self::duplicate_complex_components(twiddles[0]);
174        let (twiddles1_re, twiddles1_im) = Self::duplicate_complex_components(twiddles[1]);
175
176        let twiddled1_mid = Self::fmadd(twiddles0_re, sum1, rows[0]);
177        let twiddled2_mid = Self::fmadd(twiddles1_re, sum1, rows[0]);
178        let twiddled3_mid = Self::mul(twiddles1_im, rotated4);
179        let twiddled4_mid = Self::mul(twiddles0_im, rotated4);
180
181        let twiddled1 = Self::fmadd(twiddles1_re, sum2, twiddled1_mid);
182        let twiddled2 = Self::fmadd(twiddles0_re, sum2, twiddled2_mid);
183        let twiddled3 = Self::fnmadd(twiddles0_im, rotated3, twiddled3_mid); // fnmadd instead of fmadd because we're actually re-using twiddle0 here. remember that this algorithm is all about factoring out conjugated multiplications -- this negation of the twiddle0 imaginaries is a reflection of one of those conugations
184        let twiddled4 = Self::fmadd(twiddles1_im, rotated3, twiddled4_mid);
185
186        // Post-processing to mix the twiddle factors between the rest of the output
187        let [output1, output4] = Self::column_butterfly2([twiddled1, twiddled4]);
188        let [output2, output3] = Self::column_butterfly2([twiddled2, twiddled3]);
189
190        [output0, output1, output2, output3, output4]
191    }

194    unsafe fn column_butterfly7(rows: [Self; 7], twiddles: [Self; 3]) -> [Self; 7] {
195        // This algorithm is derived directly from the definition of the Dft of size 7
196        // We'd theoretically have to do 36 complex multiplications for the Dft, but many of the twiddles we'd be multiplying by are conjugates of each other
197        // By doing some algebra to expand the complex multiplications and factor out the real multiplications, we get this faster formula where we only do the equivalent of 9 multiplications
198
199        // do some prep work before we can start applying twiddle factors
200        let [sum1, diff6] = Self::column_butterfly2([rows[1], rows[6]]);
201        let [sum2, diff5] = Self::column_butterfly2([rows[2], rows[5]]);
202        let [sum3, diff4] = Self::column_butterfly2([rows[3], rows[4]]);
203
204        let rotation = Self::make_rotation90(FftDirection::Inverse);
205        let rotated4 = Self::rotate90(diff4, rotation);
206        let rotated5 = Self::rotate90(diff5, rotation);
207        let rotated6 = Self::rotate90(diff6, rotation);
208
209        // to compute the first output, compute the sum of all elements. sum1, sum2, and sum3 already have the sum of 1+6 and 2+5 and 3+4 respectively, so if we add them, we'll get the sum of all 6
210        let output0_left = Self::add(sum1, sum2);
211        let output0_right = Self::add(sum3, rows[0]);
212        let output0 = Self::add(output0_left, output0_right);
213
214        // apply twiddle factors. This is probably pushing the limit of how much we should do with this technique.
215        // We probably shouldn't do a size-11 FFT with this technique, for example, because this block of multiplies would grow quadratically
216        let (twiddles0_re, twiddles0_im) = Self::duplicate_complex_components(twiddles[0]);
217        let (twiddles1_re, twiddles1_im) = Self::duplicate_complex_components(twiddles[1]);
218        let (twiddles2_re, twiddles2_im) = Self::duplicate_complex_components(twiddles[2]);
219
220        // Let's do a plain 7-point Dft
221        // | X0 |   | W0  W0  W0  W0  W0  W0  W0  |   | x0 |
222        // | X1 |   | W0  W1  W2  W3  W4  W5  W6  |   | x1 |
223        // | X2 |   | W0  W2  W4  W6  W8  W10 W12 |   | x2 |
224        // | X3 |   | W0  W3  W6  W9  W12 W15 W18 |   | x3 |
225        // | X4 |   | W0  W4  W8  W12 W16 W20 W24 |   | x4 |
226        // | X5 |   | W0  W5  W10 W15 W20 W25 W30 |   | x5 |
227        // | X6 |   | W0  W6  W12 W18 W24 W30 W36 |   | x6 |
228        // where Wn = exp(-2*pi*n/7) for a forward transform, and exp(+2*pi*n/7) for an inverse.
229
230        // Next, take advantage of the fact that twiddle factor indexes for a size-7 Dft are cyclical mod 7
231        // | X0 |   | W0  W0  W0  W0  W0  W0  W0  |   | x0 |
232        // | X1 |   | W0  W1  W2  W3  W4  W5  W6  |   | x1 |
233        // | X2 |   | W0  W2  W4  W6  W1  W3  W5  |   | x2 |
234        // | X3 |   | W0  W3  W6  W2  W5  W1  W4  |   | x3 |
235        // | X4 |   | W0  W4  W1  W5  W2  W6  W3  |   | x4 |
236        // | X5 |   | W0  W5  W3  W1  W6  W4  W2  |   | x5 |
237        // | X6 |   | W0  W6  W5  W4  W3  W2  W1  |   | x6 |
238
239        // Finally, take advantage of the fact that for a size-7 Dft,
240        // twiddles 4 through 6 are conjugates of twiddes 3 through 0 (Asterisk marks conjugates)
241        // | X0 |   | W0  W0  W0  W0  W0  W0  W0  |   | x0 |
242        // | X1 |   | W0  W1  W2  W3  W3* W2* W1* |   | x1 |
243        // | X2 |   | W0  W2  W3* W1* W1  W3  W2* |   | x2 |
244        // | X3 |   | W0  W3  W1* W2  W2* W1  W3* |   | x3 |
245        // | X4 |   | W0  W3* W1  W2* W2  W1* W3  |   | x4 |
246        // | X5 |   | W0  W2* W3  W1  W1* W3* W2  |   | x5 |
247        // | X6 |   | W0  W1* W2* W3* W3  W2  W1  |   | x6 |
248
249        let twiddled1_mid = Self::fmadd(twiddles0_re, sum1, rows[0]);
250        let twiddled2_mid = Self::fmadd(twiddles1_re, sum1, rows[0]);
251        let twiddled3_mid = Self::fmadd(twiddles2_re, sum1, rows[0]);
252        let twiddled4_mid = Self::mul(twiddles2_im, rotated6);
253        let twiddled5_mid = Self::mul(twiddles1_im, rotated6);
254        let twiddled6_mid = Self::mul(twiddles0_im, rotated6);
255
256        let twiddled1_mid2 = Self::fmadd(twiddles1_re, sum2, twiddled1_mid);
257        let twiddled2_mid2 = Self::fmadd(twiddles2_re, sum2, twiddled2_mid);
258        let twiddled3_mid2 = Self::fmadd(twiddles0_re, sum2, twiddled3_mid);
259        let twiddled4_mid2 = Self::fnmadd(twiddles0_im, rotated5, twiddled4_mid); // fnmadd instead of fmadd because we're actually re-using twiddle0 here. remember that this algorithm is all about factoring out conjugated multiplications -- this negation of the twiddle0 imaginaries is a reflection of one of those conugations
260        let twiddled5_mid2 = Self::fnmadd(twiddles2_im, rotated5, twiddled5_mid);
261        let twiddled6_mid2 = Self::fmadd(twiddles1_im, rotated5, twiddled6_mid);
262
263        let twiddled1 = Self::fmadd(twiddles2_re, sum3, twiddled1_mid2);
264        let twiddled2 = Self::fmadd(twiddles0_re, sum3, twiddled2_mid2);
265        let twiddled3 = Self::fmadd(twiddles1_re, sum3, twiddled3_mid2);
266        let twiddled4 = Self::fmadd(twiddles1_im, rotated4, twiddled4_mid2);
267        let twiddled5 = Self::fnmadd(twiddles0_im, rotated4, twiddled5_mid2);
268        let twiddled6 = Self::fmadd(twiddles2_im, rotated4, twiddled6_mid2);
269
270        // Post-processing to mix the twiddle factors between the rest of the output
271        let [output1, output6] = Self::column_butterfly2([twiddled1, twiddled6]);
272        let [output2, output5] = Self::column_butterfly2([twiddled2, twiddled5]);
273        let [output3, output4] = Self::column_butterfly2([twiddled3, twiddled4]);
274
275        [
276            output0, output1, output2, output3, output4, output5, output6,
277        ]
278    }

601        // if we merge some of our 128 registers into 256 registers, we can do 2 inner butterfly3's instead of 3
602        let rows12 = Self::merge(rows[1], rows[2]);
603        let rows45 = Self::merge(rows[4], rows[5]);
604        let rows78 = Self::merge(rows[7], rows[8]);

608        let mut mid12 =
609            Self::FullVector::column_butterfly3([rows12, rows45, rows78], butterfly3_twiddles);

620        // Re-merge our half vectors into different, transposed full vectors. Thankfully the compiler is smart enough to combine these inserts and extracts into permutes
621        let transposed12 = Self::merge(mid0[1], mid0[2]);
622        let transposed45 = Self::merge(mid1_1, mid1_2);
623        let transposed78 = Self::merge(mid2_1, mid2_2);

628            [transposed12, transposed45, transposed78],

630        );
631
632        // Finally, extract our second set of merged columns
633        let (output3, output6) = Self::split(output36);
634        let (output4, output7) = Self::split(output47);
635        let (output5, output8) = Self::split(output58);

2174/// Multiply the complex numbers in `left` by the complex numbers in `right`.
2175/// This is exactly the same as `mul_complex` in `AvxVector`, but this implementation also conjugates the `left` input before multiplying
2176#[inline(always)]
2177unsafe fn mul_complex_conjugated<V: AvxVector>(left: V, right: V) -> V {
2178    // Extract the real and imaginary components from left into 2 separate registers
2179    let (left_real, left_imag) = V::duplicate_complex_components(left);
2180
2181    // create a shuffled version of right where the imaginary values are swapped with the reals
2182    let right_shuffled = V::swap_complex_components(right);
2183
2184    // multiply our duplicated imaginary left vector by our shuffled right vector. that will give us the right side of the traditional complex multiplication formula
2185    let output_right = V::mul(left_imag, right_shuffled);
2186
2187    // use a FMA instruction to multiply together left side of the complex multiplication formula, then alternatingly add and subtract the left side from the right
2188    // By using subadd instead of addsub, we can conjugate the left side for free.
2189    V::fmsubadd(left_real, right, output_right)
2190}
