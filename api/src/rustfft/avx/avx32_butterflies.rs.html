245macro_rules! gen_butterfly_twiddles_interleaved_columns {
246    ($num_rows:expr, $num_cols:expr, $skip_cols:expr, $direction: expr) => {{
247        const FFT_LEN: usize = $num_rows * $num_cols;
248        const TWIDDLE_ROWS: usize = $num_rows - 1;
249        const TWIDDLE_COLS: usize = $num_cols - $skip_cols;
250        const TWIDDLE_VECTOR_COLS: usize = TWIDDLE_COLS / 4;
251        const TWIDDLE_VECTOR_COUNT: usize = TWIDDLE_VECTOR_COLS * TWIDDLE_ROWS;
252        let mut twiddles = [AvxVector::zero(); TWIDDLE_VECTOR_COUNT];
253        for index in 0..TWIDDLE_VECTOR_COUNT {
254            let y = (index / TWIDDLE_VECTOR_COLS) + 1;
255            let x = (index % TWIDDLE_VECTOR_COLS) * 4 + $skip_cols;
256
257            twiddles[index] = AvxVector::make_mixedradix_twiddle_chunk(x, y, FFT_LEN, $direction);
258        }
259        twiddles
260    }};
261}

303impl<T> Butterfly5Avx<T> {
304    #[target_feature(enable = "avx", enable = "fma")]
305    unsafe fn perform_fft_f32(&self, mut buffer: impl AvxArrayMut<f32>) {
306        let input0 = _mm_castpd_ps(_mm_load1_pd(buffer.input_ptr() as *const f64)); // load the first element of the input, and duplicate it into both complex number slots of input0
307        let input12 = buffer.load_partial2_complex(1);
308        let input34 = buffer.load_partial2_complex(3);
309
310        // swap elements for inputs 3 and 4
311        let input43 = AvxVector::reverse_complex_elements(input34);
312
313        // do some prep work before we can start applying twiddle factors
314        let [sum12, diff43] = AvxVector::column_butterfly2([input12, input43]);
315
316        let rotation = AvxVector::make_rotation90(FftDirection::Inverse);
317        let rotated43 = AvxVector::rotate90(diff43, rotation);
318
319        let [mid14, mid23] = AvxVector::unpack_complex([sum12, rotated43]);
320
321        // to compute the first output, compute the sum of all elements. mid14[0] and mid23[0] already have the sum of 1+4 and 2+3 respectively, so if we add them, we'll get the sum of all 4
322        let sum1234 = AvxVector::add(mid14, mid23);
323        let output0 = AvxVector::add(input0, sum1234);
324
325        // apply twiddle factors
326        let twiddled14_mid = AvxVector::mul(mid14, self.twiddles[0]);
327        let twiddled23_mid = AvxVector::mul(mid14, self.twiddles[1]);
328        let twiddled14 = AvxVector::fmadd(mid23, self.twiddles[1], twiddled14_mid);
329        let twiddled23 = AvxVector::fmadd(mid23, self.twiddles[2], twiddled23_mid);
330
331        // unpack the data for the last butterfly 2
332        let [twiddled12, twiddled43] = AvxVector::unpack_complex([twiddled14, twiddled23]);
333        let [output12, output43] = AvxVector::column_butterfly2([twiddled12, twiddled43]);
334
335        // swap the elements in output43 before writing them out, and add the first input to everything
336        let final12 = AvxVector::add(input0, output12);
337        let output34 = AvxVector::reverse_complex_elements(output43);
338        let final34 = AvxVector::add(input0, output34);
339
340        buffer.store_partial1_complex(output0, 0);
341        buffer.store_partial2_complex(final12, 1);
342        buffer.store_partial2_complex(final34, 3);
343    }
344}

371impl<T> Butterfly7Avx<T> {
372    #[target_feature(enable = "avx", enable = "fma")]
373    unsafe fn perform_fft_f32(&self, mut buffer: impl AvxArrayMut<f32>) {
374        // load the first element of the input, and duplicate it into both complex number slots of input0
375        let input0 = _mm_castpd_ps(_mm_load1_pd(buffer.input_ptr() as *const f64));
376
377        // we want to load 3 elements into 123 and 3 elements into 456, but we can only load 4, so we're going to do slightly overlapping reads here
378        // we have to reverse 456 immediately after loading, and that'll be easiest if we load the 456 into the latter 3 slots of the register, rather than the front 3 slots
379        // as a bonus, that also means we don't need masked reads or anything
380        let input123 = buffer.load_complex(1);
381        let input456 = buffer.load_complex(3);
382
383        // reverse the order of input456
384        let input654 = AvxVector::reverse_complex_elements(input456);
385
386        // do some prep work before we can start applying twiddle factors
387        let [sum123, diff654] = AvxVector::column_butterfly2([input123, input654]);
388        let rotation = AvxVector::make_rotation90(FftDirection::Inverse);
389        let rotated654 = AvxVector::rotate90(diff654, rotation);
390
391        let [mid1634, mid25] = AvxVector::unpack_complex([sum123, rotated654]);
392
393        let mid16 = mid1634.lo();
394        let mid25 = mid25.lo();
395        let mid34 = mid1634.hi();
396
397        // to compute the first output, compute the sum of all elements. mid16[0], mid25[0], and mid34[0] already have the sum of 1+6, 2+5 and 3+4 respectively, so if we add them, we'll get 1+2+3+4+5+6
398        let output0_left = AvxVector::add(mid16, mid25);
399        let output0_right = AvxVector::add(input0, mid34);
400        let output0 = AvxVector::add(output0_left, output0_right);
401        buffer.store_partial1_complex(output0, 0);
402
403        _mm256_zeroupper();
404
405        // apply twiddle factors
406        let twiddled16_intermediate1 = AvxVector::mul(mid16, self.twiddles[0]);
407        let twiddled25_intermediate1 = AvxVector::mul(mid16, self.twiddles[1]);
408        let twiddled34_intermediate1 = AvxVector::mul(mid16, self.twiddles[2]);
409
410        let twiddled16_intermediate2 =
411            AvxVector::fmadd(mid25, self.twiddles[1], twiddled16_intermediate1);
412        let twiddled25_intermediate2 =
413            AvxVector::fmadd(mid25, self.twiddles[3], twiddled25_intermediate1);
414        let twiddled34_intermediate2 =
415            AvxVector::fmadd(mid25, self.twiddles[4], twiddled34_intermediate1);
416
417        let twiddled16 = AvxVector::fmadd(mid34, self.twiddles[2], twiddled16_intermediate2);
418        let twiddled25 = AvxVector::fmadd(mid34, self.twiddles[4], twiddled25_intermediate2);
419        let twiddled34 = AvxVector::fmadd(mid34, self.twiddles[1], twiddled34_intermediate2);
420
421        // unpack the data for the last butterfly 2
422        let [twiddled12, twiddled65] = AvxVector::unpack_complex([twiddled16, twiddled25]);
423        let [twiddled33, twiddled44] = AvxVector::unpack_complex([twiddled34, twiddled34]);
424
425        // we can save one add if we add input0 to twiddled33 now. normally we'd add input0 to the final output, but the arrangement of data makes that a little awkward
426        let twiddled033 = AvxVector::add(twiddled33, input0);
427
428        let [output12, output65] = AvxVector::column_butterfly2([twiddled12, twiddled65]);
429        let [output033, output044] = AvxVector::column_butterfly2([twiddled033, twiddled44]);
430        let output56 = AvxVector::reverse_complex_elements(output65);
431
432        buffer.store_partial2_complex(AvxVector::add(output12, input0), 1);
433        buffer.store_partial1_complex(output033, 3);
434        buffer.store_partial1_complex(output044, 4);
435        buffer.store_partial2_complex(AvxVector::add(output56, input0), 5);
436    }

521        let mid47: __m256 = AvxVector256::merge(mid2947.hi(), mid2947.hi());

534        // but we only want to add it once, so zero the second element

611impl<T> Butterfly8Avx<T> {
612    #[target_feature(enable = "avx", enable = "fma")]
613    unsafe fn perform_fft_f32(&self, mut buffer: impl AvxArrayMut<f32>) {
614        let row0 = buffer.load_complex(0);
615        let row1 = buffer.load_complex(4);
616
617        // Do our butterfly 2's down the columns
618        let [intermediate0, intermediate1_pretwiddle] = AvxVector::column_butterfly2([row0, row1]);
619
620        // Apply the size-8 twiddle factors
621        let intermediate1 = AvxVector::mul_complex(intermediate1_pretwiddle, self.twiddles);
622
623        // Rearrange the data before we do our butterfly 4s. This swaps the last 2 elements of butterfly0 with the first two elements of butterfly1
624        // The result is that we can then do a 4x butterfly 2, apply twiddles, use unpack instructions to transpose to the final output, then do another 4x butterfly 2
625        let permuted0 = _mm256_permute2f128_ps(intermediate0, intermediate1, 0x20);
626        let permuted1 = _mm256_permute2f128_ps(intermediate0, intermediate1, 0x31);
627
628        // Do the first set of butterfly 2's
629        let [postbutterfly0, postbutterfly1_pretwiddle] =
630            AvxVector::column_butterfly2([permuted0, permuted1]);
631
632        // Which negative we blend in depends on whether we're forward or direction
633        // Our goal is to swap the reals with the imaginaries, then negate either the reals or the imaginaries, based on whether we're an direction or not
634        // but we can't use the AvxVector swap_complex_components function, because we only want to swap the odd reals with the odd imaginaries
635        let elements_swapped = _mm256_permute_ps(postbutterfly1_pretwiddle, 0xB4);
636
637        // We can negate the elements we want by xoring the row with a pre-set vector
638        let postbutterfly1 = AvxVector::xor(elements_swapped, self.twiddles_butterfly4);
639
640        // use unpack instructions to transpose, and to prepare for the final butterfly 2's
641        let unpermuted0 = _mm256_permute2f128_ps(postbutterfly0, postbutterfly1, 0x20);
642        let unpermuted1 = _mm256_permute2f128_ps(postbutterfly0, postbutterfly1, 0x31);
643        let unpacked = AvxVector::unpack_complex([unpermuted0, unpermuted1]);
644
645        let [output0, output1] = AvxVector::column_butterfly2(unpacked);
646
647        buffer.store_complex(output0, 0);
648        buffer.store_complex(output1, 4);
649    }

676impl<T> Butterfly9Avx<T> {
677    #[target_feature(enable = "avx", enable = "fma")]
678    unsafe fn perform_fft_f32(&self, mut buffer: impl AvxArrayMut<f32>) {
679        // we're going to load these elements in a peculiar way. instead of loading a row into the first 3 element of each register and leaving the last element empty
680        // we're leaving the first element empty and putting the data in the last 3 elements. this will let us do 3 total complex multiplies instead of 4.
681
682        let input0_lo = _mm_castpd_ps(_mm_load1_pd(buffer.input_ptr() as *const f64));
683        let input0_hi = buffer.load_partial2_complex(1);
684        let input0 = AvxVector256::merge(input0_lo, input0_hi);
685        let input1 = buffer.load_complex(2);
686        let input2 = buffer.load_complex(5);
687
688        // We're going to treat our input as a 3x3 2d array. First, do 3 butterfly 3's down the columns of that array.
689        let [mid0, mid1, mid2] =
690            AvxVector::column_butterfly3([input0, input1, input2], self.twiddles_butterfly3);
691
692        // merge the twiddle-able data into a single avx vector
693        let twiddle_data = _mm256_permute2f128_ps(mid1, mid2, 0x31);
694        let twiddled = AvxVector::mul_complex(twiddle_data, self.twiddles);
695
696        // Transpose our 3x3 array. We could use the 4x4 transpose with an empty bottom row, which would result in an empty last column
697        // but it turns out that it'll make our packing process later simpler if we duplicate the second row into the last row
698        // which will result in duplicating the second column into the last column after the transpose
699        let permute0 = _mm256_permute2f128_ps(mid0, mid2, 0x20);
700        let permute1 = _mm256_permute2f128_ps(mid1, mid1, 0x20);
701        let permute2 = _mm256_permute2f128_ps(mid0, twiddled, 0x31);
702        let permute3 = _mm256_permute2f128_ps(twiddled, twiddled, 0x20);
703
704        let transposed0 = AvxVector::unpackhi_complex([permute0, permute1]);
705        let [transposed1, transposed2] = AvxVector::unpack_complex([permute2, permute3]);
706
707        // more size 3 buterflies
708        let output_rows = AvxVector::column_butterfly3(
709            [transposed0, transposed1, transposed2],
710            self.twiddles_butterfly3,
711        );
712
713        // the elements of row 1 are in pretty much the worst possible order, thankfully we can fix that with just a couple instructions
714        let swapped1 = _mm256_permute_ps(output_rows[1], 0x4E); // swap even and odd complex numbers
715        let packed1 = _mm256_permute2f128_ps(swapped1, output_rows[2], 0x21);
716        buffer.store_complex(packed1, 4);
717
718        // merge just the high element of swapped_lo into the high element of row 0
719        let zero_swapped1_lo = AvxVector256::merge(AvxVector::zero(), swapped1.lo());
720        let packed0 = _mm256_blend_ps(output_rows[0], zero_swapped1_lo, 0xC0);
721        buffer.store_complex(packed0, 0);
722
723        // The last element can just be written on its own
724        buffer.store_partial1_complex(output_rows[2].hi(), 8);
725    }

767    unsafe fn perform_fft_f32(&self, mut buffer: impl AvxArrayMut<f32>) {
768        // we're going to load these elements in a peculiar way. instead of loading a row into the first 3 element of each register and leaving the last element empty
769        // we're leaving the first element empty and putting the data in the last 3 elements. this will save us a complex multiply.
770
771        // for everything but the first element, we can do overlapping reads. for the first element, an "overlapping read" would have us reading from index -1, so instead we have to shuffle some data around
772        let input0_lo = _mm_castpd_ps(_mm_load1_pd(buffer.input_ptr() as *const f64));
773        let input0_hi = buffer.load_partial2_complex(1);
774        let input_rows = [
775            AvxVector256::merge(input0_lo, input0_hi),
776            buffer.load_complex(2),
777            buffer.load_complex(5),
778            buffer.load_complex(8),
779        ];
780
781        // 3 butterfly 4's down the columns
782        let mut mid = AvxVector::column_butterfly4(input_rows, self.twiddles_butterfly4);
783
784        // Multiply in our twiddle factors. mid2 will be normal, but for mid1 and mid3, we're going to merge the twiddle-able parts into a single vector,
785        // and do a single complex multiply on it. this transformation saves a complex multiply and costs nothing,
786        // because we needthe second halves of mid1 and mid3 in a single vector for the transpose afterward anyways, so we would have done this permute2f128 operation either way
787        mid[2] = AvxVector::mul_complex(mid[2], self.twiddles[0]);
788        let merged_mid13 = _mm256_permute2f128_ps(mid[1], mid[3], 0x31);
789        let twiddled13 = AvxVector::mul_complex(self.twiddles[1], merged_mid13);
790
791        // Transpose our 3x4 array into a 4x3. we're doing a custom transpose here because we have to re-distribute the merged twiddled23 back out, and we can roll that into the transpose to make it free
792        let transposed = {
793            let permute0 = _mm256_permute2f128_ps(mid[0], mid[2], 0x20);
794            let permute1 = _mm256_permute2f128_ps(mid[1], mid[3], 0x20);
795            let permute2 = _mm256_permute2f128_ps(mid[0], mid[2], 0x31);
796            let permute3 = twiddled13; // normally we'd need to do a permute here, but we can skip it because we already did it for twiddle factors
797
798            let unpacked1 = AvxVector::unpackhi_complex([permute0, permute1]);
799            let [unpacked2, unpacked3] = AvxVector::unpack_complex([permute2, permute3]);
800
801            [unpacked1, unpacked2, unpacked3]
802        };
803
804        // Do 4 butterfly 3's down the columns of our transposed array
805        let output_rows = AvxVector::column_butterfly3(transposed, self.twiddles_butterfly3);
806
807        buffer.store_complex(output_rows[0], 0);
808        buffer.store_complex(output_rows[1], 4);
809        buffer.store_complex(output_rows[2], 8);
810    }

1446impl<T> Butterfly128Avx<T> {
1447    #[target_feature(enable = "avx", enable = "fma")]
1448    unsafe fn column_butterflies_and_transpose(
1449        &self,
1450        input: &[Complex<f32>],
1451        mut output: &mut [Complex<f32>],
1452    ) {
1453        // A size-128 FFT is way too big to fit in registers, so instead we're going to compute it in two phases, storing in scratch in between.
1454
1455        // First phase is to treat this size-128 array like a 16x8 2D array, and do butterfly 8's down the columns
1456        // Then, apply twiddle factors, and finally transpose into the scratch space
1457
1458        // But again, we don't have enough registers to load it all at once, so only load one column of AVX vectors at a time
1459        for columnset in 0..4 {
1460            let mut rows = [AvxVector::zero(); 8];
1461            for r in 0..8 {
1462                rows[r] = input.load_complex(columnset * 4 + 16 * r);
1463            }
1464            // apply butterfly 8
1465            let mut mid = AvxVector::column_butterfly8(rows, self.twiddles_butterfly4);
1466
1467            // apply twiddle factors
1468            for r in 1..8 {
1469                mid[r] = AvxVector::mul_complex(mid[r], self.twiddles[r - 1 + 7 * columnset]);
1470            }
1471
1472            // transpose
1473            let transposed = AvxVector::transpose8_packed(mid);
1474
1475            // write out
1476            for i in 0..4 {
1477                output.store_complex(transposed[i * 2], columnset * 32 + i * 8);
1478                output.store_complex(transposed[i * 2 + 1], columnset * 32 + i * 8 + 4);
1479            }
1480        }
1481    }
1482
1483    #[target_feature(enable = "avx", enable = "fma")]
1484    unsafe fn row_butterflies(&self, mut buffer: impl AvxArrayMut<f32>) {
1485        // Second phase: Butterfly 16's down the columns of our transposed array.
1486        // Thankfully, during the first phase, we set everything up so that all we have to do here is compute the size-16 FFT columns and write them back out where we got them
1487        // We're also using a customized butterfly16 function that is smarter about when it loads/stores data, to reduce register spilling
1488        for columnset in 0usize..2 {
1489            column_butterfly16_loadfn!(
1490                |index: usize| buffer.load_complex(columnset * 4 + index * 8),
1491                |data, index| buffer.store_complex(data, columnset * 4 + index * 8),
1492                self.twiddles_butterfly16,
1493                self.twiddles_butterfly4
1494            );
1495        }
1496    }

1608impl<T> Butterfly512Avx<T> {
1609    #[target_feature(enable = "avx", enable = "fma")]
1610    unsafe fn column_butterflies_and_transpose(
1611        &self,
1612        input: &[Complex<f32>],
1613        mut output: &mut [Complex<f32>],
1614    ) {
1615        // A size-512 FFT is way too big to fit in registers, so instead we're going to compute it in two phases, storing in scratch in between.
1616
1617        // First phase is to treat this size-512 array like a 32x16 2D array, and do butterfly 16's down the columns
1618        // Then, apply twiddle factors, and finally transpose into the scratch space
1619
1620        // But again, we don't have enough registers to load it all at once, so only load one column of AVX vectors at a time
1621        // We're also using a customized butterfly16 function that is smarter about when it loads/stores data, to reduce register spilling
1622        const TWIDDLES_PER_COLUMN: usize = 15;
1623        for (columnset, twiddle_chunk) in
1624            self.twiddles.chunks_exact(TWIDDLES_PER_COLUMN).enumerate()
1625        {
1626            // Sadly we have to use MaybeUninit here. If we init an array like normal with AvxVector::Zero(), the compiler can't seem to figure out that it can
1627            // eliminate the dead stores of zeroes to the stack. By using uninit here, we avoid those unnecessary writes
1628            let mut mid_uninit: [MaybeUninit<__m256>; 16] = [MaybeUninit::<__m256>::uninit(); 16];
1629
1630            column_butterfly16_loadfn!(
1631                |index: usize| input.load_complex(columnset * 4 + 32 * index),
1632                |data, index: usize| {
1633                    mid_uninit[index].as_mut_ptr().write(data);
1634                },
1635                self.twiddles_butterfly16,
1636                self.twiddles_butterfly4
1637            );
1638
1639            // Apply twiddle factors, transpose, and store. Traditionally we apply all the twiddle factors at once and then do all the transposes at once,
1640            // But our data is pushing the limit of what we can store in registers, so the idea here is to get the data out the door with as few spills to the stack as possible
1641            for chunk in 0..4 {
1642                let twiddled = [
1643                    if chunk > 0 {
1644                        AvxVector::mul_complex(
1645                            mid_uninit[4 * chunk].assume_init(),
1646                            twiddle_chunk[4 * chunk - 1],
1647                        )
1648                    } else {
1649                        mid_uninit[4 * chunk].assume_init()
1650                    },
1651                    AvxVector::mul_complex(
1652                        mid_uninit[4 * chunk + 1].assume_init(),
1653                        twiddle_chunk[4 * chunk],
1654                    ),
1655                    AvxVector::mul_complex(
1656                        mid_uninit[4 * chunk + 2].assume_init(),
1657                        twiddle_chunk[4 * chunk + 1],
1658                    ),
1659                    AvxVector::mul_complex(
1660                        mid_uninit[4 * chunk + 3].assume_init(),
1661                        twiddle_chunk[4 * chunk + 2],
1662                    ),
1663                ];
1664
1665                let transposed = AvxVector::transpose4_packed(twiddled);
1666
1667                output.store_complex(transposed[0], columnset * 64 + 0 * 16 + 4 * chunk);
1668                output.store_complex(transposed[1], columnset * 64 + 1 * 16 + 4 * chunk);
1669                output.store_complex(transposed[2], columnset * 64 + 2 * 16 + 4 * chunk);
1670                output.store_complex(transposed[3], columnset * 64 + 3 * 16 + 4 * chunk);
1671            }
1672        }
1673    }
1674
1675    #[target_feature(enable = "avx", enable = "fma")]
1676    unsafe fn row_butterflies(&self, mut buffer: impl AvxArrayMut<f32>) {
1677        // Second phase: Butterfly 32's down the columns of our transposed array.
1678        // Thankfully, during the first phase, we set everything up so that all we have to do here is compute the size-32 FFT columns and write them back out where we got them
1679        // We're also using a customized butterfly32 function that is smarter about when it loads/stores data, to reduce register spilling

1682                |index: usize| buffer.load_complex(columnset * 4 + index * 16),
1683                |data, index| buffer.store_complex(data, columnset * 4 + index * 16),

1687        }
