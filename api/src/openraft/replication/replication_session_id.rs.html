9/// Uniquely identifies a replication session.
10///
11/// A replication session is a group of replication stream, e.g., the leader(node-1) in a cluster of
12/// 3 nodes may have a replication config `{2,3}`.
13///
14/// Replication state can not be shared between two leaders(different `vote`) or between two
15/// membership configs: E.g. given 3 membership log:
16/// - `log_id=1, members={a,b,c}`
17/// - `log_id=5, members={a,b}`
18/// - `log_id=10, members={a,b,c}`
19///
20/// When log_id=1 is appended, openraft spawns a replication to node `c`.
21/// Then log_id=1 is replicated to node `c`.
22/// Then a replication state update message `{target=c, matched=log_id-1}` is piped in message
23/// queue(`tx_api`), waiting the raft core to process.
24///
25/// Then log_id=5 is appended, replication to node `c` is dropped.
26///
27/// Then log_id=10 is appended, another replication to node `c` is spawned.
28/// Now node `c` is a new empty node, no log is replicated to it.
29/// But the delayed message `{target=c, matched=log_id-1}` may be process by raft core and make raft
30/// core believe node `c` already has `log_id=1`, and commit it.
31#[derive(Debug, Clone, Copy)]
32pub(crate) struct ReplicationSessionId<NID: NodeId> {
33    /// The Leader or Candidate this replication belongs to.
34    pub(crate) vote: Vote<NID>,
35
36    /// The log id of the membership log this replication works for.
37    pub(crate) membership_log_id: Option<LogId<NID>>,
38}
39
40impl<NID: NodeId> Display for ReplicationSessionId<NID> {
41    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
42        write!(f, "{}/{}", self.vote, self.membership_log_id.summary())
43    }
44}
45
46impl<NID: NodeId> ReplicationSessionId<NID> {
47    pub(crate) fn new(vote: Vote<NID>, membership_log_id: Option<LogId<NID>>) -> Self {
48        Self {
49            vote,
50            membership_log_id,
51        }
52    }
53
54    pub(crate) fn vote_ref(&self) -> &Vote<NID> {
55        &self.vote
56    }
57}
