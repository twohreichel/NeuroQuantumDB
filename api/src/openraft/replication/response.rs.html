13/// The response of replication command.
14#[derive(Debug)]
15pub(crate) enum Response<C>
16where C: RaftTypeConfig
17{
18    // /// Logs that are submitted to append has been persisted to disk.
19    // LogPersisted {},
20    /// Update the `matched` log id of a replication target.
21    /// Sent by a replication task `ReplicationCore`.
22    Progress {
23        /// The ID of the target node for which the match index is to be updated.
24        target: C::NodeId,
25
26        /// The id of the subject that submit this replication action.
27        request_id: RequestId,
28
29        /// The request by this leader has been successfully handled by the target node,
30        /// or an error in string.
31        ///
32        /// A successful result can still be log matching or log conflicting.
33        /// In either case, the request is considered accepted, i.e., this leader is still valid to
34        /// the target node.
35        ///
36        /// The result also track the time when this request is sent.
37        result: Result<ReplicationResult<C>, String>,
38
39        /// In which session this message is sent.
40        ///
41        /// This session id identifies a certain leader(by vote) that is replicating to a certain
42        /// group of nodes.
43        ///
44        /// A message should be discarded if it does not match the present vote and
45        /// membership_log_id.
46        session_id: ReplicationSessionId<C::NodeId>,
47    },
48
49    /// [`StorageError`] error has taken place locally(not on remote node) when replicating, and
50    /// [`RaftCore`](`crate::core::RaftCore`) needs to shutdown. Sent by a replication task
51    /// [`crate::replication::ReplicationCore`].
52    StorageError { error: StorageError<C::NodeId> },
53
54    /// ReplicationCore has seen a higher `vote`.
55    /// Sent by a replication task `ReplicationCore`.
56    HigherVote {
57        /// The ID of the target node from which the new term was observed.
58        target: C::NodeId,
59
60        /// The higher vote observed.
61        higher: Vote<C::NodeId>,
62
63        /// Which state(a Leader or Candidate) sent this message
64        sender_vote: Vote<C::NodeId>,
65        // TODO: need this?
66        // /// The cluster this replication works for.
67        // membership_log_id: Option<LogId<C::NodeId>>,
68    },
69}

71impl<C> MessageSummary<Response<C>> for Response<C>
72where C: RaftTypeConfig
73{
74    fn summary(&self) -> String {
75        match self {
76            Self::Progress {
77                ref target,
78                request_id: ref id,
79                ref result,
80                ref session_id,
81            } => {
82                format!(
83                    "UpdateReplicationProgress: target: {}, id: {}, result: {:?}, session_id: {}",
84                    target, id, result, session_id,
85                )
86            }
87
88            Self::StorageError { error } => format!("ReplicationStorageError: {}", error),
89
90            Self::HigherVote {
91                target,
92                higher,
93                sender_vote,
94            } => {
95                format!(
96                    "Seen a higher vote: target: {}, vote: {}, server_state_vote: {}",
97                    target, higher, sender_vote
98                )
99            }
100        }
101    }
102}

104/// Result of an append-entries replication
105#[derive(Clone, Debug)]
106pub(crate) struct ReplicationResult<C: RaftTypeConfig> {
107    /// The timestamp when this request is sent.
108    ///
109    /// It is used to update the lease for leader.
110    pub(crate) sending_time: InstantOf<C>,
111
112    /// Ok for matching, Err for conflict.
113    pub(crate) result: Result<Option<LogIdOf<C>>, LogIdOf<C>>,
114}
115
116impl<C> fmt::Display for ReplicationResult<C>
117where C: RaftTypeConfig
118{
119    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
120        write!(f, "{{time:{:?}, result:", self.sending_time)?;
121
122        match &self.result {
123            Ok(matching) => write!(f, "Match:{}", matching.display())?,
124            Err(conflict) => write!(f, "Conflict:{}", conflict)?,
125        }
126
127        write!(f, "}}")
128    }
129}
130
131impl<C> ReplicationResult<C>
132where C: RaftTypeConfig
133{
134    pub(crate) fn new(sending_time: InstantOf<C>, result: Result<Option<LogIdOf<C>>, LogIdOf<C>>) -> Self {
135        Self { sending_time, result }
136    }
137}
