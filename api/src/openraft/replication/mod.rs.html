77/// A task responsible for sending replication events to a target follower in the Raft cluster.
78///
79/// NOTE: we do not stack replication requests to targets because this could result in
80/// out-of-order delivery. We always buffer until we receive a success response, then send the
81/// next payload from the buffer.

83where
84    C: RaftTypeConfig,
85    N: RaftNetworkFactory<C>,
86    LS: RaftLogStorage<C>,

92    session_id: ReplicationSessionId<C::NodeId>,
93
94    /// A channel for sending events to the RaftCore.
95    #[allow(clippy::type_complexity)]
96    tx_raft_core: mpsc::UnboundedSender<Notify<C>>,
97
98    /// A channel for receiving events from the RaftCore and snapshot transmitting task.
99    rx_event: mpsc::UnboundedReceiver<Replicate<C>>,
100

102    ///
103    /// Because 1) ReplicationCore replies on the `close` event to shutdown.
104    /// 2) ReplicationCore holds this tx; It is made a weak so that when

157    #[tracing::instrument(level = "trace", skip_all,fields(target=display(&target), session_id=display(&session_id)))]
158    #[allow(clippy::type_complexity)]
159    #[allow(clippy::too_many_arguments)]
160    pub(crate) fn spawn(
161        target: C::NodeId,
162        session_id: ReplicationSessionId<C::NodeId>,
163        config: Arc<Config>,
164        committed: Option<LogId<C::NodeId>>,
165        matching: Option<LogId<C::NodeId>>,
166        network: N::Network,
167        snapshot_network: N::Network,
168        log_reader: LS::LogReader,
169        snapshot_reader: SnapshotReader<C>,
170        tx_raft_core: mpsc::UnboundedSender<Notify<C>>,
171        span: tracing::Span,
172    ) -> ReplicationHandle<C> {
173        tracing::debug!(
174            session_id = display(&session_id),
175            target = display(&target),
176            committed = display(committed.summary()),
177            matching = debug(&matching),
178            "spawn replication"
179        );
180
181        // other component to ReplicationStream
182        let (tx_event, rx_event) = mpsc::unbounded_channel();
183
184        let this = Self {
185            target,
186            session_id,
187            network,
188            snapshot_network: Arc::new(Mutex::new(snapshot_network)),
189            snapshot_state: None,
190            backoff: None,
191            log_reader,
192            snapshot_reader,
193            config,
194            committed,
195            matching,
196            tx_raft_core,
197            rx_event,
198            weak_tx_event: tx_event.downgrade(),
199            next_action: None,
200            entries_hint: Default::default(),
201        };
202
203        let join_handle = C::AsyncRuntime::spawn(this.main().instrument(span));
204
205        ReplicationHandle {
206            join_handle,
207            tx_repl: tx_event,
208        }

211    #[tracing::instrument(level="debug", skip(self), fields(session=%self.session_id, target=display(&self.target), cluster=%self.config.cluster_name))]
212    async fn main(mut self) -> Result<(), ReplicationClosed> {
213        loop {
214            let action = self.next_action.take();
215
216            let Some(d) = action else {
217                self.drain_events_with_backoff().await?;
218                continue;
219            };
220
221            // Backup the log data for retrying.
222            let mut log_data = None;
223
224            tracing::debug!(replication_data = display(&d), "{} send replication RPC", func_name!());
225
226            let request_id = d.request_id();
227
228            let res = match d {
229                Data::Heartbeat => {
230                    let m = &self.matching;
231                    // request_id==None will be ignored by RaftCore.
232                    let d = DataWithId::new(RequestId::new_heartbeat(), LogIdRange::new(m.clone(), m.clone()));
233
234                    log_data = Some(d.clone());
235                    self.send_log_entries(d).await
236                }
237                Data::Logs(log) => {
238                    log_data = Some(log.clone());
239                    self.send_log_entries(log).await
240                }
241                Data::Snapshot(snap) => self.stream_snapshot(snap).await,
242                Data::SnapshotCallback(resp) => self.handle_snapshot_callback(resp),
243            };
244
245            tracing::debug!(res = debug(&res), "replication action done");
246
247            match res {
248                Ok(next) => {
249                    // reset backoff at once if replication succeeds
250                    self.backoff = None;
251
252                    // If the RPC was successful but not finished, continue.
253                    if let Some(next) = next {
254                        self.next_action = Some(next);
255                    }
256                }
257                Err(err) => {
258                    tracing::warn!(error=%err, "error replication to target={}", self.target);
259
260                    match err {
261                        ReplicationError::Closed(closed) => {
262                            return Err(closed);
263                        }
264                        ReplicationError::HigherVote(h) => {
265                            let _ = self.tx_raft_core.send(Notify::Network {
266                                response: Response::HigherVote {
267                                    target: self.target,
268                                    higher: h.higher,
269                                    sender_vote: self.session_id.vote_ref().clone(),
270                                },
271                            });
272                            return Ok(());
273                        }
274                        ReplicationError::StorageError(error) => {
275                            tracing::error!(error=%error, "error replication to target={}", self.target);
276
277                            // TODO: report this error
278                            let _ = self.tx_raft_core.send(Notify::Network {
279                                response: Response::StorageError { error },
280                            });
281                            return Ok(());
282                        }
283                        ReplicationError::RPCError(err) => {
284                            tracing::error!(err = display(&err), "RPCError");
285
286                            let retry = match &err {
287                                RPCError::Timeout(_) => false,
288                                RPCError::Unreachable(_unreachable) => {
289                                    // If there is an [`Unreachable`] error, we will backoff for a
290                                    // period of time. Backoff will be reset if there is a
291                                    // successful RPC is sent.
292                                    if self.backoff.is_none() {
293                                        self.backoff = Some(self.network.backoff());
294                                    }
295                                    false
296                                }
297                                RPCError::PayloadTooLarge(too_large) => {
298                                    self.update_hint(too_large);
299
300                                    // PayloadTooLarge is a retryable error: retry at once.
301                                    self.next_action = Some(Data::Logs(log_data.unwrap()));
302                                    true
303                                }
304                                RPCError::Network(_) => false,
305                                RPCError::RemoteError(_) => false,
306                            };
307
308                            if retry {
309                                debug_assert!(self.next_action.is_some(), "next_action must be Some");
310                            } else {
311                                self.send_progress_error(request_id, err);
312                            }
313                        }
314                    };
315                }
316            };
317
318            self.drain_events_with_backoff().await?;
319        }
320    }

322    async fn drain_events_with_backoff(&mut self) -> Result<(), ReplicationClosed> {
323        if let Some(b) = &mut self.backoff {
324            let duration = b.next().unwrap_or_else(|| {
325                tracing::warn!("backoff exhausted, using default");
326                Duration::from_millis(500)
327            });
328
329            self.backoff_drain_events(C::now() + duration).await?;
330        }
331
332        self.drain_events().await?;
333        Ok(())
334    }

336    /// When a [`PayloadTooLarge`] error is received, update the hint for the next several RPC.
337    fn update_hint(&mut self, too_large: &PayloadTooLarge) {
338        const DEFAULT_ENTRIES_HINT_TTL: u64 = 10;
339
340        match too_large.action() {
341            RPCTypes::Vote => {
342                unreachable!("Vote RPC should not be too large")
343            }
344            RPCTypes::AppendEntries => {
345                self.entries_hint = ReplicationHint::new(too_large.entries_hint(), DEFAULT_ENTRIES_HINT_TTL);
346                tracing::debug!(entries_hint = debug(&self.entries_hint), "updated entries hint");
347            }
348            RPCTypes::InstallSnapshot => {
349                // TODO: handle too large
350                tracing::error!("InstallSnapshot RPC is too large, but it is not supported yet");
351            }
352        }
353    }

355    /// Send an AppendEntries RPC to the target.
356    ///
357    /// This request will timeout if no response is received within the
358    /// configured heartbeat interval.
359    ///
360    /// If an RPC is made but not completely finished, it returns the next action expected to do.
361    #[tracing::instrument(level = "debug", skip_all)]
362    async fn send_log_entries(
363        &mut self,
364        log_ids: DataWithId<LogIdRange<C::NodeId>>,
365    ) -> Result<Option<Data<C>>, ReplicationError<C::NodeId, C::Node>> {
366        let request_id = log_ids.request_id();
367
368        tracing::debug!(
369            request_id = display(request_id),
370            log_id_range = display(log_ids.data()),
371            "send_log_entries",
372        );
373
374        // Series of logs to send, and the last log id to send
375        let (logs, sending_range) = {
376            let rng = log_ids.data();
377
378            // The log index start and end to send.
379            let (start, end) = {
380                let start = rng.prev.next_index();
381                let end = rng.last.next_index();
382
383                if let Some(hint) = self.entries_hint.get() {
384                    let hint_end = start + hint;
385                    (start, std::cmp::min(end, hint_end))
386                } else {
387                    (start, end)
388                }
389            };
390
391            if start == end {
392                // Heartbeat RPC, no logs to send, last log id is the same as prev_log_id
393                let r = LogIdRange::new(rng.prev.clone(), rng.prev.clone());
394                (vec![], r)
395            } else {
396                // limited_get_log_entries will return logs smaller than the range [start, end).
397                let logs = self.log_reader.limited_get_log_entries(start, end).await?;
398
399                let first = logs.first().map(|x| x.get_log_id().clone()).unwrap();
400                let last = logs.last().map(|x| x.get_log_id().clone()).unwrap();
401
402                debug_assert!(
403                    !logs.is_empty() && logs.len() <= (end - start) as usize,
404                    "expect logs âŠ† [{}..{}) but got {} entries, first: {}, last: {}",
405                    start,
406                    end,
407                    logs.len(),
408                    first,
409                    last
410                );
411
412                let r = LogIdRange::new(rng.prev.clone(), Some(last));
413                (logs, r)
414            }
415        };
416
417        let leader_time = C::now();
418
419        // Build the heartbeat frame to be sent to the follower.
420        let payload = AppendEntriesRequest {
421            vote: self.session_id.vote_ref().clone(),
422            prev_log_id: sending_range.prev.clone(),
423            leader_commit: self.committed.clone(),
424            entries: logs,
425        };
426
427        // Send the payload.
428        tracing::debug!(
429            payload=%payload.summary(),
430            now = debug(leader_time),
431            "start sending append_entries, timeout: {:?}",
432            self.config.heartbeat_interval
433        );
434
435        let the_timeout = Duration::from_millis(self.config.heartbeat_interval);
436        let option = RPCOption::new(the_timeout);
437        let res = C::timeout(the_timeout, self.network.append_entries(payload, option)).await;
438
439        tracing::debug!("append_entries res: {:?}", res);
440
441        let append_res = res.map_err(|_e| {
442            let to = Timeout {
443                action: RPCTypes::AppendEntries,
444                id: self.session_id.vote_ref().leader_id().voted_for().unwrap(),
445                target: self.target.clone(),
446                timeout: the_timeout,
447            };
448            RPCError::Timeout(to)
449        })?; // return Timeout error
450
451        let append_resp = DecomposeResult::<C, _, _>::decompose_infallible(append_res)?;
452
453        tracing::debug!(
454            req = display(&sending_range),
455            resp = display(&append_resp),
456            "append_entries resp"
457        );
458
459        match append_resp {
460            AppendEntriesResponse::Success => {
461                let matching = sending_range.last;
462                let next = self.finish_success_append(matching, leader_time, log_ids);
463                Ok(next)
464            }
465            AppendEntriesResponse::PartialSuccess(matching) => {
466                Self::debug_assert_partial_success(&sending_range, &matching);
467                let next = self.finish_success_append(matching, leader_time, log_ids);
468                Ok(next)
469            }
470            AppendEntriesResponse::HigherVote(vote) => {
471                debug_assert!(
472                    &vote > self.session_id.vote_ref(),
473                    "higher vote({}) should be greater than leader's vote({})",
474                    vote,
475                    self.session_id.vote_ref(),
476                );
477                tracing::debug!(%vote, "append entries failed. converting to follower");
478
479                Err(ReplicationError::HigherVote(HigherVote {
480                    higher: vote,
481                    sender_vote: self.session_id.vote_ref().clone(),
482                }))
483            }
484            AppendEntriesResponse::Conflict => {
485                let conflict = sending_range.prev.clone();
486                debug_assert!(conflict.is_some(), "prev_log_id=None never conflict");
487
488                let conflict = conflict.unwrap();
489                self.send_progress(request_id, ReplicationResult::new(leader_time, Err(conflict)));
490
491                Ok(None)
492            }
493        }
494    }

509    /// Send the success replication result(log matching or conflict) to RaftCore.
510    fn send_progress(&mut self, request_id: RequestId, replication_result: ReplicationResult<C>) {
511        tracing::debug!(
512            request_id = display(request_id),
513            target = display(&self.target),
514            curr_matching = display(self.matching.display()),
515            result = display(&replication_result),
516            "{}",
517            func_name!()

519
520        match &replication_result.result {
521            Ok(matching) => {
522                self.validate_matching(matching.clone());
523                self.matching = matching.clone();
524            }
525            Err(_conflict) => {
526                // Conflict is not allowed to be less than the current matching.
527            }
528        }
529
530        let _ = self.tx_raft_core.send({
531            Notify::Network {
532                response: Response::Progress {
533                    session_id: self.session_id.clone(),
534                    request_id,
535                    target: self.target.clone(),
536                    result: Ok(replication_result),
537                },
538            }
539        });

542    /// Validate the value for updating matching log id.
543    ///
544    /// If the matching log id is reverted to a smaller value:
545    /// - log a warning message if [`loosen-follower-log-revert`] feature flag is enabled;
546    /// - otherwise panic, consider it as a bug.
547    ///
548    /// [`loosen-follower-log-revert`]: crate::docs::feature_flags#feature-flag-loosen-follower-log-revert
549    fn validate_matching(&self, matching: Option<LogId<C::NodeId>>) {
550        if cfg!(feature = "loosen-follower-log-revert") {
551            if self.matching > matching {
552                tracing::warn!(
553                    "follower log is reverted from {} to {}; with 'loosen-follower-log-revert' enabled, this is allowed",
554                    self.matching.display(),
555                    matching.display(),
556                );
557            }
558        } else {
559            debug_assert!(
560                self.matching <= matching,
561                "follower log is reverted from {} to {}",
562                self.matching.display(),
563                matching.display(),
564            );
565        }
566    }

568    /// Drain all events in the channel in backoff mode, i.e., there was an un-retry-able error and
569    /// should not send out anything before backoff interval expired.
570    ///
571    /// In the backoff period, we should not send out any RPCs, but we should still receive events,
572    /// in case the channel is closed, it should quit at once.
573    #[tracing::instrument(level = "debug", skip(self))]
574    pub async fn backoff_drain_events(&mut self, until: InstantOf<C>) -> Result<(), ReplicationClosed> {
575        let d = until - C::now();
576        tracing::warn!(
577            interval = debug(d),
578            "{} backoff mode: drain events without processing them",
579            func_name!()
580        );
581
582        loop {
583            let sleep_duration = until - C::now();
584            let sleep = C::sleep(sleep_duration);
585
586            let recv = self.rx_event.recv();
587
588            tracing::debug!("backoff timeout: {:?}", sleep_duration);
589
590            select! {
591                _ = sleep => {
592                    tracing::debug!("backoff timeout");
593                    return Ok(());
594                }
595                recv_res = recv => {
596                    let event = recv_res.ok_or(ReplicationClosed::new("RaftCore closed replication"))?;
597                    self.process_event(event);
598                }
599            }
600        }
601    }

603    /// Receive and process events from RaftCore, until `next_action` is filled.
604    ///
605    /// It blocks until at least one event is received.
606    #[tracing::instrument(level = "trace", skip_all)]
607    pub async fn drain_events(&mut self) -> Result<(), ReplicationClosed> {
608        tracing::debug!("drain_events");
609
610        // If there is next action to run, do not block waiting for events,
611        // instead, just try the best to drain all events.
612        if self.next_action.is_none() {
613            let event =
614                self.rx_event.recv().await.ok_or(ReplicationClosed::new("rx_repl is closed in drain_event()"))?;
615            self.process_event(event);
616        }
617
618        // Returning from process_event(), next_action is never None.
619
620        self.try_drain_events().await?;
621
622        Ok(())
623    }

625    #[tracing::instrument(level = "trace", skip(self))]
626    pub async fn try_drain_events(&mut self) -> Result<(), ReplicationClosed> {
627        tracing::debug!("{}", func_name!());
628
629        // Just drain all events in the channel.
630        // There should NOT be more than one `Replicate::Data` event in the channel.
631        // Looping it just collect all commit events and heartbeat events.
632        loop {
633            let maybe_res = self.rx_event.recv().now_or_never();
634
635            let Some(recv_res) = maybe_res else {
636                // No more event found in self.repl_rx
637                return Ok(());
638            };
639
640            let event = recv_res.ok_or(ReplicationClosed::new("rx_repl is closed in try_drain_event"))?;
641
642            self.process_event(event);
643        }
644    }

646    #[tracing::instrument(level = "trace", skip_all)]
647    pub fn process_event(&mut self, event: Replicate<C>) {
648        tracing::debug!(event=%event.summary(), "process_event");
649
650        match event {
651            Replicate::Committed(c) => {
652                // RaftCore may send a committed equals to the initial value.
653                debug_assert!(
654                    c >= self.committed,
655                    "expect new committed {} > self.committed {}",
656                    c.summary(),
657                    self.committed.summary()
658                );
659
660                self.committed = c;
661
662                // If there is no action, fill in an heartbeat action to send committed index.
663                if self.next_action.is_none() {
664                    self.next_action = Some(Data::new_heartbeat());
665                }
666            }
667            Replicate::Heartbeat => {
668                // Never overwrite action with payload.
669                if self.next_action.is_none() {
670                    self.next_action = Some(Data::new_heartbeat());
671                }
672            }
673            Replicate::Data(d) => {
674                // TODO: Currently there is at most 1 in flight data. But in future RaftCore may send next data
675                //       actions without waiting for the previous to finish.
676                debug_assert!(
677                    !self.next_action.as_ref().map(|d| d.has_payload()).unwrap_or(false),
678                    "there can not be two actions with payload in flight, curr: {}",
679                    self.next_action.as_ref().map(|d| d.to_string()).display()
680                );
681
682                if cfg!(debug_assertions) {
683                    match &d {
684                        Data::SnapshotCallback(_) => {
685                            debug_assert!(
686                                self.snapshot_state.is_some(),
687                                "snapshot state must be Some to receive callback"
688                            );
689                        }
690                        _ => {
691                            debug_assert!(
692                                self.snapshot_state.is_none(),
693                                "can not send other data while sending snapshot"
694                            );
695                        }
696                    }
697                }
698
699                self.next_action = Some(d);
700            }
701        }
702    }

704    #[tracing::instrument(level = "info", skip_all)]
705    async fn stream_snapshot(
706        &mut self,
707        snapshot_req: DataWithId<Option<LogIdOf<C>>>,
708    ) -> Result<Option<Data<C>>, ReplicationError<C::NodeId, C::Node>> {
709        let request_id = snapshot_req.request_id();
710
711        tracing::info!(request_id = display(request_id), "{}", func_name!());
712
713        let snapshot = self.snapshot_reader.get_snapshot().await.map_err(|reason| {
714            tracing::warn!(error = display(&reason), "failed to get snapshot from state machine");
715            ReplicationClosed::new(reason)
716        })?;
717
718        tracing::info!(
719            "received snapshot: request_id={}; meta:{}",
720            request_id,
721            snapshot.as_ref().map(|x| &x.meta).summary()
722        );
723
724        let snapshot = match snapshot {
725            None => {
726                let io_err = StorageIOError::read_snapshot(None, AnyError::error("snapshot not found"));
727                let sto_err = StorageError::IO { source: io_err };
728                return Err(ReplicationError::StorageError(sto_err));
729            }
730            Some(x) => x,
731        };
732
733        let mut option = RPCOption::new(self.config.install_snapshot_timeout());
734        option.snapshot_chunk_size = Some(self.config.snapshot_max_chunk_size as usize);
735
736        let (tx_cancel, rx_cancel) = oneshot::channel();
737
738        let jh = C::spawn(Self::send_snapshot(
739            request_id,
740            self.snapshot_network.clone(),
741            self.session_id.vote_ref().clone(),
742            snapshot,
743            option,
744            rx_cancel,
745            self.weak_tx_event.clone(),
746        ));
747
748        // When self.rx_event is dropped:
749        // 1) ReplicationCore will return from the main loop;
750        // 2) and tx_cancel is dropped;
751        // 3) and the snapshot task will be notified.
752        self.snapshot_state = Some((tx_cancel, jh));
753        Ok(None)
754    }
755
756    async fn send_snapshot(
757        request_id: RequestId,
758        network: Arc<Mutex<N::Network>>,
759        vote: Vote<C::NodeId>,
760        snapshot: Snapshot<C>,
761        option: RPCOption,
762        cancel: oneshot::Receiver<()>,
763        weak_tx: mpsc::WeakUnboundedSender<Replicate<C>>,
764    ) {
765        let meta = snapshot.meta.clone();
766
767        let mut net = network.lock().await;
768
769        let start_time = C::now();
770
771        let cancel = async move {
772            let _ = cancel.await;
773            ReplicationClosed::new("ReplicationCore is dropped")
774        };
775
776        let res = net.full_snapshot(vote, snapshot, cancel, option).await;
777        if let Err(e) = &res {
778            tracing::warn!(error = display(e), "failed to send snapshot");
779        }
780
781        let res = res.decompose_infallible();
782
783        if let Some(tx_noty) = weak_tx.upgrade() {
784            let data = Data::new_snapshot_callback(request_id, start_time, meta, res);
785            let send_res = tx_noty.send(Replicate::new_data(data));
786            if send_res.is_err() {
787                tracing::warn!("weak_tx failed to send snapshot result to ReplicationCore");
788            }
789        } else {
790            tracing::warn!("weak_tx is dropped, no response is sent to ReplicationCore");
791        }
792    }

794    fn handle_snapshot_callback(
795        &mut self,
796        callback: DataWithId<SnapshotCallback<C>>,
797    ) -> Result<Option<Data<C>>, ReplicationError<C::NodeId, C::Node>> {
798        tracing::debug!(
799            request_id = debug(callback.request_id()),
800            response = display(callback.data()),
801            matching = display(self.matching.display()),
802            "handle_snapshot_response"
803        );
804
805        self.snapshot_state = None;
806
807        let request_id = callback.request_id();
808        let SnapshotCallback {
809            start_time,
810            result,
811            snapshot_meta,
812        } = callback.into_data();
813
814        let resp = result?;
815
816        // Handle response conditions.
817        let sender_vote = self.session_id.vote_ref().clone();
818        if resp.vote > sender_vote {
819            return Err(ReplicationError::HigherVote(HigherVote {
820                higher: resp.vote,
821                sender_vote,
822            }));
823        }
824
825        self.send_progress(
826            request_id,
827            ReplicationResult::new(start_time, Ok(snapshot_meta.last_log_id)),
828        );
829
830        Ok(None)
831    }

833    /// Update matching and build a return value for a successful append-entries RPC.
834    ///
835    /// If there are more logs to send, it returns a new `Some(Data::Logs)` to send.
836    fn finish_success_append(
837        &mut self,
838        matching: Option<LogId<C::NodeId>>,
839        leader_time: InstantOf<C>,
840        log_ids: DataWithId<LogIdRange<C::NodeId>>,
841    ) -> Option<Data<C>> {
842        self.send_progress(
843            log_ids.request_id(),
844            ReplicationResult::new(leader_time, Ok(matching.clone())),
845        );
846
847        if matching < log_ids.data().last {
848            Some(Data::new_logs(
849                log_ids.request_id(),
850                LogIdRange::new(matching, log_ids.data().last.clone()),
851            ))
852        } else {
853            None
854        }
855    }

857    /// Check if partial success result(`matching`) is valid for a given log range to send.
858    fn debug_assert_partial_success(to_send: &LogIdRange<C::NodeId>, matching: &Option<LogId<C::NodeId>>) {
859        debug_assert!(
860            matching <= &to_send.last,
861            "matching ({}) should be <= last_log_id ({})",
862            matching.display(),
863            to_send.last.display()
864        );
865        debug_assert!(
866            matching.index() <= to_send.last.index(),
867            "matching.index ({}) should be <= last_log_id.index ({})",
868            matching.index().display(),
869            to_send.last.index().display()
870        );
871        debug_assert!(
872            matching >= &to_send.prev,
873            "matching ({}) should be >= prev_log_id ({})",
874            matching.display(),
875            to_send.prev.display()
876        );
877        debug_assert!(
878            matching.index() >= to_send.prev.index(),
879            "matching.index ({}) should be >= prev_log_id.index ({})",
880            matching.index().display(),
881            to_send.prev.index().display()
882        );
883    }
