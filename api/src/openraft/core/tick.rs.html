21/// Emit RaftMsg::Tick event at regular `interval`.
22pub(crate) struct Tick<C>
23where C: RaftTypeConfig
24{
25    interval: Duration,
26
27    tx: mpsc::UnboundedSender<Notify<C>>,
28
29    /// Emit event or not
30    enabled: Arc<AtomicBool>,
31}
32
33pub(crate) struct TickHandle<C>
34where C: RaftTypeConfig
35{
36    enabled: Arc<AtomicBool>,
37    shutdown: Mutex<Option<oneshot::Sender<()>>>,
38    join_handle: Mutex<Option<JoinHandleOf<C, ()>>>,
39}
40
41impl<C> Drop for TickHandle<C>
42where C: RaftTypeConfig
43{
44    /// Signal the tick loop to stop, without waiting for it to stop.
45    fn drop(&mut self) {
46        if self.shutdown.lock().unwrap().is_none() {
47            return;
48        }
49        let _ = self.shutdown();
50    }
51}
52
53impl<C> Tick<C>
54where C: RaftTypeConfig
55{
56    pub(crate) fn spawn(interval: Duration, tx: mpsc::UnboundedSender<Notify<C>>, enabled: bool) -> TickHandle<C> {
57        let enabled = Arc::new(AtomicBool::from(enabled));
58        let this = Self {
59            interval,
60            enabled: enabled.clone(),
61            tx,
62        };
63
64        let (shutdown, shutdown_rx) = oneshot::channel();
65
66        let shutdown = Mutex::new(Some(shutdown));
67
68        let join_handle = C::spawn(this.tick_loop(shutdown_rx).instrument(tracing::span!(
69            parent: &Span::current(),
70            Level::DEBUG,
71            "tick"
72        )));
73
74        TickHandle {
75            enabled,
76            shutdown,
77            join_handle: Mutex::new(Some(join_handle)),
78        }
79    }
80
81    pub(crate) async fn tick_loop(self, mut cancel_rx: oneshot::Receiver<()>) {
82        let mut i = 0;
83
84        let mut cancel = std::pin::pin!(cancel_rx);
85
86        loop {
87            let at = C::now() + self.interval;
88            let mut sleep_fut = C::sleep_until(at);
89            let sleep_fut = std::pin::pin!(sleep_fut);
90            let cancel_fut = cancel.as_mut();
91
92            match futures::future::select(cancel_fut, sleep_fut).await {
93                Either::Left((_canceled, _)) => {
94                    tracing::info!("TickLoop received cancel signal, quit");
95                    return;
96                }
97                Either::Right((_, _)) => {
98                    // sleep done
99                }
100            }
101
102            if !self.enabled.load(Ordering::Relaxed) {
103                continue;
104            }
105
106            i += 1;
107
108            let send_res = self.tx.send(Notify::Tick { i });
109            if let Err(_e) = send_res {
110                tracing::info!("Stopping tick_loop(), main loop terminated");
111                break;
112            } else {
113                tracing::debug!("Tick sent: {}", i)
114            }
115        }
116    }
117}
118
119impl<C> TickHandle<C>
120where C: RaftTypeConfig
121{
122    pub(crate) fn enable(&self, enabled: bool) {
123        self.enabled.store(enabled, Ordering::Relaxed);
124    }
125
126    /// Signal the tick loop to stop. And return a JoinHandle to wait for the loop to stop.
127    ///
128    /// If it is called twice, the second call will return None.
129    pub(crate) fn shutdown(&self) -> Option<JoinHandleOf<C, ()>> {
130        {
131            let shutdown = {
132                let mut x = self.shutdown.lock().unwrap();
133                x.take()
134            };
135
136            if let Some(shutdown) = shutdown {
137                let send_res = shutdown.send(());
138                tracing::info!("Timer shutdown signal sent: {send_res:?}");
139            } else {
140                tracing::warn!("Double call to Raft::shutdown()");
141            }
142        }
143
144        let jh = {
145            let mut x = self.join_handle.lock().unwrap();
146            x.take()
147        };
148        jh
149    }
150}

177    // AsyncRuntime::spawn is `spawn_local` with singlethreaded enabled.
178    // It will result in a panic:
179    // `spawn_local` called from outside of a `task::LocalSet`.
180    #[cfg(not(feature = "singlethreaded"))]
181    #[tokio::test]
182    async fn test_shutdown() -> anyhow::Result<()> {
183        let (tx, mut rx) = tokio::sync::mpsc::unbounded_channel();
184        let th = Tick::<TickUTConfig>::spawn(Duration::from_millis(100), tx, true);
185
186        TickUTConfig::sleep(Duration::from_millis(500)).await;
187        let _ = th.shutdown().unwrap().await;
188        TickUTConfig::sleep(Duration::from_millis(500)).await;
189
190        let mut received = vec![];
191        while let Some(x) = rx.recv().await {
192            received.push(x);
193        }
194
195        assert!(
196            received.len() < 10,
197            "no more tick will be received after shutdown: {}",
198            received.len()
199        );
200
201        Ok(())
202    }
