51impl<C> RaftInner<C>
52where C: RaftTypeConfig
53{
54    /// Send a RaftMsg to RaftCore
55    pub(crate) async fn send_msg(&self, mes: RaftMsg<C>) -> Result<(), Fatal<C::NodeId>> {
56        let send_res = self.tx_api.send(mes);
57
58        if let Err(e) = send_res {
59            let fatal = self.get_core_stopped_error("sending RaftMsg to RaftCore", Some(e.0.summary())).await;
60            return Err(fatal);
61        }
62        Ok(())
63    }
64
65    /// Receive a message from RaftCore, return error if RaftCore has stopped.
66    pub(crate) async fn recv_msg<T, E>(&self, rx: impl Future<Output = Result<T, E>>) -> Result<T, Fatal<C::NodeId>>
67    where
68        T: OptionalSend,
69        E: OptionalSend,
70    {
71        let recv_res = rx.await;
72        tracing::debug!("{} receives result is error: {:?}", func_name!(), recv_res.is_err());
73
74        match recv_res {
75            Ok(x) => Ok(x),
76            Err(_) => {
77                let fatal = self.get_core_stopped_error("receiving rx from RaftCore", None::<&'static str>).await;
78                tracing::error!(error = debug(&fatal), "error when {}", func_name!());
79                Err(fatal)
80            }
81        }
82    }
83
84    /// Invoke RaftCore by sending a RaftMsg and blocks waiting for response.
85    #[tracing::instrument(level = "debug", skip_all)]
86    pub(crate) async fn call_core<T, E>(
87        &self,
88        mes: RaftMsg<C>,
89        rx: <C::AsyncRuntime as AsyncRuntime>::OneshotReceiver<Result<T, E>>,
90    ) -> Result<T, RaftError<C::NodeId, E>>
91    where
92        E: Debug + OptionalSend,
93        T: OptionalSend,
94    {
95        let sum = if tracing::enabled!(Level::DEBUG) {
96            Some(mes.summary())
97        } else {
98            None
99        };
100
101        self.send_msg(mes).await?;
102
103        let recv_res = rx.await;
104        tracing::debug!("call_core receives result is error: {:?}", recv_res.is_err());
105
106        match recv_res {
107            Ok(x) => x.map_err(|e| RaftError::APIError(e)),
108            Err(_) => {
109                let fatal = self.get_core_stopped_error("receiving rx from RaftCore", sum).await;
110                tracing::error!(error = debug(&fatal), "core_call fatal error");
111                Err(RaftError::Fatal(fatal))
112            }
113        }
114    }

116    /// Send an [`ExternalCommand`] to RaftCore to execute in the `RaftCore` thread.
117    ///
118    /// It returns at once.
119    pub(in crate::raft) async fn send_external_command(
120        &self,
121        cmd: ExternalCommand<C>,
122        cmd_desc: impl fmt::Display + Default,
123    ) -> Result<(), Fatal<C::NodeId>> {
124        let send_res = self.tx_api.send(RaftMsg::ExternalCommand { cmd });
125
126        if send_res.is_err() {
127            let fatal = self.get_core_stopped_error("sending external command to RaftCore", Some(cmd_desc)).await;
128            return Err(fatal);
129        }
130        Ok(())
131    }

133    /// Get the error that caused RaftCore to stop.
134    pub(in crate::raft) async fn get_core_stopped_error(
135        &self,
136        when: impl fmt::Display,
137        message_summary: Option<impl fmt::Display + Default>,
138    ) -> Fatal<C::NodeId> {
139        // Wait for the core task to finish.
140        self.join_core_task().await;
141
142        // Retrieve the result.
143        let core_res = {
144            let state = self.core_state.lock().await;
145            if let CoreState::Done(core_task_res) = &*state {
146                core_task_res.clone()
147            } else {
148                unreachable!("RaftCore should have already quit")
149            }
150        };
151
152        tracing::error!(
153            core_result = debug(&core_res),
154            "failure {}; message: {}",
155            when,
156            message_summary.unwrap_or_default()
157        );
158
159        // Safe unwrap: Infallible is unreachable
160        core_res.unwrap_err()
161    }
162
163    /// Wait for `RaftCore` task to finish and record the returned value from the task.
164    #[tracing::instrument(level = "debug", skip_all)]
165    pub(in crate::raft) async fn join_core_task(&self) {
166        let mut state = self.core_state.lock().await;
167        match &mut *state {
168            CoreState::Running(handle) => {
169                let res = handle.await;
170                tracing::info!(res = debug(&res), "RaftCore exited");
171
172                let core_task_res = match res {
173                    Err(err) => {
174                        if C::AsyncRuntime::is_panic(&err) {
175                            Err(Fatal::Panicked)
176                        } else {
177                            Err(Fatal::Stopped)
178                        }
179                    }
180                    Ok(returned_res) => returned_res,
181                };
182
183                *state = CoreState::Done(core_task_res);
184            }
185            CoreState::Done(_) => {
186                // RaftCore has already quit, nothing to do
187            }
188        }
189    }
