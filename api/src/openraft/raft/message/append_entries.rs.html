52/// The response to an `AppendEntriesRequest`.
53///
54/// [`RaftNetwork::append_entries`] returns this type only when received an RPC reply.
55/// Otherwise it should return [`RPCError`].
56///
57/// [`RPCError`]: crate::error::RPCError
58/// [`RaftNetwork::append_entries`]: crate::network::RaftNetwork::append_entries
59#[derive(Debug)]
60#[derive(PartialEq, Eq)]
61#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize), serde(bound = ""))]
62pub enum AppendEntriesResponse<NID: NodeId> {
63    /// Successfully replicated all log entries to the target node.
64    Success,
65
66    /// Successfully sent the first portion of log entries.
67    ///
68    /// [`RaftNetwork::append_entries`] can return a partial success.
69    /// For example, it tries to send log entries `[1-2..3-10]`, the application is allowed to send
70    /// just `[1-2..1-3]` and return `PartialSuccess(1-3)`,
71    ///
72    /// ### Caution
73    ///
74    /// The returned matching log id must be **greater than or equal to** the first log
75    /// id([`AppendEntriesRequest::prev_log_id`]) of the entries to send. If no RPC reply is
76    /// received, [`RaftNetwork::append_entries`] must return an [`RPCError`] to inform
77    /// Openraft that the first log id([`AppendEntriesRequest::prev_log_id`]) may not match on
78    /// the remote target node.
79    ///
80    /// [`RPCError`]: crate::error::RPCError
81    /// [`RaftNetwork::append_entries`]: crate::network::RaftNetwork::append_entries
82    PartialSuccess(Option<LogId<NID>>),
83
84    /// The first log id([`AppendEntriesRequest::prev_log_id`]) of the entries to send does not
85    /// match on the remote target node.
86    Conflict,
87
88    /// Seen a vote `v` that does not hold `mine_vote >= v`.
89    /// And a leader's vote(committed vote) must be total order with other vote.
90    /// Therefore it has to be a higher vote: `mine_vote < v`
91    HigherVote(Vote<NID>),
92}
93
94impl<NID: NodeId> AppendEntriesResponse<NID> {
95    pub fn is_success(&self) -> bool {
96        matches!(*self, AppendEntriesResponse::Success)
97    }
98
99    pub fn is_conflict(&self) -> bool {
100        matches!(*self, AppendEntriesResponse::Conflict)
101    }
102}
103
104impl<NID: NodeId> fmt::Display for AppendEntriesResponse<NID> {
105    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
106        match self {
107            AppendEntriesResponse::Success => write!(f, "Success"),
108            AppendEntriesResponse::PartialSuccess(m) => {
109                write!(f, "PartialSuccess({})", m.display())
110            }
111            AppendEntriesResponse::HigherVote(vote) => write!(f, "Higher vote, {}", vote),
112            AppendEntriesResponse::Conflict => write!(f, "Conflict"),
113        }
114    }
115}
