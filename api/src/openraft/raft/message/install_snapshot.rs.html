9/// An RPC sent by the Raft leader to send chunks of a snapshot to a follower (ยง7).
10#[derive(Clone, Debug)]
11#[derive(PartialEq, Eq)]
12#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize), serde(bound = ""))]
13pub struct InstallSnapshotRequest<C: RaftTypeConfig> {
14    pub vote: Vote<C::NodeId>,
15
16    /// Metadata of a snapshot: snapshot_id, last_log_ed membership etc.
17    pub meta: SnapshotMeta<C::NodeId, C::Node>,
18
19    /// The byte offset where this chunk of data is positioned in the snapshot file.
20    pub offset: u64,
21    /// The raw bytes of the snapshot chunk, starting at `offset`.
22    pub data: Vec<u8>,
23
24    /// Will be `true` if this is the last chunk in the snapshot.
25    pub done: bool,
26}
27
28impl<C: RaftTypeConfig> fmt::Display for InstallSnapshotRequest<C> {
29    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
30        write!(
31            f,
32            "InstallSnapshotRequest {{ vote:{}, meta:{}, offset:{}, len:{}, done:{} }}",
33            self.vote,
34            self.meta,
35            self.offset,
36            self.data.len(),
37            self.done
38        )
39    }
40}
