14/// Candidate: voting state.
15#[derive(Clone, Debug)]
16#[derive(PartialEq, Eq)]
17pub(crate) struct Candidate<C, QS>
18where
19    C: RaftTypeConfig,
20    QS: QuorumSet<C::NodeId>,
21{
22    /// When the voting is started.
23    starting_time: InstantOf<C>,
24
25    /// The vote.
26    vote: Vote<C::NodeId>,
27
28    last_log_id: Option<LogId<C::NodeId>>,
29
30    /// Which nodes have granted the the vote at certain time point.
31    progress: VecProgress<C::NodeId, bool, bool, QS>,
32
33    quorum_set: QS,
34
35    learner_ids: Vec<C::NodeId>,
36}
37
38impl<C, QS> fmt::Display for Candidate<C, QS>
39where
40    C: RaftTypeConfig,
41    QS: QuorumSet<C::NodeId> + fmt::Debug + 'static,
42{
43    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
44        write!(
45            f,
46            "{{{}@{}, last_log_id:{} progress:{}}}",
47            self.vote,
48            self.starting_time.display(),
49            self.last_log_id.display(),
50            self.progress
51        )
52    }
53}
54
55impl<C, QS> Candidate<C, QS>
56where
57    C: RaftTypeConfig,
58    QS: QuorumSet<C::NodeId> + fmt::Debug + Clone + 'static,
59{
60    pub(crate) fn new(
61        starting_time: InstantOf<C>,
62        vote: Vote<C::NodeId>,
63        last_log_id: Option<LogId<C::NodeId>>,
64        quorum_set: QS,
65        learner_ids: impl IntoIterator<Item = C::NodeId>,
66    ) -> Self {
67        Self {
68            starting_time,
69            vote,
70            last_log_id,
71            progress: VecProgress::new(quorum_set.clone(), [], false),
72            quorum_set,
73            learner_ids: learner_ids.into_iter().collect::<Vec<_>>(),
74        }
75    }
76
77    pub(crate) fn vote_ref(&self) -> &Vote<C::NodeId> {
78        &self.vote
79    }
80
81    pub(crate) fn last_log_id(&self) -> Option<&LogId<C::NodeId>> {
82        self.last_log_id.as_ref()
83    }
84
85    pub(crate) fn progress(&self) -> &VecProgress<C::NodeId, bool, bool, QS> {
86        &self.progress
87    }
88
89    /// Grant the vote by a node.
90    pub(crate) fn grant_by(&mut self, target: &C::NodeId) -> bool {
91        let granted = *self.progress.update(target, true).expect("target not in quorum set");
92
93        tracing::info!(voting = display(&self), "{}", func_name!());
94
95        granted
96    }
97
98    /// Return the node ids that has granted this vote.
99    #[allow(dead_code)]
100    pub(crate) fn granters(&self) -> impl Iterator<Item = C::NodeId> + '_ {
101        self.progress().iter().filter(|(_, granted)| *granted).map(|(target, _)| target.clone())
102    }
103
104    pub(crate) fn into_leader(self) -> Leader<C, QS> {
105        // Mark the vote as committed, i.e., being granted and saved by a quorum.
106        let vote = {
107            let mut vote = self.vote_ref().clone();
108            debug_assert!(!vote.is_committed());
109            vote.commit();
110            vote
111        };
112
113        let last_leader_log_ids = self.last_log_id().cloned().into_iter().collect::<Vec<_>>();
114
115        Leader::new(vote, self.quorum_set.clone(), self.learner_ids, &last_leader_log_ids)
116    }
117}
