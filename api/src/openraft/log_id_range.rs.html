14/// A log id range of continuous series of log entries.
15///
16/// The range of log to send is left open right close: `(prev, last]`.
17#[derive(Clone, Copy, Debug)]
18#[derive(PartialEq, Eq)]
19pub(crate) struct LogIdRange<NID: NodeId> {
20    /// The prev log id before the first to send, exclusive.
21    pub(crate) prev: Option<LogId<NID>>,
22
23    /// The last log id to send, inclusive.
24    pub(crate) last: Option<LogId<NID>>,
25}
26
27impl<NID: NodeId> Display for LogIdRange<NID> {
28    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
29        write!(f, "({}, {}]", self.prev.display(), self.last.display())
30    }
31}
32
33impl<NID: NodeId> Validate for LogIdRange<NID> {
34    fn validate(&self) -> Result<(), Box<dyn Error>> {
35        validit::less_equal!(&self.prev, &self.last);
36        Ok(())
37    }
38}
39
40impl<NID: NodeId> LogIdRange<NID> {
41    pub(crate) fn new(prev: Option<LogId<NID>>, last: Option<LogId<NID>>) -> Self {
42        Self { prev, last }
43    }
44
45    #[allow(dead_code)]
46    pub(crate) fn len(&self) -> u64 {
47        self.last.next_index() - self.prev.next_index()
48    }
49}
