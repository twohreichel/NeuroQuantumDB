16/// Builds a log id, for testing purposes.
17pub fn log_id<NID: crate::NodeId>(term: u64, node_id: NID, index: u64) -> LogId<NID> {
18    LogId::<NID> {
19        leader_id: CommittedLeaderId::new(term, node_id),
20        index,
21    }
22}
23
24/// Create a blank log entry for test.
25pub fn blank_ent<C: RaftTypeConfig>(term: u64, node_id: C::NodeId, index: u64) -> crate::Entry<C> {
26    crate::Entry::<C>::new_blank(LogId::new(CommittedLeaderId::new(term, node_id), index))
27}
28
29/// Create a membership log entry without learner config for test.
30pub fn membership_ent<C: RaftTypeConfig>(
31    term: u64,
32    node_id: C::NodeId,
33    index: u64,
34    config: Vec<BTreeSet<C::NodeId>>,
35) -> crate::Entry<C> {
36    crate::Entry::new_membership(
37        LogId::new(CommittedLeaderId::new(term, node_id), index),
38        crate::Membership::new(config, None),
39    )
