5/// LeaderId is identifier of a `leader`.
6///
7/// In raft spec that in a term there is at most one leader, thus a `term` is enough to
8/// differentiate leaders. That is why raft uses `(term, index)` to uniquely identify a log
9/// entry.
10///
11/// Under this dirty simplification, a `Leader` is actually identified by `(term,
12/// voted_for:Option<node_id>)`.
13/// By introducing `LeaderId {term, node_id}`, things become easier to understand.
14#[derive(Debug, Default, Clone, Copy, PartialEq, Eq)]
15#[derive(PartialOrd, Ord)]
16#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize), serde(bound = ""))]
17pub struct LeaderId<NID>
18where NID: NodeId
19{
20    pub term: u64,
21    pub node_id: NID,
22}
23
24impl<NID: NodeId> LeaderId<NID> {
25    pub fn new(term: u64, node_id: NID) -> Self {
26        Self { term, node_id }
27    }
28
29    pub fn get_term(&self) -> u64 {
30        self.term
31    }
32
33    pub fn voted_for(&self) -> Option<NID> {
34        Some(self.node_id.clone())
35    }
36
37    #[allow(clippy::wrong_self_convention)]
38    pub(crate) fn to_committed(&self) -> CommittedLeaderId<NID> {
39        self.clone()
40    }
41
42    /// Return if it is the same leader as the committed leader id.
43    ///
44    /// A committed leader may have less info than a non-committed.
45    pub(crate) fn is_same_as_committed(&self, other: &CommittedLeaderId<NID>) -> bool {
46        self == other
47    }
48}
49
50impl<NID: NodeId> fmt::Display for LeaderId<NID> {
51    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
52        write!(f, "T{}-N{}", self.term, self.node_id)
53    }
54}
55
56/// The unique identifier of a leader that is already granted by a quorum in phase-1(voting).
57///
58/// [`CommittedLeaderId`] may contains less information than [`LeaderId`], because it implies the
59/// constraint that **a quorum has granted it**.
60///
61/// For a total order `LeaderId`, the [`CommittedLeaderId`] is the same.
62///
63/// For a partial order `LeaderId`, we know that all of the granted
64/// leader-id must be a total order set. Therefor once it is granted by a quorum, it only keeps the
65/// information that makes leader-ids a correct total order set, e.g., in standard raft, `voted_for:
66/// Option<node_id>` can be removed from `(term, voted_for)` once it is granted. This is why
67/// standard raft stores just a `term` in log entry.
68pub type CommittedLeaderId<NID> = LeaderId<NID>;

70#[cfg(test)]
71mod tests {
72    use crate::LeaderId;
73
74    #[cfg(feature = "serde")]
75    #[test]
76    fn test_committed_leader_id_serde() -> anyhow::Result<()> {
77        use crate::CommittedLeaderId;
78
79        let c = CommittedLeaderId::<u32>::new(5, 10);
80        let s = serde_json::to_string(&c)?;
81        assert_eq!(r#"{"term":5,"node_id":10}"#, s);
82
83        let c2: CommittedLeaderId<u32> = serde_json::from_str(&s)?;
84        assert_eq!(CommittedLeaderId::new(5, 10), c2);
85
86        Ok(())
87    }
88
89    #[test]
90    fn test_leader_id_partial_order() -> anyhow::Result<()> {
91        #[allow(clippy::redundant_closure)]
92        let lid = |term, node_id| LeaderId::<u64>::new(term, node_id);
93
94        // Compare term first
95        assert!(lid(2, 2) > lid(1, 2));
96        assert!(lid(1, 2) < lid(2, 2));
97
98        // Equal term
99        assert!(lid(2, 2) > lid(2, 1));
100        assert!(lid(2, 1) < lid(2, 2));
101
102        // Equal term, node_id
103        assert!(lid(2, 2) == lid(2, 2));
104        assert!(lid(2, 2) >= lid(2, 2));
105        assert!(lid(2, 2) <= lid(2, 2));
106
107        Ok(())
108    }
