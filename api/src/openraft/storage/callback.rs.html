14/// A oneshot callback for completion of log io operation.
15pub struct LogFlushed<C>
16where C: RaftTypeConfig
17{
18    log_io_id: LogIOId<C::NodeId>,
19    tx: OneshotSenderOf<C, Result<LogIOId<C::NodeId>, io::Error>>,
20}
21
22impl<C> LogFlushed<C>
23where C: RaftTypeConfig
24{
25    pub(crate) fn new(
26        log_io_id: LogIOId<C::NodeId>,
27        tx: OneshotSenderOf<C, Result<LogIOId<C::NodeId>, io::Error>>,
28    ) -> Self {
29        Self { log_io_id, tx }
30    }
31
32    /// Report log io completion event.
33    ///
34    /// It will be called when the log is successfully appended to the storage or an error occurs.
35    pub fn log_io_completed(self, result: Result<(), io::Error>) {
36        let res = if let Err(e) = result {
37            tracing::error!("LogFlush error: {}, while flushing upto {}", e, self.log_io_id);
38            self.tx.send(Err(e))
39        } else {
40            self.tx.send(Ok(self.log_io_id))
41        };
42
43        if let Err(e) = res {
44            tracing::error!("failed to send log io completion event: {:?}", e);
45        }
46    }
47}

49/// A oneshot callback for completion of applying logs to state machine.
50pub struct LogApplied<C>
51where C: RaftTypeConfig
52{
53    last_log_id: LogId<C::NodeId>,
54    tx: oneshot::Sender<Result<(LogId<C::NodeId>, Vec<C::R>), StorageIOError<C::NodeId>>>,
55}
56
57impl<C> LogApplied<C>
58where C: RaftTypeConfig
59{
60    #[allow(dead_code)]
61    pub(crate) fn new(
62        last_log_id: LogId<C::NodeId>,
63        tx: oneshot::Sender<Result<(LogId<C::NodeId>, Vec<C::R>), StorageIOError<C::NodeId>>>,
64    ) -> Self {
65        Self { last_log_id, tx }
66    }
67
68    /// Report apply io completion event.
69    ///
70    /// It will be called when the log is successfully applied to the state machine or an error
71    /// occurs.
72    pub fn completed(self, result: Result<Vec<C::R>, StorageIOError<C::NodeId>>) {
73        let res = match result {
74            Ok(x) => {
75                tracing::debug!("LogApplied upto {}", self.last_log_id);
76                let resp = (self.last_log_id.clone(), x);
77                self.tx.send(Ok(resp))
78            }
79            Err(e) => {
80                tracing::error!("LogApplied error: {}, while applying upto {}", e, self.last_log_id);
81                self.tx.send(Err(e))
82            }
83        };
84
85        if let Err(_e) = res {
86            tracing::error!("failed to send apply complete event, last_log_id: {}", self.last_log_id);
87        }
88    }
89}
