8/// A trait defining the interface for a Raft network factory to create connections between cluster
9/// members.
10///
11/// See the [network chapter of the guide](crate::docs::getting_started#4-implement-raftnetwork)
12/// for details and discussion on this trait and how to implement it.
13///
14/// Typically, the network implementation as such will be hidden behind a `Box<T>` or `Arc<T>` and
15/// this interface implemented on the `Box<T>` or `Arc<T>`.
16#[add_async_trait]
17pub trait RaftNetworkFactory<C>: OptionalSend + OptionalSync + 'static
18where C: RaftTypeConfig
19{
20    /// Actual type of the network handling a single connection.
21    type Network: RaftNetwork<C>;
22
23    /// Create a new network instance sending RPCs to the target node.
24    ///
25    /// This function should **not** create a connection but rather a client that will connect when
26    /// required. Therefore there is chance it will build a client that is unable to send out
27    /// anything, e.g., in case the Node network address is configured incorrectly. But this method
28    /// does not return an error because openraft can only ignore it.
29    ///
30    /// The method is intentionally async to give the implementation a chance to use asynchronous
31    /// sync primitives to serialize access to the common internal object, if needed.
32    async fn new_client(&mut self, target: C::NodeId, node: &C::Node) -> Self::Network;
33}
