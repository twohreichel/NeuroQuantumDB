24/// A trait defining the interface for a Raft network between cluster members.
25///
26/// See the [network chapter of the guide](crate::docs::getting_started#4-implement-raftnetwork)
27/// for details and discussion on this trait and how to implement it.
28///
29/// A single network instance is used to connect to a single target node. The network instance is
30/// constructed by the [`RaftNetworkFactory`](`crate::network::RaftNetworkFactory`).
31///
32/// [Ensure connection to correct node][correct-node]
33///
34/// [`RaftNetwork`]: crate::network::v1::RaftNetwork
35/// [correct-node]: `crate::docs::cluster_control::dynamic_membership#ensure-connection-to-the-correct-node`
36#[add_async_trait]
37pub trait RaftNetwork<C>: OptionalSend + OptionalSync + 'static
38where C: RaftTypeConfig
39{
40    /// Send an AppendEntries RPC to the target.
41    async fn append_entries(
42        &mut self,
43        rpc: AppendEntriesRequest<C>,
44        option: RPCOption,
45    ) -> Result<AppendEntriesResponse<C::NodeId>, RPCError<C::NodeId, C::Node, RaftError<C::NodeId>>>;
46
47    /// Send an InstallSnapshot RPC to the target.
48    #[cfg(feature = "generic-snapshot-data")]
49    #[deprecated(
50        since = "0.9.0",
51        note = "with `generic-snapshot-data` enabled, use `full_snapshot()` instead to send full snapshot"
52    )]
53    async fn install_snapshot(
54        &mut self,
55        _rpc: crate::raft::InstallSnapshotRequest<C>,
56        _option: RPCOption,
57    ) -> Result<
58        crate::raft::InstallSnapshotResponse<C::NodeId>,
59        RPCError<C::NodeId, C::Node, RaftError<C::NodeId, crate::error::InstallSnapshotError>>,
60    > {
61        unimplemented!()
62    }
63
64    /// Send an InstallSnapshot RPC to the target.
65    #[cfg(not(feature = "generic-snapshot-data"))]
66    async fn install_snapshot(
67        &mut self,
68        _rpc: crate::raft::InstallSnapshotRequest<C>,
69        _option: RPCOption,
70    ) -> Result<
71        crate::raft::InstallSnapshotResponse<C::NodeId>,
72        RPCError<C::NodeId, C::Node, RaftError<C::NodeId, crate::error::InstallSnapshotError>>,
73    >;
74
75    /// Send a RequestVote RPC to the target.
76    async fn vote(
77        &mut self,
78        rpc: VoteRequest<C::NodeId>,
79        option: RPCOption,
80    ) -> Result<VoteResponse<C::NodeId>, RPCError<C::NodeId, C::Node, RaftError<C::NodeId>>>;
81
82    /// Send a complete Snapshot to the target.
83    ///
84    /// This method is responsible to fragment the snapshot and send it to the target node.
85    /// Before returning from this method, the snapshot should be completely transmitted and
86    /// installed on the target node, or rejected because of `vote` being smaller than the
87    /// remote one.
88    ///
89    /// The default implementation just calls several `install_snapshot` RPC for each fragment.
90    ///
91    /// The `vote` is the leader vote which is used to check if the leader is still valid by a
92    /// follower.
93    /// When the follower finished receiving snapshot, it calls `Raft::install_full_snapshot()`
94    /// with this vote.
95    ///
96    /// `cancel` get `Ready` when the caller decides to cancel this snapshot transmission.
97    ///
98    /// To implement a more generic snapshot transmission, you can use the `generic-snapshot-data`
99    /// feature. Enabling this feature allows you to send any type of snapshot data.
100    /// See the [generic snapshot
101    /// data](crate::docs::feature_flags#feature-flag-generic-snapshot-data) chapter for
102    /// details.
103    #[cfg(feature = "generic-snapshot-data")]
104    async fn full_snapshot(
105        &mut self,
106        vote: Vote<C::NodeId>,
107        snapshot: Snapshot<C>,
108        cancel: impl Future<Output = ReplicationClosed> + OptionalSend + 'static,
109        option: RPCOption,
110    ) -> Result<SnapshotResponse<C::NodeId>, StreamingError<C, Fatal<C::NodeId>>>;
111
112    /// Send a complete Snapshot to the target.
113    ///
114    /// This method is responsible to fragment the snapshot and send it to the target node.
115    /// Before returning from this method, the snapshot should be completely transmitted and
116    /// installed on the target node, or rejected because of `vote` being smaller than the
117    /// remote one.
118    ///
119    /// The default implementation just calls several `install_snapshot` RPC for each fragment.
120    ///
121    /// The `vote` is the leader vote which is used to check if the leader is still valid by a
122    /// follower.
123    /// When the follower finished receiving snapshot, it calls `Raft::install_full_snapshot()`
124    /// with this vote.
125    ///
126    /// `cancel` get `Ready` when the caller decides to cancel this snapshot transmission.
127    ///
128    /// To implement a more generic snapshot transmission, you can use the `generic-snapshot-data`
129    /// feature. Enabling this feature allows you to send any type of snapshot data.
130    /// See the [generic snapshot
131    /// data](crate::docs::feature_flags#feature-flag-generic-snapshot-data) chapter for
132    /// details.
133    // If generic-snapshot-data disabled,
134    // provide a default implementation that relies on AsyncRead + AsyncSeek + Unpin
135    #[cfg(not(feature = "generic-snapshot-data"))]
136    async fn full_snapshot(
137        &mut self,
138        vote: Vote<C::NodeId>,
139        snapshot: Snapshot<C>,
140        cancel: impl Future<Output = ReplicationClosed> + OptionalSend + 'static,
141        option: RPCOption,
142    ) -> Result<SnapshotResponse<C::NodeId>, StreamingError<C, Fatal<C::NodeId>>> {
143        use crate::network::snapshot_transport::Chunked;
144        use crate::network::snapshot_transport::SnapshotTransport;
145
146        let resp = Chunked::send_snapshot(self, vote, snapshot, cancel, option).await?;
147        Ok(resp)
148    }
149
150    /// Build a backoff instance if the target node is temporarily(or permanently) unreachable.
151    ///
152    /// When a [`Unreachable`](`crate::error::Unreachable`) error is returned from the `Network`
153    /// methods, Openraft does not retry connecting to a node immediately. Instead, it sleeps
154    /// for a while and retries. The duration of the sleep is determined by the backoff
155    /// instance.
156    ///
157    /// The backoff is an infinite iterator that returns the ith sleep interval before the ith
158    /// retry. The returned instance will be dropped if a successful RPC is made.
159    ///
160    /// By default it returns a constant backoff of 500 ms.
161    fn backoff(&self) -> Backoff {
162        Backoff::new(std::iter::repeat(Duration::from_millis(500)))
163    }
