1use std::time::Duration;
2
3use crate::engine::time_state;
4use crate::AsyncRuntime;
5use crate::Config;
6use crate::NodeId;

9/// Config for Engine
10#[derive(Clone, Debug)]
11#[derive(PartialEq, Eq)]
12pub(crate) struct EngineConfig<NID: NodeId> {
13    /// The id of this node.
14    pub(crate) id: NID,
15
16    /// The snapshot policy to use for a Raft node.
17    pub(crate) snapshot_policy: SnapshotPolicy,
18
19    /// The maximum number of applied logs to keep before purging.
20    pub(crate) max_in_snapshot_log_to_keep: u64,
21
22    /// The minimal number of applied logs to purge in a batch.
23    pub(crate) purge_batch_size: u64,
24
25    /// The maximum number of entries per payload allowed to be transmitted during replication
26    pub(crate) max_payload_entries: u64,
27
28    pub(crate) timer_config: time_state::Config,
29}
30
31impl<NID: NodeId> Default for EngineConfig<NID> {
32    fn default() -> Self {
33        Self {
34            id: NID::default(),
35            snapshot_policy: SnapshotPolicy::LogsSinceLast(5000),
36            max_in_snapshot_log_to_keep: 1000,
37            purge_batch_size: 256,
38            max_payload_entries: 300,
39            timer_config: time_state::Config::default(),
40        }
41    }
42}
43
44impl<NID: NodeId> EngineConfig<NID> {
45    pub(crate) fn new<RT: AsyncRuntime>(id: NID, config: &Config) -> Self {
46        let election_timeout = Duration::from_millis(config.new_rand_election_timeout::<RT>());
47        Self {
48            id,
49            snapshot_policy: config.snapshot_policy.clone(),
50            max_in_snapshot_log_to_keep: config.max_in_snapshot_log_to_keep,
51            purge_batch_size: config.purge_batch_size,
52            max_payload_entries: config.max_payload_entries,
53            timer_config: time_state::Config {
54                election_timeout,
55                smaller_log_timeout: Duration::from_millis(config.election_timeout_max * 2),
56                leader_lease: Duration::from_millis(config.election_timeout_max),
57            },
58        }
59    }
60}
