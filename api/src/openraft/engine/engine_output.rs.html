1use std::collections::VecDeque;
2
3use crate::core::sm::CommandSeq;
4use crate::engine::Command;
5use crate::RaftTypeConfig;
6
7/// The entry of output from Engine to the runtime.
8#[derive(Debug, Default)]
9pub(crate) struct EngineOutput<C>
10where C: RaftTypeConfig
11{
12    /// A Engine level sequence number for identifying a command.
13    pub(crate) seq: CommandSeq,
14
15    /// Command queue that need to be executed by `RaftRuntime`.
16    pub(crate) commands: VecDeque<Command<C>>,
17}
18
19impl<C> EngineOutput<C>
20where C: RaftTypeConfig
21{
22    /// Generate the next command seq of an sm::Command.
23    pub(crate) fn next_sm_seq(&mut self) -> CommandSeq {
24        self.seq += 1;
25        self.seq
26    }
27
28    /// Get the last used sm::Command seq
29    pub(crate) fn last_sm_seq(&self) -> CommandSeq {
30        self.seq
31    }
32
33    pub(crate) fn new(command_buffer_size: usize) -> Self {
34        Self {
35            seq: 0,
36            commands: VecDeque::with_capacity(command_buffer_size),
37        }
38    }
39
40    /// Push a command to the queue.
41    pub(crate) fn push_command(&mut self, mut cmd: Command<C>) {
42        tracing::debug!("push command: {:?}", cmd);
43
44        match &mut cmd {
45            Command::StateMachine { command } => {
46                let seq = self.next_sm_seq();
47                tracing::debug!("next_seq: {}", seq);
48                command.set_seq(seq);
49            }
50            Command::BecomeLeader => {}
51            Command::QuitLeader => {}
52            Command::AppendInputEntries { .. } => {}
53            Command::ReplicateCommitted { .. } => {}
54            Command::Commit { .. } => {}
55            Command::Replicate { .. } => {}
56            Command::RebuildReplicationStreams { .. } => {}
57            Command::SaveVote { .. } => {}
58            Command::SendVote { .. } => {}
59            Command::PurgeLog { .. } => {}
60            Command::DeleteConflictLog { .. } => {}
61            Command::Respond { .. } => {}
62        }
63        self.commands.push_back(cmd)
64    }
65
66    /// Put back the command to the head of the queue.
67    ///
68    /// This will be used when the command is not ready to be executed.
69    pub(crate) fn postpone_command(&mut self, cmd: Command<C>) {
70        tracing::debug!("postpone command: {:?}", cmd);
71        self.commands.push_front(cmd)
72    }
73
74    /// Pop the first command to run from the queue.
75    pub(crate) fn pop_command(&mut self) -> Option<Command<C>> {
76        self.commands.pop_front()
77    }
78
79    /// Iterate all queued commands.
80    pub(crate) fn iter_commands(&self) -> impl Iterator<Item = &Command<C>> {
81        self.commands.iter()
82    }
83
84    /// Take all queued commands and clear the queue.
85    #[cfg(test)]
86    pub(crate) fn take_commands(&mut self) -> Vec<Command<C>> {
87        self.commands.drain(..).collect()
88    }
89
90    /// Clear all queued commands.
91    #[cfg(test)]
92    pub(crate) fn clear_commands(&mut self) {
93        self.commands.clear()
94    }
95}
