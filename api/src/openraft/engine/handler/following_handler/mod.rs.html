40/// Receive replication request and deal with them.
41///
42/// It mainly implements the logic of a follower/learner
43pub(crate) struct FollowingHandler<'x, C>
44where C: RaftTypeConfig
45{
46    pub(crate) config: &'x mut EngineConfig<C::NodeId>,
47    pub(crate) state: &'x mut RaftState<C::NodeId, C::Node, <C::AsyncRuntime as AsyncRuntime>::Instant>,
48    pub(crate) output: &'x mut EngineOutput<C>,
49}

51impl<C> FollowingHandler<'_, C>
52where C: RaftTypeConfig

54    /// Append entries to follower/learner.
55    ///
56    /// Also clean conflicting entries and update membership state.
57    #[tracing::instrument(level = "debug", skip_all)]
58    pub(crate) fn append_entries(&mut self, prev_log_id: Option<LogId<C::NodeId>>, entries: Vec<C::Entry>) {
59        tracing::debug!(
60            prev_log_id = display(prev_log_id.summary()),
61            entries = display(DisplaySlice::<_>(&entries)),
62            "append-entries request"
63        );
64        tracing::debug!(
65            my_last_log_id = display(self.state.last_log_id().summary()),
66            my_committed = display(self.state.committed().summary()),
67            "local state"
68        );
69
70        if let Some(x) = entries.first() {
71            debug_assert!(x.get_log_id().index == prev_log_id.next_index());
72        }
73
74        tracing::debug!(
75            committed = display(self.state.committed().summary()),
76            entries = display(DisplaySlice::<_>(&entries)),
77            "prev_log_id matches, skip matching entries",
78        );
79
80        let last_log_id = entries.last().map(|x| x.get_log_id().clone());
81
82        self.state.update_accepted(std::cmp::max(prev_log_id, last_log_id));
83
84        let l = entries.len();
85        let since = self.state.first_conflicting_index(&entries);
86        if since < l {
87            // Before appending, if an entry overrides an conflicting one,
88            // the entries after it has to be deleted first.
89            // Raft requires log ids are in total order by (term,index).
90            // Otherwise the log id with max index makes committed entry invisible in election.
91            self.truncate_logs(entries[since].get_log_id().index);
92        }
93
94        self.do_append_entries(entries, since);
95    }

97    /// Ensures the log to replicate is consecutive to the local log.
98    ///
99    /// If not, truncate the local log and return an error.
100    pub(crate) fn ensure_log_consecutive(
101        &mut self,
102        prev_log_id: Option<LogId<C::NodeId>>,
103    ) -> Result<(), RejectAppendEntries<C::NodeId>> {
104        if let Some(ref prev) = prev_log_id {
105            if !self.state.has_log_id(prev) {
106                let local = self.state.get_log_id(prev.index);
107                tracing::debug!(local = display(DisplayOption(&local)), "prev_log_id does not match");
108
109                self.truncate_logs(prev.index);
110                return Err(RejectAppendEntries::ByConflictingLogId {
111                    local,
112                    expect: prev.clone(),
113                });
114            }
115        }
116
117        // else `prev_log_id.is_none()` means replicating logs from the very beginning.
118
119        Ok(())
120    }

122    /// Follower/Learner appends `entries[since..]`.
123    ///
124    /// It assumes:
125    /// - Previous entries all match the leader's.
126    /// - conflicting entries are deleted.
127    ///
128    /// Membership config changes are also detected and applied here.
129    #[tracing::instrument(level = "debug", skip(self, entries))]
130    pub(crate) fn do_append_entries(&mut self, mut entries: Vec<C::Entry>, since: usize) {
131        let l = entries.len();
132
133        if since == l {
134            return;
135        }
136
137        let entries = entries.split_off(since);
138
139        debug_assert_eq!(
140            entries[0].get_log_id().index,
141            self.state.log_ids.last().cloned().next_index(),
142        );
143
144        debug_assert!(Some(entries[0].get_log_id()) > self.state.log_ids.last());
145
146        self.state.extend_log_ids(&entries);
147        self.append_membership(entries.iter());
148
149        self.output.push_command(Command::AppendInputEntries {
150            // A follower should always use the node's vote.
151            vote: self.state.vote_ref().clone(),
152            entries,
153        });
154    }

156    /// Commit entries that are already committed by the leader.
157    #[tracing::instrument(level = "debug", skip_all)]
158    pub(crate) fn commit_entries(&mut self, leader_committed: Option<LogId<C::NodeId>>) {
159        let accepted = self.state.accepted().cloned();
160        let committed = std::cmp::min(accepted.clone(), leader_committed.clone());
161
162        tracing::debug!(
163            leader_committed = display(DisplayOption(&leader_committed)),
164            accepted = display(DisplayOption(&accepted)),
165            committed = display(DisplayOption(&committed)),
166            "{}",
167            func_name!()
168        );
169
170        if let Some(prev_committed) = self.state.update_committed(&committed) {
171            let seq = self.output.next_sm_seq();
172            self.output.push_command(Command::Commit {
173                seq,
174                // TODO(xp): when restart, commit is reset to None. Use last_applied instead.
175                already_committed: prev_committed,
176                upto: committed.unwrap(),
177            });
178
179            if self.config.snapshot_policy.should_snapshot(&self.state) {
180                self.snapshot_handler().trigger_snapshot();
181            }
182        }
183    }

185    /// Delete log entries since log index `since`, inclusive, when the log at `since` is found
186    /// conflict with the leader.
187    ///
188    /// And revert effective membership to the last committed if it is from the conflicting logs.
189    #[tracing::instrument(level = "debug", skip(self))]
190    fn truncate_logs(&mut self, since: u64) {
191        tracing::debug!(since = since, "truncate_logs");
192
193        debug_assert!(since >= self.state.last_purged_log_id().next_index());
194
195        let since_log_id = match self.state.get_log_id(since) {
196            None => {
197                tracing::debug!("trying to delete absent log at: {}", since);
198                return;
199            }
200            Some(x) => x,
201        };
202
203        self.state.log_ids.truncate(since);
204        self.output.push_command(Command::DeleteConflictLog { since: since_log_id });
205
206        let changed = self.state.membership_state.truncate(since);
207        if let Some(_c) = changed {
208            self.server_state_handler().update_server_state_if_changed();
209        }
210    }

212    /// Append membership log if membership config entries are found, after appending entries to
213    /// log.
214    fn append_membership<'a>(&mut self, entries: impl DoubleEndedIterator<Item = &'a C::Entry>)
215    where C::Entry: 'a {
216        let memberships = Self::last_two_memberships(entries);
217        if memberships.is_empty() {
218            return;
219        }
220
221        // Update membership state with the last 2 membership configs found in new log entries.
222        // Other membership log can be just ignored.
223        for (i, m) in memberships.into_iter().enumerate() {
224            tracing::debug!(
225                last = display(m.summary()),
226                "applying {}-th new membership configs received from leader",
227                i
228            );
229            self.state.membership_state.append(Arc::new(EffectiveMembership::new_from_stored_membership(m)));
230        }
231
232        tracing::debug!(
233            membership_state = display(&self.state.membership_state.summary()),
234            "updated membership state"
235        );
236
237        self.server_state_handler().update_server_state_if_changed();
238    }

240    /// Update membership state with a committed membership config
241    #[tracing::instrument(level = "debug", skip_all)]
242    fn update_committed_membership(&mut self, membership: EffectiveMembership<C::NodeId, C::Node>) {
243        tracing::debug!("update committed membership: {}", membership.summary());
244
245        let m = Arc::new(membership);
246
247        // TODO: if effective membership changes, call `update_replication()`, if a follower has replication
248        //       streams. Now we don't have replication streams for follower, so it's ok to not call
249        //       `update_replication()`.
250        let _effective_changed = self.state.membership_state.update_committed(m);
251
252        self.server_state_handler().update_server_state_if_changed();
253    }

255    /// Installs a full snapshot on a follower or learner node.
256    ///
257    /// Refer to [`snapshot_replication`](crate::docs::protocol::replication::snapshot_replication)
258    /// for the reason the following workflow is needed.
259    ///
260    /// The method processes the given `snapshot` and updates the internal state of the node based
261    /// on the snapshot's metadata. It checks if the `snapshot` is newer than the currently
262    /// committed snapshot. If not, it does nothing.
263    ///
264    /// It returns the condition about when the snapshot is installed and can proceed the commands
265    /// that depends on the state of snapshot.
266    ///
267    /// It returns an `Option<Condition<C>>` indicating the next action:
268    /// - `Some(Condition::StateMachineCommand { command_seq })` if the snapshot will be installed.
269    ///   Further commands that depend on snapshot state should use this condition so that these
270    ///   command block until the condition is satisfied(`RaftCore` receives a `Notify`).
271    /// - Otherwise `None` if the snapshot will not be installed (e.g., if it is not newer than the
272    ///   current state).
273    #[tracing::instrument(level = "debug", skip_all)]
274    pub(crate) fn install_full_snapshot(&mut self, snapshot: Snapshot<C>) -> Option<Condition<C::NodeId>> {
275        let meta = &snapshot.meta;
276        tracing::info!("install_full_snapshot: meta:{:?}", meta);
277
278        let snap_last_log_id = meta.last_log_id.clone();
279
280        if snap_last_log_id.as_ref() <= self.state.committed() {
281            tracing::info!(
282                "No need to install snapshot; snapshot last_log_id({}) <= committed({})",
283                snap_last_log_id.summary(),
284                self.state.committed().summary()
285            );
286            return None;
287        }
288
289        // snapshot_last_log_id can not be None
290        let snap_last_log_id = snap_last_log_id.unwrap();
291
292        // 1. Truncate all logs if conflict.
293        // 2. Install snapshot.
294        // 3. Purge logs the snapshot covers.
295
296        let mut snap_handler = self.snapshot_handler();
297        let updated = snap_handler.update_snapshot(meta.clone());
298        if !updated {
299            return None;
300        }
301
302        let local = self.state.get_log_id(snap_last_log_id.index);
303        if let Some(local) = local {
304            if local != snap_last_log_id {
305                // Conflict, delete all non-committed logs.
306                self.truncate_logs(self.state.committed().next_index());
307            }
308        }
309
310        self.state.update_accepted(Some(snap_last_log_id.clone()));
311        self.state.committed = Some(snap_last_log_id.clone());
312        self.update_committed_membership(EffectiveMembership::new_from_stored_membership(
313            meta.last_membership.clone(),
314        ));
315
316        self.output.push_command(Command::from(sm::Command::install_full_snapshot(snapshot)));
317        let last_sm_seq = self.output.last_sm_seq();
318
319        self.state.purge_upto = Some(snap_last_log_id);
320        self.log_handler().purge_log();
321
322        Some(Condition::StateMachineCommand {
323            command_seq: last_sm_seq,
324        })
325    }

327    /// Find the last 2 membership entries in a list of entries.
328    ///
329    /// A follower/learner reverts the effective membership to the previous one,
330    /// when conflicting logs are found.
331    fn last_two_memberships<'a>(
332        entries: impl DoubleEndedIterator<Item = &'a C::Entry>,
333    ) -> Vec<StoredMembership<C::NodeId, C::Node>>
334    where C::Entry: 'a {
335        let mut memberships = vec![];
336
337        // Find the last 2 membership config entries: the committed and the effective.
338        for ent in entries.rev() {
339            if let Some(m) = ent.get_membership() {
340                memberships.insert(0, StoredMembership::new(Some(ent.get_log_id().clone()), m.clone()));
341                if memberships.len() == 2 {
342                    break;
343                }
344            }
345        }
346
347        memberships
348    }
