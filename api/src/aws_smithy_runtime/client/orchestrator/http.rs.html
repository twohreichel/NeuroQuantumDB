6use aws_smithy_runtime_api::client::orchestrator::{HttpResponse, SensitiveOutput};
7use aws_smithy_types::body::SdkBody;
8use aws_smithy_types::config_bag::ConfigBag;
9use bytes::Bytes;
10use pin_utils::pin_mut;
11use tracing::trace;
12
13const LOG_SENSITIVE_BODIES: &str = "LOG_SENSITIVE_BODIES";
14
15async fn body_to_bytes(body: SdkBody) -> Result<Bytes, <SdkBody as http_body_1x::Body>::Error> {
16    use http_body_util::BodyExt;
17    pin_mut!(body);
18    let collected = body.collect().await?;
19
20    Ok(collected.to_bytes())
21}
22
23pub(crate) async fn read_body(
24    response: &mut HttpResponse,
25) -> Result<(), <SdkBody as http_body_1x::Body>::Error> {
26    let mut body = SdkBody::taken();
27    std::mem::swap(&mut body, response.body_mut());
28
29    let bytes = body_to_bytes(body).await?;
30    let mut body = SdkBody::from(bytes);
31    std::mem::swap(&mut body, response.body_mut());
32
33    Ok(())
34}
35
36pub(crate) fn log_response_body(response: &HttpResponse, cfg: &ConfigBag) {
37    if cfg.load::<SensitiveOutput>().is_none()
38        || std::env::var(LOG_SENSITIVE_BODIES)
39            .map(|v| v.eq_ignore_ascii_case("true"))
40            .unwrap_or_default()
41    {
42        trace!(response = ?response, "read HTTP response body");
43    } else {
44        trace!(
45            response = "** REDACTED **. To print, set LOG_SENSITIVE_BODIES=true",
46            "read HTTP response body"
47        )
48    }
49}
