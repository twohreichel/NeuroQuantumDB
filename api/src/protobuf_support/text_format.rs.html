1pub fn escape_bytes_to(bytes: &[u8], buf: &mut String) {
2    for &c in bytes {
3        match c {
4            b'\n' => buf.push_str(r"\n"),
5            b'\r' => buf.push_str(r"\r"),
6            b'\t' => buf.push_str(r"\t"),
7            b'\'' => buf.push_str("\\\'"),
8            b'"' => buf.push_str("\\\""),
9            b'\\' => buf.push_str(r"\\"),
10            b'\x20'..=b'\x7e' => buf.push(c as char),
11            _ => {
12                buf.push('\\');
13                buf.push((b'0' + (c >> 6)) as char);
14                buf.push((b'0' + ((c >> 3) & 7)) as char);
15                buf.push((b'0' + (c & 7)) as char);
16            }
17        }
18    }
19}

21pub fn quote_bytes_to(bytes: &[u8], buf: &mut String) {
22    buf.push('"');
23    escape_bytes_to(bytes, buf);
24    buf.push('"');
25}
