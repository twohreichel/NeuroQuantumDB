1/*!
2
3Platform-independent error model.
4
5There is an escape hatch here for surfacing platform-specific
6error information returned by the platform-specific storage provider,
7but the concrete objects returned must be `Send` so they can be
8moved from one thread to another. (Since most platform errors
9are integer error codes, this requirement
10is not much of a burden on the platform-specific store providers.)
11 */
12
13use crate::Credential;

15#[derive(Debug)]
16/// Each variant of the `Error` enum provides a summary of the error.
17/// More details, if relevant, are contained in the associated value,
18/// which may be platform-specific.
19///
20/// This enum is non-exhaustive so that more values can be added to it
21/// without a SemVer break. Clients should always have default handling
22/// for variants they don't understand.
23#[non_exhaustive]
24pub enum Error {
25    /// This indicates runtime failure in the underlying
26    /// platform storage system.  The details of the failure can
27    /// be retrieved from the attached platform error.
28    PlatformFailure(Box<dyn std::error::Error + Send + Sync>),
29    /// This indicates that the underlying secure storage
30    /// holding saved items could not be accessed.  Typically this
31    /// is because of access rules in the platform; for example, it
32    /// might be that the credential store is locked.  The underlying
33    /// platform error will typically give the reason.
34    NoStorageAccess(Box<dyn std::error::Error + Send + Sync>),
35    /// This indicates that there is no underlying credential
36    /// entry in the platform for this entry.  Either one was
37    /// never set, or it was deleted.
38    NoEntry,
39    /// This indicates that the retrieved password blob was not
40    /// a UTF-8 string.  The underlying bytes are available
41    /// for examination in the attached value.
42    BadEncoding(Vec<u8>),
43    /// This indicates that one of the entry's credential
44    /// attributes exceeded a
45    /// length limit in the underlying platform.  The
46    /// attached values give the name of the attribute and
47    /// the platform length limit that was exceeded.
48    TooLong(String, u32),
49    /// This indicates that one of the entry's required credential
50    /// attributes was invalid.  The
51    /// attached value gives the name of the attribute
52    /// and the reason it's invalid.
53    Invalid(String, String),
54    /// This indicates that there is more than one credential found in the store
55    /// that matches the entry.  Its value is a vector of the matching credentials.
56    Ambiguous(Vec<Box<Credential>>),
57}
58
59pub type Result<T> = std::result::Result<T, Error>;
60
61impl std::fmt::Display for Error {
62    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
63        match self {
64            Error::PlatformFailure(err) => write!(f, "Platform secure storage failure: {err}"),
65            Error::NoStorageAccess(err) => {
66                write!(f, "Couldn't access platform secure storage: {err}")
67            }
68            Error::NoEntry => write!(f, "No matching entry found in secure storage"),
69            Error::BadEncoding(_) => write!(f, "Data is not UTF-8 encoded"),
70            Error::TooLong(name, len) => write!(
71                f,
72                "Attribute '{name}' is longer than platform limit of {len} chars"
73            ),
74            Error::Invalid(attr, reason) => {
75                write!(f, "Attribute {attr} is invalid: {reason}")
76            }
77            Error::Ambiguous(items) => {
78                write!(
79                    f,
80                    "Entry is matched by {} credentials: {items:?}",
81                    items.len(),
82                )
83            }
84        }
85    }
86}
87
88impl std::error::Error for Error {
89    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
90        match self {
91            Error::PlatformFailure(err) => Some(err.as_ref()),
92            Error::NoStorageAccess(err) => Some(err.as_ref()),
93            _ => None,
94        }
95    }
96}

98/// Try to interpret a byte vector as a password string
99pub fn decode_password(bytes: Vec<u8>) -> Result<String> {
100    String::from_utf8(bytes).map_err(|err| Error::BadEncoding(err.into_bytes()))
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    #[test]
108    fn test_bad_password() {
109        // malformed sequences here taken from:
110        // https://www.cl.cam.ac.uk/~mgk25/ucs/examples/UTF-8-test.txt
111        for bytes in [b"\x80".to_vec(), b"\xbf".to_vec(), b"\xed\xa0\xa0".to_vec()] {
112            match decode_password(bytes.clone()) {
113                Err(Error::BadEncoding(str)) => assert_eq!(str, bytes),
114                Err(other) => panic!("Bad password ({bytes:?}) decode gave wrong error: {other}"),
115                Ok(s) => panic!("Bad password ({bytes:?}) decode gave results: {s:?}"),
116            }
117        }
118    }
