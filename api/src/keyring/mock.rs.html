3# Mock credential store
4
5To facilitate testing of clients, this crate provides a Mock credential store
6that is platform-independent, provides no persistence, and allows the client
7to specify the return values (including errors) for each call. The credentials
8in this store have no attributes at all.
9
10To use this credential store instead of the default, make this call during
11application startup _before_ creating any entries:
12```rust
13# use keyring::{set_default_credential_builder, mock};
14set_default_credential_builder(mock::default_credential_builder());
15```
16
17You can then create entries as you usually do, and call their usual methods
18to set, get, and delete passwords.  There is no persistence other than
19in the entry itself, so getting a password before setting it will always result
20in a [NoEntry](Error::NoEntry) error.
21
22If you want a method call on an entry to fail in a specific way, you can
23downcast the entry to a [MockCredential] and then call [set_error](MockCredential::set_error)
24with the appropriate error.  The next entry method called on the credential
25will fail with the error you set.  The error will then be cleared, so the next
26call on the mock will operate as usual.  Here's a complete example:
27```rust
28# use keyring::{Entry, Error, mock, mock::MockCredential};
29# keyring::set_default_credential_builder(mock::default_credential_builder());
30let entry = Entry::new("service", "user").unwrap();
31let mock: &MockCredential = entry.get_credential().downcast_ref().unwrap();
32mock.set_error(Error::Invalid("mock error".to_string(), "takes precedence".to_string()));
33entry.set_password("test").expect_err("error will override");
34entry.set_password("test").expect("error has been cleared");
35```
36 */

45/// The concrete mock credential
46///
47/// Mocks use an internal mutability pattern since entries are read-only.
48/// The mutex is used to make sure these are Sync.
49#[derive(Debug)]
50pub struct MockCredential {
51    pub inner: Mutex<RefCell<MockData>>,
52}
53
54impl Default for MockCredential {
55    fn default() -> Self {
56        Self {
57            inner: Mutex::new(RefCell::new(Default::default())),
58        }
59    }
60}
61
62/// The (in-memory) persisted data for a mock credential.
63///
64/// We keep a password, but unlike most keystores
65/// we also keep an intended error to return on the next call.
66///
67/// (Everything about this structure is public for transparency.
68/// Most keystore implementation hide their internals.)
69#[derive(Debug, Default)]
70pub struct MockData {
71    pub secret: Option<Vec<u8>>,
72    pub error: Option<Error>,
73}
74
75impl CredentialApi for MockCredential {
76    /// Set a password on a mock credential.
77    ///
78    /// If there is an error in the mock, it will be returned
79    /// and the password will _not_ be set.  The error will
80    /// be cleared, so calling again will set the password.
81    fn set_password(&self, password: &str) -> Result<()> {
82        let mut inner = self.inner.lock().expect("Can't access mock data for set");
83        let data = inner.get_mut();
84        let err = data.error.take();
85        match err {
86            None => {
87                data.secret = Some(password.as_bytes().to_vec());
88                Ok(())
89            }
90            Some(err) => Err(err),
91        }
92    }
93
94    /// Set a password on a mock credential.
95    ///
96    /// If there is an error in the mock, it will be returned
97    /// and the password will _not_ be set.  The error will
98    /// be cleared, so calling again will set the password.
99    fn set_secret(&self, secret: &[u8]) -> Result<()> {
100        let mut inner = self.inner.lock().expect("Can't access mock data for set");
101        let data = inner.get_mut();
102        let err = data.error.take();
103        match err {
104            None => {
105                data.secret = Some(secret.to_vec());
106                Ok(())
107            }
108            Some(err) => Err(err),
109        }
110    }
111
112    /// Get the password from a mock credential, if any.
113    ///
114    /// If there is an error set in the mock, it will
115    /// be returned instead of a password.
116    fn get_password(&self) -> Result<String> {
117        let mut inner = self.inner.lock().expect("Can't access mock data for get");
118        let data = inner.get_mut();
119        let err = data.error.take();
120        match err {
121            None => match &data.secret {
122                None => Err(Error::NoEntry),
123                Some(val) => decode_password(val.clone()),
124            },
125            Some(err) => Err(err),
126        }
127    }
128
129    /// Get the password from a mock credential, if any.
130    ///
131    /// If there is an error set in the mock, it will
132    /// be returned instead of a password.
133    fn get_secret(&self) -> Result<Vec<u8>> {
134        let mut inner = self.inner.lock().expect("Can't access mock data for get");
135        let data = inner.get_mut();
136        let err = data.error.take();
137        match err {
138            None => match &data.secret {
139                None => Err(Error::NoEntry),
140                Some(val) => Ok(val.clone()),
141            },
142            Some(err) => Err(err),
143        }
144    }
145
146    /// Delete the password in a mock credential
147    ///
148    /// If there is an error, it will be returned and
149    /// the deletion will not happen.
150    ///
151    /// If there is no password, a [NoEntry](Error::NoEntry) error
152    /// will be returned.
153    fn delete_credential(&self) -> Result<()> {
154        let mut inner = self
155            .inner
156            .lock()
157            .expect("Can't access mock data for delete");
158        let data = inner.get_mut();
159        let err = data.error.take();
160        match err {
161            None => match data.secret {
162                Some(_) => {
163                    data.secret = None;
164                    Ok(())
165                }
166                None => Err(Error::NoEntry),
167            },
168            Some(err) => Err(err),
169        }
170    }
171
172    /// Return this mock credential concrete object
173    /// wrapped in the [Any](std::any::Any) trait,
174    /// so it can be downcast.
175    fn as_any(&self) -> &dyn std::any::Any {
176        self
177    }
178
179    /// Expose the concrete debug formatter for use via the [Credential] trait
180    fn debug_fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
181        std::fmt::Debug::fmt(self, f)
182    }
183}

185impl MockCredential {
186    /// Make a new mock credential.
187    ///
188    /// Since mocks have no persistence between sessions,
189    /// new mocks always have no password.
190    fn new_with_target(_target: Option<&str>, _service: &str, _user: &str) -> Result<Self> {
191        Ok(Default::default())
192    }
193
194    /// Set an error to be returned from this mock credential.
195    ///
196    /// Error returns always take precedence over the normal
197    /// behavior of the mock.  But once an error has been
198    /// returned it is removed, so the mock works thereafter.
199    pub fn set_error(&self, err: Error) {
200        let mut inner = self
201            .inner
202            .lock()
203            .expect("Can't access mock data for set_error");
204        let data = inner.get_mut();
205        data.error = Some(err);
206    }
207}

209/// The builder for mock credentials.
210pub struct MockCredentialBuilder {}
211
212impl CredentialBuilderApi for MockCredentialBuilder {
213    /// Build a mock credential for the given target, service, and user.
214    ///
215    /// Since mocks don't persist between sessions,  all mocks
216    /// start off without passwords.
217    fn build(&self, target: Option<&str>, service: &str, user: &str) -> Result<Box<Credential>> {
218        let credential = MockCredential::new_with_target(target, service, user)?;
219        Ok(Box::new(credential))
220    }
221
222    /// Get an [Any][std::any::Any] reference to the mock credential builder.
223    fn as_any(&self) -> &dyn std::any::Any {
224        self
225    }
226
227    /// This keystore keeps the password in the entry!
228    fn persistence(&self) -> CredentialPersistence {
229        CredentialPersistence::EntryOnly
230    }
231}
232
233/// Return a mock credential builder for use by clients.
234pub fn default_credential_builder() -> Box<CredentialBuilder> {
235    Box::new(MockCredentialBuilder {})
236}

292    #[test]
293    fn test_set_error() {
294        let name = generate_random_string();
295        let entry = entry_new(&name, &name);
296        let password = "test ascii password";
297        let mock: &MockCredential = entry
298            .inner
299            .as_any()
300            .downcast_ref()
301            .expect("Downcast failed");
302        mock.set_error(Error::Invalid(
303            "mock error".to_string(),
304            "is an error".to_string(),
305        ));
306        assert!(
307            matches!(entry.set_password(password), Err(Error::Invalid(_, _))),
308            "set: No error"
309        );
310        entry
311            .set_password(password)
312            .expect("set: Error not cleared");
313        mock.set_error(Error::NoEntry);
314        assert!(
315            matches!(entry.get_password(), Err(Error::NoEntry)),
316            "get: No error"
317        );
318        let stored_password = entry.get_password().expect("get: Error not cleared");
319        assert_eq!(
320            stored_password, password,
321            "Retrieved and set ascii passwords don't match"
322        );
323        mock.set_error(Error::TooLong("mock".to_string(), 3));
324        assert!(
325            matches!(entry.delete_credential(), Err(Error::TooLong(_, 3))),
326            "delete: No error"
327        );
328        entry
329            .delete_credential()
330            .expect("delete: Error not cleared");
331        assert!(
332            matches!(entry.get_password(), Err(Error::NoEntry)),
333            "Able to read a deleted ascii password"
334        )
335    }
