Main database engine that integrates all components

Note: NeuroQuantumDB is intentionally not Clone. For shared access across multiple tasks/threads, wrap it in Arc<tokio::sync::RwLock<NeuroQuantumDB>>. This prevents accidental cloning of large internal data structures and ensures consistent cache state across all accessors.

ยงExample

use std::sync::Arc;
use tokio::sync::RwLock;
use neuroquantum_core::NeuroQuantumDBBuilder;

let db = NeuroQuantumDBBuilder::new().build().await?;
let shared_db = Arc::new(RwLock::new(db));

// Clone the Arc for sharing, not the database itself
let db_clone = shared_db.clone();
